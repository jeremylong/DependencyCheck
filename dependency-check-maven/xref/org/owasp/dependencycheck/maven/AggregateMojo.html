35  /**
36   * Maven Plugin that checks project dependencies and the dependencies of all
37   * child modules to see if they have any known published vulnerabilities.
38   *
39   * @author Jeremy Long
40   */
41  @Mojo(
42          name = "aggregate",
43          defaultPhase = LifecyclePhase.VERIFY,
44          aggregator = true,
45          threadSafe = true,
46          requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME,
47          requiresOnline = true
48  )
49  public class AggregateMojo extends BaseDependencyCheckMojo {
50  
51      /**
52       * The name of the report in the site.
53       */
54      @SuppressWarnings("CanBeFinal")
55      @Parameter(property = "name", defaultValue = "dependency-check:aggregate", required = true)
56      private String name = "dependency-check:aggregate";
57  
58      /**
59       * Scans the dependencies of the projects in aggregate.
60       *
61       * @param engine the engine used to perform the scanning
62       * @return a collection of exceptions
63       * @throws MojoExecutionException thrown if a fatal exception occurs
64       */
65      @Override
66      protected ExceptionCollection scanDependencies(final Engine engine) throws MojoExecutionException {
67          ExceptionCollection exCol = scanArtifacts(getProject(), engine, true);
68          for (MavenProject childProject : getDescendants(this.getProject())) {
69              //TODO consider the following as to whether a child should be skipped per #2152
70              //childProject.getBuildPlugins().get(0).getExecutions().get(0).getConfiguration()
71              final ExceptionCollection ex = scanArtifacts(childProject, engine, true);
72              if (ex != null) {
73                  if (exCol == null) {
74                      exCol = ex;
75                  }
76                  exCol.getExceptions().addAll(ex.getExceptions());
77                  if (ex.isFatal()) {
78                      exCol.setFatal(true);
79                      final String msg = String.format("Fatal exception(s) analyzing %s", childProject.getName());
80                      if (this.isFailOnError()) {
81                          throw new MojoExecutionException(msg, exCol);
82                      }
83                      getLog().error(msg);
84                      if (getLog().isDebugEnabled()) {
85                          getLog().debug(exCol);
86                      }
87                  }
88              }
89          }
90          return exCol;
91      }
92  
93      /**
94       * Returns a set containing all the descendant projects of the given
95       * project.
96       *
97       * @param project the project for which all descendants will be returned
98       * @return the set of descendant projects
99       */
100     protected Set<MavenProject> getDescendants(MavenProject project) {
101         if (project == null) {
102             return Collections.emptySet();
103         }
104         final Set<MavenProject> descendants = new HashSet<>();
105         int size;
106         if (getLog().isDebugEnabled()) {
107             getLog().debug(String.format("Collecting descendants of %s", project.getName()));
108         }
109         for (String m : project.getModules()) {
110             for (MavenProject mod : getReactorProjects()) {
111                 try {
112                     File mpp = new File(project.getBasedir(), m);
113                     mpp = mpp.getCanonicalFile();
114                     if (mpp.compareTo(mod.getBasedir()) == 0 && descendants.add(mod)
115                             && getLog().isDebugEnabled()) {
116                         getLog().debug(String.format("Descendant module %s added", mod.getName()));
117 
118                     }
119                 } catch (IOException ex) {
120                     if (getLog().isDebugEnabled()) {
121                         getLog().debug("Unable to determine module path", ex);
122                     }
123                 }
124             }
125         }
126         do {
127             size = descendants.size();
128             for (MavenProject p : getReactorProjects()) {
129                 if (project.equals(p.getParent()) || descendants.contains(p.getParent())) {
130                     if (descendants.add(p) && getLog().isDebugEnabled()) {
131                         getLog().debug(String.format("Descendant %s added", p.getName()));
132 
133                     }
134                     for (MavenProject modTest : getReactorProjects()) {
135                         if (p.getModules() != null && p.getModules().contains(modTest.getName())
136                                 && descendants.add(modTest)
137                                 && getLog().isDebugEnabled()) {
138                             getLog().debug(String.format("Descendant %s added", modTest.getName()));
139                         }
140                     }
141                 }
142                 final Set<MavenProject> addedDescendants = new HashSet<>();
143                 for (MavenProject dec : descendants) {
144                     for (String mod : dec.getModules()) {
145                         try {
146                             File mpp = new File(dec.getBasedir(), mod);
147                             mpp = mpp.getCanonicalFile();
148                             if (mpp.compareTo(p.getBasedir()) == 0) {
149                                 addedDescendants.add(p);
150                             }
151                         } catch (IOException ex) {
152                             if (getLog().isDebugEnabled()) {
153                                 getLog().debug("Unable to determine module path", ex);
154                             }
155                         }
156                     }
157                 }
158                 for (MavenProject addedDescendant : addedDescendants) {
159                     if (descendants.add(addedDescendant) && getLog().isDebugEnabled()) {
160                         getLog().debug(String.format("Descendant module %s added", addedDescendant.getName()));
161                     }
162                 }
163             }
164         } while (size != 0 && size != descendants.size());
165         if (getLog().isDebugEnabled()) {
166             getLog().debug(String.format("%s has %d children", project, descendants.size()));
167         }
168         return descendants;
169     }
170 
171     /**
172      * Test if the project has pom packaging
173      *
174      * @param mavenProject Project to test
175      * @return <code>true</code> if it has a pom packaging; otherwise
176      * <code>false</code>
177      */
178     protected boolean isMultiModule(MavenProject mavenProject) {
179         return "pom".equals(mavenProject.getPackaging());
180     }
181 
182     @Override
183     public boolean canGenerateReport() {
184         return true; //aggregate always returns true for now - we can look at a more complicated/accurate solution later
185     }
186 
187     /**
188      * Returns the report name.
189      *
190      * @param locale the location
191      * @return the report name
192      */
193     @Override
194     public String getName(Locale locale) {
195         return name;
196     }
197 
198     /**
199      * Gets the description of the Dependency-Check report to be displayed in
200      * the Maven Generated Reports page.
201      *
202      * @param locale The Locale to get the description for
203      * @return the description
204      */
205     @Override
206     public String getDescription(Locale locale) {
207         return "Generates an aggregate report of all child Maven projects providing details on any "
208                 + "published vulnerabilities within project dependencies. This report is a best "
209                 + "effort and may contain false positives and false negatives.";
210     }
