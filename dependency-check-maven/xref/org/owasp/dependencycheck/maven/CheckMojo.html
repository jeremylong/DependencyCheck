18  package org.owasp.dependencycheck.maven;
19  
20  import java.util.Locale;
21  import org.apache.maven.artifact.Artifact;
22  import org.apache.maven.plugin.MojoExecutionException;
23  import org.apache.maven.plugins.annotations.LifecyclePhase;
24  import org.apache.maven.plugins.annotations.Mojo;
25  import org.apache.maven.plugins.annotations.Parameter;
26  import org.apache.maven.plugins.annotations.ResolutionScope;
27  import org.owasp.dependencycheck.Engine;
28  import org.owasp.dependencycheck.exception.ExceptionCollection;
29  
30  /**
31   * Maven Plugin that checks the project dependencies to see if they have any
32   * known published vulnerabilities.
33   *
34   * @author Jeremy Long
35   */
36  @Mojo(
37          name = "check",
38          defaultPhase = LifecyclePhase.VERIFY,
39          threadSafe = true,
40          requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME,
41          requiresOnline = true
42  )
43  public class CheckMojo extends BaseDependencyCheckMojo {
44  
45      /**
46       * The name of the report in the site.
47       */
48      @SuppressWarnings("CanBeFinal")
49      @Parameter(property = "name", defaultValue = "dependency-check", required = true)
50      private String name = "dependency-check";
51  
52      /**
53       * Returns whether or not a the report can be generated.
54       *
55       * @return <code>true</code> if the report can be generated; otherwise
56       * <code>false</code>
57       */
58      @Override
59      public boolean canGenerateReport() {
60          populateSettings();
61          boolean isCapable = false;
62          for (Artifact a : getProject().getArtifacts()) {
63              if (!getArtifactScopeExcluded().passes(a.getScope())) {
64                  isCapable = true;
65                  break;
66              }
67          }
68          return isCapable;
69      }
70  
71      /**
72       * Returns the report name.
73       *
74       * @param locale the location
75       * @return the report name
76       */
77      @Override
78      public String getName(Locale locale) {
79          return name;
80      }
81  
82      /**
83       * Gets the description of the Dependency-Check report to be displayed in
84       * the Maven Generated Reports page.
85       *
86       * @param locale The Locale to get the description for
87       * @return the description
88       */
89      @Override
90      public String getDescription(Locale locale) {
91          return "Generates a report providing details on any published vulnerabilities within project dependencies. "
92                  + "This report is a best effort and may contain false positives and false negatives.";
93      }
94  
95      /**
96       * Scans the dependencies of the project.
97       *
98       * @param engine the engine used to perform the scanning
99       * @return a collection of exceptions
100      * @throws MojoExecutionException thrown if a fatal exception occurs
101      */
102     @Override
103     protected ExceptionCollection scanDependencies(final Engine engine) throws MojoExecutionException {
104         return scanArtifacts(getProject(), engine);
105     }
106 
107 }
