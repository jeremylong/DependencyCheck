public enum EcosystemHintNature {
    /**
     * Hint is from the file extension.
     */
    FILE_EXTENSION,
    /**
     * Hint is from a keyword.
     */
    KEYWORD,
    /**
     * Hint is from the host in a URL.
     */
    URL_HOST,
    /**
     * Hint is from the URL path.
     */
    URL_PATH
}
