35      /**
36       * Test of equals method, of class VulnerableSoftware.
37       *
38       * @throws CpeValidationException
39       */
40      @Test
41      public void testEquals() throws CpeValidationException {
42          VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
43          VulnerableSoftware obj = null;
44          VulnerableSoftware instance = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1").build();
45          assertFalse(instance.equals(obj));
46  
47          obj = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1.0").build();
48          instance = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1").build();
49          assertFalse(instance.equals(obj));
50  
51          obj = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1.0").build();
52          instance = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1.0").build();
53          assertTrue(instance.equals(obj));
54      }

56      /**
57       * Test of compareTo method, of class VulnerableSoftware.
58       *
59       * @throws CpeValidationException
60       */
61      @Test
62      public void testCompareTo() throws CpeValidationException {
63          VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
64          VulnerableSoftware obj = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1.0").build();
65          VulnerableSoftware instance = builder.part(Part.APPLICATION).vendor("mortbay").product("jetty").version("6.1").build();
66          int result = instance.compareTo(obj);
67          assertTrue(result < 0);
68  
69          obj = builder.part(Part.APPLICATION).vendor("yahoo").product("toolbar").version("3.1.0.20130813024103").build();
70          instance = builder.part(Part.APPLICATION).vendor("yahoo").product("toolbar").version("3.1.0.20130813024104").build();
71          result = instance.compareTo(obj);
72          assertTrue(result > 0);
73      }

75      @Test
76      public void testCompareVersionRange() throws CpeValidationException {
77          VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
78          VulnerableSoftware instance = builder.version("2.0.0").build();
79          assertTrue(instance.compareVersionRange("2.0.0"));
80          assertFalse(instance.compareVersionRange("2.0.1"));
81  
82          instance = builder.version(LogicalValue.ANY).build();
83          assertTrue(instance.compareVersionRange("2.0.1"));
84  
85          instance = builder.version(LogicalValue.NA).build();
86          assertFalse(instance.compareVersionRange("2.0.1"));
87  
88          instance = builder.version(LogicalValue.ANY).versionEndIncluding("2.0.1").build();
89          assertTrue(instance.compareVersionRange("2.0.1"));
90          assertFalse(instance.compareVersionRange("2.0.2"));
91  
92          instance = builder.version(LogicalValue.ANY).versionEndExcluding("2.0.2").build();
93          assertTrue(instance.compareVersionRange("2.0.1"));
94          assertFalse(instance.compareVersionRange("2.0.2"));
95  
96          instance = builder.version(LogicalValue.ANY).versionStartIncluding("1.0.1").build();
97          assertTrue(instance.compareVersionRange("1.0.1"));
98          assertFalse(instance.compareVersionRange("1.0.0"));
99  
100         instance = builder.version(LogicalValue.ANY).versionStartExcluding("1.0.0").build();
101         assertTrue(instance.compareVersionRange("1.0.1"));
102         assertFalse(instance.compareVersionRange("1.0.0"));
103     }
