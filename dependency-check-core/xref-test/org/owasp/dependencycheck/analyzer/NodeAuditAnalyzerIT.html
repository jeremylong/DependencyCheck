18      @Test
19      public void testAnalyzePackage() throws AnalysisException, InitializationException, InvalidSettingException {
20          Assume.assumeThat(getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED), is(true));
21          try (Engine engine = new Engine(getSettings())) {
22              NodeAuditAnalyzer analyzer = new NodeAuditAnalyzer();
23              analyzer.setFilesMatched(true);
24              analyzer.initialize(getSettings());
25              analyzer.prepare(engine);
26              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, "nodeaudit/package-lock.json"));
27              analyzer.analyze(toScan, engine);
28              boolean found = false;
29              assertTrue("Mpre then 1 dependency should be identified", 1 < engine.getDependencies().length);
30              for (Dependency result : engine.getDependencies()) {
31                  if ("package-lock.json?uglify-js".equals(result.getFileName())) {
32                      found = true;
33                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("uglify-js"));
34                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("uglify-js"));
35                      assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("2.4.24"));
36                      assertTrue(result.isVirtual());
37                  }
38              }
39              assertTrue("Uglify was not found", found);
40          }
41      }
