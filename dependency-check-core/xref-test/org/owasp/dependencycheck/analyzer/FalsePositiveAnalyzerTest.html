32  /**
33   *
34   * @author Jeremy Long
35   */
36  public class FalsePositiveAnalyzerTest extends BaseTest {
37  
38      /**
39       * A CPE builder object.
40       */
41      private final CpeBuilder builder = new CpeBuilder();
42  
43      /**
44       * Test of getName method, of class FalsePositiveAnalyzer.
45       */
46      @Test
47      public void testGetName() {
48          FalsePositiveAnalyzer instance = new FalsePositiveAnalyzer();
49          String expResult = "False Positive Analyzer";
50          String result = instance.getName();
51          assertEquals(expResult, result);
52      }
53  
54      /**
55       * Test of getAnalysisPhase method, of class FalsePositiveAnalyzer.
56       */
57      @Test
58      public void testGetAnalysisPhase() {
59          FalsePositiveAnalyzer instance = new FalsePositiveAnalyzer();
60          AnalysisPhase expResult = AnalysisPhase.POST_IDENTIFIER_ANALYSIS;
61          AnalysisPhase result = instance.getAnalysisPhase();
62          assertEquals(expResult, result);
63      }
64  
65      /**
66       * Test of getAnalyzerEnabledSettingKey method, of class
67       * FalsePositiveAnalyzer.
68       */
69      @Test
70      public void testGetAnalyzerEnabledSettingKey() {
71          FalsePositiveAnalyzer instance = new FalsePositiveAnalyzer();
72          String expResult = Settings.KEYS.ANALYZER_FALSE_POSITIVE_ENABLED;
73          String result = instance.getAnalyzerEnabledSettingKey();
74          assertEquals(expResult, result);
75      }
76  
77      /**
78       * Test of analyzeDependency method, of class FalsePositiveAnalyzer.
79       */
80      @Test
81      public void testAnalyzeDependency() throws Exception {
82          Dependency dependency = new Dependency();
83          dependency.setFileName("pom.xml");
84          dependency.setFilePath("pom.xml");
85          Cpe cpe = builder.part(Part.APPLICATION).vendor("file").product("file").version("1.2.1").build();
86          CpeIdentifier id = new CpeIdentifier(cpe, "http://some.org/url", Confidence.HIGHEST);
87          dependency.addVulnerableSoftwareIdentifier(id);
88          Engine engine = null;
89          FalsePositiveAnalyzer instance = new FalsePositiveAnalyzer();
90          int before = dependency.getVulnerableSoftwareIdentifiers().size();
91          instance.analyze(dependency, engine);
92          int after = dependency.getVulnerableSoftwareIdentifiers().size();
93          assertTrue(before > after);
94      }
95  
96      /**
97       * Test of removeBadMatches method, of class FalsePositiveAnalyzer.
98       */
99      @Test
100     public void testRemoveBadMatches() throws Exception {
101         Dependency dependency = new Dependency();
102         dependency.setFileName("some.jar");
103         dependency.setFilePath("some.jar");
104         Cpe cpe = builder.part(Part.APPLICATION).vendor("m-core").product("m-core").build();
105         CpeIdentifier id = new CpeIdentifier(cpe, Confidence.HIGHEST);
106         dependency.addVulnerableSoftwareIdentifier(id);
107 
108         assertEquals(1, dependency.getVulnerableSoftwareIdentifiers().size());
109 
110         FalsePositiveAnalyzer instance = new FalsePositiveAnalyzer();
111         instance.removeBadMatches(dependency);
112 
113         assertEquals(0, dependency.getVulnerableSoftwareIdentifiers().size());
114         dependency.addVulnerableSoftwareIdentifier(id);
115         dependency.addEvidence(EvidenceType.PRODUCT, "test", "name", "m-core", Confidence.HIGHEST);
116 
117         instance.removeBadMatches(dependency);
118         assertEquals(1, dependency.getVulnerableSoftwareIdentifiers().size());
119     }
120 }
