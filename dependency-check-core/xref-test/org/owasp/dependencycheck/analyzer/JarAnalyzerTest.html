45       * Test of inspect method, of class JarAnalyzer.
46       *
47       * @throws Exception is thrown when an exception occurs.

49      @Test
50      public void testAnalyze() throws Exception {
51          //File file = new File(this.getClass().getClassLoader().getResource("struts2-core-2.1.2.jar").getPath());
52          File file = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
53          Dependency result = new Dependency(file);
54          JarAnalyzer instance = new JarAnalyzer();
55          instance.initialize(getSettings());
56          instance.prepareFileTypeAnalyzer(null);
57          instance.analyze(result, null);
58          assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("apache"));
59          assertTrue(result.getVendorWeightings().contains("apache"));
60  
61          file = BaseTest.getResourceAsFile(this, "dwr.jar");
62          result = new Dependency(file);
63          instance.analyze(result, null);
64          assertEquals(JarAnalyzer.DEPENDENCY_ECOSYSTEM, result.getEcosystem());
65          boolean found = false;
66          for (Evidence e : result.getEvidence(EvidenceType.VENDOR)) {
67              if (e.getName().equals("url")) {
68                  assertEquals("Project url was not as expected in dwr.jar", "http://getahead.ltd.uk/dwr", e.getValue());
69                  found = true;
70                  break;
71              }
72          }
73          assertTrue("Project url was not found in dwr.jar", found);
74  
75          //file = new File(this.getClass().getClassLoader().getResource("org.mortbay.jetty.jar").getPath());
76          file = BaseTest.getResourceAsFile(this, "org.mortbay.jetty.jar");
77          result = new Dependency(file);
78          instance.analyze(result, null);
79          found = false;
80          for (Evidence e : result.getEvidence(EvidenceType.PRODUCT)) {
81              if (e.getName().equalsIgnoreCase("package-title")
82                      && e.getValue().equalsIgnoreCase("org.mortbay.http")) {
83                  found = true;
84                  break;
85              }
86          }
87          assertTrue("package-title of org.mortbay.http not found in org.mortbay.jetty.jar", found);
88  
89          found = false;
90          for (Evidence e : result.getEvidence(EvidenceType.VENDOR)) {
91              if (e.getName().equalsIgnoreCase("implementation-url")
92                      && e.getValue().equalsIgnoreCase("http://jetty.mortbay.org")) {
93                  found = true;
94                  break;
95              }
96          }
97          assertTrue("implementation-url of http://jetty.mortbay.org not found in org.mortbay.jetty.jar", found);
98  
99          found = false;
100         for (Evidence e : result.getEvidence(EvidenceType.VERSION)) {
101             if (e.getName().equalsIgnoreCase("Implementation-Version")
102                     && e.getValue().equalsIgnoreCase("4.2.27")) {
103                 found = true;
104                 break;
105             }
106         }
107         assertTrue("implementation-version of 4.2.27 not found in org.mortbay.jetty.jar", found);
108 
109         //file = new File(this.getClass().getClassLoader().getResource("org.mortbay.jmx.jar").getPath());
110         file = BaseTest.getResourceAsFile(this, "org.mortbay.jmx.jar");
111         result = new Dependency(file);
112         instance.analyze(result, null);
113         assertEquals("org.mortbar.jmx.jar has version evidence?", 0, result.getEvidence(EvidenceType.VERSION).size());
114     }

116     @Test
117     public void testAddMatchingValues() throws Exception {
118         File file = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
119         Dependency dependency = new Dependency(file);
120         JarAnalyzer instance = new JarAnalyzer();
121         instance.initialize(getSettings());
122         instance.prepareFileTypeAnalyzer(null);
123 
124         final List<JarAnalyzer.ClassNameInformation> classNames = instance.collectClassNames(dependency);
125 
126         JarAnalyzer.addMatchingValues(classNames, "thevelocity", dependency, EvidenceType.VENDOR);
127         JarAnalyzer.addMatchingValues(classNames, "freemarkercore", dependency, EvidenceType.VENDOR);
128         JarAnalyzer.addMatchingValues(classNames, "the defaultpropertiesprovidertest", dependency, EvidenceType.VENDOR);
129         JarAnalyzer.addMatchingValues(classNames, "thedefaultpropertiesprovider test", dependency, EvidenceType.VENDOR);
130         JarAnalyzer.addMatchingValues(classNames, "thedefaultpropertiesprovidertest", dependency, EvidenceType.VENDOR);
131 
132         assertFalse(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("velocity"));
133         assertFalse(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("freemarker"));
134         assertFalse(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("defaultpropertiesprovider"));
135 
136         JarAnalyzer.addMatchingValues(classNames, "strutsexception", dependency, EvidenceType.VENDOR);
137         JarAnalyzer.addMatchingValues(classNames, "the velocity", dependency, EvidenceType.VENDOR);
138         JarAnalyzer.addMatchingValues(classNames, "freemarker core", dependency, EvidenceType.VENDOR);
139         JarAnalyzer.addMatchingValues(classNames, "the defaultpropertiesprovider test", dependency, EvidenceType.VENDOR);
140 
141         assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("strutsexception"));
142         assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("velocity"));
143         assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("freemarker"));
144         assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("defaultpropertiesprovider"));
145     }

215     @Test
216     public void testAnalyzeDependency_SkipsMacOSMetaDataFile() throws Exception {
217         JarAnalyzer instance = new JarAnalyzer();
218         Dependency macOSMetaDataFile = new Dependency();
219         macOSMetaDataFile
220                 .setActualFilePath(FileUtils.getFile("src", "test", "resources", "._avro-ipc-1.5.0.jar").getAbsolutePath());
221         macOSMetaDataFile.setFileName("._avro-ipc-1.5.0.jar");
222         Dependency actualJarFile = new Dependency();
223         actualJarFile.setActualFilePath(BaseTest.getResourceAsFile(this, "avro-ipc-1.5.0.jar").getAbsolutePath());
224         actualJarFile.setFileName("avro-ipc-1.5.0.jar");
225         try (Engine engine = new Engine(getSettings())) {
226             engine.setDependencies(Arrays.asList(macOSMetaDataFile, actualJarFile));
227             instance.analyzeDependency(macOSMetaDataFile, engine);
228             assertEquals(1, engine.getDependencies().length);
229         }
