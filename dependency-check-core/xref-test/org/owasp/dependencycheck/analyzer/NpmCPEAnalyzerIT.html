40      @Test
41      public void testAnalyzeDependency() throws Exception {
42  
43          NpmCPEAnalyzer instance = new NpmCPEAnalyzer();
44          try (Engine engine = new Engine(getSettings())) {
45              // Opening the database in read-only mode always copies the whole database -> open it only once
46              engine.openDatabase(true, true);
47              instance.initialize(getSettings());
48              instance.prepare(engine);
49  
50              //callAnalyzeDependency("negotiator", "negotiator", "0.3.0", "cpe:2.3:a:negotiator_project:negotiator:0.3.0:*:*:*:*:*:*:*", instance, engine);
51              callAnalyzeDependency("mime", "mime", "1.2.11", "cpe:2.3:a:mime_project:mime:1.2.11:*:*:*:*:*:*:*", instance, engine);
52  
53              instance.close();
54          }
55      }

57      /**
58       * Executes the test call against AnalyzeDependency.
59       *
60       * @param vendor
61       * @param product
62       * @param version
63       * @param expectedCpe
64       * @param cpeAnalyzer
65       * @param engine
66       * @throws Exception
67       */
68      private void callAnalyzeDependency(String vendor, String product, String version, String expectedCpe, NpmCPEAnalyzer cpeAnalyzer, Engine engine) throws Exception {
69          Dependency dep = new Dependency(true);
70          dep.addEvidence(EvidenceType.VENDOR, "test", "vendor", vendor, Confidence.HIGHEST);
71          dep.addEvidence(EvidenceType.PRODUCT, "test", "product", product, Confidence.HIGHEST);
72          dep.addEvidence(EvidenceType.VERSION, "test", "version", version, Confidence.HIGHEST);
73          dep.setVersion(version);
74          dep.setEcosystem("npm");
75          dep.setMd5sum("");
76          dep.setSha1sum("");
77          dep.setSha256sum("");
78  
79          cpeAnalyzer.analyzeDependency(dep, engine);
80  
81          boolean found = dep.getVulnerableSoftwareIdentifiers().stream().anyMatch(id -> {
82              System.out.println(id.getValue());
83              return expectedCpe.equals(id.getValue());
84          });
85          assertTrue(String.format("%s:%s:%s identifier not found", vendor, product, version), found);
86      }
