46  /**
47   * Tests for the CentralAnalyzer.
48   */
49  public class CentralAnalyzerTest extends BaseTest {
50  
51      private static final String SHA1_SUM = "my-sha1-sum";
52      final CentralSearch centralSearch = mock(CentralSearch.class);
53      final Dependency dependency = mock(Dependency.class);
54  
55      @Test
56      @SuppressWarnings("PMD.NonStaticInitializer")
57      public void testFetchMavenArtifactsWithoutException() throws IOException, TooManyRequestsException {
58              CentralAnalyzer instance = new CentralAnalyzer();
59              instance.setCentralSearch(centralSearch);
60              when(dependency.getSha1sum()).thenReturn(SHA1_SUM);
61              when(centralSearch.searchSha1(SHA1_SUM)).thenReturn(Collections.emptyList());
62  
63              final List<MavenArtifact> actualMavenArtifacts = instance.fetchMavenArtifacts(dependency);
64  
65              assertTrue(actualMavenArtifacts.isEmpty());
66      }
67  
68      @Test(expected = FileNotFoundException.class)
69      @SuppressWarnings("PMD.NonStaticInitializer")
70      public void testFetchMavenArtifactsRethrowsFileNotFoundException()
71              throws IOException, TooManyRequestsException {
72          CentralAnalyzer instance = new CentralAnalyzer();
73          instance.setCentralSearch(centralSearch);
74          when(dependency.getSha1sum()).thenReturn(SHA1_SUM);
75          when(centralSearch.searchSha1(SHA1_SUM)).thenThrow(FileNotFoundException.class);
76          instance.fetchMavenArtifacts(dependency);
77      }
78  
79      @Test(expected = IOException.class)
80      @SuppressWarnings("PMD.NonStaticInitializer")
81      public void testFetchMavenArtifactsAlwaysThrowsIOException()
82              throws IOException, TooManyRequestsException {
83          getSettings().setInt(Settings.KEYS.ANALYZER_CENTRAL_RETRY_COUNT, 1);
84          getSettings().setBoolean(Settings.KEYS.ANALYZER_CENTRAL_USE_CACHE, false);
85          CentralAnalyzer instance = new CentralAnalyzer();
86          instance.initialize(getSettings());
87          
88          instance.setCentralSearch(centralSearch);
89          when(dependency.getSha1sum()).thenReturn(SHA1_SUM);
90          when(centralSearch.searchSha1(SHA1_SUM)).thenThrow(IOException.class);
91  
92          instance.fetchMavenArtifacts(dependency);
93      }
94  }
