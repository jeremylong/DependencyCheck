30  import static org.junit.Assert.assertTrue;
31  
32  public class YarnAuditAnalyzerIT extends BaseTest {
33  
34      @Test
35      public void testAnalyzePackageYarnClassic() throws AnalysisException, InitializationException, InvalidSettingException {
36          testAnalyzePackageYarn("yarn-classic-audit/yarn.lock");
37      }
38  
39      @Test
40      public void testAnalyzePackageYarnBerry() throws AnalysisException, InitializationException, InvalidSettingException {
41          testAnalyzePackageYarn("yarn-berry-audit/yarn.lock");
42      }
43  
44      private void testAnalyzePackageYarn(String yarnLockFile) throws AnalysisException {
45          //Assume.assumeThat(getSettings().getBoolean(Settings.KEYS.ANALYZER_YARN_AUDIT_ENABLED), is(true));
46          try (Engine engine = new Engine(getSettings())) {
47              var analyzer = new YarnAuditAnalyzer();
48              analyzer.setFilesMatched(true);
49              analyzer.initialize(getSettings());
50              analyzer.prepare(engine);
51              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, yarnLockFile));
52              analyzer.analyze(toScan, engine);
53              boolean found = false;
54              assertTrue("More then 1 dependency should be identified", 1 < engine.getDependencies().length);
55              for (Dependency result : engine.getDependencies()) {
56                  if ("yarn.lock?uglify-js".equals(result.getFileName())) {
57                      found = true;
58                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("uglify-js"));
59                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("uglify-js"));
60                      assertTrue("Unable to find version 2.4.24: " + result.getEvidence(EvidenceType.VERSION).toString(), result.getEvidence(EvidenceType.VERSION).toString().contains("2.4.24"));
61                      assertTrue(result.isVirtual());
62                  }
63              }
64              assertTrue("Uglify was not found", found);
65          } catch (InitializationException ex) {
66              //yarn is not installed - skip the test case.
67              Assume.assumeNoException(ex);
68          }
69      }
70  }
