18  package org.owasp.dependencycheck.analyzer;
19  
20  import org.junit.Assume;
21  import org.junit.Test;
22  import org.owasp.dependencycheck.BaseTest;
23  import org.owasp.dependencycheck.Engine;
24  import org.owasp.dependencycheck.analyzer.exception.AnalysisException;
25  import org.owasp.dependencycheck.dependency.Dependency;
26  import org.owasp.dependencycheck.dependency.EvidenceType;
27  import org.owasp.dependencycheck.exception.InitializationException;
28  import org.owasp.dependencycheck.utils.InvalidSettingException;
29  import static org.junit.Assert.assertTrue;
30  
31  public class YarnAuditAnalyzerIT extends BaseTest {
32  
33      @Test
34      public void testAnalyzePackageYarn() throws AnalysisException, InitializationException, InvalidSettingException {
35  
36          //Assume.assumeThat(getSettings().getBoolean(Settings.KEYS.ANALYZER_YARN_AUDIT_ENABLED), is(true));
37          try (Engine engine = new Engine(getSettings())) {
38              YarnAuditAnalyzer analyzer = new YarnAuditAnalyzer();
39              analyzer.setFilesMatched(true);
40              analyzer.initialize(getSettings());
41              analyzer.prepare(engine);
42              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, "yarnaudit/yarn.lock"));
43              analyzer.analyze(toScan, engine);
44              boolean found = false;
45              assertTrue("More then 1 dependency should be identified", 1 < engine.getDependencies().length);
46              for (Dependency result : engine.getDependencies()) {
47                  if ("yarn.lock?uglify-js".equals(result.getFileName())) {
48                      found = true;
49                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("uglify-js"));
50                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("uglify-js"));
51                      assertTrue("Unable to find version 2.4.24: " + result.getEvidence(EvidenceType.VERSION).toString(), result.getEvidence(EvidenceType.VERSION).toString().contains("2.4.24"));
52                      assertTrue(result.isVirtual());
53                  }
54              }
55              assertTrue("Uglify was not found", found);
56          } catch (InitializationException ex) {
57              //yarn is not installed - skip the test case.
58              Assume.assumeNoException(ex);
59          }
60      }
61  }
