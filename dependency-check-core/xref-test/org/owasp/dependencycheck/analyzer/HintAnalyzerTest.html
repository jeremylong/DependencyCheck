62      /**
63       * Test of analyze method, of class HintAnalyzer.
64       */
65      @Test
66      public void testAnalyze() throws Exception {
67          //File guice = new File(this.getClass().getClassLoader().getResource("guice-3.0.jar").getPath());
68          File guice = BaseTest.getResourceAsFile(this, "guice-3.0.jar");
69          //Dependency guice = new EngineDependency(fileg);
70          //File spring = new File(this.getClass().getClassLoader().getResource("spring-core-3.0.0.RELEASE.jar").getPath());
71          File spring = BaseTest.getResourceAsFile(this, "spring-core-3.0.0.RELEASE.jar");
72          //Dependency spring = new Dependency(files);
73          getSettings().setBoolean(Settings.KEYS.AUTO_UPDATE, false);
74          getSettings().setBoolean(Settings.KEYS.ANALYZER_NEXUS_ENABLED, false);
75          getSettings().setBoolean(Settings.KEYS.ANALYZER_CENTRAL_ENABLED, false);
76          try (Engine engine = new Engine(getSettings())) {
77              engine.scan(guice);
78              engine.scan(spring);
79              engine.analyzeDependencies();
80              Dependency gdep = null;
81              Dependency sdep = null;
82              for (Dependency d : engine.getDependencies()) {
83                  if (d.getActualFile().equals(guice)) {
84                      gdep = d;
85                  } else if (d.getActualFile().equals(spring)) {
86                      sdep = d;
87                  }
88              }
89              final Evidence springTest1 = new Evidence("hint analyzer", "product", "springsource_spring_framework", Confidence.HIGHEST, true);
90              final Evidence springTest2 = new Evidence("hint analyzer", "vendor", "SpringSource", Confidence.HIGHEST, true);
91              final Evidence springTest3 = new Evidence("hint analyzer", "vendor", "vmware", Confidence.HIGHEST, true);
92              final Evidence springTest4 = new Evidence("hint analyzer", "product", "springsource_spring_framework", Confidence.HIGHEST, true);
93              final Evidence springTest5 = new Evidence("hint analyzer", "vendor", "vmware", Confidence.HIGHEST, true);
94  
95              assertFalse(gdep.contains(EvidenceType.PRODUCT, springTest1));
96              assertFalse(gdep.contains(EvidenceType.VENDOR, springTest2));
97              assertFalse(gdep.contains(EvidenceType.VENDOR, springTest3));
98              assertFalse(gdep.contains(EvidenceType.PRODUCT, springTest4));
99              assertFalse(gdep.contains(EvidenceType.VENDOR, springTest5));
100 
101             assertTrue(sdep.contains(EvidenceType.PRODUCT, springTest1));
102             assertTrue(sdep.contains(EvidenceType.VENDOR, springTest2));
103             assertTrue(sdep.contains(EvidenceType.VENDOR, springTest3));
104             //assertTrue(evidence.contains(springTest4));
105             //assertTrue(evidence.contains(springTest5));
106         }
107     }

109     /**
110      * Test of analyze method, of class HintAnalyzer.
111      */
112     @Test
113     public void testAnalyze_1() throws Exception {
114         File path = BaseTest.getResourceAsFile(this, "hints_12.xml");
115         getSettings().setString(Settings.KEYS.HINTS_FILE, path.getPath());
116         HintAnalyzer instance = new HintAnalyzer();
117         instance.initialize(getSettings());
118         instance.prepare(null);
119         Dependency d = new Dependency();
120         d.addEvidence(EvidenceType.VERSION, "version source", "given version name", "1.2.3", Confidence.HIGH);
121         d.addEvidence(EvidenceType.VERSION, "hint analyzer", "remove version name", "value", Confidence.HIGH);
122         d.addEvidence(EvidenceType.VENDOR, "hint analyzer", "remove vendor name", "vendor", Confidence.HIGH);
123         d.addEvidence(EvidenceType.PRODUCT, "hint analyzer", "remove product name", "product", Confidence.HIGH);
124         d.addEvidence(EvidenceType.VERSION, "hint analyzer", "other version name", "value", Confidence.HIGH);
125         d.addEvidence(EvidenceType.VENDOR, "hint analyzer", "other vendor name", "vendor", Confidence.HIGH);
126         d.addEvidence(EvidenceType.PRODUCT, "hint analyzer", "other product name", "product", Confidence.HIGH);
127 
128         assertEquals("vendor evidence mismatch", 2, d.getEvidence(EvidenceType.VENDOR).size());
129         assertEquals("product evidence mismatch", 2, d.getEvidence(EvidenceType.PRODUCT).size());
130         assertEquals("version evidence mismatch", 3, d.getEvidence(EvidenceType.VERSION).size());
131         instance.analyze(d, null);
132         assertEquals("vendor evidence mismatch", 1, d.getEvidence(EvidenceType.VENDOR).size());
133         assertEquals("product evidence mismatch", 1, d.getEvidence(EvidenceType.PRODUCT).size());
134         assertEquals("version evidence mismatch", 2, d.getEvidence(EvidenceType.VERSION).size());
135 
136     }
