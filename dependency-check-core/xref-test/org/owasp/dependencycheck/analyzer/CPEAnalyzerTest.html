75      @Test
76      public void testAddEvidenceWithoutDuplicateTerms() {
77          Map<String, MutableInt> terms = new HashMap<>();
78          List<Evidence> evidence = new ArrayList<>();
79          evidence.add(new Evidence("test case", "value", "test", Confidence.HIGHEST));
80          CPEAnalyzer instance = new CPEAnalyzer();
81          String expResult = "test";
82          int expValue = 1;
83          instance.collectTerms(terms, evidence);
84          assertTrue(terms.containsKey(expResult));
85          assertEquals(expValue, terms.get(expResult).intValue());
86  
87          evidence = new ArrayList<>();
88          evidence.add(new Evidence("test case", "value", "other", Confidence.HIGHEST));
89          instance.collectTerms(terms, evidence);
90          expResult = "other";
91          expValue = 1;
92          assertTrue(terms.containsKey(expResult));
93          assertEquals(expValue, terms.get(expResult).intValue());
94  
95          evidence.clear();
96          evidence.add(new Evidence("test case", "value", "other", Confidence.HIGHEST));
97          instance.collectTerms(terms, evidence);
98          expResult = "other";
99          expValue = 2;
100         assertTrue(terms.containsKey(expResult));
101         assertEquals(expValue, terms.get(expResult).intValue());
102 
103         StringBuilder sb = new StringBuilder();
104         StringBuilder expect = new StringBuilder();
105         for (int x = 0; x < 500; x++) {
106             sb.append("items ");
107             if (expect.length() + 5 < 1000) {
108                 expect.append("items ");
109             }
110         }
111         evidence.clear();
112         terms.clear();
113         evidence.add(new Evidence("test case", "value", sb.toString(), Confidence.HIGHEST));
114         instance.collectTerms(terms, evidence);
115         expResult = expect.toString().trim();
116         expValue = 1;
117         assertTrue(terms.containsKey(expResult));
118         assertEquals(expValue, terms.get(expResult).intValue());
119 
120         sb = new StringBuilder();
121         expect = new StringBuilder();
122         for (int x = 0; x < 500; x++) {
123             sb.append("items.");
124             if (expect.length() + 5 < 1000) {
125                 expect.append("items.");
126             }
127         }
128         evidence.clear();
129         terms.clear();
130         evidence.add(new Evidence("test case", "value", sb.toString(), Confidence.HIGHEST));
131         instance.collectTerms(terms, evidence);
132         expect.setLength(expect.length() - 1);
133         expResult = expect.toString();
134         expValue = 1;
135         assertTrue(terms.containsKey(expResult));
136         assertEquals(expValue, terms.get(expResult).intValue());
137 
138         sb = new StringBuilder();
139         expect = new StringBuilder();
140         for (int x = 0; x < 500; x++) {
141             sb.append("items-");
142             if (expect.length() + 5 < 1000) {
143                 expect.append("items-");
144             }
145         }
146         evidence.clear();
147         terms.clear();
148         evidence.add(new Evidence("test case", "value", sb.toString(), Confidence.HIGHEST));
149         instance.collectTerms(terms, evidence);
150         expect.setLength(expect.length() - 1);
151         expResult = expect.toString();
152         expValue = 1;
153         assertTrue(terms.containsKey(expResult));
154         assertEquals(expValue, terms.get(expResult).intValue());
155 
156         sb = new StringBuilder();
157         expect = new StringBuilder();
158         for (int x = 0; x < 500; x++) {
159             sb.append("items_");
160             if (expect.length() + 5 < 1000) {
161                 expect.append("items_");
162             }
163         }
164         evidence.clear();
165         terms.clear();
166         evidence.add(new Evidence("test case", "value", sb.toString(), Confidence.HIGHEST));
167         instance.collectTerms(terms, evidence);
168         expect.setLength(expect.length() - 1);
169         expResult = expect.toString();
170         expValue = 1;
171         assertTrue(terms.containsKey(expResult));
172         assertEquals(expValue, terms.get(expResult).intValue());
173 
174         sb = new StringBuilder();
175         expect = new StringBuilder();
176         for (int x = 0; x < 500; x++) {
177             sb.append("items/");
178             if (expect.length() + 5 < 1000) {
179                 expect.append("items/");
180             }
181         }
182         evidence.clear();
183         terms.clear();
184         evidence.add(new Evidence("test case", "value", sb.toString(), Confidence.HIGHEST));
185         instance.collectTerms(terms, evidence);
186         expect.setLength(expect.length() - 1);
187         expResult = expect.toString();
188         expValue = 1;
189         assertTrue(terms.containsKey(expResult));
190         assertEquals(expValue, terms.get(expResult).intValue());
191     }

203     /**
204      * Test of buildSearch method, of class CPEAnalyzer.
205      */
206     @Test
207     public void testBuildSearch() {
208         Map<String, MutableInt> vendor = new HashMap<>();
209         Map<String, MutableInt> product = new HashMap<>();
210         vendor.put("apache software foundation", new MutableInt(1));
211         product.put("lucene index", new MutableInt(1));
212         Set<String> vendorWeighting = new HashSet<>();
213         Set<String> productWeightings = new HashSet<>();
214 
215         CPEAnalyzer instance = new CPEAnalyzer();
216         String expResult = "product:(lucene index) AND vendor:(apache software foundation)";
217         String result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
218         assertEquals(expResult, result);
219 
220         vendorWeighting.add("apache");
221         expResult = "product:(lucene index) AND vendor:(apache^2 software foundation)";
222         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
223         assertEquals(expResult, result);
224 
225         productWeightings.add("lucene");
226         expResult = "product:(lucene^2 index) AND vendor:(apache^2 software foundation)";
227         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
228         assertEquals(expResult, result);
229 
230         productWeightings.add("ignored");
231         expResult = "product:(lucene^2 index) AND vendor:(apache^2 software foundation)";
232         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
233         assertEquals(expResult, result);
234 
235         vendorWeighting.clear();
236         expResult = "product:(lucene^2 index) AND vendor:(apache software foundation)";
237         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
238         assertEquals(expResult, result);
239 
240         vendorWeighting.add("ignored");
241         productWeightings.clear();
242         expResult = "product:(lucene index) AND vendor:(apache software foundation)";
243         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
244         assertEquals(expResult, result);
245 
246         vendor.put("apache software foundation", new MutableInt(3));
247         product.put("lucene index", new MutableInt(2));
248 
249         expResult = "product:(lucene^2 index^2) AND vendor:(apache^3 software^3 foundation^3)";
250         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
251         assertEquals(expResult, result);
252 
253         vendorWeighting.add("apache");
254         expResult = "product:(lucene^2 index^2) AND vendor:(apache^4 software^3 foundation^3)";
255         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
256         assertEquals(expResult, result);
257 
258         productWeightings.add("lucene");
259         expResult = "product:(lucene^3 index^2) AND vendor:(apache^4 software^3 foundation^3)";
260         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
261         assertEquals(expResult, result);
262 
263         productWeightings.clear();
264         productWeightings.add("lucene2");
265         expResult = "product:(lucene^3 index^2 lucene2^3) AND vendor:(apache^4 software^3 foundation^3)";
266         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
267         assertEquals(expResult, result);
268 
269         vendor.put("apache software foundation", new MutableInt(1));
270         vendor.put("apache", new MutableInt(2));
271         product.put("lucene index", new MutableInt(1));
272         product.put("lucene", new MutableInt(2));
273         vendorWeighting.clear();
274         productWeightings.clear();
275         productWeightings.add("lucene2");
276         expResult = "product:(lucene^3 lucene2^3 lucene^2 index lucene2^2) AND vendor:(apache^2 apache software foundation)";
277         result = instance.buildSearch(vendor, product, vendorWeighting, productWeightings);
278         assertEquals(expResult, result);
279     }
