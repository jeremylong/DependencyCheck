65      /**
66       * Test of analyze method, of class DependencyBundlingAnalyzer. The actually
67       * passed dependency does not matter. The analyzer only runs once.
68       */
69      @Test
70      public void testAnalyze() throws Exception {
71          DependencyBundlingAnalyzer instance = new DependencyBundlingAnalyzer();
72  
73          // the actual dependency does not matter
74          assertFalse(instance.getAnalyzed());
75          instance.analyze(null, engineMock);
76  
77          // the second runs basically does nothing
78          assertTrue(instance.getAnalyzed());
79          instance.analyze(null, engineMock);
80          instance.analyze(null, engineMock);
81          instance.analyze(null, engineMock);
82          assertTrue(instance.getAnalyzed());
83  
84          new Verifications() {
85              {
86                  engineMock.getDependencies();
87                  times = 1;
88              }
89          };
90      }

155     @Test
156     public void testIsShaded() throws MalformedPackageURLException {
157         DependencyBundlingAnalyzer instance = new DependencyBundlingAnalyzer();
158 
159         Dependency left = null;
160         Dependency right = null;
161 
162         boolean expResult = false;
163         boolean result = instance.isShadedJar(left, right);
164         assertEquals(expResult, result);
165 
166         left = new Dependency();
167         expResult = false;
168         result = instance.isShadedJar(left, right);
169         assertEquals(expResult, result);
170 
171         left = new Dependency(new File("/path/jar.jar"), true);
172         expResult = false;
173         result = instance.isShadedJar(left, right);
174         assertEquals(expResult, result);
175 
176         right = new Dependency();
177         expResult = false;
178         result = instance.isShadedJar(left, right);
179         assertEquals(expResult, result);
180 
181         right = new Dependency(new File("/path/pom.xml"), true);
182         expResult = false;
183         result = instance.isShadedJar(left, right);
184         assertEquals(expResult, result);
185 
186         left.addSoftwareIdentifier(new PurlIdentifier("maven", "test", "test", "1.0", Confidence.HIGHEST));
187         expResult = false;
188         result = instance.isShadedJar(left, right);
189         assertEquals(expResult, result);
190 
191         right.addSoftwareIdentifier(new PurlIdentifier("maven", "next", "next", "1.0", Confidence.HIGHEST));
192         expResult = false;
193         result = instance.isShadedJar(left, right);
194         assertEquals(expResult, result);
195 
196         left.addSoftwareIdentifier(new PurlIdentifier("maven", "next", "next", "1.0", Confidence.HIGHEST));
197         expResult = true;
198         result = instance.isShadedJar(left, right);
199         assertEquals(expResult, result);
200 
201         left = new Dependency(new File("/path/pom.xml"), true);
202         left.addSoftwareIdentifier(new PurlIdentifier("maven", "test", "test", "1.0", Confidence.HIGHEST));
203         right = new Dependency(new File("/path/jar.jar"), true);
204         right.addSoftwareIdentifier(new PurlIdentifier("maven", "next", "next", "1.0", Confidence.HIGHEST));
205         expResult = false;
206         result = instance.isShadedJar(left, right);
207         assertEquals(expResult, result);
208 
209         right.addSoftwareIdentifier(new PurlIdentifier("maven", "test", "test", "1.0", Confidence.HIGHEST));
210         expResult = true;
211         result = instance.isShadedJar(left, right);
212         assertEquals(expResult, result);
213 
214         left = new Dependency(new File("/path/other.jar"), true);
215         left.addSoftwareIdentifier(new PurlIdentifier("maven", "test", "test", "1.0", Confidence.HIGHEST));
216         right = new Dependency(new File("/path/jar.jar"), true);
217         right.addSoftwareIdentifier(new PurlIdentifier("maven", "next", "next", "1.0", Confidence.HIGHEST));
218         expResult = false;
219         result = instance.isShadedJar(left, right);
220         assertEquals(expResult, result);
221     }

224     public void testIsWebJar() throws MalformedPackageURLException {
225         DependencyBundlingAnalyzer instance = new DependencyBundlingAnalyzer();
226 
227         Dependency left = null;
228         Dependency right = null;
229 
230         boolean expResult = false;
231         boolean result = instance.isWebJar(left, right);
232         assertEquals(expResult, result);
233 
234         left = new Dependency();
235         expResult = false;
236         result = instance.isWebJar(left, right);
237         assertEquals(expResult, result);
238 
239         left = new Dependency(new File("/path/jquery.jar"), true);
240         expResult = false;
241         result = instance.isWebJar(left, right);
242         assertEquals(expResult, result);
243 
244         right = new Dependency();
245         expResult = false;
246         result = instance.isWebJar(left, right);
247         assertEquals(expResult, result);
248 
249         right = new Dependency(new File("/path/jquery.js"), true);
250         expResult = false;
251         result = instance.isWebJar(left, right);
252         assertEquals(expResult, result);
253 
254         right = new Dependency(new File("/path/jquery.js"), true);
255         right.setFileName("jquery.jar: jquery.js");
256         expResult = false;
257         result = instance.isWebJar(left, right);
258         assertEquals(expResult, result);
259 
260         left.addSoftwareIdentifier(new PurlIdentifier("maven", "org.webjars", "jquery", "1.0", Confidence.HIGHEST));
261         expResult = false;
262         result = instance.isWebJar(left, right);
263         assertEquals(expResult, result);
264 
265         right.addSoftwareIdentifier(new PurlIdentifier("javascript", "bootstrap", "1.0", Confidence.HIGHEST));
266         expResult = false;
267         result = instance.isWebJar(left, right);
268         assertEquals(expResult, result);
269 
270         right = new Dependency(new File("/path/jquery.js"), true);
271         right.setFileName("jquery.jar: jquery.js");
272         right.addSoftwareIdentifier(new PurlIdentifier("javascript", "jquery", "1.0", Confidence.HIGHEST));
273         expResult = true;
274         result = instance.isWebJar(left, right);
275         assertEquals(expResult, result);
276         
277         
278         left = new Dependency(new File("/path/spring-core.jar"), true);
279         left.addSoftwareIdentifier(new PurlIdentifier("maven", "org.springframework", "spring-core", "3.0.0", Confidence.HIGHEST));
280         expResult = false;
281         result = instance.isWebJar(left, right);
282         assertEquals(expResult, result);
283     }
