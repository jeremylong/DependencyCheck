19  @RunWith(MockitoJUnitRunner.class)
20  public class AnalysisTaskTest extends BaseTest {
21  
22      @Mock
23      private FileTypeAnalyzer fileTypeAnalyzer;
24  
25      @Mock
26      private Dependency dependency;
27  
28      @Mock
29      private Engine engine;
30  
31  
32      @Test
33      public void shouldAnalyzeReturnsTrueForNonFileTypeAnalyzers() {
34          AnalysisTask instance = new AnalysisTask(new HintAnalyzer(), null, null, null);
35          boolean shouldAnalyze = instance.shouldAnalyze();
36          assertTrue(shouldAnalyze);
37      }
38  
39      @Test
40      public void shouldAnalyzeReturnsTrueIfTheFileTypeAnalyzersAcceptsTheDependency() {
41          final File dependencyFile = new File("");
42          when(dependency.getActualFile()).thenReturn(dependencyFile);
43          when(fileTypeAnalyzer.accept(dependencyFile)).thenReturn(true);
44  
45          AnalysisTask analysisTask = new AnalysisTask(fileTypeAnalyzer, dependency, null, null);
46  
47          boolean shouldAnalyze = analysisTask.shouldAnalyze();
48          assertTrue(shouldAnalyze);
49      }
50  
51      @Test
52      public void shouldAnalyzeReturnsFalseIfTheFileTypeAnalyzerDoesNotAcceptTheDependency() {
53          final File dependencyFile = new File("");
54          when(dependency.getActualFile()).thenReturn(dependencyFile);
55          when(fileTypeAnalyzer.accept(dependencyFile)).thenReturn(false);
56  
57          AnalysisTask analysisTask = new AnalysisTask(fileTypeAnalyzer, dependency, null, null);
58  
59          boolean shouldAnalyze = analysisTask.shouldAnalyze();
60          assertFalse(shouldAnalyze);
61      }
62  
63      @Test
64      public void taskAnalyzes() throws Exception {
65          final AnalysisTask analysisTask = new AnalysisTask(fileTypeAnalyzer, dependency, engine, null);
66          when(fileTypeAnalyzer.accept(dependency.getActualFile())).thenReturn(true);
67  
68          analysisTask.call();
69  
70          verify(fileTypeAnalyzer, times(1)).analyze(dependency, engine);
71      }
72  
73      @Test
74      public void taskDoesNothingIfItShouldNotAnalyze() throws Exception {
75          final AnalysisTask analysisTask = new AnalysisTask(fileTypeAnalyzer, dependency, engine, null);
76          when(fileTypeAnalyzer.accept(dependency.getActualFile())).thenReturn(false);
77  
78          analysisTask.call();
79  
80          verify(fileTypeAnalyzer, times(0)).analyze(dependency, engine);
81      }
