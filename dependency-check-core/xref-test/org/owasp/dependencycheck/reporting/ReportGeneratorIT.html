156     public void generateReport(Settings settings, File writeTo, File writeJsonTo, File writeHtmlTo, File writeJunitTo, File writeCsvTo, File writeSarifTo, File suppressionFile){
157         try {
158             //first check parent folder
159             createParentFolder(writeTo);
160             createParentFolder(writeJsonTo);
161             createParentFolder(writeHtmlTo);
162             createParentFolder(writeJunitTo);
163             createParentFolder(writeCsvTo);
164             createParentFolder(writeSarifTo);
165 
166             File struts = new File(this.getClass().getClassLoader().getResource("struts2-core-2.1.2.jar").getPath());
167             File war = BaseTest.getResourceAsFile(this, "war-4.0.war");
168             File cfu = BaseTest.getResourceAsFile(this, "commons-fileupload-1.2.1.jar");
169             
170             //File axis = new File(this.getClass().getClassLoader().getResource("axis2-adb-1.4.1.jar").getPath());
171             File axis = BaseTest.getResourceAsFile(this, "axis2-adb-1.4.1.jar");
172             //File jetty = new File(this.getClass().getClassLoader().getResource("org.mortbay.jetty.jar").getPath());
173             File jetty = BaseTest.getResourceAsFile(this, "org.mortbay.jetty.jar");
174 
175             File nodeTest = BaseTest.getResourceAsFile(this, "nodejs");
176             int vulnCount;
177             try (Engine engine = new Engine(settings)) {
178                 engine.scan(struts);
179                 engine.scan(war);
180                 engine.scan(cfu);
181                 engine.scan(axis);
182                 engine.scan(jetty);
183                 engine.scan(nodeTest);
184                 engine.analyzeDependencies();
185 
186                 vulnCount = countVulns(engine.getDependencies());
187                 ExceptionCollection exceptions = new ExceptionCollection();
188                 exceptions.addException(new DownloadFailedException("test exception 1"));
189                 DownloadFailedException sub = new DownloadFailedException("test cause exception - nested");
190                 DownloadFailedException inner = new DownloadFailedException("Unable to download test file", sub.fillInStackTrace());
191                 UpdateException ex = new UpdateException("Test Exception 2", inner.fillInStackTrace());
192                 exceptions.addException(ex);
193                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeTo, "XML", exceptions);
194                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeJsonTo, "JSON", exceptions);
195                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeHtmlTo, "HTML", exceptions);
196                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeCsvTo, "CSV", exceptions);
197                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeJunitTo, "JUNIT", exceptions);
198                 engine.writeReports("Test Report", "org.owasp", "dependency-check-core", "1.4.8", writeSarifTo, "SARIF", exceptions);
199             }
200             //Test XML
201             InputStream xsdStream = ReportGenerator.class.getClassLoader().getResourceAsStream("schema/dependency-check.4.0.xsd");
202             StreamSource xsdSource = new StreamSource(xsdStream);
203             StreamSource xmlSource = new StreamSource(writeTo);
204             SchemaFactory sf = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
205             Schema schema = sf.newSchema(xsdSource);
206             Validator validator = schema.newValidator();
207             validator.validate(xmlSource);
208 
209             //Test CSV
210             int linesWritten = countLines(writeCsvTo);
211             Assert.assertEquals(vulnCount + 1, linesWritten);
212         } catch (DatabaseException | ExceptionCollection | ReportException | SAXException | IOException ex) {
213             fail(ex.getMessage());
214         }
215     }

217     /**
218      * create the parent folder if doesn't exist
219      * @param file the file
220      * @return true if all fine ?
221      */
222     private boolean createParentFolder(File file){
223         if (!file.getParentFile().exists()) {
224             return file.getParentFile().mkdir();
225         }
226         return true;
227     }

231     /**
232      * Counts the lines in a file. Copied from
233      * https://stackoverflow.com/a/14411695.
234      *
235      * @param file the file path
236      * @return the count of the lines in the file
237      * @throws IOException thrown if the file can't be read
238      */
239     private int countLines(File file) throws IOException {
240         try (InputStream is = new BufferedInputStream(new FileInputStream(file))) {
241             byte[] c = new byte[1024];
242             int count = 0;
243             int readChars = 0;
244             boolean endsWithoutNewLine = false;
245             while ((readChars = is.read(c)) != -1) {
246                 for (int i = 0; i < readChars; ++i) {
247                     if (c[i] == '\n') {
248                         ++count;
249                     }
250                 }
251                 endsWithoutNewLine = (c[readChars - 1] != '\n');
252             }
253             if (endsWithoutNewLine) {
254                 ++count;
255             }
256             return count;
257         }
258     }

260     private int countVulns(Dependency[] dependencies) {
261         return Arrays.stream(dependencies)
262                 .mapToInt(d -> d.getVulnerabilities().size())
263                 .sum();
264     }
