145     /**
146      * Test of csvIdentifiers method, of class EscapeTool.
147      */
148     @Test
149     public void testCsvIdentifiers() {
150         EscapeTool instance = new EscapeTool();
151         Set<Identifier> ids = null;
152         String expResult = "\"\"";
153         String result = instance.csvIdentifiers(ids);
154         assertEquals(expResult, result);
155 
156         ids = new HashSet<>();
157         expResult = "\"\"";
158         result = instance.csvIdentifiers(ids);
159         assertEquals(expResult, result);
160 
161         ids = new HashSet<>();
162         ids.add(new GenericIdentifier("somegroup:something:1.0", Confidence.HIGH));
163         expResult = "somegroup:something:1.0";
164         result = instance.csvIdentifiers(ids);
165         assertEquals(expResult, result);
166 
167         ids = new HashSet<>();
168         ids.add(new GenericIdentifier("somegroup:something:1.0", Confidence.HIGH));
169         ids.add(new GenericIdentifier("somegroup2:something:1.2", Confidence.HIGH));
170         expResult = "\"somegroup:something:1.0, somegroup2:something:1.2\"";
171         String expResult2 = "\"somegroup2:something:1.2, somegroup:something:1.0\"";
172         result = instance.csvIdentifiers(ids);
173         assertTrue(expResult.equals(result) || expResult2.equals(result));
174     }

178      */
179     @Test
180     public void testCsvCpeConfidence() {
181         EscapeTool instance = new EscapeTool();
182         Set<Identifier> ids = null;
183         String expResult = "\"\"";
184         String result = instance.csvCpeConfidence(ids);
185         assertEquals(expResult, result);
186 
187         ids = new HashSet<>();
188         expResult = "\"\"";
189         result = instance.csvCpeConfidence(ids);
190         assertEquals(expResult, result);
191 
192         ids = new HashSet<>();
193         GenericIdentifier i1 = new GenericIdentifier("cpe:/a:somegroup:something:1.0", Confidence.HIGH);
194         ids.add(i1);
195         expResult = "HIGH";
196         result = instance.csvCpeConfidence(ids);
197         assertEquals(expResult, result);
198 
199         ids = new HashSet<>();
200         i1 = new GenericIdentifier("cpe:/a:somegroup:something:1.0", Confidence.HIGH);
201         ids.add(i1);
202         GenericIdentifier i2 = new GenericIdentifier("cpe:/a:somegroup:something2:1.0", Confidence.MEDIUM);
203         ids.add(i2);
204 
205         expResult = "\"HIGH, MEDIUM\"";
206         String expResult2 = "\"MEDIUM, HIGH\"";
207         result = instance.csvCpeConfidence(ids);
208         assertTrue(expResult.equals(result) || expResult2.equals(result));
209     }
