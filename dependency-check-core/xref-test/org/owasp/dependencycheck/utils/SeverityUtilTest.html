28  public class SeverityUtilTest {

30      /**
31       * Test of estimateCvssV2 method, of class SeverityUtil.
32       */
33      @Test
34      public void testEstimateCvssV2() {
35          String severity = null;
36          double expResult = 0.0;
37          Double result = SeverityUtil.estimateCvssV2(severity);
38          assertThat(String.format("Expected %s to be %f", severity, expResult),
39                  result, Matchers.equalTo(expResult));
40  
41          severity = "garbage";
42          expResult = 3.9;
43          result = SeverityUtil.estimateCvssV2(severity);
44          assertThat(String.format("Expected %s to be %f", severity, expResult),
45                  result, Matchers.equalTo(expResult));
46  
47          severity = "Critical";

51                  result, Matchers.equalTo(expResult));
52  
53          severity = "HIGH";

63                  result, Matchers.equalTo(expResult));
