13  public class PyPACoreMetadataParserTest {
14  
15      @Test
16      public void getProperties_should_throw_exception_for_too_large_major() throws IOException {
17          try {
18              PyPACoreMetadataParser.getProperties(new BufferedReader(new StringReader("Metadata-Version: 3.0")));
19              Assert.fail("Expected IllegalArgumentException for too large major in Metadata-Version");
20          } catch (IllegalArgumentException e) {
21              Assert.assertTrue(e.getMessage().contains("Unsupported PyPA Wheel metadata"));
22          }
23      }
24  
25      @Test
26      public void getProperties_should_properly_parse_multiline_description() throws IOException {
27          String payload = "Metadata-Version: 1.0\r\n"
28                           + "Description: This is the first line\r\n"
29                           + "       | and this the second\r\n"
30                           + "       |\r\n"
31                           + "       | and the fourth after an empty third\r\n"
32                           + "\r\n"
33                           + "This: is the body and it is ignored. It may contain an extensive description in various formats";
34          Properties props = PyPACoreMetadataParser.getProperties(new BufferedReader(new StringReader(payload)));
35          Assert.assertEquals("1.0", props.get("Metadata-Version"));
36          Assert.assertEquals("This is the first line\n"
37                              + " and this the second\n"
38                              + "\n"
39                              + " and the fourth after an empty third", props.get("Description"));
40          Assert.assertFalse("Body was parsed as a header", props.containsKey("This"));
41      }
42  
43      @Test
44      public void getProperties_should_support_colon_in_headerValue() throws IOException {
45          String payload = "Metadata-Version: 2.2\r\n"
46                           + "Description: My value contains a : colon\r\n";
47          Properties props = PyPACoreMetadataParser.getProperties(new BufferedReader(new StringReader(payload)));
48          Assert.assertEquals("2.2", props.getProperty("Metadata-Version"));
49          Assert.assertEquals("My value contains a : colon", props.getProperty("Description"));
50      }
51      @Test
52      public void getProperties_should_support_folding_in_headerValue() throws IOException {
53          String payload = "Metadata-Version: 2\r\n"
54                           + " .2\r\n"
55                           + "Description: My value\r\n"
56                           + "  contains a \r\n"
57                           + " : colon\r\n";
58          Properties props = PyPACoreMetadataParser.getProperties(new BufferedReader(new StringReader(payload)));
59          Assert.assertEquals("2.2", props.getProperty("Metadata-Version"));
60          Assert.assertEquals("My value contains a : colon", props.getProperty("Description"));
61      }
62      @Test
63      public void getProperties_should_support_newer_minors() throws IOException {
64          String payload = "Metadata-Version: 2\r\n"
65                           + " .5\r\n";
66          Properties props = PyPACoreMetadataParser.getProperties(new BufferedReader(new StringReader(payload)));
67          Assert.assertEquals("2.5", props.getProperty("Metadata-Version"));
68      }
