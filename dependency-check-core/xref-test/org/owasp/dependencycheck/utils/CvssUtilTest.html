34       * Test of vectorToCvssV2 method, of class CvssUtil.
35       */
36      @Test
37      public void testVectorToCvssV2() {
38          String vectorString = "/AV:L/AC:L/Au:N/C:N/I:N/A:C";
39          Double baseScore = 1.0;
40          CvssV2 result = CvssUtil.vectorToCvssV2(vectorString, baseScore);
41          assertEquals(CvssV2Data.Version._2_0, result.getCvssData().getVersion());
42          assertEquals(CvssV2Data.AccessVectorType.LOCAL, result.getCvssData().getAccessVector());
43          assertEquals(CvssV2Data.AccessComplexityType.LOW, result.getCvssData().getAccessComplexity());
44          assertEquals(CvssV2Data.AuthenticationType.NONE, result.getCvssData().getAuthentication());
45          assertEquals(CvssV2Data.CiaType.NONE, result.getCvssData().getConfidentialityImpact());
46          assertEquals(CvssV2Data.CiaType.NONE, result.getCvssData().getIntegrityImpact());
47          assertEquals(CvssV2Data.CiaType.COMPLETE, result.getCvssData().getAvailabilityImpact());
48          assertEquals("LOW", result.getCvssData().getBaseSeverity());
49          assertEquals(1.0, result.getCvssData().getBaseScore(), 0);
50      }

162     /**
163      * Test of vectorToCvssV3 method, of class CvssUtil.
164      */
165     @Test
166     public void testVectorToCvssV3() {
167         String vectorString = "CVSS:3.1/AV:L/AC:L/PR:L/UI:N/S:U/C:N/I:N/A:H";
168         Double baseScore = 10.0;
169         CvssV3 result = CvssUtil.vectorToCvssV3(vectorString, baseScore);
170         assertEquals(CvssV3Data.Version._3_1, result.getCvssData().getVersion());
171         assertEquals(CvssV3Data.AttackVectorType.LOCAL, result.getCvssData().getAttackVector());
172         assertEquals(CvssV3Data.AttackComplexityType.LOW, result.getCvssData().getAttackComplexity());
173         assertEquals(CvssV3Data.PrivilegesRequiredType.LOW, result.getCvssData().getPrivilegesRequired());
174         assertEquals(CvssV3Data.UserInteractionType.NONE, result.getCvssData().getUserInteraction());
175         assertEquals(CvssV3Data.ScopeType.UNCHANGED, result.getCvssData().getScope());
176         assertEquals(CvssV3Data.CiaType.NONE, result.getCvssData().getConfidentialityImpact());
177         assertEquals(CvssV3Data.CiaType.NONE, result.getCvssData().getIntegrityImpact());
178         assertEquals(CvssV3Data.CiaType.HIGH, result.getCvssData().getAvailabilityImpact());
179         assertEquals(CvssV3Data.SeverityType.CRITICAL, result.getCvssData().getBaseSeverity());
180         assertEquals(10.0, result.getCvssData().getBaseScore(), 0);
181     }
