359         assertEquals(expResult, result);
360 
361         cpe.setValue("cpe:/a:microsoft:.net_framework");
362         cpe.setCaseSensitive(false);
363         expResult = true;
364         result = instance.identifierMatches(cpe, identifier);

366 
367         cpe.setValue("cpe:/a:microsoft:.*");

373         cpe.setValue("CPE:/a:microsoft:.*");

411     /**
412      * Test of process method, of class SuppressionRule.
413      */
414     @Test
415     public void testProcess() throws CpeValidationException {
416         //File struts = new File(this.getClass().getClassLoader().getResource("struts2-core-2.1.2.jar").getPath());
417         File struts = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
418         Dependency dependency = new Dependency(struts);
419         CpeIdentifier cpeId = new CpeIdentifier("microsoft", ".net_framework", "4.5", Confidence.HIGH);
420         dependency.addVulnerableSoftwareIdentifier(cpeId);
421         String sha1 = dependency.getSha1sum();
422         dependency.setSha1sum("384FAA82E193D4E4B0546059CA09572654BC3970");
423         Vulnerability v = createVulnerability();
424         dependency.addVulnerability(v);
425 
426         //cwe
427         SuppressionRule instance = new SuppressionRule();
428         instance.setSha1(sha1);
429         instance.addCwe("287");
430         instance.process(dependency);
431         assertEquals(1, dependency.getVulnerabilities().size());
432         dependency.setSha1sum(sha1);
433         instance.process(dependency);
434         assertTrue(dependency.getVulnerabilities().isEmpty());
435         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
436 
437         //cvss
438         dependency.addVulnerability(v);
439         instance = new SuppressionRule();
440         instance.addCvssBelow(5.0);
441         instance.process(dependency);
442         assertEquals(1, dependency.getVulnerabilities().size());
443         instance.addCvssBelow(8.0);
444         instance.process(dependency);
445         assertTrue(dependency.getVulnerabilities().isEmpty());
446         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
447 
448         //cve
449         dependency.addVulnerability(v);
450         instance = new SuppressionRule();
451         instance.addCve("CVE-2012-1337");
452         instance.process(dependency);
453         assertEquals(1, dependency.getVulnerabilities().size());
454         instance.addCve("CVE-2013-1337");
455         instance.process(dependency);
456         assertTrue(dependency.getVulnerabilities().isEmpty());
457         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
458 
459         //cpe
460         instance = new SuppressionRule();
461         PropertyType pt = new PropertyType();
462         pt.setValue("cpe:/a:microsoft:.net_framework:4.0");
463         instance.addCpe(pt);
464         instance.process(dependency);
465         assertTrue(dependency.getVulnerableSoftwareIdentifiers().size() == 1);
466         pt = new PropertyType();
467         pt.setValue("cpe:/a:microsoft:.net_framework:4.5");
468         instance.addCpe(pt);
469         pt = new PropertyType();
470         pt.setValue(".*");
471         pt.setRegex(true);
472         instance.setFilePath(pt);
473         instance.process(dependency);
474         assertTrue(dependency.getVulnerableSoftwareIdentifiers().isEmpty());
475         assertEquals(1, dependency.getSuppressedIdentifiers().size());
476 
477         instance = new SuppressionRule();
478 
479         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "4.0", Confidence.HIGH));
480         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "4.5", Confidence.HIGH));
481         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "5.0", Confidence.HIGH));
482         pt = new PropertyType();
483         pt.setValue("cpe:/a:microsoft:.net_framework");
484         instance.addCpe(pt);
485         instance.setBase(true);
486         assertEquals(3, dependency.getVulnerableSoftwareIdentifiers().size());
487         assertEquals(1, dependency.getSuppressedIdentifiers().size());
488         instance.process(dependency);
489         assertTrue(dependency.getVulnerableSoftwareIdentifiers().isEmpty());
490         assertEquals(1, dependency.getSuppressedIdentifiers().size());
491     }

531     @Test
532     public void testProcessVulnerabilityNames() throws CpeValidationException, MalformedPackageURLException {
533         File spring = BaseTest.getResourceAsFile(this, "spring-security-web-3.0.0.RELEASE.jar");
534         Dependency dependency = new Dependency(spring);
535         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("vmware", "springsource_spring_security", "3.0.0", Confidence.HIGH));
536         dependency.addSoftwareIdentifier(new PurlIdentifier("maven", "org.springframework.security", "spring-security-web", "3.0.0.RELEASE", Confidence.HIGH));
537 
538         dependency.addVulnerability(createVulnerability());
539         SuppressionRule instance = new SuppressionRule();
540         //gav
541         PropertyType pt = new PropertyType();
542         pt.setValue("pkg:maven/org.springframework.security/spring-security-web@3.0.0.RELEASE");
543         pt.setRegex(false);
544         pt.setCaseSensitive(false);
545         instance.setPackageUrl(pt);
546 
547         pt = new PropertyType();
548         pt.setValue("CVE-2013-1338");
549         instance.addVulnerabilityName(pt);
550 
551         instance.process(dependency);
552         assertEquals(1, dependency.getVulnerabilities().size());
553         assertEquals(0, dependency.getSuppressedVulnerabilities().size());
554 
555         
556         pt = new PropertyType();
557         pt.setValue("CVE-2013-1337");
558         instance.addVulnerabilityName(pt);
559         
560         instance.process(dependency);
561         assertEquals(0, dependency.getVulnerabilities().size());
562         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
563     }

565     private Vulnerability createVulnerability() {
566         Vulnerability v = new Vulnerability();
567         v.addCwe("CWE-287 Improper Authentication");
568         v.setName("CVE-2013-1337");
569         
570         CvssV2 cvss = CvssUtil.vectorToCvssV2("/AV:N/AC:L/Au:N/C:P/I:P/A:P", 7.5);
571         v.setCvssV2(cvss);
572         return v;
573     }
