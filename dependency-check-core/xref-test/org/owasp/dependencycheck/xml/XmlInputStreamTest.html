33      /**
34       * Test of length method, of class XmlInputStream.
35       */
36      @Test
37      public void testLength() {
38          String data = "";
39          InputStream stream = new ByteArrayInputStream(data.getBytes(StandardCharsets.UTF_8));
40          XmlInputStream instance = new XmlInputStream(stream);
41          int expResult = 0;
42          int result = instance.length();
43          assertEquals(expResult, result);
44  
45          data = "Input data";
46          stream = new ByteArrayInputStream(data.getBytes(StandardCharsets.UTF_8));
47          instance = new XmlInputStream(stream);
48          result = instance.length();
49          assertTrue(result > 0);
50      }

52      /**
53       * Test of read method, of class XmlInputStream.
54       */
55      @Test
56      public void testRead_0args() throws Exception {
57          String data = "";
58          InputStream stream = new ByteArrayInputStream(data.getBytes(StandardCharsets.UTF_8));
59          XmlInputStream instance = new XmlInputStream(stream);
60          int expResult = -1;
61          int result = instance.read();
62          assertEquals(expResult, result);
63  
64          data = "*";
65          stream = new ByteArrayInputStream(data.getBytes(StandardCharsets.UTF_8));
66          instance = new XmlInputStream(stream);
67          expResult = 42;
68          result = instance.read();
69          assertEquals(expResult, result);
70      }

72      /**
73       * Test of read method, of class XmlInputStream.
74       */
75      @Test
76      public void testRead_3args() throws Exception {
77          byte[] data = new byte[10];
78          int offset = 0;
79          int length = 10;
80          byte[] expected = "abcdefghij".getBytes(StandardCharsets.UTF_8);
81          String text = "abcdefghijklmnopqrstuvwxyz";
82          InputStream stream = new ByteArrayInputStream(text.getBytes(StandardCharsets.UTF_8));
83          XmlInputStream instance = new XmlInputStream(stream);
84          int expResult = 10;
85          int result = instance.read(data, offset, length);
86          assertEquals(expResult, result);
87          assertArrayEquals(expected, data);
88          
89          
90          data = new byte[5];
91          offset = 0;
92          length = 5;
93          expected = "&#38;".getBytes(StandardCharsets.UTF_8);
94          text = "&amp;";
95          stream = new ByteArrayInputStream(text.getBytes(StandardCharsets.UTF_8));
96          instance = new XmlInputStream(stream);
97          expResult = 5;
98          result = instance.read(data, offset, length);
99          assertEquals(expResult, result);
100         assertArrayEquals(expected, data);
101         
102         data = new byte[10];
103         offset = 0;
104         length = 10;
105         expected = "&#38; test".getBytes(StandardCharsets.UTF_8);
106         text = "& test";
107         stream = new ByteArrayInputStream(text.getBytes(StandardCharsets.UTF_8));
108         instance = new XmlInputStream(stream);
109         expResult = 10;
110         result = instance.read(data, offset, length);
111         assertEquals(expResult, result);
112         assertArrayEquals(expected, data);
113     }

115     /**
116      * Test of toString method, of class XmlInputStream.
117      */
118     @Test
119     public void testToString() throws IOException {
120         String data = "test";
121         InputStream stream = new ByteArrayInputStream(data.getBytes(StandardCharsets.UTF_8));
122         XmlInputStream instance = new XmlInputStream(stream);
123         int r = instance.read();
124         assertEquals('t', r);
125         String expResult = "[1]-\"t\" ( 74)";
126         String result = instance.toString();
127         assertEquals(expResult, result);
128         r = instance.read();
129         assertEquals('e', r);
130         expResult = "[2]-\"te\" ( 74 65)";
131         result = instance.toString();
132         assertEquals(expResult, result);
133 
134     }
