34      /**
35       * Test of readPom method, of class PomUtils.
36       *
37       * @throws java.lang.Exception thrown when the test fails due to an
38       * exception
39       */
40      @Test
41      public void testReadPom_File() throws Exception {
42          File file = BaseTest.getResourceAsFile(this, "dwr-pom.xml");
43          String expResult = "Direct Web Remoting";
44          Model result = PomUtils.readPom(file);
45          assertEquals(expResult, result.getName());
46  
47          expResult = "get ahead";
48          assertEquals(expResult, result.getOrganization());
49          expResult = "http://getahead.ltd.uk/dwr";
50          assertEquals(expResult, result.getOrganizationUrl());
51  
52          file = BaseTest.getResourceAsFile(this, "jmockit-1.26.pom");
53          expResult = "Main Ã¸ modified to test issue #710 and #801 (&amps;)";
54          result = PomUtils.readPom(file);
55          assertEquals(expResult, result.getName());
56  
57          file = BaseTest.getResourceAsFile(this, "pom/mailapi-1.4.3_projectcomment.pom");
58          expResult = "JavaMail API jar";
59          result = PomUtils.readPom(file);
60          assertEquals(expResult, result.getName());
61      }
62  
63      @Test
64      public void testReadPom_String_File() throws Exception {
65          File fileCommonValidator = BaseTest.getResourceAsFile(this, "commons-validator-1.4.0.jar");
66          JarFile jar = new JarFile(fileCommonValidator, false);
67          String expResult = "Commons Validator";
68          Model result = PomUtils.readPom("META-INF/maven/commons-validator/commons-validator/pom.xml", jar);
69          assertEquals(expResult, result.getName());
70      }
