31          FileReader fir = new FileReader(jsonFixtureFile);

42          assertEquals("Cookie headers were not validated\n", r.getDescription());
43          assertEquals("https://github.com/elixir-plug/plug/commit/8857f8ab4acf9b9c22e80480dae2636692f5f573", r.getUrl());
