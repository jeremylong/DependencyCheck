20  import org.junit.Assume;

23  import org.junit.Test;

25  import org.owasp.dependencycheck.utils.Settings;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import java.io.FileNotFoundException;
30  
31  import static org.junit.Assert.assertEquals;
32  import static org.junit.Assert.assertNotNull;
33  
34  public class NexusV3SearchTest extends BaseTest {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(NexusV3SearchTest.class);
37      private NexusV3Search searcher;
38  
39      @Before
40      @Override
41      public void setUp() throws Exception {
42          super.setUp();
43          Settings sett = getSettings();
44  //        sett.setString(Settings.KEYS.ANALYZER_NEXUS_USER, "demo");
45  //        sett.setString(Settings.KEYS.ANALYZER_NEXUS_PASSWORD, "demo");
46  //        sett.setString(Settings.KEYS.ANALYZER_NEXUS_URL, "http://localhost/service/rest/");
47          String nexusUrl = sett.getString(Settings.KEYS.ANALYZER_NEXUS_URL);
48  
49          LOGGER.debug(nexusUrl);
50          searcher = new NexusV3Search(sett, false);
51          Assume.assumeTrue(searcher.preflightRequest());
52      }
53  
54      @Test(expected = IllegalArgumentException.class)
55      @Ignore
56      public void testNullSha1() throws Exception {
57          searcher.searchSha1(null);
58      }
59  
60      @Test(expected = IllegalArgumentException.class)
61      @Ignore
62      public void testMalformedSha1() throws Exception {
63          searcher.searchSha1("invalid");
64      }
65  
66      // This test does generate network traffic and communicates with a host
67      // you may not be able to reach. Remove the @Ignore annotation if you want to
68      // test it anyway
69      @Test
70      @Ignore
71      public void testValidSha1() throws Exception {
72          MavenArtifact ma = searcher.searchSha1("9977a8d04e75609cf01badc4eb6a9c7198c4c5ea");
73          assertEquals("Incorrect group", "org.apache.maven.plugins", ma.getGroupId());
74          assertEquals("Incorrect artifact", "maven-compiler-plugin", ma.getArtifactId());
75          assertEquals("Incorrect version", "3.1", ma.getVersion());
76          assertNotNull("URL Should not be null", ma.getArtifactUrl());
77      }
78  
79      // This test does generate network traffic and communicates with a host
80      // you may not be able to reach. Remove the @Ignore annotation if you want to
81      // test it anyway
82      @Test(expected = FileNotFoundException.class)
83      @Ignore
84      public void testMissingSha1() throws Exception {
85          searcher.searchSha1("AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA");
86      }
87  }
