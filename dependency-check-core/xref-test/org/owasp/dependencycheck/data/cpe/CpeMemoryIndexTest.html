20  import org.apache.lucene.document.Document;

22  import org.apache.lucene.search.TopDocs;
23  import org.junit.AfterClass;
24  import org.junit.Before;
25  import org.junit.Test;
26  import static org.junit.Assert.*;
27  import org.owasp.dependencycheck.BaseDBTestCase;
28  import org.owasp.dependencycheck.Engine;

30  /**
31   *
32   * @author jeremy long
33   */
34  public class CpeMemoryIndexTest extends BaseDBTestCase {
35  
36      private static final CpeMemoryIndex instance = CpeMemoryIndex.getInstance();
37      private static Engine engine = null;
38  
39      @AfterClass
40      public static void tearDownClass() {
41          if (instance.isOpen()) {
42              instance.close();
43          }
44          if (engine != null) {
45              engine.close();
46          }
47      }
48  
49      @Before
50      @Override
51      public void setUp() throws Exception {
52          super.setUp();
53          if (!instance.isOpen()) {
54              engine = new Engine(getSettings());
55              engine.openDatabase();
56              instance.open(engine.getDatabase().getVendorProductList(), this.getSettings());
57          }
58      }
59  
60      /**
61       * Test of getInstance method, of class CpeMemoryIndex.
62       */
63      @Test
64      public void testGetInstance() {
65          CpeMemoryIndex result = CpeMemoryIndex.getInstance();
66          assertNotNull(result);
67      }
68  
69      /**
70       * Test of isOpen method, of class CpeMemoryIndex.
71       */
72      @Test
73      public void testIsOpen() {
74          boolean expResult = true;
75          boolean result = instance.isOpen();
76          assertEquals(expResult, result);
77      }
78  
79      /**
80       * Test of search method, of class CpeMemoryIndex.
81       */
82      @Test
83      public void testSearch_String_int() throws Exception {
84          String searchString = "product:(commons) AND vendor:(apache)";
85          int maxQueryResults = 3;
86          TopDocs result = instance.search(searchString, maxQueryResults);
87          assertEquals(3, result.scoreDocs.length);
88          instance.close();
89      }
90  
91      /**
92       * Test of parseQuery method, of class CpeMemoryIndex.
93       */
94      @Test
95      public void testParseQuery() throws Exception {
96          String searchString = "product:(resteasy) AND vendor:(red hat)";
97  
98          String expResult = "+product:resteasy +(vendor:red vendor:redhat vendor:hat)";
99          Query result = instance.parseQuery(searchString);
100         assertEquals(expResult, result.toString());
101         instance.resetAnalyzers();
102         searchString = "product:(struts2\\-core^2 struts^3 core) AND vendor:(apache.struts apache^3 foundation)";
103 
104         expResult = "+((product:struts2 product:struts2struts product:struts product:strutscore product:core)^2.0 (product:corestruts product:struts)^3.0 (product:strutscore product:core)) +((vendor:apache vendor:apachestruts vendor:struts) (vendor:strutsapache vendor:apache)^3.0)";
105         result = instance.parseQuery(searchString);
106         assertEquals(expResult, result.toString());
107         instance.close();
108     }
109 
110     /**
111      * Test of search method, of class CpeMemoryIndex.
112      */
113     @Test
114     public void testSearch_Query_int() throws Exception {
115         String searchString = "product:(commons) AND vendor:(apache)";
116         Query query = instance.parseQuery(searchString);
117         int maxQueryResults = 3;
118         TopDocs result = instance.search(query, maxQueryResults);
119         assertEquals(3, result.scoreDocs.length);
120     }
121 
122     /**
123      * Test of getDocument method, of class CpeMemoryIndex.
124      */
125     @Test
126     public void testGetDocument() throws Exception {
127         String searchString = "product:(commons) AND vendor:(apache)";
128         int maxQueryResults = 1;
129         TopDocs docs = instance.search(searchString, maxQueryResults);
130         int documentId = docs.scoreDocs[0].doc;
131         Document result = instance.getDocument(documentId);
132         assertEquals("apache", result.get("vendor"));
133     }
134

137      */
138     @Test
139     public void testNumDocs() {
140         int result = instance.numDocs();
141         assertTrue(result > 100);
142     }
