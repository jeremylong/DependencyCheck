64       * Test of save method, of class DatabaseProperties.
65       */
66      @Test
67      public void testSave() throws Exception {
68          String key = "test";
69          String value = "something";
70          String expected = "something";
71          DatabaseProperties instance = cveDb.getDatabaseProperties();
72          instance.save(key, value);
73          instance = cveDb.reloadProperties();
74          String results = instance.getProperty(key);
75          assertEquals(expected, results);
76      }

81      @Test

84          String defaultValue = "default";
85          DatabaseProperties instance = cveDb.getDatabaseProperties();
86          String expResult = "default";
87          String result = instance.getProperty(key, defaultValue);
88          assertEquals(expResult, result);
89      }

91      /**
92       * Test of getProperty method, of class DatabaseProperties.
93       */
94      @Test
95      public void testGetProperty_String() throws DatabaseException {
96          String key = "version";
97          DatabaseProperties instance = cveDb.getDatabaseProperties();
98          String result = instance.getProperty(key);
99          
100         int major = Integer.parseInt(result.substring(0, result.indexOf('.')));
101        
102         assertTrue(major >= 5);
103     }
