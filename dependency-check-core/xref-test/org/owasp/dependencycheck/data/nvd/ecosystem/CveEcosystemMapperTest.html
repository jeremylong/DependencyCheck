37  public class CveEcosystemMapperTest {
38  
39      /**
40       * Test of getEcosystem method, of class CveEcosystemMapper.
41       */
42      @Test
43      public void testGetEcosystem() {
44          CveEcosystemMapper mapper = new CveEcosystemMapper();
45          String value = "There is a vulnerability in some.java file";
46          assertEquals(JarAnalyzer.DEPENDENCY_ECOSYSTEM, mapper.getEcosystem(asCve(value)));
47          value = "There is a vulnerability in some file";
48          assertNull(mapper.getEcosystem(asCve(value,"cpe:2.3:a:apache:struts:1.2.1:*:*:*:*:*:*:*")));
49      }
50  
51      private DefCveItem asCve(String description, String... cpe) {
52  
53          List<LangString> d = new ArrayList<>();
54          LangString str = new LangString("en", description);
55          d.add(str);
56  
57          List<Config> configurations = new ArrayList<>();
58          List<Node> nodes = new ArrayList<>();
59          List<CpeMatch> cpeMatch = new ArrayList<>();
60          for (String s : cpe) {
61              CpeMatch m = new CpeMatch(Boolean.TRUE, s, null, null, null, null, null);
62              cpeMatch.add(m);
63          }
64          Node node = new Node(Node.Operator.OR, Boolean.FALSE, cpeMatch);
65          nodes.add(node);
66          Config conf = new Config(Config.Operator.AND, Boolean.FALSE, nodes);
67          CveItem cveItem = new CveItem(null, null, null, null, null, null, null, null, null, null, null, null, d, null, null, null, configurations, null);
68          DefCveItem defCveItem = new DefCveItem(cveItem);
69  
70          return defCveItem;
71      }
