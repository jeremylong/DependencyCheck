18  package org.owasp.dependencycheck.data.artifactory;
19  
20  import org.junit.AfterClass;
21  import org.junit.Before;
22  import org.junit.BeforeClass;
23  import org.junit.Test;
24  import org.owasp.dependencycheck.BaseTest;
25  import org.owasp.dependencycheck.dependency.Dependency;
26  import org.owasp.dependencycheck.utils.Settings;
27  
28  import java.io.IOException;
29  import java.net.UnknownHostException;
30  
31  import static org.junit.Assert.assertEquals;
32  import static org.junit.Assert.assertNotNull;
33  import static org.junit.Assert.assertTrue;
34  import static org.junit.Assert.fail;
35  
36  public class ArtifactorySearchTest extends BaseTest {
37      private static String httpsProxyHostOrig;
38      private static String httpsPortOrig;
39  
40      @BeforeClass
41      public static void tinkerProxies() {
42          httpsProxyHostOrig = System.getProperty("https.proxyHost");
43          if (httpsProxyHostOrig == null) {
44              httpsProxyHostOrig = System.getenv("https.proxyHost");
45          }
46          httpsPortOrig = System.getProperty("https.proxyPort");
47          if (httpsPortOrig == null) {
48              httpsPortOrig = System.getenv("https.proxyPort");
49          }
50          System.setProperty("https.proxyHost", "");
51          System.setProperty("https.proxyPort", "");
52      }
53  
54      @AfterClass
55      public static void restoreProxies() {
56          if (httpsProxyHostOrig != null) {
57              System.setProperty("https.proxyHost", httpsProxyHostOrig);
58          }
59          if (httpsPortOrig != null) {
60              System.setProperty("https.proxyPort", httpsPortOrig);
61          }
62      }
63  
64      @Before
65      @Override
66      public void setUp() throws Exception {
67          super.setUp();
68      }
69  
70  
71      @Test
72      public void shouldFailWhenHostUnknown() throws IOException {
73          // Given
74          Dependency dependency = new Dependency();
75          dependency.setSha1sum("c5b4c491aecb72e7c32a78da0b5c6b9cda8dee0f");
76          dependency.setSha256sum("512b4bf6927f4864acc419b8c5109c23361c30ed1f5798170248d33040de068e");
77          dependency.setMd5sum("2d1dd0fc21ee96bccfab4353d5379649");
78  
79          final Settings settings = getSettings();
80          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_URL, "https://artifactory.techno.ingenico.com.invalid/artifactory");
81          final ArtifactorySearch artifactorySearch = new ArtifactorySearch(settings);
82          // When
83          try {
84              artifactorySearch.search(dependency);
85              fail("Should have thrown an UnknownHostException");
86          } catch (UnknownHostException exception) {
87              // Then
88              assertNotNull(exception.getMessage());
89              assertTrue(exception.getMessage().contains("artifactory.techno.ingenico.com.invalid"));
90          }
91      }
