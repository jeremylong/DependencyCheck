80      @Test
81      public void shouldFailWhenHostUnknown() throws IOException {
82          // Given
83          Dependency dependency = new Dependency();
84          dependency.setSha1sum("c5b4c491aecb72e7c32a78da0b5c6b9cda8dee0f");
85          dependency.setSha256sum("512b4bf6927f4864acc419b8c5109c23361c30ed1f5798170248d33040de068e");
86          dependency.setMd5sum("2d1dd0fc21ee96bccfab4353d5379649");
87  
88          final Settings settings = getSettings();
89          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_URL, "https://artifactory.techno.ingenico.com.non-existing/artifactory");
90          final ArtifactorySearch artifactorySearch = new ArtifactorySearch(settings);
91          // When
92          try {
93              artifactorySearch.search(dependency);
94              fail();
95          } catch (UnknownHostException exception) {
96              // Then
97              assertEquals("artifactory.techno.ingenico.com.non-existing", exception.getMessage());
98          } catch (SocketTimeoutException exception) {
99              // Then
100             assertEquals("connect timed out", exception.getMessage());
101         } catch (IOException ex) {
102             assertEquals("Connection refused (Connection refused)", ex.getMessage());
103         }
104     }
