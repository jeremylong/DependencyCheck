37      @Test
38      public void testWithRealInstanceUsingBearerToken() throws IOException {
39          // Given
40          Dependency dependency = new Dependency();
41          dependency.setSha1sum("c5b4c491aecb72e7c32a78da0b5c6b9cda8dee0f");
42          dependency.setSha256sum("512b4bf6927f4864acc419b8c5109c23361c30ed1f5798170248d33040de068e");
43          dependency.setMd5sum("2d1dd0fc21ee96bccfab4353d5379649");
44  
45          final Settings settings = new Settings();
46          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_URL, "https://artifactory.techno.ingenico.com/artifactory");
47  
48          // See org.owasp.dependencycheck.utils.Settings.KEYS.ANALYZER_ARTIFACTORY_BEARER_TOKEN .for how to generate a bearer token using the API
49          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_BEARER_TOKEN, "yourBearerToken");
50          final ArtifactorySearch artifactorySearch = new ArtifactorySearch(settings);
51          // When
52          final List<MavenArtifact> mavenArtifacts = artifactorySearch.search(dependency);
53  
54          // Then
55  
56          assertEquals(1, mavenArtifacts.size());
57          final MavenArtifact artifact = mavenArtifacts.get(0);
58          assertEquals("com.google.code.gson", artifact.getGroupId());
59          assertEquals("gson", artifact.getArtifactId());
60          assertEquals("2.8.5", artifact.getVersion());
61          assertEquals("https://artifactory.techno.ingenico.com/artifactory/repo1-cache/com/google/code/gson/gson/2.8.5/gson-2.8.5-sources.jar", artifact.getArtifactUrl());
62          assertEquals("https://artifactory.techno.ingenico.com/artifactory/repo1-cache/com/google/code/gson/gson/2.8.5/gson-2.8.5.pom", artifact.getPomUrl());
63      }

84      @Test
85      public void testWithRealInstanceWithUserToken() throws IOException {
86          // Given
87          Dependency dependency = new Dependency();
88          dependency.setSha1sum("0695b63d702f505b9b916e02272e3b6381bade7f");
89          dependency.setMd5sum("cde1963375c651f769d40c8023ab5876");
90  
91          final Settings settings = new Settings();
92          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_URL, "https://artifactory.techno.ingenico.com/artifactory");
93          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_API_USERNAME, "yourUserName");
94          settings.setString(Settings.KEYS.ANALYZER_ARTIFACTORY_API_TOKEN, "yourSecretApiToken");
95          final ArtifactorySearch artifactorySearch = new ArtifactorySearch(settings);
96          // When
97          final List<MavenArtifact> mavenArtifacts = artifactorySearch.search(dependency);
98  
99          // Then
100         assertEquals(1, mavenArtifacts.size());
101         final MavenArtifact artifact = mavenArtifacts.get(0);
102         assertEquals("com.google.code.gson", artifact.getGroupId());
103         assertEquals("gson", artifact.getArtifactId());
104         assertEquals("2.4", artifact.getVersion());
105         assertEquals("https://artifactory.techno.ingenico.com/artifactory/gradle-libs-cache/com/google/code/gson/gson/2.4/gson-2.4.jar", artifact.getArtifactUrl());
106         assertEquals("https://artifactory.techno.ingenico.com/artifactory/gradle-libs-cache/com/google/code/gson/gson/2.4/gson-2.4.pom", artifact.getPomUrl());
107     }
