51  public class FieldAnalyzerTest extends BaseTest {
52  
53      @Test
54      public void testAnalyzers() throws Exception {
55  
56          Analyzer analyzer = new SearchFieldAnalyzer();
57          File temp = getSettings().getTempDirectory();
58          Directory index = new MMapDirectory(temp.toPath());
59  
60          String field1 = "product";
61          String text1 = "springframework";
62  
63          String field2 = "vendor";
64          String text2 = "springsource";
65  
66          try (IndexWriter w = createIndex(analyzer, index)) {
67              addDoc(w, field1, text1, field2, text2);
68              text1 = "x-stream";
69              text2 = "xstream";
70              addDoc(w, field1, text1, field2, text2);
71          }
72  
73          //Analyzer searchingAnalyzer = new SearchFieldAnalyzer();
74          String querystr = "product:\"(Spring Framework Core)\" vendor:(SpringSource)";
75  
76          SearchFieldAnalyzer searchAnalyzerProduct = new SearchFieldAnalyzer();
77          SearchFieldAnalyzer searchAnalyzerVendor = new SearchFieldAnalyzer();
78          HashMap<String, Analyzer> map = new HashMap<>();
79          map.put(field1, searchAnalyzerProduct);
80          map.put(field2, searchAnalyzerVendor);
81          PerFieldAnalyzerWrapper wrapper = new PerFieldAnalyzerWrapper(new StandardAnalyzer(), map);
82          QueryParser parser = new QueryParser(field1, wrapper);
83  
84          Query q = parser.parse(querystr);
85  
86          int hitsPerPage = 10;
87          int hitsThreshold = 100;
88  
89          IndexReader reader = DirectoryReader.open(index);
90          IndexSearcher searcher = new IndexSearcher(reader);
91          TopScoreDocCollector collector = TopScoreDocCollector.create(hitsPerPage, hitsThreshold);
92          searcher.search(q, collector);
93          ScoreDoc[] hits = collector.topDocs().scoreDocs;
94  
95          assertEquals("Did not find 1 document?", 1, hits.length);
96          assertEquals("springframework", searcher.doc(hits[0].doc).get(field1));
97          assertEquals("springsource", searcher.doc(hits[0].doc).get(field2));
98  
99          querystr = "product:(Apache Struts) vendor:(Apache)";
100 
101         reset(searchAnalyzerProduct, searchAnalyzerVendor);
102         Query q2 = parser.parse(querystr);
103         assertFalse("second parsing contains previousWord from the TokenPairConcatenatingFilter", q2.toString().contains("core"));
104 
105         querystr = "product:(  x-stream^5 )  AND  vendor:(  thoughtworks.xstream )";
106         reset(searchAnalyzerProduct, searchAnalyzerVendor);
107         Query q3 = parser.parse(querystr);
108         collector = TopScoreDocCollector.create(hitsPerPage, hitsThreshold);
109         searcher.search(q3, collector);
110         hits = collector.topDocs().scoreDocs;
111         assertEquals("x-stream", searcher.doc(hits[0].doc).get(field1));
112         assertEquals("xstream", searcher.doc(hits[0].doc).get(field2));
113     }
114 
115     private IndexWriter createIndex(Analyzer analyzer, Directory index) throws IOException {
116         IndexWriterConfig config = new IndexWriterConfig(analyzer);
117         return new IndexWriter(index, config);
118     }
119 
120     private static void addDoc(IndexWriter w, String field1, String text1, String field2, String text2) throws IOException {
121         Document doc = new Document();
122         doc.add(new TextField(field1, text1, Field.Store.YES));
123         doc.add(new TextField(field2, text2, Field.Store.YES));
124         w.addDocument(doc);
125     }
126 
127     private void reset(SearchFieldAnalyzer searchAnalyzerProduct, SearchFieldAnalyzer searchAnalyzerVendor) throws IOException {
128         searchAnalyzerProduct.reset();
129         searchAnalyzerVendor.reset();
130     }
