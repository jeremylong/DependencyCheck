18  package org.owasp.dependencycheck.data.lucene;
19  
20  import java.io.IOException;
21  import org.apache.lucene.analysis.Analyzer;
22  import org.apache.lucene.analysis.BaseTokenStreamTestCase;
23  import org.apache.lucene.analysis.MockTokenizer;
24  import org.apache.lucene.analysis.Tokenizer;
25  import org.apache.lucene.analysis.core.KeywordTokenizer;
26  import org.junit.Test;
27  
28  /**
29   *
30   * @author Jeremy Long
31   */
32  public class UrlTokenizingFilterTest extends BaseTokenStreamTestCase {
33  
34      private final Analyzer analyzer;
35  
36      public UrlTokenizingFilterTest() {
37          analyzer = new Analyzer() {
38              @Override
39              protected TokenStreamComponents createComponents(String fieldName) {
40                  Tokenizer source = new MockTokenizer(MockTokenizer.WHITESPACE, false);
41                  return new TokenStreamComponents(source, new UrlTokenizingFilter(source));
42              }
43          };
44      }
45  
46      /**
47       * test some example domains
48       */
49      @Test
50      public void testExamples() throws IOException {
51          String[] expected = new String[2];
52          expected[0] = "domain";
53          expected[1] = "test";
54          assertAnalyzesTo(analyzer, "http://www.domain.com/test.php", expected);
55          checkOneTerm(analyzer, "https://apache.org", "apache");
56      }
57  
58      /**
59       * copied from
60       * http://svn.apache.org/repos/asf/lucene/dev/trunk/lucene/analysis/common/src/test/org/apache/lucene/analysis/en/TestEnglishMinimalStemFilter.java
61       * blast some random strings through the analyzer
62       */
63      @Test
64      public void testRandomStrings() {
65          try {
66              checkRandomData(random(), analyzer, 1000 * RANDOM_MULTIPLIER);
67          } catch (IOException ex) {
68              fail("Failed test random strings: " + ex.getMessage());
69          }
70      }
71  
72      /**
73       * copied from
74       * http://svn.apache.org/repos/asf/lucene/dev/trunk/lucene/analysis/common/src/test/org/apache/lucene/analysis/en/TestEnglishMinimalStemFilter.java
75       *
76       * @throws IOException
77       */
78      @Test
79      public void testEmptyTerm() throws IOException {
80          Analyzer a = new Analyzer() {
81              @Override
82              protected TokenStreamComponents createComponents(String fieldName) {
83                  Tokenizer tokenizer = new KeywordTokenizer();
84                  return new TokenStreamComponents(tokenizer, new UrlTokenizingFilter(tokenizer));
85              }
86          };
87          checkOneTerm(a, "", "");
88      }
89  }
