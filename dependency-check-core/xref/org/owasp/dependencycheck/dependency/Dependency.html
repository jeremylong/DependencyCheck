207     /**
208      * Calculates the checksums for the given file.
209      *
210      * @param file the file used to calculate the checksums
211      */
212     private void calculateChecksums(File file) {
213         try {
214             this.md5sum = Checksum.getMD5Checksum(file);
215             this.sha1sum = Checksum.getSHA1Checksum(file);
216             this.sha256sum = Checksum.getSHA256Checksum(file);
217         } catch (NoSuchAlgorithmException | IOException ex) {
218             LOGGER.debug(String.format("Unable to calculate checksums on %s", file), ex);
219         }
220     }

458     /**
459      * Adds an entry to the list of detected Identifiers for the dependency
460      * file.
461      *
462      * @param identifier a reference to the identifier to add
463      */
464     public synchronized void addSoftwareIdentifier(Identifier identifier) {
465         //todo the following assertion should be removed after initial testing and implementation
466         assert !(identifier instanceof CpeIdentifier) : "vulnerability identifier cannot be added to software identifiers";
467 
468         final Optional<Identifier> found = softwareIdentifiers.stream().filter(id
469                 -> id.getValue().equals(identifier.getValue())).findFirst();
470         if (found.isPresent()) {
471             //TODO - should we check for type of identifier?  I.e. could we see a Purl and GenericIdentifier with the same value
472             final Identifier existing = found.get();
473             if (existing.getConfidence().compareTo(identifier.getConfidence()) < 0) {
474                 existing.setConfidence(identifier.getConfidence());
475             }
476             if (existing.getNotes() != null && identifier.getNotes() != null) {
477                 existing.setNotes(existing.getNotes() + " " + identifier.getNotes());
478             } else if (identifier.getNotes() != null) {
479                 existing.setNotes(identifier.getNotes());
480             }
481             if (existing.getUrl() == null && identifier.getUrl() != null) {
482                 existing.setUrl(identifier.getUrl());
483             }
484         } else {
485             this.softwareIdentifiers.add(identifier);
486         }
487     }

508     /**
509      * Adds the Maven artifact as evidence.
510      *
511      * @param source The source of the evidence
512      * @param mavenArtifact The Maven artifact
513      * @param confidence The confidence level of this evidence
514      */
515     public void addAsEvidence(String source, MavenArtifact mavenArtifact, Confidence confidence) {
516         if (mavenArtifact.getGroupId() != null && !mavenArtifact.getGroupId().isEmpty()) {
517             this.addEvidence(EvidenceType.VENDOR, source, "groupid", mavenArtifact.getGroupId(), confidence);
518         }
519         if (mavenArtifact.getArtifactId() != null && !mavenArtifact.getArtifactId().isEmpty()) {
520             this.addEvidence(EvidenceType.PRODUCT, source, "artifactid", mavenArtifact.getArtifactId(), confidence);
521         }
522         if (mavenArtifact.getVersion() != null && !mavenArtifact.getVersion().isEmpty()) {
523             this.addEvidence(EvidenceType.VERSION, source, "version", mavenArtifact.getVersion(), confidence);
524         }
525         boolean found = false;
526         if (mavenArtifact.getArtifactUrl() != null && !mavenArtifact.getArtifactUrl().isEmpty()) {
527             synchronized (this) {
528                 for (Identifier i : this.softwareIdentifiers) {
529                     if (i instanceof PurlIdentifier) {
530                         final PurlIdentifier id = (PurlIdentifier) i;
531                         if (mavenArtifact.getArtifactId().equals(id.getName())
532                                 && mavenArtifact.getGroupId().equals(id.getNamespace())) {
533                             found = true;
534                             i.setConfidence(Confidence.HIGHEST);
535                             final String url = "http://search.maven.org/#search|ga|1|1%3A%22" + this.getSha1sum() + "%22";
536                             i.setUrl(url);
537                             //i.setUrl(mavenArtifact.getArtifactUrl());
538                             LOGGER.debug("Already found identifier {}. Confidence set to highest", i.getValue());
539                             break;
540                         }
541                     }
542                 }
543             }
544         }
545         if (!found && !Strings.isNullOrEmpty(mavenArtifact.getGroupId())
546                 && !Strings.isNullOrEmpty(mavenArtifact.getArtifactId())
547                 && !Strings.isNullOrEmpty(mavenArtifact.getVersion())) {
548             try {
549                 LOGGER.debug("Adding new maven identifier {}", mavenArtifact);
550                 final PackageURL p = new PackageURL("maven", mavenArtifact.getGroupId(),
551                         mavenArtifact.getArtifactId(), mavenArtifact.getVersion(), null, null);
552                 final PurlIdentifier id = new PurlIdentifier(p, Confidence.HIGHEST);
553                 this.addSoftwareIdentifier(id);
554             } catch (MalformedPackageURLException ex) {
555                 throw new UnexpectedAnalysisException(ex);
556             }
557         }
558     }

687     /**
688      * Determines the SHA1 and MD5 sum for the given file.
689      *
690      * @param hashFunction the hashing function
691      * @return the checksum
692      */
693     private String determineHashes(HashingFunction hashFunction) {
694         if (isVirtual) {
695             return null;
696         }
697         try {
698             final File file = getActualFile();
699             return hashFunction.hash(file);
700         } catch (IOException | RuntimeException ex) {
701             LOGGER.warn("Unable to read '{}' to determine hashes.", actualFilePath);
702             LOGGER.debug("", ex);
703         } catch (NoSuchAlgorithmException ex) {
704             LOGGER.warn("Unable to use MD5 or SHA1 checksums.");
705             LOGGER.debug("", ex);
706         }
707         return null;
708     }

775     /**
776      * Adds a related dependency.
777      *
778      * @param dependency a reference to the related dependency
779      */
780     @SuppressWarnings("ReferenceEquality")
781     public synchronized void addRelatedDependency(Dependency dependency) {
782         if (this == dependency) {
783             LOGGER.warn("Attempted to add a circular reference - please post the log file to issue #172 here "
784                     + "https://github.com/jeremylong/DependencyCheck/issues/172");
785             LOGGER.debug("this: {}", this);
786             LOGGER.debug("dependency: {}", dependency);
787         } else if (!relatedDependencies.add(dependency)) {
788             LOGGER.debug("Failed to add dependency, likely due to referencing the same file as another dependency in the set.");
789             LOGGER.debug("this: {}", this);
790             LOGGER.debug("dependency: {}", dependency);
791         }
792     }

831     /**
832      * Implementation of the equals method.
833      *
834      * @param obj the object to compare
835      * @return true if the objects are equal, otherwise false
836      */
837     @Override
838     public boolean equals(Object obj) {
839         if (obj == null || !(obj instanceof Dependency)) {
840             return false;
841         }
842         if (this == obj) {
843             return true;
844         }
845         final Dependency other = (Dependency) obj;
846         return new EqualsBuilder()
847                 .appendSuper(super.equals(obj))
848                 .append(this.actualFilePath, other.actualFilePath)
849                 .append(this.filePath, other.filePath)
850                 .append(this.fileName, other.fileName)
851                 .append(this.packagePath, other.packagePath)
852                 .append(this.md5sum, other.md5sum)
853                 .append(this.sha1sum, other.sha1sum)
854                 .append(this.sha256sum, other.sha256sum)
855                 .append(this.softwareIdentifiers, other.softwareIdentifiers)
856                 .append(this.vulnerabileSoftwareIdentifiers, other.vulnerabileSoftwareIdentifiers)
857                 .append(this.suppressedIdentifiers, other.suppressedIdentifiers)
858                 .append(this.description, other.description)
859                 .append(this.license, other.license)
860                 .append(this.vulnerabilities, other.vulnerabilities)
861                 .append(this.projectReferences, other.projectReferences)
862                 .append(this.availableVersions, other.availableVersions)
863                 .append(this.version, other.version)
864                 .append(this.ecosystem, other.ecosystem)
865                 .isEquals();
866     }
