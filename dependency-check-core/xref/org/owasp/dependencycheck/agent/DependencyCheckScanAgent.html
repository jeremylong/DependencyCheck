39  /**
40   * This class provides a way to easily conduct a scan solely based on existing
41   * evidence metadata rather than collecting evidence from the files themselves.
42   * This class is based on the Ant task and Maven plugin with the exception that
43   * it takes a list of dependencies that can be programmatically added from data
44   * in a spreadsheet, database or some other datasource and conduct a scan based
45   * on this pre-defined evidence.
46   *
47   * <h2>Example:</h2>
48   * <pre>
49   * List&lt;Dependency&gt; dependencies = new ArrayList&lt;Dependency&gt;();
50   * Dependency dependency = new Dependency(new File(FileUtils.getBitBucket()));
51   * dependency.addEvidence(EvidenceType.PRODUCT, "my-datasource", "name", "Jetty", Confidence.HIGH);
52   * dependency.addEvidence(EvidenceType.VERSION, "my-datasource", "version", "5.1.10", Confidence.HIGH);
53   * dependency.addEvidence(EvidenceType.VENDOR, "my-datasource", "vendor", "mortbay", Confidence.HIGH);
54   * dependencies.add(dependency);
55   *
56   * DependencyCheckScanAgent scan = new DependencyCheckScanAgent();
57   * scan.setDependencies(dependencies);
58   * scan.setReportFormat(ReportGenerator.Format.ALL);
59   * scan.setReportOutputDirectory(System.getProperty("user.home"));
60   * scan.execute();
61   * </pre>
62   *
63   * @author Steve Springett
64   */
65  @SuppressWarnings("unused")
66  @NotThreadSafe
67  public class DependencyCheckScanAgent {

853     /**
854      * Executes the Dependency-Check on the dependent libraries. <b>Note</b>,
855      * the engine object returned from this method must be closed by calling
856      * `close()`
857      *
858      * @return the Engine used to scan the dependencies.
859      * @throws ExceptionCollection a collection of one or more exceptions that
860      * occurred during analysis.
861      */
862     @SuppressWarnings("squid:S2095")
863     private Engine executeDependencyCheck() throws ExceptionCollection {
864         populateSettings();
865         final Engine engine;
866         try {
867             engine = new Engine(settings);
868         } catch (DatabaseException ex) {
869             throw new ExceptionCollection(ex, true);
870         }
871         if (this.updateOnly) {
872             try {
873                 engine.doUpdates();
874             } catch (UpdateException ex) {
875                 throw new ExceptionCollection(ex);
876             } finally {
877                 engine.close();
878             }
879         } else {
880             engine.setDependencies(this.dependencies);
881             engine.analyzeDependencies();
882         }
883         return engine;

953     /**
954      * Executes the dependency-check and generates the report.
955      *
956      * @return a reference to the engine used to perform the scan.
957      * @throws org.owasp.dependencycheck.exception.ScanAgentException thrown if
958      * there is an exception executing the scan.
959      */
960     public Engine execute() throws ScanAgentException {
961         Engine engine = null;
962         try {
963             engine = executeDependencyCheck();
964             if (!this.updateOnly) {
965                 if (this.generateReport) {
966                     generateExternalReports(engine, new File(this.reportOutputDirectory));
967                 }
968                 if (this.showSummary) {
969                     showSummary(engine.getDependencies());
970                 }
971                 if (this.failBuildOnCVSS <= 10) {
972                     checkForFailure(engine.getDependencies());
973                 }
974             }
975         } catch (ExceptionCollection ex) {
976             if (ex.isFatal()) {
977                 LOGGER.error("A fatal exception occurred during analysis; analysis has stopped. Please see the debug log for more details.");
978                 LOGGER.debug("", ex);
979             }
980             throw new ScanAgentException("One or more exceptions occurred during analysis; please see the debug log for more details.", ex);
981         } finally {
982             if (engine != null) {
983                 engine.close();
984             }
985             settings.cleanup(true);
986         }
987         return engine;
988     }

990     /**
991      * Checks to see if a vulnerability has been identified with a CVSS score
992      * that is above the threshold set in the configuration.
993      *
994      * @param dependencies the list of dependency objects
995      * @throws org.owasp.dependencycheck.exception.ScanAgentException thrown if
996      * there is an exception executing the scan.
997      */
998     private void checkForFailure(Dependency[] dependencies) throws ScanAgentException {
999         final StringBuilder ids = new StringBuilder();
1000         for (Dependency d : dependencies) {
1001             boolean addName = true;
1002             for (Vulnerability v : d.getVulnerabilities()) {
1003                 if (v.getCvssV2().getScore() >= failBuildOnCVSS) {
1004                     if (addName) {
1005                         addName = false;
1006                         ids.append(NEW_LINE).append(d.getFileName()).append(": ");
1007                         ids.append(v.getName());
1008                     } else {
1009                         ids.append(", ").append(v.getName());
1010                     }
1011                 }
1012             }
1013         }
1014         if (ids.length() > 0) {
1015             final String msg;
1016             if (showSummary) {
1017                 msg = String.format("%n%nDependency-Check Failure:%n"
1018                         + "One or more dependencies were identified with vulnerabilities that have a CVSS score greater than or equal to '%.1f': %s%n"
1019                         + "See the dependency-check report for more details.%n%n", failBuildOnCVSS, ids.toString());
1020             } else {
1021                 msg = String.format("%n%nDependency-Check Failure:%n"
1022                         + "One or more dependencies were identified with vulnerabilities.%n%n"
1023                         + "See the dependency-check report for more details.%n%n");
1024             }
1025             throw new ScanAgentException(msg);
1026         }
1027     }

1039     /**
1040      * Generates a warning message listing a summary of dependencies and their
1041      * associated CPE and CVE entries.
1042      *
1043      * @param projectName the name of the project
1044      * @param dependencies a list of dependency objects
1045      */
1046     public static void showSummary(String projectName, Dependency[] dependencies) {
1047         final StringBuilder summary = new StringBuilder();
1048         for (Dependency d : dependencies) {
1049             final String ids = d.getVulnerabilities(true).stream()
1050                     .map(v -> v.getName())
1051                     .collect(Collectors.joining(", "));
1052             if (ids.length() > 0) {
1053                 summary.append(d.getFileName()).append(" (");
1054                 summary.append(Stream.concat(d.getSoftwareIdentifiers().stream(), d.getVulnerableSoftwareIdentifiers().stream())
1055                         .map(i -> i.getValue())
1056                         .collect(Collectors.joining(", ")));
1057                 summary.append(") : ").append(ids).append(NEW_LINE);
1058             }
1059         }
1060         if (summary.length() > 0) {
1061             if (projectName == null || projectName.isEmpty()) {
1062                 LOGGER.warn("\n\nOne or more dependencies were identified with known vulnerabilities:\n\n{}\n\n"
1063                         + "See the dependency-check report for more details.\n\n",
1064                         summary.toString());
1065             } else {
1066                 LOGGER.warn("\n\nOne or more dependencies were identified with known vulnerabilities in {}:\n\n{}\n\n"
1067                         + "See the dependency-check report for more details.\n\n",
1068                         projectName,
1069                         summary.toString());
1070             }
1071         }
1072     }
