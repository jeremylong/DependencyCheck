18  package org.owasp.dependencycheck.utils;
19  
20  import org.apache.commons.lang3.StringUtils;
21  import org.owasp.dependencycheck.analyzer.exception.AnalysisException;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.io.BufferedReader;
26  import java.io.File;
27  import java.io.IOException;
28  import java.math.BigDecimal;
29  import java.nio.charset.StandardCharsets;
30  import java.nio.file.Files;
31  import java.util.Properties;
32  
33  /**
34   * A utility class to handle Python Packaging Authority (PyPA) core metadata files. It was created based on the
35   * <a href="https://packaging.python.org/en/latest/specifications/core-metadata/">specification by PyPA</a> for
36   * version 2.2
37   *
38   * @author Hans Aikema
39   */
40  public final class PyPACoreMetadataParser {
41  
42      /**
43       * The logger.
44       */
45      private static final Logger LOGGER = LoggerFactory.getLogger(PyPACoreMetadataParser.class);
46  
47      /**
48       * The largest major version considered by this parser
49       */
50      private static final int SUPPORTED_MAJOR_UPPERBOUND = 2;
51  
52      /**
53       * The largest version of the specification considered during coding of this parser
54       */
55      private static final BigDecimal MAX_SUPPORTED_VERSION = BigDecimal.valueOf(22, 1);
56  
57      private PyPACoreMetadataParser() {
58          // hide constructor for utility class
59      }
60      /**
61       * Loads all key/value pairs from PyPA metadata specificationsÂ¶.
62       *
63       * @param file
64       *         The Wheel metadata of a Python package as a File
65       *
66       * @return The metadata properties read from the file
67       */
68      public static Properties getProperties(File file) throws AnalysisException {
69          try (BufferedReader utf8Reader = Files.newBufferedReader(file.toPath(), StandardCharsets.UTF_8)) {
70              return getProperties(utf8Reader);
71          } catch (IOException | IllegalArgumentException e) {
72              throw new AnalysisException("Error parsing PyPA core-metadata file", e);
73          }
74      }
75  
76      /**
77       * Loads all key/value pairs from PyPA metadata specificationsÂ¶.
78       *
79       * @param utf8Reader
80       *         The Wheel metadata of a Python package as a BufferedReader
81       *
82       * @return The metadata properties read from the utf8Reader
83       * @throws java.io.IOException thrown if there is error reading the properties
84       */
85      public static Properties getProperties(final BufferedReader utf8Reader) throws IOException {
86          final Properties result = new Properties();
87          String line = utf8Reader.readLine();
88          StringBuilder singleHeader = null;
89          boolean inDescription = false;
90          while (line != null && !line.isEmpty()) {
91              if (inDescription && line.startsWith("       |")) {
92                  singleHeader.append('\n').append(line.substring(8));
93              } else if (singleHeader != null && line.startsWith(" ")) {
94                  singleHeader.append(line.substring(1));
95              } else {
96                  if (singleHeader != null) {
97                      parseAndAddHeader(result, singleHeader);
98                  }
99                  singleHeader = new StringBuilder(line);
100                 inDescription = line.startsWith("Description:");
101             }
102             line = utf8Reader.readLine();
103         }
104         if (singleHeader != null) {
105             parseAndAddHeader(result, singleHeader);
106         }
107         // ignore a body if any (description is allowed to be the message body)
108         return result;
109     }
110 
111     /**
112      * Add a single metadata keyvalue pair to the metadata. When the given metadataHeader cannot be parsed as a '{@code key: value}'
113      * line a warning is emitted and the line is ignored.
114      *
115      * @param metadata
116      *         The collected metadata to which the new metadataHeader must be added
117      * @param metadataHeader
118      *         A single uncollapsed header line of the metadata
119      *
120      * @throws IllegalArgumentException
121      *         When the given metadataHeader has a key {@code Metadata-Version} and the value holds a major version that is larger
122      *         than the highest supported metadata version. As defined by the specification: <blockquote>Automated tools consuming
123      *         metadata SHOULD warn if metadata_version is greater than the highest version they support, and MUST fail if
124      *         metadata_version has a greater major version than the highest version they support (as described in PEP 440, the
125      *         major version is the value before the first dot).</blockquote>
126      */
127     private static void parseAndAddHeader(final Properties metadata, final StringBuilder metadataHeader) {
128         final String[] keyValue = StringUtils.split(metadataHeader.toString(), ":", 2);
129         if (keyValue.length != 2) {
130             LOGGER.warn("Invalid mailheader format encountered in Wheel Metadata, not a \"key: value\" string");
131             return;
132         }
133         final String key = keyValue[0];
134         final String value = keyValue[1].trim();
135         if ("Metadata-Version".equals(key)) {
136             final int majorVersion = Integer.parseInt(value.substring(0, value.indexOf('.')), 10);
137             final BigDecimal version = new BigDecimal(value);
138             if (majorVersion > SUPPORTED_MAJOR_UPPERBOUND) {
139                 throw new IllegalArgumentException(String.format(
140                         "Unsupported PyPA Wheel metadata. Metadata-Version " + "is '%s', largest supported major is %d", value,
141                         SUPPORTED_MAJOR_UPPERBOUND));
142             } else if (version.compareTo(MAX_SUPPORTED_VERSION) > 0 && LOGGER.isWarnEnabled()) {
143                 LOGGER.warn(String.format("Wheel metadata Metadata-Version (%s) has a larger minor version than the highest known "
144                                           + "supported Metadata specification (%s) continuing with best effort", value,
145                                           MAX_SUPPORTED_VERSION));
146             }
147         }
148         metadata.setProperty(key, value);
149     }
