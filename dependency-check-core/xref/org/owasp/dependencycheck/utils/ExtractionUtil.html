62      /**
63       * Extracts the contents of an archive into the specified directory.
64       *
65       * @param archive an archive file such as a WAR or EAR
66       * @param extractTo a directory to extract the contents to
67       * @throws ExtractionException thrown if an exception occurs while
68       * extracting the files
69       */
70      public static void extractFiles(File archive, File extractTo) throws ExtractionException {
71          extractFiles(archive, extractTo, null);
72      }

74      /**
75       * Extracts the contents of an archive into the specified directory. The
76       * files are only extracted if they are supported by the analyzers loaded
77       * into the specified engine. If the engine is specified as null then all
78       * files are extracted.
79       *
80       * @param archive an archive file such as a WAR or EAR
81       * @param extractTo a directory to extract the contents to
82       * @param engine the scanning engine
83       * @throws ExtractionException thrown if there is an error extracting the
84       * files
85       */
86      public static void extractFiles(File archive, File extractTo, Engine engine) throws ExtractionException {
87          if (archive == null || extractTo == null) {
88              return;
89          }
90          final String destPath;
91          try {
92              destPath = extractTo.getCanonicalPath();
93          } catch (IOException ex) {
94              throw new ExtractionException("Unable to extract files to destination path", ex);
95          }
96          ZipEntry entry;
97          try (FileInputStream fis = new FileInputStream(archive);
98                  BufferedInputStream bis = new BufferedInputStream(fis);
99                  ZipInputStream zis = new ZipInputStream(bis)) {
100             while ((entry = zis.getNextEntry()) != null) {
101                 if (entry.isDirectory()) {
102                     final File d = new File(extractTo, entry.getName());
103                     if (!d.getCanonicalPath().startsWith(destPath)) {
104                         final String msg = String.format(
105                                 "Archive (%s) contains a path that would be extracted outside of the target directory.",
106                                 archive.getAbsolutePath());
107                         throw new ExtractionException(msg);
108                     }
109                     if (!d.exists() && !d.mkdirs()) {
110                         final String msg = String.format("Unable to create '%s'.", d.getAbsolutePath());
111                         throw new ExtractionException(msg);
112                     }
113                 } else {
114                     final File file = new File(extractTo, entry.getName());
115                     if (engine == null || engine.accept(file)) {
116                         if (!file.getCanonicalPath().startsWith(destPath)) {
117                             final String msg = String.format(
118                                     "Archive (%s) contains a file that would be extracted outside of the target directory.",
119                                     archive.getAbsolutePath());
120                             throw new ExtractionException(msg);
121                         }
122                         try (FileOutputStream fos = new FileOutputStream(file)) {
123                             IOUtils.copy(zis, fos);
124                         } catch (FileNotFoundException ex) {
125                             LOGGER.debug("", ex);
126                             final String msg = String.format("Unable to find file '%s'.", file.getName());
127                             throw new ExtractionException(msg, ex);
128                         } catch (IOException ex) {
129                             LOGGER.debug("", ex);
130                             final String msg = String.format("IO Exception while parsing file '%s'.", file.getName());
131                             throw new ExtractionException(msg, ex);
132                         }
133                     }
134                 }
135             }
136         } catch (IOException ex) {
137             final String msg = String.format("Exception reading archive '%s'.", archive.getName());
138             LOGGER.debug("", ex);
139             throw new ExtractionException(msg, ex);
140         }
141     }

143     /**
144      * Extracts the contents of an archive into the specified directory. The
145      * files are only extracted if they are supported by the analyzers loaded
146      * into the specified engine. If the engine is specified as null then all
147      * files are extracted.
148      *
149      * @param archive an archive file such as a WAR or EAR
150      * @param extractTo a directory to extract the contents to
151      * @throws ExtractionException thrown if there is an error extracting the
152      * files
153      */
154     public static void extractFiles(InputStream archive, File extractTo) throws ExtractionException {
155         if (archive == null || extractTo == null) {
156             return;
157         }
158         final String destPath;
159         try {
160             destPath = extractTo.getCanonicalPath();
161         } catch (IOException ex) {
162             throw new ExtractionException("Unable to extract files to destination path", ex);
163         }
164         ZipEntry entry;
165         try (BufferedInputStream bis = new BufferedInputStream(archive);
166                 ZipInputStream zis = new ZipInputStream(bis)) {
167             while ((entry = zis.getNextEntry()) != null) {
168                 if (entry.isDirectory()) {
169                     final File d = new File(extractTo, entry.getName());
170                     if (!d.getCanonicalPath().startsWith(destPath)) {
171                         throw new ExtractionException("Archive contains a path that would be extracted outside of the target directory.");
172                     }
173                     if (!d.exists() && !d.mkdirs()) {
174                         final String msg = String.format("Unable to create '%s'.", d.getAbsolutePath());
175                         throw new ExtractionException(msg);
176                     }
177                 } else {
178                     final File file = new File(extractTo, entry.getName());
179                     if (!file.getCanonicalPath().startsWith(destPath)) {
180                         LOGGER.debug("ZipSlip detected\n-Destination: " + destPath + "\n-Path: " + file);
181                         throw new ExtractionException("Archive contains a file that would be extracted outside of the target directory.");
182                     }
183                     try (FileOutputStream fos = new FileOutputStream(file)) {
184                         IOUtils.copy(zis, fos);
185                     } catch (FileNotFoundException ex) {
186                         LOGGER.debug("", ex);
187                         final String msg = String.format("Unable to find file '%s'.", file.getName());
188                         throw new ExtractionException(msg, ex);
189                     } catch (IOException ex) {
190                         LOGGER.debug("", ex);
191                         final String msg = String.format("IO Exception while parsing file '%s'.", file.getName());
192                         throw new ExtractionException(msg, ex);
193                     }
194                 }
195             }
196         } catch (IOException ex) {
197             throw new ExtractionException("Exception reading archive", ex);
198         }
199     }

201     /**
202      * Extracts the contents of an archive into the specified directory.
203      *
204      * @param archive an archive file such as a WAR or EAR
205      * @param destination a directory to extract the contents to
206      * @param filter determines which files get extracted
207      * @throws ExtractionException thrown if the archive is not found
208      */
209     public static void extractFilesUsingFilter(File archive, File destination, FilenameFilter filter) throws ExtractionException {
210         if (archive == null || destination == null) {
211             return;
212         }
213 
214         try (FileInputStream fis = new FileInputStream(archive)) {
215             extractArchive(new ZipArchiveInputStream(new BufferedInputStream(fis)), destination, filter);
216         } catch (FileNotFoundException ex) {
217             final String msg = String.format("Error extracting file `%s` with filter: %s", archive.getAbsolutePath(), ex.getMessage());
218             LOGGER.debug(msg, ex);
219             throw new ExtractionException(msg);
220         } catch (IOException | ArchiveExtractionException ex) {
221             LOGGER.warn("Exception extracting archive '{}'.", archive.getAbsolutePath());
222             LOGGER.debug("", ex);
223             throw new ExtractionException("Unable to extract from archive", ex);
224         }
225     }

227     /**
228      * Extracts files from an archive.
229      *
230      * @param input the archive to extract files from
231      * @param destination the location to write the files too
232      * @param filter determines which files get extracted
233      * @throws ArchiveExtractionException thrown if there is an exception
234      * extracting files from the archive
235      */
236     private static void extractArchive(ArchiveInputStream input,
237             File destination, FilenameFilter filter)
238             throws ArchiveExtractionException {
239         ArchiveEntry entry;
240         try {
241             final String destPath = destination.getCanonicalPath();
242 
243             while ((entry = input.getNextEntry()) != null) {
244                 if (entry.isDirectory()) {
245                     final File dir = new File(destination, entry.getName());
246                     if (!dir.getCanonicalPath().startsWith(destPath)) {
247                         final String msg = String.format(
248                                 "Archive contains a path (%s) that would be extracted outside of the target directory.",
249                                 dir.getAbsolutePath());
250                         throw new AnalysisException(msg);
251                     }
252                     if (!dir.exists() && !dir.mkdirs()) {
253                         final String msg = String.format(
254                                 "Unable to create directory '%s'.",
255                                 dir.getAbsolutePath());
256                         throw new AnalysisException(msg);
257                     }
258                 } else {
259                     extractFile(input, destination, filter, entry);
260                 }
261             }
262         } catch (IOException | AnalysisException ex) {
263             throw new ArchiveExtractionException(ex);
264         } finally {
265             FileUtils.close(input);
266         }
267     }

269     /**
270      * Extracts a file from an archive (input stream) and correctly builds the
271      * directory structure.
272      *
273      * @param input the archive input stream
274      * @param destination where to write the file
275      * @param filter the file filter to apply to the files being extracted
276      * @param entry the entry from the archive to extract
277      * @throws ExtractionException thrown if there is an error reading from the
278      * archive stream
279      */
280     private static void extractFile(ArchiveInputStream input, File destination,
281             FilenameFilter filter, ArchiveEntry entry) throws ExtractionException {
282         final File file = new File(destination, entry.getName());
283         try {
284             if (filter.accept(file.getParentFile(), file.getName())) {
285                 final String destPath = destination.getCanonicalPath();
286                 if (!file.getCanonicalPath().startsWith(destPath)) {
287                     LOGGER.debug("ZipSlip detected\n-Destination: " + destPath + "\n-Path: " + file);
288                     final String msg = String.format(
289                             "Archive contains a file (%s) that would be extracted outside of the target directory.",
290                             file.getAbsolutePath());
291                     throw new ExtractionException(msg);
292                 }
293                 LOGGER.debug("Extracting '{}'", file.getPath());
294                 createParentFile(file);
295 
296                 try (FileOutputStream fos = new FileOutputStream(file)) {
297                     IOUtils.copy(input, fos);
298                 } catch (FileNotFoundException ex) {
299                     LOGGER.debug("", ex);
300                     final String msg = String.format("Unable to find file '%s'.", file.getName());
301                     throw new ExtractionException(msg, ex);
302                 }
303             }
304         } catch (IOException ex) {
305             LOGGER.debug("", ex);
306             final String msg = String.format("IO Exception while parsing file '%s'.", file.getName());
307             throw new ExtractionException(msg, ex);
308         }
309     }

311     /**
312      * Ensures the parent path is correctly created on disk so that the file can
313      * be extracted to the correct location.
314      *
315      * @param file the file path
316      * @throws ExtractionException thrown if the parent paths could not be
317      * created
318      */
319     private static void createParentFile(final File file) throws ExtractionException {
320         final File parent = file.getParentFile();
321         if (!parent.isDirectory() && !parent.mkdirs()) {
322             final String msg = String.format(
323                     "Unable to build directory '%s'.",
324                     parent.getAbsolutePath());
325             throw new ExtractionException(msg);
326         }
327     }

329     /**
330      * Extracts the file contained in a gzip archive. The extracted file is
331      * placed in the exact same path as the file specified.
332      *
333      * @param file the archive file
334      * @throws FileNotFoundException thrown if the file does not exist
335      * @throws IOException thrown if there is an error extracting the file.
336      */
337     public static void extractGzip(File file) throws FileNotFoundException, IOException {
338         final String originalPath = file.getPath();
339         final File gzip = new File(originalPath + ".gz");
340         if (gzip.isFile() && !gzip.delete()) {
341             LOGGER.debug("Failed to delete initial temporary file when extracting 'gz' {}", gzip);
342             gzip.deleteOnExit();
343         }
344         if (!file.renameTo(gzip)) {
345             throw new IOException("Unable to rename '" + file.getPath() + "'");
346         }
347         final File newFile = new File(originalPath);
348         try (FileInputStream fis = new FileInputStream(gzip);
349                 GZIPInputStream cin = new GZIPInputStream(fis);
350                 FileOutputStream out = new FileOutputStream(newFile)) {
351             IOUtils.copy(cin, out);
352         } finally {
353             if (gzip.isFile() && !org.apache.commons.io.FileUtils.deleteQuietly(gzip)) {
354                 LOGGER.debug("Failed to delete temporary file when extracting 'gz' {}", gzip);
355                 gzip.deleteOnExit();
356             }
357         }
358     }

360     /**
361      * Extracts the file contained in a Zip archive. The extracted file is
362      * placed in the exact same path as the file specified.
363      *
364      * @param file the archive file
365      * @throws FileNotFoundException thrown if the file does not exist
366      * @throws IOException thrown if there is an error extracting the file.
367      */
368     public static void extractZip(File file) throws FileNotFoundException, IOException {
369         final String originalPath = file.getPath();
370         final File zip = new File(originalPath + ".zip");
371         if (zip.isFile() && !zip.delete()) {
372             LOGGER.debug("Failed to delete initial temporary file when extracting 'zip' {}", zip);
373             zip.deleteOnExit();
374         }
375         if (!file.renameTo(zip)) {
376             throw new IOException("Unable to rename '" + file.getPath() + "'");
377         }
378         final File newFile = new File(originalPath);
379         try (FileInputStream fis = new FileInputStream(zip);
380                 ZipInputStream cin = new ZipInputStream(fis);
381                 FileOutputStream out = new FileOutputStream(newFile)) {
382             cin.getNextEntry();
383             IOUtils.copy(cin, out);
384         } finally {
385             if (zip.isFile() && !org.apache.commons.io.FileUtils.deleteQuietly(zip)) {
386                 LOGGER.debug("Failed to delete temporary file when extracting 'zip' {}", zip);
387                 zip.deleteOnExit();
388             }
389         }
