18  package org.owasp.dependencycheck.data.lucene;
19  
20  import java.io.IOException;
21  import org.apache.lucene.analysis.Analyzer;
22  import org.apache.lucene.analysis.TokenStream;
23  import org.apache.lucene.analysis.Tokenizer;
24  import org.apache.lucene.analysis.core.LowerCaseFilter;
25  import org.apache.lucene.analysis.core.StopFilter;
26  import org.apache.lucene.analysis.core.WhitespaceTokenizer;
27  import org.apache.lucene.analysis.miscellaneous.WordDelimiterGraphFilter;
28  import org.apache.lucene.analysis.CharArraySet;
29  import org.apache.lucene.analysis.en.EnglishAnalyzer;
30  
31  /**
32   * A Lucene field analyzer used to analyzer queries against the CPE data.
33   *
34   * @author Jeremy Long
35   */
36  public class SearchFieldAnalyzer extends Analyzer {
37  
38      /**
39       * The list of additional stop words to use.
40       */
41      private static final String[] ADDITIONAL_STOP_WORDS = {"software", "framework", "inc",
42          "com", "org", "net", "www", "consulting", "ltd", "foundation", "project"};
43      /**
44       * The set of stop words to use in the analyzer.
45       */
46      private final CharArraySet stopWords;
47      /**
48       * A reference to the concatenating filter so that it can be reset/cleared.
49       */
50      private TokenPairConcatenatingFilter concatenatingFilter;
51  
52      /**
53       * Returns the set of stop words being used.
54       *
55       * @return the set of stop words being used
56       */
57      public static CharArraySet getStopWords() {
58          final CharArraySet words = StopFilter.makeStopSet(ADDITIONAL_STOP_WORDS, true);
59          words.addAll(EnglishAnalyzer.ENGLISH_STOP_WORDS_SET);
60          return words;
61      }
62  
63      /**
64       * Constructs a new SearchFieldAnalyzer.
65       *
66       */
67      public SearchFieldAnalyzer() {
68          stopWords = getStopWords();
69      }
70  
71      /**
72       * Creates a the TokenStreamComponents used to analyze the stream.
73       *
74       * @param fieldName the field that this lucene analyzer will process
75       * @return the token stream filter chain
76       */
77      @Override
78      protected TokenStreamComponents createComponents(String fieldName) {
79          //final Tokenizer source = new AlphaNumericTokenizer();
80          final Tokenizer source = new WhitespaceTokenizer();
81          TokenStream stream = source;
82  
83          stream = new UrlTokenizingFilter(stream);
84          stream = new AlphaNumericFilter(stream);
85          stream = new WordDelimiterGraphFilter(stream,
86                  WordDelimiterGraphFilter.GENERATE_WORD_PARTS
87                  //| WordDelimiterGraphFilter.GENERATE_NUMBER_PARTS
88                  | WordDelimiterGraphFilter.PRESERVE_ORIGINAL
89                  | WordDelimiterGraphFilter.SPLIT_ON_CASE_CHANGE
90                  | WordDelimiterGraphFilter.SPLIT_ON_NUMERICS
91                  | WordDelimiterGraphFilter.STEM_ENGLISH_POSSESSIVE, null);
92  
93          stream = new LowerCaseFilter(stream);
94  
95          stream = new StopFilter(stream, stopWords);
96          concatenatingFilter = new TokenPairConcatenatingFilter(stream);
97  
98          return new TokenStreamComponents(source, concatenatingFilter);
99      }
100 
101     /**
102      * Resets the analyzer. This must be manually called between searching and
103      * indexing.
104      *
105      * @throws IOException thrown if there is an error resetting the tokenizer
106      */
107     public void reset() throws IOException {
108         if (concatenatingFilter != null) {
109             concatenatingFilter.clear();
110         }
111     }
112 }
