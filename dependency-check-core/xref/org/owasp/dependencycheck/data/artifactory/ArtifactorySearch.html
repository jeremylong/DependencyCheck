51  /**
52   * Class of methods to search Artifactory for hashes and determine Maven GAV
53   * from there.
54   *
55   * Data classes copied from JFrog's artifactory-client-java project.
56   *
57   * @author nhenneaux
58   */
59  @ThreadSafe
60  @SuppressWarnings("squid:S2647")
61  public class ArtifactorySearch {
62  
63      /**
64       * Used for logging.
65       */
66      private static final Logger LOGGER = LoggerFactory.getLogger(ArtifactorySearch.class);
67  
68      /**
69       * Pattern to match the path returned by the Artifactory AQL API.
70       */
71      private static final Pattern PATH_PATTERN = Pattern.compile("^/(?<groupId>.+)/(?<artifactId>[^/]+)/(?<version>[^/]+)/[^/]+$");
72      /**
73       * Extracted duplicateArtifactorySearchIT.java comment.
74       */
75      private static final String WHILE_ACTUAL_IS = " while actual is ";
76      /**
77       * The URL for the Central service.
78       */
79      private final String rootURL;
80  
81      /**
82       * Whether to use the Proxy when making requests.
83       */
84      private final boolean useProxy;
85  
86      /**
87       * The configured settings.
88       */
89      private final Settings settings;
90  
91      /**
92       * Search result reader
93       */
94      private final ObjectReader objectReader;

96      /**
97       * Creates a NexusSearch for the given repository URL.
98       *
99       * @param settings the configured settings
100      */
101     public ArtifactorySearch(Settings settings) {
102         this.settings = settings;
103 
104         final String searchUrl = settings.getString(Settings.KEYS.ANALYZER_ARTIFACTORY_URL);
105 
106         this.rootURL = searchUrl;
107         LOGGER.debug("Artifactory Search URL {}", searchUrl);
108 
109         if (null != settings.getString(Settings.KEYS.PROXY_SERVER)) {
110             boolean useProxySettings = false;
111             try {
112                 useProxySettings = settings.getBoolean(Settings.KEYS.ANALYZER_ARTIFACTORY_USES_PROXY);
113             } catch (InvalidSettingException e) {
114                 LOGGER.error("Settings {} is invalid, only, true/false is valid", Settings.KEYS.ANALYZER_ARTIFACTORY_USES_PROXY, e);
115             }
116             this.useProxy = useProxySettings;
117             LOGGER.debug("Using proxy? {}", useProxy);
118         } else {
119             useProxy = false;
120             LOGGER.debug("Not using proxy");
121         }
122 
123         objectReader = new ObjectMapper().configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false).readerFor(FileImpl.class);
124     }

126     /**
127      * Searches the configured Central URL for the given hash (MD5, SHA1 and
128      * SHA256). If the artifact is found, a <code>MavenArtifact</code> is
129      * populated with the GAV.
130      *
131      * @param dependency the dependency for which to search (search is based on
132      * hashes)
133      * @return the populated Maven GAV.
134      * @throws FileNotFoundException if the specified artifact is not found
135      * @throws IOException if it's unable to connect to the specified repository
136      */
137     public List<MavenArtifact> search(Dependency dependency) throws IOException {
138 
139         final String sha1sum = dependency.getSha1sum();
140         final URL url = buildUrl(sha1sum);
141         final HttpURLConnection conn = connect(url);
142         final int responseCode = conn.getResponseCode();
143         if (responseCode == 200) {
144             return processResponse(dependency, conn);
145         }
146         throw new IOException("Could not connect to Artifactory " + url + " (" + responseCode + "): " + conn.getResponseMessage());
147 
148     }
149 
150     /**
151      * Makes an connection to the given URL.
152      *
153      * @param url the URL to connect to
154      * @return the HTTP URL Connection
155      * @throws IOException thrown if there is an error making the connection
156      */
157     private HttpURLConnection connect(URL url) throws IOException {
158         LOGGER.debug("Searching Artifactory url {}", url);
159 
160         // Determine if we need to use a proxy. The rules:
161         // 1) If the proxy is set, AND the setting is set to true, use the proxy
162         // 2) Otherwise, don't use the proxy (either the proxy isn't configured,
163         // or proxy is specifically set to false)
164         final URLConnectionFactory factory = new URLConnectionFactory(settings);
165         final HttpURLConnection conn = factory.createHttpURLConnection(url, useProxy);
166         conn.setDoOutput(true);
167 
168         conn.addRequestProperty("X-Result-Detail", "info");
169 
170         final String username = settings.getString(Settings.KEYS.ANALYZER_ARTIFACTORY_API_USERNAME);
171         final String apiToken = settings.getString(Settings.KEYS.ANALYZER_ARTIFACTORY_API_TOKEN);
172         if (username != null && apiToken != null) {
173             final String userpassword = username + ":" + apiToken;
174             final String encodedAuthorization = Base64.getEncoder().encodeToString(userpassword.getBytes(StandardCharsets.UTF_8));
175             conn.addRequestProperty("Authorization", "Basic " + encodedAuthorization);
176         } else {
177             final String bearerToken = settings.getString(Settings.KEYS.ANALYZER_ARTIFACTORY_BEARER_TOKEN);
178             if (bearerToken != null) {
179                 conn.addRequestProperty("Authorization", "Bearer " + bearerToken);
180             }
181         }
182 
183         conn.connect();
184         return conn;
185     }
186 
187     /**
188      * Constructs the URL using the SHA1 checksum.
189      *
190      * @param sha1sum the SHA1 checksum
191      * @return the API URL to search for the given checksum
192      * @throws MalformedURLException thrown if the URL is malformed
193      */
194     private URL buildUrl(String sha1sum) throws MalformedURLException {
195         // TODO Investigate why sha256 parameter is not working
196         // API defined https://www.jfrog.com/confluence/display/RTF/Artifactory+REST+API#ArtifactoryRESTAPI-ChecksumSearch
197         return new URL(rootURL + "/api/search/checksum?sha1=" + sha1sum);
198     }

200     /**
201      * Process the Artifactory response.
202      *
203      * @param dependency the dependency
204      * @param conn the HTTP URL Connection
205      * @return a list of the Maven Artifact information
206      * @throws IOException thrown if there is an I/O error
207      */
208     protected List<MavenArtifact> processResponse(Dependency dependency, HttpURLConnection conn) throws IOException {
209         final List<MavenArtifact> result = new ArrayList<>();
210 
211         try (InputStreamReader streamReader = new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8);
212                 JsonParser parser = objectReader.getFactory().createParser(streamReader)) {
213 
214             if (init(parser) && parser.nextToken() == com.fasterxml.jackson.core.JsonToken.START_OBJECT) {
215                 // at least one result
216                 do {
217                     final FileImpl file = objectReader.readValue(parser);
218 
219                     checkHashes(dependency, file.getChecksums());
220 
221                     final Matcher pathMatcher = PATH_PATTERN.matcher(file.getPath());
222                     if (!pathMatcher.matches()) {
223                         throw new IllegalStateException("Cannot extract the Maven information from the path "
224                                 + "retrieved in Artifactory " + file.getPath());
225                     }
226 
227                     final String groupId = pathMatcher.group("groupId").replace('/', '.');
228                     final String artifactId = pathMatcher.group("artifactId");
229                     final String version = pathMatcher.group("version");
230 
231                     result.add(new MavenArtifact(groupId, artifactId, version, file.getDownloadUri(),
232                             MavenArtifact.derivePomUrl(artifactId, version, file.getDownloadUri())));
233 
234                 } while (parser.nextToken() == com.fasterxml.jackson.core.JsonToken.START_OBJECT);
235             } else {
236                 throw new FileNotFoundException("Artifact " + dependency + " not found in Artifactory");
237             }
238         }
239 
240         return result;
241     }
242 
243     protected boolean init(JsonParser parser) throws IOException {
244         com.fasterxml.jackson.core.JsonToken nextToken = parser.nextToken();
245         if (nextToken != com.fasterxml.jackson.core.JsonToken.START_OBJECT) {
246             throw new IOException("Expected " + com.fasterxml.jackson.core.JsonToken.START_OBJECT + ", got " + nextToken);
247         }
248 
249         do {
250             nextToken = parser.nextToken();
251             if (nextToken == null) {
252                 break;
253             }
254 
255             if (nextToken.isStructStart()) {
256                 if (nextToken == com.fasterxml.jackson.core.JsonToken.START_ARRAY && Objects.equal("results", parser.currentName())) {
257                     return true;
258                 } else {
259                     parser.skipChildren();
260                 }
261             }
262         } while (true);
263 
264         return false;
265     }

267     /**
268      * Validates the hashes of the dependency.
269      *
270      * @param dependency the dependency
271      * @param checksums the collection of checksums (md5, sha1, sha256)
272      * @throws FileNotFoundException thrown if one of the checksums does not
273      * match
274      */
275     private void checkHashes(Dependency dependency, ChecksumsImpl checksums) throws FileNotFoundException {
276         final String md5sum = dependency.getMd5sum();
277         if (!checksums.getMd5().equals(md5sum)) {
278             throw new FileNotFoundException("Artifact found by API is not matching the md5 "
279                     + "of the artifact (repository hash is " + checksums.getMd5() + WHILE_ACTUAL_IS + md5sum + ") !");
280         }
281         final String sha1sum = dependency.getSha1sum();
282         if (!checksums.getSha1().equals(sha1sum)) {
283             throw new FileNotFoundException("Artifact found by API is not matching the SHA1 "
284                     + "of the artifact (repository hash is " + checksums.getSha1() + WHILE_ACTUAL_IS + sha1sum + ") !");
285         }
286         final String sha256sum = dependency.getSha256sum();
287         if (checksums.getSha256() != null && !checksums.getSha256().equals(sha256sum)) {
288             throw new FileNotFoundException("Artifact found by API is not matching the SHA-256 "
289                     + "of the artifact (repository hash is " + checksums.getSha256() + WHILE_ACTUAL_IS + sha256sum + ") !");
290         }
291     }
292 
293     /**
294      * Performs a pre-flight request to ensure the Artifactory service is
295      * reachable.
296      *
297      * @return <code>true</code> if Artifactory could be reached; otherwise
298      * <code>false</code>.
299      */
300     public boolean preflightRequest() {
301         try {
302             final URL url = buildUrl(Checksum.getSHA1Checksum(UUID.randomUUID().toString()));
303             final HttpURLConnection connection = connect(url);
304             if (connection.getResponseCode() != 200) {
305                 LOGGER.warn("Expected 200 result from Artifactory ({}), got {}", url, connection.getResponseCode());
306                 return false;
307             }
308             return true;
309         } catch (IOException e) {
310             LOGGER.error("Cannot connect to Artifactory", e);
311             return false;
312         }

314     }
