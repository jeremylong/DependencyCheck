18  package org.owasp.dependencycheck.data.cwe;
19  
20  import java.util.HashMap;
21  import javax.annotation.concurrent.NotThreadSafe;
22  import org.xml.sax.Attributes;
23  import org.xml.sax.SAXException;
24  import org.xml.sax.helpers.DefaultHandler;
25  
26  /**
27   * A SAX Handler that will parse the CWE XML.
28   *
29   * @author Jeremy Long
30   */
31  @NotThreadSafe
32  public class CweHandler extends DefaultHandler {
33  
34      /**
35       * a HashMap containing the CWE data.
36       */
37      private final HashMap<String, String> cwe = new HashMap<>();
38  
39      /**
40       * Returns the HashMap of CWE entries (CWE-ID, Full CWE Name).
41       *
42       * @return a HashMap of CWE entries &lt;String, String&gt;
43       */
44      public HashMap<String, String> getCwe() {
45          return cwe;
46      }
47  
48      @Override
49      public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
50  
51          if ("Weakness".equals(qName) || "Category".equals(qName)) {
52              final String id = "CWE-" + attributes.getValue("ID");
53              final String name = attributes.getValue("Name");
54              cwe.put(id, name);
55          }
56      }
57  }
