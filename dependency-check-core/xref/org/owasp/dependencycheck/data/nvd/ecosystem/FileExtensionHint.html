18  package org.owasp.dependencycheck.data.nvd.ecosystem;
19  
20  public enum FileExtensionHint implements EcosystemHint {
21  
22      // note: all must be lowercase
23      /**
24       * PHP file extension hint.
25       */
26      PHP(".php", Ecosystem.PHP),
27      /**
28       * Perl file extension hint.
29       */
30      PERL_PM(".pm", Ecosystem.PERL),
31      /**
32       * Perl file extension hint.
33       */
34      PERL_PL(".pl", Ecosystem.PERL),
35      /**
36       * Java file extension hint.
37       */
38      JAR_JAVA(".java", Ecosystem.JAVA),
39      /**
40       * Perl file extension hint.
41       */
42      JAR_JSP(".jsp", Ecosystem.JAVA),
43      /**
44       * Ruby file extension hint.
45       */
46      RUBY(".rb", Ecosystem.RUBY),
47      /**
48       * Python file extension hint.
49       */
50      PYTON(".py", Ecosystem.PYTHON),
51      /**
52       * C++ file extension hint.
53       */
54      CMAKE_CPP(".cpp", Ecosystem.NATIVE),
55      /**
56       * C file extension hint.
57       */
58      CMAKE_C(".c", Ecosystem.NATIVE),
59      /**
60       * C file extension hint.
61       */
62      CMAKE_H(".h", Ecosystem.NATIVE);
63  
64      /**
65       * The file extension for the ecosystem.
66       */
67      private final String extension;
68      /**
69       * The ecosystem for the given file extension.
70       */
71      private final String ecosystem;
72  
73      /**
74       * Constructs a new hint for the given file extension and ecosystem.
75       *
76       * @param extension the file extension to identify the given ecosystem
77       * @param ecosystem the ecosystem for the given file extension
78       */
79      FileExtensionHint(String extension, String ecosystem) {
80          this.extension = extension;
81          this.ecosystem = ecosystem;
82      }
83  
84      @Override
85      public String getEcosystem() {
86          return ecosystem;
87      }
88  
89      public String getExtension() {
90          return extension;
91      }
92  
93      @Override
94      public EcosystemHintNature getNature() {
95          return EcosystemHintNature.FILE_EXTENSION;
96      }
97  
98      @Override
99      public String getValue() {
100         return getExtension();
101     }
102 }
