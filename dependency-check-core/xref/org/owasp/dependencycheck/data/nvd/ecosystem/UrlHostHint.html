18  package org.owasp.dependencycheck.data.nvd.ecosystem;
19  
20  public enum UrlHostHint implements EcosystemHint {
21  
22      // note: all must be lowercase
23      /**
24       * Ruby ecosystem.
25       */
26      RUBY("ruby-lang.org", Ecosystem.RUBY),
27      /**
28       * Python ecosystem.
29       */
30      PYTHON("python.org", Ecosystem.PYTHON),
31      /**
32       * PHP ecosystem.
33       */
34      DRUPAL("drupal.org", Ecosystem.PHP),
35      /**
36       * Node.js ecosystem.
37       */
38      NODEJS("nodejs.org", Ecosystem.NODEJS),
39      /**
40       * Node.js ecosystem.
41       */
42      NODE_SECURITY("nodesecurity.io", Ecosystem.NODEJS),
43      /**
44       * Rust ecosystem.
45       */
46      RUST("rustsec.org", Ecosystem.RUST);
47  
48      /**
49       * The keyword to identify the ecosystem.
50       */
51      private final String keyword;
52  
53      /**
54       * The ecosystem for the keyword.
55       */
56      private final String ecosystem;
57  
58      /**
59       * Constructs a new URL Host Hint to map the keyword to the ecosystem.
60       *
61       * @param keyword the keyword
62       * @param ecosystem the ecosystem
63       */
64      UrlHostHint(String keyword, String ecosystem) {
65          this.keyword = keyword;
66          this.ecosystem = ecosystem;
67      }
68  
69      @Override
70      public String getEcosystem() {
71          return ecosystem;
72      }
73  
74      public String getKeyword() {
75          return keyword;
76      }
77  
78      @Override
79      public EcosystemHintNature getNature() {
80          return EcosystemHintNature.URL_HOST;
81      }
82  
83      @Override
84      public String getValue() {
85          return getKeyword();
86      }
87  
88  }
