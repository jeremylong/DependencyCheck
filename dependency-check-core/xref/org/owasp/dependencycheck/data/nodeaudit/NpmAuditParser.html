18  package org.owasp.dependencycheck.data.nodeaudit;
19  
20  import org.json.JSONArray;
21  import org.json.JSONObject;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import java.util.ArrayList;
25  import java.util.Iterator;
26  import java.util.List;
27  
28  /**
29   * Parser for NPM Audit API response. This parser is derived from:
30   * https://github.com/DependencyTrack/dependency-track/blob/master/src/main/java/org/owasp/dependencytrack/parser/npm/audit/NpmAuditParser.java
31   *
32   * @author Steve Springett
33   */
34  public class NpmAuditParser {
35  
36      /**
37       * The logger.
38       */
39      private static final Logger LOGGER = LoggerFactory.getLogger(NpmAuditParser.class);
40  
41      /**
42       * Parses the JSON response from the NPM Audit API.
43       *
44       * @param jsonResponse the JSON node to parse
45       * @return an AdvisoryResults object
46       */
47      public List<Advisory> parse(JSONObject jsonResponse) {
48          LOGGER.debug("Parsing JSON node");
49          final List<Advisory> advisories = new ArrayList<>();
50          final JSONObject jsonAdvisories = jsonResponse.getJSONObject("advisories");
51          final Iterator<?> keys = jsonAdvisories.keys();
52          while (keys.hasNext()) {
53              final String key = (String) keys.next();
54              final Advisory advisory = parseAdvisory(jsonAdvisories.getJSONObject(key));
55              advisories.add(advisory);
56          }
57          return advisories;
58      }
59  
60      /**
61       * Parses the advisory from Node Audit.
62       *
63       * @param object the JSON object containing the advisory
64       * @return the Advisory object
65       */
66      private Advisory parseAdvisory(JSONObject object) {
67          final Advisory advisory = new Advisory();
68          advisory.setId(object.getInt("id"));
69          advisory.setOverview(object.optString("overview", null));
70          advisory.setReferences(object.optString("references", null));
71          advisory.setCreated(object.optString("created", null));
72          advisory.setUpdated(object.optString("updated", null));
73          advisory.setRecommendation(object.optString("recommendation", null));
74          advisory.setTitle(object.optString("title", null));
75          //advisory.setFoundBy(object.optString("author", null));
76          //advisory.setReportedBy(object.optString("author", null));
77          advisory.setModuleName(object.optString("module_name", null));
78          advisory.setVulnerableVersions(object.optString("vulnerable_versions", null));
79          advisory.setPatchedVersions(object.optString("patched_versions", null));
80          advisory.setAccess(object.optString("access", null));
81          advisory.setSeverity(object.optString("severity", null));
82          advisory.setCwe(object.optString("cwe", null));
83  
84          final JSONArray findings = object.optJSONArray("findings");
85          for (int i = 0; i < findings.length(); i++) {
86              final JSONObject finding = findings.getJSONObject(i);
87              final String version = finding.optString("version", null);
88              final JSONArray paths = finding.optJSONArray("paths");
89              for (int j = 0; j < paths.length(); j++) {
90                  final String path = paths.getString(j);
91                  if (path != null && path.equals(advisory.getModuleName())) {
92                      advisory.setVersion(version);
93                  }
94              }
95          }
96  
97          final JSONArray jsonCves = object.optJSONArray("cves");
98          final List<String> stringCves = new ArrayList<>();
99          if (jsonCves != null) {
100             for (int j = 0; j < jsonCves.length(); j++) {
101                 stringCves.add(jsonCves.getString(j));
102             }
103             advisory.setCves(stringCves);
104         }
105         return advisory;
106     }
107 }
