42  /**
43   * Class of methods to search Nexus repositories.
44   *
45   * @author colezlaw
46   */
47  @ThreadSafe
48  public class NexusSearch {
49  
50      /**
51       * The root URL for the Nexus repository service.
52       */
53      private final URL rootURL;
54  
55      /**
56       * Whether to use the Proxy when making requests.
57       */
58      private final boolean useProxy;
59      /**
60       * The configured settings.
61       */
62      private final Settings settings;
63      /**
64       * Used for logging.
65       */
66      private static final Logger LOGGER = LoggerFactory.getLogger(NexusSearch.class);
67  
68      /**
69       * Creates a NexusSearch for the given repository URL.
70       *
71       * @param settings the configured settings
72       * @param useProxy flag indicating if the proxy settings should be used
73       * @throws java.net.MalformedURLException thrown if the configured URL is
74       * invalid
75       */
76      public NexusSearch(Settings settings, boolean useProxy) throws MalformedURLException {
77          this.settings = settings;
78          this.useProxy = useProxy;
79  
80          final String searchUrl = settings.getString(Settings.KEYS.ANALYZER_NEXUS_URL);
81          LOGGER.debug("Nexus Search URL: {}", searchUrl);
82          this.rootURL = new URL(searchUrl);
83  
84      }
85  
86      /**
87       * Searches the configured Nexus repository for the given sha1 hash. If the
88       * artifact is found, a <code>MavenArtifact</code> is populated with the
89       * coordinate information.
90       *
91       * @param sha1 The SHA-1 hash string for which to search
92       * @return the populated Maven coordinates
93       * @throws IOException if it's unable to connect to the specified repository
94       * or if the specified artifact is not found.
95       */
96      public MavenArtifact searchSha1(String sha1) throws IOException {
97          if (null == sha1 || !sha1.matches("^[0-9A-Fa-f]{40}$")) {
98              throw new IllegalArgumentException("Invalid SHA1 format");
99          }
100 
101         final URL url = new URL(rootURL, String.format("identify/sha1/%s",
102                 sha1.toLowerCase()));
103 
104         LOGGER.debug("Searching Nexus url {}", url);
105 
106         // Determine if we need to use a proxy. The rules:
107         // 1) If the proxy is set, AND the setting is set to true, use the proxy
108         // 2) Otherwise, don't use the proxy (either the proxy isn't configured,
109         // or proxy is specifically set to false
110         final HttpURLConnection conn;
111         final URLConnectionFactory factory = new URLConnectionFactory(settings);
112         conn = factory.createHttpURLConnection(url, useProxy);
113         conn.setDoOutput(true);
114         final String authHeader = buildHttpAuthHeaderValue();
115         if (!authHeader.isEmpty()) {
116             conn.addRequestProperty("Authorization", authHeader);
117         }
118 
119         // JSON would be more elegant, but there's not currently a dependency
120         // on JSON, so don't want to add one just for this
121         conn.addRequestProperty("Accept", "application/xml");
122         conn.connect();
123 
124         switch (conn.getResponseCode()) {
125             case 200:
126                 try {
127                     final DocumentBuilder builder = XmlUtils.buildSecureDocumentBuilder();
128                     final Document doc = builder.parse(conn.getInputStream());
129                     final XPath xpath = XPathFactory.newInstance().newXPath();
130                     final String groupId = xpath
131                             .evaluate(
132                                     "/org.sonatype.nexus.rest.model.NexusArtifact/groupId",
133                                     doc);
134                     final String artifactId = xpath.evaluate(
135                             "/org.sonatype.nexus.rest.model.NexusArtifact/artifactId",
136                             doc);
137                     final String version = xpath
138                             .evaluate(
139                                     "/org.sonatype.nexus.rest.model.NexusArtifact/version",
140                                     doc);
141                     final String link = xpath
142                             .evaluate(
143                                     "/org.sonatype.nexus.rest.model.NexusArtifact/artifactLink",
144                                     doc);
145                     final String pomLink = xpath
146                             .evaluate(
147                                     "/org.sonatype.nexus.rest.model.NexusArtifact/pomLink",
148                                     doc);
149                     final MavenArtifact ma = new MavenArtifact(groupId, artifactId, version);
150                     if (link != null && !link.isEmpty()) {
151                         ma.setArtifactUrl(link);
152                     }
153                     if (pomLink != null && !pomLink.isEmpty()) {
154                         ma.setPomUrl(pomLink);
155                     }
156                     return ma;
157                 } catch (ParserConfigurationException | IOException | SAXException | XPathExpressionException e) {
158                     // Anything else is jacked-up XML stuff that we really can't recover
159                     // from well
160                     throw new IOException(e.getMessage(), e);
161                 }
162             case 404:
163                 throw new FileNotFoundException("Artifact not found in Nexus");
164             default:
165                 LOGGER.debug("Could not connect to Nexus received response code: {} {}",
166                         conn.getResponseCode(), conn.getResponseMessage());
167                 throw new IOException("Could not connect to Nexus");
168         }
169     }
170 
171     /**
172      * Do a preflight request to see if the repository is actually working.
173      *
174      * @return whether the repository is listening and returns the /status URL
175      * correctly
176      */
177     public boolean preflightRequest() {
178         final HttpURLConnection conn;
179         try {
180             final URL url = new URL(rootURL, "status");
181             final URLConnectionFactory factory = new URLConnectionFactory(settings);
182             conn = factory.createHttpURLConnection(url, useProxy);
183             conn.addRequestProperty("Accept", "application/xml");
184             final String authHeader = buildHttpAuthHeaderValue();
185             if (!authHeader.isEmpty()) {
186                 conn.addRequestProperty("Authorization", authHeader);
187             }
188             conn.connect();
189             if (conn.getResponseCode() != 200) {
190                 LOGGER.warn("Expected 200 result from Nexus, got {}", conn.getResponseCode());
191                 return false;
192             }
193             final DocumentBuilder builder = XmlUtils.buildSecureDocumentBuilder();
194 
195             final Document doc = builder.parse(conn.getInputStream());
196             if (!"status".equals(doc.getDocumentElement().getNodeName())) {
197                 LOGGER.warn("Expected root node name of status, got {}", doc.getDocumentElement().getNodeName());
198                 return false;
199             }
200         } catch (IOException | ParserConfigurationException | SAXException e) {
201             LOGGER.warn("Pre-flight request to Nexus failed: ", e);
202             return false;
203         }
204         return true;
205     }
206 
207     /**
208      * Constructs the base64 encoded basic authentication header value.
209      *
210      * @return the base64 encoded basic authentication header value
211      */
212     private String buildHttpAuthHeaderValue() {
213         final String user = settings.getString(Settings.KEYS.ANALYZER_NEXUS_USER, "");
214         final String pass = settings.getString(Settings.KEYS.ANALYZER_NEXUS_PASSWORD, "");
215         String result = "";
216         if (user.isEmpty() || pass.isEmpty()) {
217             LOGGER.debug("Skip authentication as user and/or password for nexus is empty");
218         } else {
219             final String auth = user + ':' + pass;
220             final String base64Auth = Base64.getEncoder().encodeToString(auth.getBytes(StandardCharsets.UTF_8));
221             result = "Basic " + base64Auth;
222         }
223         return result;
224     }
