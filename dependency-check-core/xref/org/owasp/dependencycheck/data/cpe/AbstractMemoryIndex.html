54  /**
55   * <p>
56   * An in memory Lucene index that contains the vendor/product combinations from
57   * the CPE (application) identifiers within the NVD CVE data.</p>
58   *
59   * This is the last remaining singleton in dependency-check-core; The use of
60   * this singleton - while it may not technically be thread-safe (one database
61   * used to build this index may not have the same entries as another) the risk
62   * of this is currently believed to be small. As this memory index consumes a
63   * large amount of memory we will remain using the singleton pattern for now.
64   *
65   * @author Jeremy Long
66   */
67  @ThreadSafe
68  public abstract class AbstractMemoryIndex implements MemoryIndex {
69  
70      /**
71       * The logger.
72       */
73      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractMemoryIndex.class);
74      /**
75       * The in memory Lucene index.
76       */
77      private MMapDirectory index;
78      /**
79       * The Lucene IndexReader.
80       */
81      private IndexReader indexReader;
82      /**
83       * The Lucene IndexSearcher.
84       */
85      private IndexSearcher indexSearcher;
86      /**
87       * The Lucene Analyzer used for Searching.
88       */
89      private Analyzer searchingAnalyzer;
90      /**
91       * The Lucene QueryParser used for Searching.
92       */
93      private QueryParser queryParser;
94      /**
95       * The product field analyzer.
96       */
97      private SearchFieldAnalyzer productFieldAnalyzer;
98      /**
99       * The vendor field analyzer.
100      */
101     private SearchFieldAnalyzer vendorFieldAnalyzer;
102     /**
103      * Track the number of current users of the Lucene index; used to track it
104      * it is okay to actually close the index.
105      */
106     private final AtomicInteger usageCount = new AtomicInteger(0);
107 
108     /**
109      * Returns a reference to the instance.
110      *
111      * @return a reference to the instance
112      */
113     protected abstract AbstractMemoryIndex instance();
114 
115     /**
116      * Creates and loads data into an in memory index.
117      *
118      * @param data the CPE data
119      * @param settings a reference to the dependency-check settings
120      * @throws IndexException thrown if there is an error creating the index
121      */
122     @Override
123     public synchronized void open(Set<Pair<String, String>> data, Settings settings) throws IndexException {
124         if (instance().usageCount.addAndGet(1) == 1) {
125             try {
126                 final File temp = settings.getTempDirectory();
127                 index = new MMapDirectory(temp.toPath());
128                 buildIndex(data);
129 
130                 indexReader = DirectoryReader.open(index);
131             } catch (IOException ex) {
132                 throw new IndexException(ex);
133             }
134             indexSearcher = new IndexSearcher(indexReader);
135             searchingAnalyzer = createSearchingAnalyzer();
136             queryParser = new QueryParser(Fields.DOCUMENT_KEY, searchingAnalyzer);
137         }
138     }
139 
140     /**
141      * returns whether or not the index is open.
142      *
143      * @return whether or not the index is open
144      */
145     public synchronized boolean isOpen() {
146         return instance().usageCount.get() > 0;
147     }
148 
149     /**
150      * Creates an Analyzer for searching the CPE Index.
151      *
152      * @return the CPE Analyzer.
153      */
154     private Analyzer createSearchingAnalyzer() {
155         final Map<String, Analyzer> fieldAnalyzers = new HashMap<>();
156         fieldAnalyzers.put(Fields.DOCUMENT_KEY, new KeywordAnalyzer());
157         productFieldAnalyzer = new SearchFieldAnalyzer();
158         vendorFieldAnalyzer = new SearchFieldAnalyzer();
159         fieldAnalyzers.put(Fields.PRODUCT, productFieldAnalyzer);
160         fieldAnalyzers.put(Fields.VENDOR, vendorFieldAnalyzer);
161 
162         return new PerFieldAnalyzerWrapper(new KeywordAnalyzer(), fieldAnalyzers);
163     }
164 
165     /**
166      * Closes the CPE Index.
167      */
168     @Override
169     public synchronized void close() {
170         if (instance().usageCount.addAndGet(-1) == 0) {
171             if (searchingAnalyzer != null) {
172                 searchingAnalyzer.close();
173                 searchingAnalyzer = null;
174             }
175             if (indexReader != null) {
176                 try {
177                     indexReader.close();
178                 } catch (IOException ex) {
179                     LOGGER.trace("", ex);
180                 }
181                 indexReader = null;
182             }
183             queryParser = null;
184             indexSearcher = null;
185             if (index != null) {
186                 try {
187                     index.close();
188                 } catch (IOException ex) {
189                     LOGGER.trace("", ex);
190                 }
191                 index = null;
192             }
193         }
194     }
195 
196     /**
197      * Builds the CPE Lucene Index based off of the data within the CveDB.
198      *
199      * @param data the CPE data
200      * @throws IndexException thrown if there is an issue creating the index
201      */
202     private void buildIndex(Set<Pair<String, String>> data) throws IndexException {
203         try (Analyzer analyzer = createSearchingAnalyzer();
204                 IndexWriter indexWriter = new IndexWriter(index,
205                         new IndexWriterConfig(analyzer))) {
206 
207             final FieldType ft = new FieldType(TextField.TYPE_STORED);
208             //ignore term frequency
209             ft.setIndexOptions(IndexOptions.DOCS);
210             //ignore field length normalization
211             ft.setOmitNorms(true);
212             // Tip: reuse the Document and Fields for performance...
213             // See "Re-use Document and Field instances" from
214             // http://wiki.apache.org/lucene-java/ImproveIndexingSpeed
215             final Document doc = new Document();
216             final Field v = new Field(Fields.VENDOR, Fields.VENDOR, ft);
217             final Field p = new Field(Fields.PRODUCT, Fields.PRODUCT, ft);
218 
219             doc.add(v);
220             doc.add(p);
221 
222             for (Pair<String, String> pair : data) {
223                 if (pair.getLeft() != null && pair.getRight() != null) {
224                     v.setStringValue(pair.getLeft());
225                     p.setStringValue(pair.getRight());
226                     indexWriter.addDocument(doc);
227                     resetAnalyzers();
228                 }
229             }
230             indexWriter.commit();
231 
232         } catch (DatabaseException ex) {
233             LOGGER.debug("", ex);
234             throw new IndexException("Error reading CPE data", ex);
235         } catch (IOException ex) {
236             throw new IndexException("Unable to close an in-memory index", ex);
237         }
238     }
239 
240     /**
241      * Searches the index using the given search string.
242      *
243      * @param searchString the query text
244      * @param maxQueryResults the maximum number of documents to return
245      * @return the TopDocs found by the search
246      * @throws ParseException thrown when the searchString is invalid
247      * @throws IndexException thrown when there is an internal error resetting
248      * the search analyzer
249      * @throws IOException is thrown if there is an issue with the underlying
250      * Index
251      */
252     @Override
253     public synchronized TopDocs search(String searchString, int maxQueryResults) throws ParseException, IndexException, IOException {
254         final Query query = parseQuery(searchString);
255         return search(query, maxQueryResults);
256     }
257 
258     /**
259      * Parses the given string into a Lucene Query.
260      *
261      * @param searchString the search text
262      * @return the Query object
263      * @throws ParseException thrown if the search text cannot be parsed
264      * @throws IndexException thrown if there is an error resetting the
265      * analyzers
266      */
267     @Override
268     public synchronized Query parseQuery(String searchString) throws ParseException, IndexException {
269         if (searchString == null || searchString.trim().isEmpty()
270                 || "product:() AND vendor:()".equals(searchString)) {
271             throw new ParseException("Query is null or empty");
272         }
273         LOGGER.debug(searchString);
274 
275         Query query;
276         try {
277             query = queryParser.parse(searchString);
278         } catch (BooleanQuery.TooManyClauses ex) {
279             BooleanQuery.setMaxClauseCount(Integer.MAX_VALUE);
280             query = queryParser.parse(searchString);
281         } catch (ParseException ex) {
282             if (ex.getMessage() != null && ex.getMessage().contains("too many boolean clauses")) {
283                 BooleanQuery.setMaxClauseCount(Integer.MAX_VALUE);
284                 query = queryParser.parse(searchString);
285             } else {
286                 LOGGER.debug("Parse Excepction", ex);
287                 throw ex;
288             }
289         }
290         try {
291             resetAnalyzers();
292         } catch (IOException ex) {
293             throw new IndexException("Unable to reset the analyzer after parsing", ex);
294         }
295         return query;
296     }
297 
298     /**
299      * Searches the index using the given query.
300      *
301      * @param query the query used to search the index
302      * @param maxQueryResults the max number of results to return
303      * @return the TopDocs found be the query
304      * @throws CorruptIndexException thrown if the Index is corrupt
305      * @throws IOException thrown if there is an IOException
306      */
307     @Override
308     public synchronized TopDocs search(Query query, int maxQueryResults) throws CorruptIndexException, IOException {
309         return indexSearcher.search(query, maxQueryResults);
310     }
311 
312     /**
313      * Retrieves a document from the Index.
314      *
315      * @param documentId the id of the document to retrieve
316      * @return the Document
317      * @throws IOException thrown if there is an IOException
318      */
319     @Override
320     public synchronized Document getDocument(int documentId) throws IOException {
321         return indexSearcher.doc(documentId);
322     }
323 
324     /**
325      * Returns the number of CPE entries stored in the index.
326      *
327      * @return the number of CPE entries stored in the index
328      */
329     public synchronized int numDocs() {
330         if (indexReader == null) {
331             return -1;
332         }
333         return indexReader.numDocs();
334     }
335 
336     /**
337      * Method to explain queries matches.
338      *
339      * @param query the query used
340      * @param doc the document matched
341      * @return the expalanation
342      * @throws IOException thrown if there is an index error
343      */
344     public synchronized String explain(Query query, int doc) throws IOException {
345         return indexSearcher.explain(query, doc).toString();
346     }
347 
348     /**
349      * Common method to reset the searching analyzers.
350      *
351      * @throws IOException thrown if there is an index error
352      */
353     protected synchronized void resetAnalyzers() throws IOException {
354         productFieldAnalyzer.reset();
355         vendorFieldAnalyzer.reset();
356     }
