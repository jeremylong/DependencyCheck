18  package org.owasp.dependencycheck.data.golang;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  import javax.annotation.concurrent.ThreadSafe;
26  import javax.json.Json;
27  import javax.json.JsonArray;
28  import javax.json.JsonException;
29  import javax.json.JsonObject;
30  import javax.json.JsonReader;
31  import javax.json.JsonReaderFactory;
32  import javax.json.stream.JsonParsingException;
33  
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import org.owasp.dependencycheck.analyzer.exception.AnalysisException;
38  import org.owasp.dependencycheck.utils.JsonArrayFixingInputStream;
39  
40  /**
41   * Parses json output from `go list -json -m all`.
42   *
43   * @author Matthijs van den Bos
44   */
45  @ThreadSafe
46  public final class GoModJsonParser {
47  
48      /**
49       * The LOGGER
50       */
51      private static final Logger LOGGER = LoggerFactory.getLogger(GoModJsonParser.class);
52  
53      private GoModJsonParser() {
54      }
55  
56      /**
57       * Process the input stream to create the list of dependencies.
58       *
59       * @param inputStream the InputStream to parse
60       * @return the list of dependencies
61       * @throws AnalysisException thrown when there is an error parsing the
62       * results of `go mod`
63       */
64      public static List<GoModDependency> process(InputStream inputStream) throws AnalysisException {
65          LOGGER.debug("Beginning go.mod processing");
66  
67          final List<GoModDependency> goModDependencies = new ArrayList<>();
68          try (JsonArrayFixingInputStream jsonStream = new JsonArrayFixingInputStream(inputStream)) {
69              final JsonReaderFactory factory = Json.createReaderFactory(null);
70              try (JsonReader reader = factory.createReader(jsonStream, java.nio.charset.StandardCharsets.UTF_8)) {
71                  final JsonArray modules = reader.readArray();
72                  modules.getValuesAs(JsonObject.class).forEach((module) -> {
73                      final String path = module.getString("Path");
74                      String version = module.getString("Version", null);
75                      if (version != null && version.startsWith("v")) {
76                          version = version.substring(1);
77                      }
78                      String dir = null;
79                      if (module.getJsonString("Dir") != null) {
80                          dir = module.getString("Dir");
81                      }
82                      goModDependencies.add(new GoModDependency(path, version, dir));
83                  });
84              }
85          } catch (JsonParsingException jsonpe) {
86              throw new AnalysisException("Error parsing output from `go list -json -m all`", jsonpe);
87          } catch (JsonException jsone) {
88              throw new AnalysisException("Error reading output from `go list -json -m all`", jsone);
89          } catch (IllegalStateException ise) {
90              throw new AnalysisException("Illegal state in go mod stream", ise);
91          } catch (ClassCastException cce) {
92              throw new AnalysisException("JSON not exactly matching output of `go list -json -m all`", cce);
93          } catch (IOException ex) {
94              throw new AnalysisException("Error reading output of `go list -json -m all`", ex);
95          }
96          return goModDependencies;
97      }
98  }
