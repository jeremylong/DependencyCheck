18  package org.owasp.dependencycheck.data.golang;
19  
20  import com.github.packageurl.MalformedPackageURLException;
21  import com.github.packageurl.PackageURLBuilder;
22  import org.apache.commons.lang3.StringUtils;
23  import org.owasp.dependencycheck.dependency.Confidence;
24  import org.owasp.dependencycheck.dependency.Dependency;
25  import org.owasp.dependencycheck.dependency.EvidenceType;
26  import org.owasp.dependencycheck.dependency.naming.GenericIdentifier;
27  import org.owasp.dependencycheck.dependency.naming.Identifier;
28  import org.owasp.dependencycheck.dependency.naming.PurlIdentifier;
29  import org.owasp.dependencycheck.utils.Checksum;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import static org.owasp.dependencycheck.analyzer.GolangModAnalyzer.DEPENDENCY_ECOSYSTEM;
34  import static org.owasp.dependencycheck.analyzer.GolangModAnalyzer.GO_MOD;
35  
36  /**
37   * Represents a Go module dependency.
38   *
39   * @author Matthijs van den Bos
40   */
41  public class GoModDependency {
42  
43      /**
44       * A reference to the logger.
45       */
46      private static final Logger LOGGER = LoggerFactory.getLogger(GoModDependency.class);
47      /**
48       * The module path.
49       */
50      private final String modulePath;
51      /**
52       * The version.
53       */
54      private final String version;
55  
56      /**
57       * A Package-URL builder.
58       */
59      private final PackageURLBuilder packageURLBuilder;
60  
61      /**
62       * Constructs a new GoModDependency.
63       *
64       * @param modulePath the module path
65       * @param version the dependency version
66       */
67      GoModDependency(String modulePath, String version) {
68          this.modulePath = modulePath;
69          this.version = version;
70  
71          packageURLBuilder = PackageURLBuilder.aPackageURL().withType("golang");
72      }
73  
74      /**
75       * Converts the GoModDependency into a Dependency object.
76       *
77       * @param parentDependency the parent dependency
78       * @return the resulting Dependency object
79       */
80      public Dependency toDependency(Dependency parentDependency) {
81          return createDependency(parentDependency, modulePath, version);
82      }
83  
84      /**
85       * Builds a dependency object based on the given data.
86       *
87       * @param parentDependency a reference to the parent dependency
88       * @param name the name of the dependency
89       * @param version the version of the dependency
90       * @return a new dependency object
91       */
92      private Dependency createDependency(Dependency parentDependency, String name, String version) {
93          final Dependency dep = new Dependency(parentDependency.getActualFile(), true);
94  
95          String namespace = null;
96          String vendor = null;
97          String moduleName = null;
98          String packageNamespace = null;
99  
100         // separate the product from the vendor
101         final int lastSlash = name.lastIndexOf("/");
102         if (lastSlash > 0) {
103             packageNamespace = name.substring(0, lastSlash);
104             final int pos = packageNamespace.indexOf("/");
105             if (pos > 0) {
106                 namespace = packageNamespace.substring(0, pos);
107                 vendor = packageNamespace.substring(pos + 1);
108             }
109             moduleName = name.substring(lastSlash + 1);
110         } else {
111             moduleName = name;
112         }
113 
114         final String filePath = String.format("%s:%s/%s/%s", parentDependency.getFilePath(), packageNamespace, moduleName, version);
115 
116         packageURLBuilder.withName(moduleName);
117         packageURLBuilder.withNamespace(packageNamespace);
118         if (StringUtils.isNotBlank(version)) {
119             packageURLBuilder.withVersion(version);
120         }
121         dep.setEcosystem(DEPENDENCY_ECOSYSTEM);
122         dep.setDisplayFileName(name + ":" + version);
123         dep.setName(moduleName);
124         if (StringUtils.isNotBlank(version)) {
125             dep.setVersion(version);
126             dep.setPackagePath(String.format("%s:%s", name, version));
127         }
128         dep.setFilePath(filePath);
129         dep.setSha1sum(Checksum.getSHA1Checksum(filePath));
130         dep.setSha256sum(Checksum.getSHA256Checksum(filePath));
131         dep.setMd5sum(Checksum.getMD5Checksum(filePath));
132 
133         if (vendor != null) {
134             dep.addEvidence(EvidenceType.VENDOR, GO_MOD, "vendor", vendor, Confidence.HIGHEST);
135             dep.addEvidence(EvidenceType.VENDOR, GO_MOD, "vendor", vendor, Confidence.MEDIUM);
136         }
137         if (namespace != null && !"golang.org".equals(namespace)) {
138             dep.addEvidence(EvidenceType.VENDOR, GO_MOD, "namespace", namespace, Confidence.LOW);
139         }
140         dep.addEvidence(EvidenceType.PRODUCT, GO_MOD, "name", moduleName, Confidence.HIGHEST);
141         dep.addEvidence(EvidenceType.VENDOR, GO_MOD, "name", moduleName, Confidence.HIGH);
142         if (StringUtils.isNotBlank(version)) {
143             dep.addEvidence(EvidenceType.VERSION, GO_MOD, "version", version, Confidence.HIGHEST);
144         }
145         Identifier id;
146         try {
147             id = new PurlIdentifier(packageURLBuilder.build(), Confidence.HIGHEST);
148         } catch (MalformedPackageURLException ex) {
149             LOGGER.warn("Unable to create package-url identifier for `{}` in `{}` - reason: {}",
150                     name, parentDependency.getFilePath(), ex.getMessage());
151             final StringBuilder value = new StringBuilder(name);
152             if (StringUtils.isNotBlank(version)) {
153                 value.append("@").append(version);
154             }
155             id = new GenericIdentifier(value.toString(), Confidence.HIGH);
156         }
157         dep.addSoftwareIdentifier(id);
158         return dep;
159     }
160 
161     @Override
162     public String toString() {
163         return modulePath + ": " + version;
164     }
165 }
