18  package org.owasp.dependencycheck.data.nvdcve;
19  
20  import java.time.Instant;
21  import java.time.ZoneId;
22  import java.time.ZonedDateTime;
23  import java.time.format.DateTimeFormatter;
24  import java.util.Map;
25  import java.util.Map.Entry;
26  import java.util.Properties;
27  import java.util.TreeMap;
28  import javax.annotation.concurrent.ThreadSafe;
29  
30  import org.owasp.dependencycheck.data.update.nvd.NvdCveInfo;
31  import org.owasp.dependencycheck.data.update.exception.UpdateException;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  /**
36   * This is a wrapper around a set of properties that are stored in the database.
37   * This class is safe to be accessed from multiple threads in parallel.
38   *
39   * @author Jeremy Long
40   */
41  @ThreadSafe
42  public class DatabaseProperties {
43  
44      /**
45       * The Logger.
46       */
47      private static final Logger LOGGER = LoggerFactory.getLogger(DatabaseProperties.class);
48      /**
49       * Modified key word, used as a key to store information about the modified
50       * file (i.e. the containing the last 8 days of updates)..
51       */
52      public static final String MODIFIED = "Modified";
53      /**
54       * The properties file key for the last checked field - used to store the
55       * last check time of the Modified NVD CVE xml file.
56       */
57      public static final String LAST_CHECKED = "NVD CVE Checked";
58      /**
59       * The properties file key for the last updated field - used to store the
60       * last updated time of the Modified NVD CVE xml file.
61       */
62      public static final String LAST_UPDATED = "NVD CVE Modified";
63      /**
64       * Stores the last updated time for each of the NVD CVE files. These
65       * timestamps should be updated if we process the modified file within 7
66       * days of the last update.
67       */
68      public static final String LAST_UPDATED_BASE = "NVD CVE ";
69      /**
70       * The key for the last time the CPE data was updated.
71       */
72      public static final String LAST_CPE_UPDATE = "LAST_CPE_UPDATE";
73      /**
74       * The key for the database schema version.
75       */
76      public static final String VERSION = "version";
77  
78      /**
79       * A collection of properties about the data.
80       */
81      private final Properties properties;
82      /**
83       * A reference to the database.
84       */
85      private final CveDB cveDB;
86  
87      /**
88       * Constructs a new data properties object.
89       *
90       * @param cveDB the database object holding the properties
91       */
92      DatabaseProperties(CveDB cveDB) {
93          this.cveDB = cveDB;
94          this.properties = cveDB.getProperties();
95      }
96  
97      /**
98       * Returns whether or not any properties are set.
99       *
100      * @return whether or not any properties are set
101      */
102     public synchronized boolean isEmpty() {
103         return properties == null || properties.isEmpty();
104     }
105 
106     /**
107      * Saves the last updated information to the properties file.
108      *
109      * @param updatedValue the updated NVD CVE entry
110      * @throws UpdateException is thrown if there is an update exception
111      */
112     public synchronized void save(NvdCveInfo updatedValue) throws UpdateException {
113         if (updatedValue == null) {
114             return;
115         }
116         save(LAST_UPDATED_BASE + updatedValue.getId(), String.valueOf(updatedValue.getTimestamp()));
117     }
118 
119     /**
120      * Saves the key value pair to the properties store.
121      *
122      * @param key the property key
123      * @param value the property value
124      * @throws UpdateException is thrown if there is an update exception
125      */
126     public synchronized void save(String key, String value) throws UpdateException {
127         properties.put(key, value);
128         cveDB.saveProperty(key, value);
129     }
130 
131     /**
132      * Returns the property value for the given key. If the key is not contained
133      * in the underlying properties null is returned.
134      *
135      * @param key the property key
136      * @return the value of the property
137      */
138     public synchronized String getProperty(String key) {
139         return properties.getProperty(key);
140     }
141 
142     /**
143      * Returns the property value for the given key. If the key is not contained
144      * in the underlying properties the default value is returned.
145      *
146      * @param key the property key
147      * @param defaultValue the default value
148      * @return the value of the property
149      */
150     public synchronized String getProperty(String key, String defaultValue) {
151         return properties.getProperty(key, defaultValue);
152     }
153 
154     /**
155      * Returns the collection of Database Properties as a properties collection.
156      *
157      * @return the collection of Database Properties
158      */
159     public synchronized Properties getProperties() {
160         return properties;
161     }
162 
163     /**
164      * Returns a map of the meta data from the database properties. This
165      * primarily contains timestamps of when the NVD CVE information was last
166      * updated.
167      *
168      * @return a map of the database meta data
169      */
170     public synchronized Map<String, String> getMetaData() {
171         final Map<String, String> map = new TreeMap<>();
172         for (Entry<Object, Object> entry : properties.entrySet()) {
173             final String key = (String) entry.getKey();
174             if (!"version".equals(key)) {
175                 if ("NVD CVE Modified".equals(key) || "NVD CVE Checked".equals(key) || "VersionCheckOn".equals(key)) {
176                     try {
177                         final long epoch = Long.parseLong((String) entry.getValue());
178                         final ZonedDateTime dateTime = Instant.ofEpochSecond(epoch).atZone(ZoneId.systemDefault());
179                         final String formatted = DateTimeFormatter.ISO_LOCAL_DATE_TIME.format(dateTime);
180                         map.put(key, formatted);
181                     } catch (Throwable ex) { //deliberately being broad in this catch clause
182                         LOGGER.debug("Unable to parse timestamp from DB", ex);
183                         map.put(key, (String) entry.getValue());
184                     }
185                 } else if (!key.startsWith("NVD CVE ")) {
186                     map.put(key, (String) entry.getValue());
187                 }
188             }
189         }
190         return map;
191     }
192 }
