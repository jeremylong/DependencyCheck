38  /**
39   * Produces {@link OssindexClient} instances.
40   *
41   * @author Jason Dillon
42   * @since 5.0.0
43   */
44  public final class OssindexClientFactory {
45  
46      /**
47       * Static logger.
48       */
49      private static final Logger LOGGER = LoggerFactory.getLogger(OssindexClientFactory.class);
50  
51      static {
52          // prefer pkg scheme vs scheme-less variant
53          RenderFlavor.setDefault(RenderFlavor.SCHEME);
54      }
55  
56      /**
57       * Private constructor for utility class.
58       */
59      private OssindexClientFactory() {
60          //private constructor for utility class
61      }
62  
63      /**
64       * Constructs a new OSS Index Client.
65       *
66       * @param settings the configured settings
67       * @return a new OSS Index Client
68       */
69      public static OssindexClient create(final Settings settings) {
70          final OssindexClientConfiguration config = new OssindexClientConfiguration();
71  
72          final String baseUrl = settings.getString(Settings.KEYS.ANALYZER_OSSINDEX_URL, null);
73          if (baseUrl != null) {
74              config.setBaseUrl(baseUrl);
75          }
76  
77          final String username = settings.getString(Settings.KEYS.ANALYZER_OSSINDEX_USER);
78          final String password = settings.getString(Settings.KEYS.ANALYZER_OSSINDEX_PASSWORD);
79  
80          if (username != null && password != null) {
81              final AuthConfiguration auth = new AuthConfiguration(username, password);
82              config.setAuthConfiguration(auth);
83          }
84  
85          final int batchSize = settings.getInt(Settings.KEYS.ANALYZER_OSSINDEX_BATCH_SIZE, OssindexClientConfiguration.DEFAULT_BATCH_SIZE);
86          config.setBatchSize(batchSize);
87  
88          // proxy likely does not need to be configured here as we are using the
89          // URLConnectionFactory#createHttpURLConnection() which automatically configures
90          // the proxy on the connection.
91  //        ProxyConfiguration proxy = new ProxyConfiguration();
92  //        settings.getString(Settings.KEYS.PROXY_PASSWORD);
93  //        config.setProxyConfiguration(proxy);
94          if (settings.getBoolean(Settings.KEYS.ANALYZER_OSSINDEX_USE_CACHE, true)) {
95              final DirectoryCache.Configuration cache = new DirectoryCache.Configuration();
96              final File data;
97              try {
98                  data = settings.getDataDirectory();
99                  final File cacheDir = new File(data, "oss_cache");
100                 if (cacheDir.isDirectory() || cacheDir.mkdirs()) {
101                     cache.setBaseDir(cacheDir.toPath());
102                     cache.setExpireAfter(Duration.standardHours(24));
103                     config.setCacheConfiguration(cache);
104                     LOGGER.debug("OSS Index Cache: " + cache);
105                 } else {
106                     LOGGER.warn("Unable to use a cache for the OSS Index");
107                 }
108             } catch (IOException ex) {
109                 LOGGER.warn("Unable to use a cache for the OSS Index", ex);
110             }
111         }
112         // customize User-Agent for use with dependency-check
113         final UserAgentSupplier userAgent = new UserAgentSupplier(
114                 "dependency-check",
115                 settings.getString(Settings.KEYS.APPLICATION_VERSION, "unknown")
116         );
117 
118         final Transport transport = new ODCConnectionTransport(settings, config, userAgent);
119 
120         final Marshaller marshaller = new GsonMarshaller();
121 
122         return new OssindexClientImpl(config, transport, marshaller);
123     }
124 }
