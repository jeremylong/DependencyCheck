18  package org.owasp.dependencycheck.data.ossindex;
19  
20  import static com.google.common.base.Preconditions.checkNotNull;
21  import static com.google.common.net.HttpHeaders.AUTHORIZATION;
22  import java.io.IOException;
23  import java.net.HttpURLConnection;
24  import java.net.URL;
25  import org.owasp.dependencycheck.utils.Settings;
26  import org.owasp.dependencycheck.utils.URLConnectionFactory;
27  import org.sonatype.ossindex.service.client.OssindexClientConfiguration;
28  import org.sonatype.ossindex.service.client.transport.BasicAuthHelper;
29  import org.sonatype.ossindex.service.client.transport.HttpUrlConnectionTransport;
30  import org.sonatype.ossindex.service.client.transport.UserAgentSupplier;
31  
32  /**
33   * ODC connection transport is used instead of HttpUrlConnectionTransport
34   * because the proxy information is already configured.
35   *
36   * @author Jeremy Long
37   */
38  public class ODCConnectionTransport extends HttpUrlConnectionTransport {
39  
40      /**
41       * The OSS Index client configuration.
42       */
43      private final OssindexClientConfiguration configuration;
44      /**
45       * The URL Connection factory.
46       */
47      private final URLConnectionFactory connectionFactory;
48      /**
49       * The user agent to send in the HTTP connection.
50       */
51      private final UserAgentSupplier userAgent;
52  
53      /**
54       * Constructs a new transport object to connect to the OSS Index.
55       *
56       * @param settings the ODC settings
57       * @param config the OSS client configuration
58       * @param userAgent the user agent to send to OSS Index
59       */
60      public ODCConnectionTransport(Settings settings, OssindexClientConfiguration config, UserAgentSupplier userAgent) {
61          super(userAgent);
62          this.userAgent = checkNotNull(userAgent);
63          this.configuration = checkNotNull(config);
64          connectionFactory = new URLConnectionFactory(settings);
65      }
66  
67      @Override
68      protected HttpURLConnection connect(final URL url) throws IOException {
69          final HttpURLConnection connection = connectionFactory.createHttpURLConnection(url);
70          connection.setRequestProperty("User-Agent", userAgent.get());
71  
72          final String authorization = BasicAuthHelper.authorizationHeader(configuration.getAuthConfiguration());
73          if (authorization != null) {
74              connection.setRequestProperty(AUTHORIZATION, authorization);
75          }
76          return connection;
77      }
78  
79  }
