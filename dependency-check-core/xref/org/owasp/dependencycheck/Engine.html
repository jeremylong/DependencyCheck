214     /**
215      * Loads the analyzers specified in the configuration file (or system
216      * properties).
217      */
218     private void loadAnalyzers() {
219         if (!analyzers.isEmpty()) {
220             return;
221         }
222         mode.getPhases().forEach((phase) -> analyzers.put(phase, new ArrayList<>()));
223         final AnalyzerService service = new AnalyzerService(serviceClassLoader, settings);
224         final List<Analyzer> iterator = service.getAnalyzers(mode.getPhases());
225         iterator.forEach((a) -> {
226             a.initialize(this.settings);
227             analyzers.get(a.getAnalysisPhase()).add(a);
228             if (a instanceof FileTypeAnalyzer) {
229                 this.fileTypeAnalyzers.add((FileTypeAnalyzer) a);
230             }
231         });
232     }

498                     }
499                 } else {
500                     final Dependency d = scanFile(f, projectReference);
501                     if (d != null) {
502                         deps.add(d);
503                     }
504                 }
505             }
506         }
507         return deps;

521     /**
522      * Scans a specified file. If a dependency is identified it is added to the
523      * dependency collection.
524      *
525      * @param file The file to scan
526      * @param projectReference the name of the project or scope in which the
527      * dependency was identified
528      * @return the scanned dependency
529      * @since v1.4.4
530      */
531     protected synchronized Dependency scanFile(@NotNull final File file, @Nullable final String projectReference) {
532         Dependency dependency = null;
533         if (file.isFile()) {
534             if (accept(file)) {
535                 dependency = new Dependency(file);
536                 if (projectReference != null) {
537                     dependency.addProjectReference(projectReference);
538                 }
539                 final String sha1 = dependency.getSha1sum();
540                 boolean found = false;
541 
542                 if (sha1 != null) {
543                     for (Dependency existing : dependencies) {
544                         if (sha1.equals(existing.getSha1sum())) {
545                             if (existing.getFileName().contains(": ") || dependency.getFileName().contains(": ")) {
546                                 //TODO this won't be quite right 100% of the time. Its possible that the ": " would get added later
547                                 continue;
548                             }
549                             found = true;
550                             if (projectReference != null) {
551                                 existing.addProjectReference(projectReference);
552                             }
553                             if (existing.getActualFilePath() != null && dependency.getActualFilePath() != null
554                                     && !existing.getActualFilePath().equals(dependency.getActualFilePath())) {
555                                 existing.addRelatedDependency(dependency);
556                             } else {
557                                 dependency = existing;
558                             }
559                             break;
560                         }
561                     }
562                 }
563                 if (!found) {
564                     dependencies.add(dependency);
565                     dependenciesExternalView = null;
566                 }
567             }
568         } else {
569             LOGGER.debug("Path passed to scanFile(File) is not a file that can be scanned by dependency-check: {}. Skipping the file.", file);
570         }
571         return dependency;
572     }

574     /**
575      * Runs the analyzers against all of the dependencies. Since the mutable
576      * dependencies list is exposed via {@link #getDependencies()}, this method
577      * iterates over a copy of the dependencies list. Thus, the potential for
578      * {@link java.util.ConcurrentModificationException}s is avoided, and
579      * analyzers may safely add or remove entries from the dependencies list.
580      * <p>
581      * Every effort is made to complete analysis on the dependencies. In some
582      * cases an exception will occur with part of the analysis being performed
583      * which may not affect the entire analysis. If an exception occurs it will
584      * be included in the thrown exception collection.
585      *
586      * @throws ExceptionCollection a collections of any exceptions that occurred
587      * during analysis
588      */
589     public void analyzeDependencies() throws ExceptionCollection {
590         final List<Throwable> exceptions = Collections.synchronizedList(new ArrayList<>());
591 
592         initializeAndUpdateDatabase(exceptions);
593 
594         //need to ensure that data exists
595         try {
596             ensureDataExists();
597         } catch (NoDataException ex) {
598             throwFatalExceptionCollection("Unable to continue dependency-check analysis.", ex, exceptions);
599         }
600         LOGGER.info("\n\nDependency-Check is an open source tool performing a best effort analysis of 3rd party dependencies; false positives and "
601                 + "false negatives may exist in the analysis performed by the tool. Use of the tool and the reporting provided constitutes "
602                 + "acceptance for use in an AS IS condition, and there are NO warranties, implied or otherwise, with regard to the analysis "
603                 + "or its use. Any use of the tool and the reporting provided is at the users risk. In no event shall the copyright holder "
604                 + "or OWASP be held liable for any damages whatsoever arising out of or in connection with the use of this tool, the analysis "
605                 + "performed, or the resulting report.\n\n");
606         LOGGER.debug("\n----------------------------------------------------\nBEGIN ANALYSIS\n----------------------------------------------------");
607         LOGGER.info("Analysis Started");
608         final long analysisStart = System.currentTimeMillis();
609 
610         // analysis phases
611         for (AnalysisPhase phase : mode.getPhases()) {
612             final List<Analyzer> analyzerList = analyzers.get(phase);
613 
614             for (final Analyzer analyzer : analyzerList) {
615                 final long analyzerStart = System.currentTimeMillis();
616                 try {
617                     initializeAnalyzer(analyzer);
618                 } catch (InitializationException ex) {
619                     exceptions.add(ex);
620                     if (ex.isFatal()) {
621                         continue;
622                     }
623                 }
624 
625                 if (analyzer.isEnabled()) {
626                     executeAnalysisTasks(analyzer, exceptions);
627 
628                     final long analyzerDurationMillis = System.currentTimeMillis() - analyzerStart;
629                     final long analyzerDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(analyzerDurationMillis);
630                     LOGGER.info("Finished {} ({} seconds)", analyzer.getName(), analyzerDurationSeconds);
631                 } else {
632                     LOGGER.debug("Skipping {} (not enabled)", analyzer.getName());
633                 }
634             }
635         }
636         mode.getPhases().stream()
637                 .map((phase) -> analyzers.get(phase))
638                 .forEach((analyzerList) -> analyzerList.forEach((a) -> closeAnalyzer(a)));
639 
640         LOGGER.debug("\n----------------------------------------------------\nEND ANALYSIS\n----------------------------------------------------");
641         final long analysisDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(System.currentTimeMillis() - analysisStart);
642         LOGGER.info("Analysis Complete ({} seconds)", analysisDurationSeconds);
643         if (exceptions.size() > 0) {
644             throw new ExceptionCollection(exceptions);
645         }
646     }

648     /**
649      * Performs any necessary updates and initializes the database.
650      *
651      * @param exceptions a collection to store non-fatal exceptions
652      * @throws ExceptionCollection thrown if fatal exceptions occur
653      */
654     private void initializeAndUpdateDatabase(@NotNull final List<Throwable> exceptions) throws ExceptionCollection {
655         if (!mode.isDatabaseRequired()) {
656             return;
657         }
658         final boolean autoUpdate;
659         autoUpdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE, true);
660         if (autoUpdate) {
661             try {
662                 doUpdates(true);
663             } catch (UpdateException ex) {
664                 exceptions.add(ex);
665                 LOGGER.warn("Unable to update 1 or more Cached Web DataSource, using local "
666                         + "data instead. Results may not include recent vulnerabilities.");
667                 LOGGER.debug("Update Error", ex);
668             } catch (DatabaseException ex) {
669                 throwFatalDatabaseException(ex, exceptions);
670             }
671         } else {
672             try {
673                 if (ConnectionFactory.isH2Connection(settings) && !ConnectionFactory.h2DataFileExists(settings)) {
674                     throw new ExceptionCollection(new NoDataException("Autoupdate is disabled and the database does not exist"), true);
675                 } else {
676                     openDatabase(true, true);
677                 }
678             } catch (IOException ex) {
679                 throw new ExceptionCollection(new DatabaseException("Autoupdate is disabled and unable to connect to the database"), true);
680             } catch (DatabaseException ex) {
681                 throwFatalDatabaseException(ex, exceptions);
682             }
683         }
684     }

692      * as a fatal exception

694     private void throwFatalDatabaseException(DatabaseException ex, final List<Throwable> exceptions) throws ExceptionCollection {
695         final String msg;
696         if (ex.getMessage().contains("Unable to connect") && ConnectionFactory.isH2Connection(settings)) {
697             msg = "Unable to connect to the database - if this error persists it may be "
698                     + "due to a corrupt database. Consider running `purge` to delete the existing database";
699         } else {
700             msg = "Unable to connect to the dependency-check database";
701         }
702         exceptions.add(new DatabaseException(msg, ex));
703         throw new ExceptionCollection(exceptions, true);
704     }

706     /**
707      * Executes executes the analyzer using multiple threads.
708      *
709      * @param exceptions a collection of exceptions that occurred during
710      * analysis
711      * @param analyzer the analyzer to execute
712      * @throws ExceptionCollection thrown if exceptions occurred during analysis
713      */
714     protected void executeAnalysisTasks(@NotNull final Analyzer analyzer, List<Throwable> exceptions) throws ExceptionCollection {
715         LOGGER.debug("Starting {}", analyzer.getName());
716         final List<AnalysisTask> analysisTasks = getAnalysisTasks(analyzer, exceptions);
717         final ExecutorService executorService = getExecutorService(analyzer);
718 
719         try {
720             final int timeout = settings.getInt(Settings.KEYS.ANALYSIS_TIMEOUT, 20);
721             final List<Future<Void>> results = executorService.invokeAll(analysisTasks, timeout, TimeUnit.MINUTES);
722 
723             // ensure there was no exception during execution
724             for (Future<Void> result : results) {
725                 try {
726                     result.get();
727                 } catch (ExecutionException e) {
728                     throwFatalExceptionCollection("Analysis task failed with a fatal exception.", e, exceptions);
729                 } catch (CancellationException e) {
730                     throwFatalExceptionCollection("Analysis task was cancelled.", e, exceptions);
731                 }
732             }
733         } catch (InterruptedException e) {
734             Thread.currentThread().interrupt();
735             throwFatalExceptionCollection("Analysis has been interrupted.", e, exceptions);
736         } finally {
737             executorService.shutdown();
738         }
739     }

756      *
757      * @param analyzer the analyzer to obtain an executor
758      * @return the executor service
759      */
760     protected ExecutorService getExecutorService(Analyzer analyzer) {
761         if (analyzer.supportsParallelProcessing()) {
762             final int maximumNumberOfThreads = Runtime.getRuntime().availableProcessors();
763             LOGGER.debug("Parallel processing with up to {} threads: {}.", maximumNumberOfThreads, analyzer.getName());
764             return Executors.newFixedThreadPool(maximumNumberOfThreads);
765         } else {
766             LOGGER.debug("Parallel processing is not supported: {}.", analyzer.getName());
767             return Executors.newSingleThreadExecutor();

823      * @throws UpdateException thrown if the operation fails
824      * @throws DatabaseException if the operation fails due to a local database

831     /**
832      * Cycles through the cached web data sources and calls update on all of
833      * them.
834      *
835      * @param remainOpen whether or not the database connection should remain
836      * open
837      * @throws UpdateException thrown if the operation fails
838      * @throws DatabaseException if the operation fails due to a local database
839      * failure
840      */
841     public void doUpdates(boolean remainOpen) throws UpdateException, DatabaseException {
842         if (mode.isDatabaseRequired()) {
843             try (WriteLock dblock = new WriteLock(getSettings(), ConnectionFactory.isH2Connection(getSettings()))) {
844                 //lock is not needed as we already have the lock held
845                 openDatabase(false, false);
846                 LOGGER.info("Checking for updates");
847                 final long updateStart = System.currentTimeMillis();
848                 final UpdateService service = new UpdateService(serviceClassLoader);
849                 final Iterator<CachedWebDataSource> iterator = service.getDataSources();
850                 boolean dbUpdatesMade = false;
851                 UpdateException updateException = null;
852                 while (iterator.hasNext()) {
853                     try {
854                         final CachedWebDataSource source = iterator.next();
855                         dbUpdatesMade |= source.update(this);
856                     } catch (UpdateException ex) {
857                         updateException = ex;
858                         LOGGER.error(ex.getMessage(), ex);
859                     }
860                 }
861                 if (dbUpdatesMade) {
862                     database.defrag();
863                 }
864                 database.close();
865                 database = null;
866                 if (updateException != null) {
867                     throw updateException;
868                 }
869                 LOGGER.info("Check for updates complete ({} ms)", System.currentTimeMillis() - updateStart);
870                 if (remainOpen) {
871                     //lock is not needed as we already have the lock held
872                     openDatabase(true, false);
873                 }
874             } catch (WriteLockException ex) {
875                 throw new UpdateException("Unable to obtain an exclusive lock on the H2 database to perform updates", ex);
876             }
877         } else {
878             LOGGER.info("Skipping update check in evidence collection mode.");
879         }
880     }

882     /**
883      * Purges the cached web data sources.
884      *
885      * @return <code>true</code> if the purge was successful; otherwise
886      * <code>false</code>
887      */
888     public boolean purge() {
889         boolean result = true;
890         final UpdateService service = new UpdateService(serviceClassLoader);
891         final Iterator<CachedWebDataSource> iterator = service.getDataSources();
892         while (iterator.hasNext()) {
893             result &= iterator.next().purge(this);
894         }
895         try {
896             final File cache = new File(settings.getDataDirectory(), "cache");
897             if (cache.exists()) {
898                 if (FileUtils.deleteQuietly(cache)) {
899                     LOGGER.info("Cache directory purged");
900                 }
901             }
902         } catch (IOException ex) {
903             throw new RuntimeException(ex);
904         }
905         try {
906             final File cache = new File(settings.getDataDirectory(), "oss_cache");
907             if (cache.exists()) {
908                 if (FileUtils.deleteQuietly(cache)) {
909                     LOGGER.info("OSS Cache directory purged");
910                 }
911             }
912         } catch (IOException ex) {
913             throw new RuntimeException(ex);
914         }
915 
916         return result;
917     }

933     /**
934      * <p>
935      * This method is only public for unit/integration testing. This method
936      * should not be called by any integration that uses
937      * dependency-check-core.</p>
938      * <p>
939      * Opens the database connection; if readOnly is true a copy of the database
940      * will be made.</p>
941      *
942      * @param readOnly whether or not the database connection should be readonly
943      * @param lockRequired whether or not a lock needs to be acquired when
944      * opening the database
945      * @throws DatabaseException if the database connection could not be created
946      */
947     public void openDatabase(boolean readOnly, boolean lockRequired) throws DatabaseException {
948         if (mode.isDatabaseRequired() && database == null) {
949             try (WriteLock dblock = new WriteLock(getSettings(), lockRequired && ConnectionFactory.isH2Connection(settings))) {
950                 if (readOnly
951                         && ConnectionFactory.isH2Connection(settings)
952                         && settings.getString(Settings.KEYS.DB_CONNECTION_STRING).contains("file:%s")) {
953                     final File db = ConnectionFactory.getH2DataFile(settings);
954                     if (db.isFile()) {
955                         final File temp = settings.getTempDirectory();
956                         final File tempDB = new File(temp, db.getName());
957                         LOGGER.debug("copying database {} to {}", db.toPath(), temp.toPath());
958                         Files.copy(db.toPath(), tempDB.toPath());
959                         settings.setString(Settings.KEYS.H2_DATA_DIRECTORY, temp.getPath());
960                         final String connStr = settings.getString(Settings.KEYS.DB_CONNECTION_STRING);
961                         if (!connStr.contains("ACCESS_MODE_DATA")) {
962                             settings.setString(Settings.KEYS.DB_CONNECTION_STRING, connStr + "ACCESS_MODE_DATA=r");
963                         }
964                         database = new CveDB(settings);
965                     } else {
966                         throw new DatabaseException("Unable to open database - configured database file does not exist: " + db.toString());
967                     }
968                 } else {
969                     database = new CveDB(settings);
970                 }
971             } catch (IOException ex) {
972                 throw new DatabaseException("Unable to open database in read only mode", ex);
973             } catch (WriteLockException ex) {
974                 throw new DatabaseException("Failed to obtain lock - unable to open database", ex);
975             }
976         }
977     }
