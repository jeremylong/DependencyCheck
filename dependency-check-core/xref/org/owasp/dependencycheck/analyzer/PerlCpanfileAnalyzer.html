54  /**
55   * <p>
56   * Used to analyze Perl CPAN files. The analyzer does not yet differentiate
57   * developer and test dependencies from required dependencies. Nor does the
58   * analyzer support `cpanfile.snapshot` files yet. Finally, version ranges are
59   * not yet correctly handled either.</p>
60   * <p>
61   * Future enhancements should include supporting the snapshot files (which
62   * should not have version ranges) and correctly parsing the cpanfile DSL so
63   * that one can differentiate developer and test dependencies - which one may
64   * not want to include in the analysis.</p>
65   *
66   * @author Harjit Sandhu
67   * @author Jeremy Long
68   */
69  @ThreadSafe
70  @Experimental
71  public class PerlCpanfileAnalyzer extends AbstractFileTypeAnalyzer {
72  
73      /**
74       * The logger.
75       */
76      private static final Logger LOGGER = LoggerFactory.getLogger(PerlCpanfileAnalyzer.class);
77      /**
78       * The filter used to identify CPAN files.
79       */
80      private static final FileFilter PACKAGE_FILTER = FileFilterBuilder.newInstance().addFilenames("cpanfile").build();
81      /**
82       * The pattern to extract version numbers from CPAN files.
83       */
84      private static final Pattern VERSION_PATTERN = Pattern.compile("([0-9\\.]+)");
85  
86      /**
87       * Create a new Perl CPAN File Analyzer.
88       */
89      public PerlCpanfileAnalyzer() {
90          //empty constructor
91      }
92  
93      @Override
94      protected FileFilter getFileFilter() {
95          return PACKAGE_FILTER;
96      }
97  
98      @Override
99      public String getName() {
100         return "Perl cpanfile Analyzer";
101     }
102 
103     @Override
104     public AnalysisPhase getAnalysisPhase() {
105         return AnalysisPhase.INFORMATION_COLLECTION;
106     }
107 
108     @Override
109     protected String getAnalyzerEnabledSettingKey() {
110         return Settings.KEYS.ANALYZER_CPANFILE_ENABLED;
111     }
112 
113     @Override
114     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
115         //nothing to prepare
116     }
117 
118     @Override
119     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
120         engine.removeDependency(dependency);
121         final String contents = tryReadFile(dependency.getActualFile());
122         final List<String> requires = prepareContents(contents);
123         if (requires != null && !requires.isEmpty()) {
124             processFileContents(requires, dependency.getFilePath(), engine);
125         }
126     }
127 
128     private String tryReadFile(File file) throws AnalysisException {
129         try {
130             return FileUtils.readFileToString(file, Charset.defaultCharset()).trim();
131         } catch (IOException e) {
132             throw new AnalysisException("Problem occurred while reading dependency file.", e);
133         }
134     }
135 
136     protected List<String> prepareContents(String contents) {
137         final Pattern pattern = Pattern.compile(";");
138         return Arrays.stream(contents.split("\n"))
139                 .map(r -> r.indexOf("#") > 0 ? r.substring(0, r.indexOf("#")) : r)
140                 .flatMap(r -> pattern.splitAsStream(r))
141                 .map(r -> r.trim())
142                 .filter(r -> r.startsWith("requires"))
143                 .collect(toCollection(ArrayList::new));
144     }
145 
146     protected void processFileContents(List<String> fileLines, String filePath, Engine engine) throws AnalysisException {
147         fileLines.stream()
148                 .map(fileLine -> fileLine.split("(,|=>)"))
149                 .map(requires -> {
150                     //LOGGER.debug("perl scanning file:" + fileLine);
151                     final String fqName = requires[0].substring(8)
152                             .replace("'", "")
153                             .replace("\"", "")
154                             .trim();
155                     final String version;
156                     if (requires.length == 1) {
157                         version = "0";
158                     } else {
159                         final Matcher matcher = VERSION_PATTERN.matcher(requires[1]);
160                         if (matcher.find()) {
161                             version = matcher.group(1);
162                         } else {
163                             version = "0";
164                         }
165                     }
166                     final int pos = fqName.lastIndexOf("::");
167                     final String namespace;
168                     final String name;
169                     if (pos > 0) {
170                         namespace = fqName.substring(0, pos);
171                         name = fqName.substring(pos + 2);
172                     } else {
173                         namespace = null;
174                         name = fqName;
175                     }
176                     final Dependency dependency = new Dependency(true);
177                     final File f = new File(filePath);
178                     dependency.setFileName(f.getName());
179                     dependency.setFilePath(filePath);
180                     dependency.setActualFilePath(filePath);
181                     dependency.setDisplayFileName("'" + fqName + "', '" + version + "'");
182                     dependency.setEcosystem(Ecosystem.PERL);
183                     dependency.addEvidence(EvidenceType.VENDOR, "cpanfile", "requires", fqName, Confidence.HIGHEST);
184                     dependency.addEvidence(EvidenceType.PRODUCT, "cpanfile", "requires", fqName, Confidence.HIGHEST);
185                     dependency.addEvidence(EvidenceType.VERSION, "cpanfile", "requires", version, Confidence.HIGHEST);
186                     Identifier id = null;
187                     try {
188                         //note - namespace might be null and that's okay.
189                         final PackageURL purl = PackageURLBuilder.aPackageURL()
190                                 .withType("cpan")
191                                 .withNamespace(namespace)
192                                 .withName(name)
193                                 .withVersion(version)
194                                 .build();
195                         id = new PurlIdentifier(purl, Confidence.HIGH);
196                     } catch (MalformedPackageURLException ex) {
197                         LOGGER.debug("Error building package url for " + fqName + "; using generic identifier instead.", ex);
198                         id = new GenericIdentifier("cpan:" + fqName + "::" + version, Confidence.HIGH);
199                     }
200                     dependency.setVersion(version);
201                     dependency.setName(fqName);
202                     dependency.addSoftwareIdentifier(id);
203                     //sha1sum is used for anchor links in the HtML report
204                     dependency.setSha1sum(Checksum.getSHA1Checksum(id.getValue()));
205                     return dependency;
206                 }).forEachOrdered(dependency -> {
207             engine.addDependency(dependency);
208         });
209     }
