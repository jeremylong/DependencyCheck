104     /**
105      * Builds the beginnings of a List for ProcessBuilder
106      *
107      * @return the list of arguments to begin populating the ProcessBuilder
108      */
109     protected List<String> buildArgumentList() {
110         // Use file.separator as a wild guess as to whether this is Windows
111         final List<String> args = new ArrayList<>();
112         if (!StringUtils.isEmpty(getSettings().getString(Settings.KEYS.ANALYZER_ASSEMBLY_DOTNET_PATH))) {
113             args.add(getSettings().getString(Settings.KEYS.ANALYZER_ASSEMBLY_DOTNET_PATH));
114         } else if (isDotnetPath()) {
115             args.add("dotnet");
116         } else {
117             return null;
118         }
119         args.add(grokAssembly.getPath());
120         return args;
121     }

123     /**
124      * Performs the analysis on a single Dependency.
125      *
126      * @param dependency the dependency to analyze
127      * @param engine the engine to perform the analysis under
128      * @throws AnalysisException if anything goes sideways
129      */
130     @Override
131     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
132         final File test = new File(dependency.getActualFilePath());
133         if (!test.isFile()) {
134             throw new AnalysisException(String.format("%s does not exist and cannot be analyzed by dependency-check",
135                     dependency.getActualFilePath()));
136         }
137         if (grokAssembly == null) {
138             LOGGER.warn("GrokAssembly didn't get deployed");
139             return;
140         }
141         if (baseArgumentList == null) {
142             LOGGER.warn("Assembly Analyzer was unable to execute");
143             return;
144         }
145         final List<String> args = new ArrayList<>(baseArgumentList);
146         args.add(dependency.getActualFilePath());
147         final ProcessBuilder pb = new ProcessBuilder(args);
148         try {
149             final Process proc = pb.start();
150             final GrokParser parser = new GrokParser();
151             final AssemblyData data = parser.parse(proc.getInputStream());
152 
153             // Try evacuating the error stream
154             final String errorStream = new String(ByteStreams.toByteArray(proc.getErrorStream()), StandardCharsets.UTF_8);
155             if (null != errorStream && !errorStream.isEmpty()) {
156                 LOGGER.warn("Error from GrokAssembly: {}", errorStream);
157             }
158 
159             final int rc;
160             try {
161                 rc = proc.waitFor();
162             } catch (InterruptedException ie) {
163                 Thread.currentThread().interrupt();
164                 return;
165             }
166             if (rc == 3) {
167                 LOGGER.debug("{} is not a .NET assembly or executable and as such cannot be analyzed by dependency-check",
168                         dependency.getActualFilePath());
169                 return;
170             } else if (rc != 0) {
171                 LOGGER.debug("Return code {} from GrokAssembly; dependency-check is unable to analyze the library: {}",
172                         rc, dependency.getActualFilePath());
173                 return;
174             }
175 
176             // First, see if there was an error
177             final String error = data.getError();
178             if (error != null && !error.isEmpty()) {
179                 throw new AnalysisException(error);
180             }
181             if (data.getWarning() != null) {
182                 LOGGER.debug("Grok Assembly - could not get namespace on dependency `{}` - ()", dependency.getActualFilePath(), data.getWarning());
183             }
184 
185             final StringBuilder sb = new StringBuilder();
186             if (!StringUtils.isEmpty(data.getFileDescription())) {
187                 sb.append(data.getFileDescription());
188             }
189             if (!StringUtils.isEmpty(data.getComments())) {
190                 if (sb.length() > 0) {
191                     sb.append("\n\n");
192                 }
193                 sb.append(data.getComments());
194             }
195             if (!StringUtils.isEmpty(data.getLegalCopyright())) {
196                 if (sb.length() > 0) {
197                     sb.append("\n\n");
198                 }
199                 sb.append(data.getLegalCopyright());
200             }
201             if (!StringUtils.isEmpty(data.getLegalTrademarks())) {
202                 if (sb.length() > 0) {
203                     sb.append("\n");
204                 }
205                 sb.append(data.getLegalTrademarks());
206             }
207             final String description = sb.toString();
208             if (description.length() > 0) {
209                 dependency.setDescription(description);
210                 addMatchingValues(data.getNamespaces(), description, dependency, EvidenceType.VENDOR);
211                 addMatchingValues(data.getNamespaces(), description, dependency, EvidenceType.PRODUCT);
212             }
213 
214             if (!StringUtils.isEmpty(data.getProductVersion())) {
215                 dependency.addEvidence(EvidenceType.VERSION, "grokassembly", "ProductVersion", data.getProductVersion(), Confidence.HIGHEST);
216             }
217             if (!StringUtils.isEmpty(data.getFileVersion())) {
218                 dependency.addEvidence(EvidenceType.VERSION, "grokassembly", "FileVersion", data.getFileVersion(), Confidence.HIGH);
219             }
220 
221             if (data.getFileVersion() != null && data.getProductVersion() != null) {
222                 final int max = data.getFileVersion().length() > data.getProductVersion().length()
223                         ? data.getProductVersion().length() : data.getFileVersion().length();
224                 int pos;
225                 for (pos = 0; pos < max; pos++) {
226                     if (data.getFileVersion().charAt(pos) != data.getProductVersion().charAt(pos)) {
227                         break;
228                     }
229                 }
230                 final DependencyVersion fileVersion = DependencyVersionUtil.parseVersion(data.getFileVersion(), true);
231                 final DependencyVersion productVersion = DependencyVersionUtil.parseVersion(data.getProductVersion(), true);
232                 if (pos > 0) {
233                     final DependencyVersion matchingVersion = DependencyVersionUtil.parseVersion(data.getFileVersion().substring(0, pos), true);
234                     if (fileVersion != null && data.getFileVersion() != null
235                             && fileVersion.toString().length() == data.getFileVersion().length()) {
236                         if (matchingVersion != null && matchingVersion.getVersionParts().size() > 2) {
237                             dependency.addEvidence(EvidenceType.VERSION, "AssemblyAnalyzer", "FilteredVersion",
238                                     matchingVersion.toString(), Confidence.HIGHEST);
239                             dependency.setVersion(matchingVersion.toString());
240                         }
241                     }
242                 }
243                 if (dependency.getVersion() == null) {
244                     if (data.getFileVersion() != null && data.getProductVersion() != null
245                             && data.getFileVersion().length() >= data.getProductVersion().length()) {
246                         if (fileVersion != null && fileVersion.toString().length() == data.getFileVersion().length()) {
247                             dependency.setVersion(fileVersion.toString());
248                         } else if (productVersion != null && productVersion.toString().length() == data.getProductVersion().length()) {
249                             dependency.setVersion(productVersion.toString());
250                         }
251                     } else {
252                         if (productVersion != null && productVersion.toString().length() == data.getProductVersion().length()) {
253                             dependency.setVersion(productVersion.toString());
254                         } else if (fileVersion != null && fileVersion.toString().length() == data.getFileVersion().length()) {
255                             dependency.setVersion(fileVersion.toString());
256                         }
257                     }
258                 }
259             }
260             if (dependency.getVersion() == null && data.getFileVersion() != null) {
261                 final DependencyVersion version = DependencyVersionUtil.parseVersion(data.getFileVersion(), true);
262                 if (version != null) {
263                     dependency.setVersion(version.toString());
264                 }
265             }
266             if (dependency.getVersion() == null && data.getProductVersion() != null) {
267                 final DependencyVersion version = DependencyVersionUtil.parseVersion(data.getProductVersion(), true);
268                 if (version != null) {
269                     dependency.setVersion(version.toString());
270                 }
271             }
272 
273             if (!StringUtils.isEmpty(data.getCompanyName())) {
274                 dependency.addEvidence(EvidenceType.VENDOR, "grokassembly", "CompanyName", data.getCompanyName(), Confidence.HIGHEST);
275                 addMatchingValues(data.getNamespaces(), data.getCompanyName(), dependency, EvidenceType.VENDOR);
276             }
277             if (!StringUtils.isEmpty(data.getProductName())) {
278                 dependency.addEvidence(EvidenceType.PRODUCT, "grokassembly", "ProductName", data.getProductName(), Confidence.HIGHEST);
279                 addMatchingValues(data.getNamespaces(), data.getProductName(), dependency, EvidenceType.PRODUCT);
280             }
281             if (!StringUtils.isEmpty(data.getFileDescription())) {
282                 dependency.addEvidence(EvidenceType.PRODUCT, "grokassembly", "FileDescription", data.getFileDescription(), Confidence.HIGH);
283                 addMatchingValues(data.getNamespaces(), data.getFileDescription(), dependency, EvidenceType.PRODUCT);
284             }
285 
286             final String internalName = data.getInternalName();
287             if (!StringUtils.isEmpty(internalName)) {
288                 dependency.addEvidence(EvidenceType.PRODUCT, "grokassembly", "InternalName", internalName, Confidence.MEDIUM);
289                 addMatchingValues(data.getNamespaces(), internalName, dependency, EvidenceType.PRODUCT);
290                 addMatchingValues(data.getNamespaces(), internalName, dependency, EvidenceType.VENDOR);
291                 if (dependency.getName() == null && StringUtils.containsIgnoreCase(dependency.getActualFile().getName(), internalName)) {
292                     final String ext = FileUtils.getFileExtension(internalName);
293                     if (ext != null) {
294                         dependency.setName(internalName.substring(0, internalName.length() - ext.length() - 1));
295                     } else {
296                         dependency.setName(internalName);
297                     }
298                 }
299             }
300 
301             final String originalFilename = data.getOriginalFilename();
302             if (!StringUtils.isEmpty(originalFilename)) {
303                 dependency.addEvidence(EvidenceType.PRODUCT, "grokassembly", "OriginalFilename", originalFilename, Confidence.MEDIUM);
304                 addMatchingValues(data.getNamespaces(), originalFilename, dependency, EvidenceType.PRODUCT);
305                 if (dependency.getName() == null && StringUtils.containsIgnoreCase(dependency.getActualFile().getName(), originalFilename)) {
306                     final String ext = FileUtils.getFileExtension(originalFilename);
307                     if (ext != null) {
308                         dependency.setName(originalFilename.substring(0, originalFilename.length() - ext.length() - 1));
309                     } else {
310                         dependency.setName(originalFilename);
311                     }
312                 }
313             }
314             if (dependency.getName() != null && dependency.getVersion() != null) {
315                 try {
316                     dependency.addSoftwareIdentifier(new PurlIdentifier("generic", dependency.getName(), dependency.getVersion(), Confidence.MEDIUM));
317                 } catch (MalformedPackageURLException ex) {
318                     LOGGER.debug("Unable to create Package URL Identifier for " + dependency.getName(), ex);
319                     dependency.addSoftwareIdentifier(new GenericIdentifier(
320                             String.format("%s@%s", dependency.getName(), dependency.getVersion()),
321                             Confidence.MEDIUM));
322                 }
323             }
324             dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
325         } catch (GrokParseException saxe) {
326             LOGGER.error("----------------------------------------------------");
327             LOGGER.error("Failed to read the Assembly Analyzer results.");
328             LOGGER.error("----------------------------------------------------");
329             throw new AnalysisException("Couldn't parse Assembly Analyzer results (GrokAssembly)", saxe);
330         } catch (IOException ioe) {
331             throw new AnalysisException(ioe);
332         }
333     }

335     /**
336      * Initialize the analyzer. In this case, extract GrokAssembly.dll to a
337      * temporary location.
338      *
339      * @param engine a reference to the dependency-check engine
340      * @throws InitializationException thrown if anything goes wrong
341      */
342     @Override
343     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
344         final File location;
345         try (InputStream in = FileUtils.getResourceAsStream("GrokAssembly.zip")) {
346             if (in == null) {
347                 throw new InitializationException("Unable to extract GrokAssembly.dll - file not found");
348             }
349             location = FileUtils.createTempDirectory(getSettings().getTempDirectory());
350             ExtractionUtil.extractFiles(in, location);
351         } catch (ExtractionException ex) {
352             throw new InitializationException("Unable to extract GrokAssembly.dll", ex);
353         } catch (IOException ex) {
354             throw new InitializationException("Unable to create temp directory for GrokAssembly", ex);
355         }
356 
357         grokAssembly = new File(location, "GrokAssembly.dll");
358 
359         // Now, need to see if GrokAssembly actually runs from this location.
360         baseArgumentList = buildArgumentList();
361         //TODO this creates an "unreported" error - if someone doesn't look
362         // at the command output this could easily be missed (especially in an
363         // Ant or Maven build.
364         //
365         // We need to create a non-fatal warning error type that will
366         // get added to the report.
367         //TODO this idea needs to get replicated to the bundle audit analyzer.
368         if (baseArgumentList == null) {
369             setEnabled(false);
370             LOGGER.error("----------------------------------------------------");
371             LOGGER.error(".NET Assembly Analyzer could not be initialized and at least one "
372                     + "'exe' or 'dll' was scanned. The 'dotnet' executable could not be found on "
373                     + "the path; either disable the Assembly Analyzer or add the path to dotnet "
374                     + "core in the configuration.");
375             LOGGER.error("----------------------------------------------------");
376             return;
377         }
378         try {
379             final ProcessBuilder pb = new ProcessBuilder(baseArgumentList);
380             final Process p = pb.start();
381             // Try evacuating the error stream
382             ByteStreams.copy(p.getErrorStream(), NullOutputStream.NULL_OUTPUT_STREAM);
383 
384             final GrokParser grok = new GrokParser();
385             final AssemblyData data = grok.parse(p.getInputStream());
386             if (p.waitFor() != 1 || data == null || StringUtils.isEmpty(data.getError())) {
387                 LOGGER.warn("An error occurred with the .NET AssemblyAnalyzer, please see the log for more details.");
388                 LOGGER.debug("GrokAssembly.dll is not working properly");
389                 grokAssembly = null;
390                 setEnabled(false);
391                 throw new InitializationException("Could not execute .NET AssemblyAnalyzer");
392             }
393         } catch (InitializationException e) {
394             setEnabled(false);
395             throw e;
396         } catch (InterruptedException e) {
397             Thread.currentThread().interrupt();
398             LOGGER.warn("An error occurred with the .NET AssemblyAnalyzer;\n"
399                     + "this can be ignored unless you are scanning .NET DLLs. Please see the log for more details.");
400             LOGGER.debug("Could not execute GrokAssembly {}", e.getMessage());
401             setEnabled(false);
402             throw new InitializationException("An error occurred with the .NET AssemblyAnalyzer", e);
403         } catch (IOException e) {
404             LOGGER.warn("An error occurred with the .NET AssemblyAnalyzer;\n"
405                     + "this can be ignored unless you are scanning .NET DLLs. Please see the log for more details.");
406             LOGGER.debug("Could not execute GrokAssembly {}", e.getMessage());
407             setEnabled(false);
408             throw new InitializationException("An error occurred with the .NET AssemblyAnalyzer", e);
409         }
410     }

458     /**
459      * Tests to see if a file is in the system path.
460      *
461      * @return <code>true</code> if dotnet could be found in the path; otherwise
462      * <code>false</code>
463      */
464     private boolean isDotnetPath() {
465         final String[] args = new String[2];
466         args[0] = "dotnet";
467         args[1] = "--version";
468         final ProcessBuilder pb = new ProcessBuilder(args);
469         try {
470             final Process proc = pb.start();
471             final int retCode = proc.waitFor();
472             if (retCode == 0) {
473                 return true;
474             }
475             final byte[] version = new byte[50];
476             final int read = proc.getInputStream().read(version);
477             if (read > 0) {
478                 final String v = new String(version, UTF_8);
479                 if (v.length() > 0) {
480                     return true;
481                 }
482             }
483         } catch (InterruptedException ex) {
484             Thread.currentThread().interrupt();
485             LOGGER.debug("Path search failed for dotnet", ex);
486         } catch (IOException ex) {
487             LOGGER.debug("Path search failed for dotnet", ex);
488         }
489         return false;
490     }

492     /**
493      * Cycles through the collection of class name information to see if parts
494      * of the package names are contained in the provided value. If found, it
495      * will be added as the HIGHEST confidence evidence because we have more
496      * then one source corroborating the value.
497      *
498      * @param packages a collection of class name information
499      * @param value the value to check to see if it contains a package name
500      * @param dep the dependency to add new entries too
501      * @param type the type of evidence (vendor, product, or version)
502      */
503     protected static void addMatchingValues(List<String> packages, String value, Dependency dep, EvidenceType type) {
504         if (value == null || value.isEmpty() || packages == null || packages.isEmpty()) {
505             return;
506         }
507         for (String key : packages) {
508             final int pos = StringUtils.indexOfIgnoreCase(value, key);
509             if ((pos == 0 && (key.length() == value.length() || (key.length() < value.length()
510                     && !Character.isLetterOrDigit(value.charAt(key.length())))))
511                     || (pos > 0 && !Character.isLetterOrDigit(value.charAt(pos - 1))
512                     && (pos + key.length() == value.length() || (key.length() < value.length()
513                     && !Character.isLetterOrDigit(value.charAt(pos + key.length())))))) {
514                 dep.addEvidence(type, "dll", "namespace", key, Confidence.HIGHEST);
515             }
516 
517         }
518     }
