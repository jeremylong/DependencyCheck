48  /**

142     @Override
143     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
144         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
145         String contents;
146         try {
147             contents = FileUtils.readFileToString(dependency.getActualFile(), Charset.defaultCharset());
148         } catch (IOException e) {
149             throw new AnalysisException(
150                     "Problem occurred while reading dependency file.", e);
151         }
152         final Matcher matcher = GEMSPEC_BLOCK_INIT.matcher(contents);
153         if (matcher.find()) {
154             contents = contents.substring(matcher.end());
155             final String blockVariable = matcher.group(1);
156 
157             final String name = addStringEvidence(dependency, EvidenceType.PRODUCT, contents, blockVariable, "name", "name", Confidence.HIGHEST);
158             if (!name.isEmpty()) {
159                 dependency.addEvidence(EvidenceType.VENDOR, GEMSPEC, "name_project", name + "_project", Confidence.LOW);
160                 dependency.setName(name);
161             }
162             final String description = addStringEvidence(dependency, EvidenceType.PRODUCT, contents, blockVariable,
163                     "summary", "summary", Confidence.LOW);
164             if (description != null && !description.isEmpty()) {
165                 dependency.setDescription(description);
166             }
167             addStringEvidence(dependency, EvidenceType.VENDOR, contents, blockVariable,
168                     "author", "authors?", Confidence.HIGHEST);
169             addStringEvidence(dependency, EvidenceType.VENDOR, contents, blockVariable,
170                     "email", "emails?", Confidence.MEDIUM);
171             addStringEvidence(dependency, EvidenceType.VENDOR, contents, blockVariable,
172                     "homepage", "homepage", Confidence.HIGHEST);
173             final String license = addStringEvidence(dependency, EvidenceType.VENDOR, contents, blockVariable,
174                     "license", "licen[cs]es?", Confidence.HIGHEST);
175             if (license != null && !license.isEmpty()) {
176                 dependency.setLicense(license);
177             }
178             final String value = addStringEvidence(dependency, EvidenceType.VERSION, contents,
179                     blockVariable, "version", "version", Confidence.HIGHEST);
180             if (value.length() < 1) {
181                 final String version = addEvidenceFromVersionFile(dependency, EvidenceType.VERSION, dependency.getActualFile());
182                 if (version != null) {
183                     dependency.setVersion(version);
184                 }
185             } else {
186                 dependency.setVersion(value);
187             }
188         }
189         if (dependency.getName() != null && dependency.getVersion() != null) {
190             dependency.setDisplayFileName(String.format("%s:%s", dependency.getName(), dependency.getVersion()));
191         }
192 
193         try {
194             final PackageURLBuilder builder = PackageURLBuilder.aPackageURL().withType("gem").withName(dependency.getName());
195             if (dependency.getVersion() != null) {
196                 builder.withVersion(dependency.getVersion());
197             }
198             final PackageURL purl = builder.build();
199             dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
200         } catch (MalformedPackageURLException ex) {
201             LOGGER.debug("Unable to build package url for python", ex);
202             final GenericIdentifier id;
203             if (dependency.getVersion() != null) {
204                 id = new GenericIdentifier("gem:" + dependency.getName() + "@" + dependency.getVersion(), Confidence.HIGHEST);
205             } else {
206                 id = new GenericIdentifier("gem:" + dependency.getName(), Confidence.HIGHEST);
207             }
208             dependency.addSoftwareIdentifier(id);
209         }
210 
211         setPackagePath(dependency);
212     }

214     /**
215      * Adds the specified evidence to the given evidence collection.
216      *
217      * @param dependency the dependency being analyzed
218      * @param type the type of evidence to add
219      * @param contents the evidence contents
220      * @param blockVariable the variable
221      * @param field the field
222      * @param fieldPattern the field pattern
223      * @param confidence the confidence of the evidence
224      * @return the evidence string value added
225      */
226     private String addStringEvidence(Dependency dependency, EvidenceType type, String contents,
227             String blockVariable, String field, String fieldPattern, Confidence confidence) {
228         String value = "";
229 
230         //capture array value between [ ]
231         final Matcher arrayMatcher = Pattern.compile(
232                 String.format("\\s*?%s\\.%s\\s*?=\\s*?\\[(.*?)\\]", blockVariable, fieldPattern), Pattern.CASE_INSENSITIVE).matcher(contents);
233         if (arrayMatcher.find()) {
234             final String arrayValue = arrayMatcher.group(1);
235             value = arrayValue.replaceAll("['\"]", "").trim(); //strip quotes
236         } else { //capture single value between quotes
237             final Matcher matcher = Pattern.compile(
238                     String.format("\\s*?%s\\.%s\\s*?=\\s*?(['\"])(.*?)\\1", blockVariable, fieldPattern), Pattern.CASE_INSENSITIVE).matcher(contents);
239             if (matcher.find()) {
240                 value = matcher.group(2);
241             }
242         }
243         if (value.length() > 0) {
244             dependency.addEvidence(type, GEMSPEC, field, value, confidence);
245         }
246 
247         return value;
248     }

250     /**
251      * Adds evidence from the version file.
252      *
253      * @param dependency the dependency being analyzed
254      * @param type the type of evidence to add
255      * @param dependencyFile the dependency being analyzed
256      * @return the version number added
257      */
258     private String addEvidenceFromVersionFile(Dependency dependency, EvidenceType type, File dependencyFile) {
259         final File parentDir = dependencyFile.getParentFile();
260         String version = null;
261         int versionCount = 0;
262         if (parentDir != null) {
263             final File[] matchingFiles = parentDir.listFiles(new FilenameFilter() {
264                 @Override
265                 public boolean accept(File dir, String name) {
266                     return name.contains(VERSION_FILE_NAME);
267                 }
268             });
269             if (matchingFiles == null) {
270                 return null;
271             }
272             for (File f : matchingFiles) {
273                 try {
274                     final List<String> lines = FileUtils.readLines(f, Charset.defaultCharset());
275                     if (lines.size() == 1) { //TODO other checking?
276                         final String value = lines.get(0).trim();
277                         if (version == null || !version.equals(value)) {
278                             version = value;
279                             versionCount++;
280                         }
281 
282                         dependency.addEvidence(type, GEMSPEC, "version", value, Confidence.HIGH);
283                     }
284                 } catch (IOException e) {
285                     LOGGER.debug("Error reading gemspec", e);
286                 }
287             }
288         }
289         if (versionCount == 1) {
290             return version;
291         }
292         return null;
293     }

295     /**
296      * Sets the package path on the dependency.
297      *
298      * @param dep the dependency to alter
299      */
300     private void setPackagePath(Dependency dep) {
301         final File file = new File(dep.getFilePath());
302         final String parent = file.getParent();
303         if (parent != null) {
304             dep.setPackagePath(parent);
305         }
