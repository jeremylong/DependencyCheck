57  /**
58   * Analyzer which will attempt to locate a dependency, and the GAV information,
59   * by querying Central for the dependency's SHA-1 digest.
60   *
61   * @author colezlaw
62   */
63  @ThreadSafe
64  public class CentralAnalyzer extends AbstractFileTypeAnalyzer {
65  
66      /**
67       * The logger.
68       */
69      private static final Logger LOGGER = LoggerFactory.getLogger(CentralAnalyzer.class);
70  
71      /**
72       * The name of the analyzer.
73       */
74      private static final String ANALYZER_NAME = "Central Analyzer";
75  
76      /**
77       * The phase in which this analyzer runs.
78       */
79      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
80  
81      /**
82       * The types of files on which this will work.
83       */
84      private static final String SUPPORTED_EXTENSIONS = "jar";
85  
86      /**
87       * The file filter used to determine which files this analyzer supports.
88       */
89      private static final FileFilter FILTER = FileFilterBuilder.newInstance().addExtensions(SUPPORTED_EXTENSIONS).build();
90  
91      /**
92       * The base wait time between retrying a failed connection to Central.
93       */
94      private static final int BASE_RETRY_WAIT = 1500;
95      /**
96       * There may be temporary issues when connecting to MavenCentral. In order
97       * to compensate for 99% of the issues, we perform a retry before finally
98       * failing the analysis.
99       */
100     private static int numberOfRetries = 7;
101 
102     /**
103      * The searcher itself.
104      */
105     private CentralSearch searcher;
106     /**
107      * A reference to the cache for POM model data collected from Central.
108      */
109     private DataCache<Model> cache;
110 
111     /**
112      * Initializes the analyzer with the configured settings.
113      *
114      * @param settings the configured settings to use
115      */
116     @Override
117     public synchronized void initialize(Settings settings) {
118         super.initialize(settings);
119         setEnabled(checkEnabled());
120         numberOfRetries = getSettings().getInt(Settings.KEYS.ANALYZER_CENTRAL_RETRY_COUNT, numberOfRetries);
121         if (settings.getBoolean(Settings.KEYS.ANALYZER_CENTRAL_USE_CACHE, true)) {
122             try {
123                 final DataCacheFactory factory = new DataCacheFactory(settings);
124                 cache = factory.getPomCache();
125             } catch (CacheException ex) {
126                 settings.setBoolean(Settings.KEYS.ANALYZER_CENTRAL_USE_CACHE, false);
127                 LOGGER.debug("Error creating cache, disabling caching", ex);
128             }
129         }
130     }
131 
132     /**
133      * Whether the analyzer is configured to support parallel processing.
134      *
135      * @return true if configured to support parallel processing; otherwise
136      * false
137      */
138     @Override
139     public boolean supportsParallelProcessing() {
140         return getSettings().getBoolean(Settings.KEYS.ANALYZER_CENTRAL_PARALLEL_ANALYSIS, true);
141     }

143     /**
144      * Determines if this analyzer is enabled.
145      *
146      * @return <code>true</code> if the analyzer is enabled; otherwise
147      * <code>false</code>
148      */
149     private boolean checkEnabled() {
150         try {
151             return getSettings().getBoolean(Settings.KEYS.ANALYZER_CENTRAL_ENABLED);
152         } catch (InvalidSettingException ise) {
153             LOGGER.warn("Invalid setting. Disabling the Central analyzer");
154         }
155         return false;
156     }
157 
158     /**
159      * Initializes the analyzer once before any analysis is performed.
160      *
161      * @param engine a reference to the dependency-check engine
162      * @throws InitializationException if there's an error during initialization
163      */
164     @Override
165     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
166         LOGGER.debug("Initializing Central analyzer");
167         LOGGER.debug("Central analyzer enabled: {}", isEnabled());
168         if (isEnabled()) {
169             try {
170                 searcher = new CentralSearch(getSettings());
171             } catch (MalformedURLException ex) {
172                 setEnabled(false);
173                 throw new InitializationException("The configured URL to Maven Central is malformed", ex);
174             }
175         }
176     }

214     /**
215      * Performs the analysis.
216      *
217      * @param dependency the dependency to analyze
218      * @param engine the engine
219      * @throws AnalysisException when there's an exception during analysis
220      */
221     @Override
222     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
223         for (Evidence e : dependency.getEvidence(EvidenceType.VENDOR)) {
224             if ("pom".equals(e.getSource())) {
225                 return;
226             }
227         }
228         try {
229             final List<MavenArtifact> mas = fetchMavenArtifacts(dependency);
230             final Confidence confidence = mas.size() > 1 ? Confidence.HIGH : Confidence.HIGHEST;
231             for (MavenArtifact ma : mas) {
232                 LOGGER.debug("Central analyzer found artifact ({}) for dependency ({})", ma, dependency.getFileName());
233                 dependency.addAsEvidence("central", ma, confidence);
234 
235                 if (ma.getPomUrl() != null) {
236                     File pomFile = null;
237                     try {
238                         final File baseDir = getSettings().getTempDirectory();
239                         pomFile = File.createTempFile("pom", ".xml", baseDir);
240                         if (!pomFile.delete()) {
241                             LOGGER.warn("Unable to fetch pom.xml for {} from Central; "
242                                     + "this could result in undetected CPE/CVEs.", dependency.getFileName());
243                             LOGGER.debug("Unable to delete temp file");
244                         }
245                         final Downloader downloader = new Downloader(getSettings());
246                         final int maxAttempts = this.getSettings().getInt(Settings.KEYS.ANALYZER_CENTRAL_RETRY_COUNT, 3);
247                         int retryCount = 0;
248                         long sleepingTimeBetweenRetriesInMillis = BASE_RETRY_WAIT;
249                         boolean success = false;
250                         Model model = null;
251                         if (cache != null) {
252                             model = cache.get(ma.getPomUrl());
253                         }
254                         if (model != null) {
255                             success = true;
256                             LOGGER.debug("Cache hit for {}", ma.getPomUrl());
257                         } else {
258                             LOGGER.debug("Downloading {}", ma.getPomUrl());
259                             do {
260                                 //CSOFF: NestedTryDepth
261                                 try {
262                                     downloader.fetchFile(new URL(ma.getPomUrl()), pomFile);
263                                     success = true;
264                                 } catch (DownloadFailedException ex) {
265                                     try {
266                                         Thread.sleep(sleepingTimeBetweenRetriesInMillis);
267                                     } catch (InterruptedException ex1) {
268                                         Thread.currentThread().interrupt();
269                                         throw new UnexpectedAnalysisException(ex1);
270                                     }
271                                     sleepingTimeBetweenRetriesInMillis *= 2;
272                                 } catch (ResourceNotFoundException ex) {
273                                     LOGGER.debug("pom.xml does not exist in Central for {}", dependency.getFileName());
274                                     return;
275                                 }
276                                 //CSON: NestedTryDepth
277                             } while (!success && retryCount++ < maxAttempts);
278                         }
279                         if (success) {
280                             if (model == null) {
281                                 model = PomUtils.readPom(pomFile);
282                                 if (cache != null) {
283                                     cache.put(ma.getPomUrl(), model);
284                                 }
285                             }
286                             final boolean isMainPom = mas.size() == 1 || dependency.getActualFilePath().contains(ma.getVersion());
287                             JarAnalyzer.setPomEvidence(dependency, model, null, isMainPom);
288                         } else {
289                             LOGGER.warn("Unable to download pom.xml for {} from Central; "
290                                     + "this could result in undetected CPE/CVEs.", dependency.getFileName());
291                         }
292 
293                     } catch (AnalysisException ex) {
294                         LOGGER.warn(MessageFormat.format("Unable to analyze pom.xml for {0} from Central; "
295                                 + "this could result in undetected CPE/CVEs.", dependency.getFileName()), ex);
296 
297                     } finally {
298                         if (pomFile != null && pomFile.exists() && !FileUtils.deleteQuietly(pomFile)) {
299                             LOGGER.debug("Failed to delete temporary pom file {}", pomFile);
300                             pomFile.deleteOnExit();
301                         }
302                     }
303                 }
304             }
305         } catch (TooManyRequestsException tre) {
306             this.setEnabled(false);
307             final String message = "Connections to Central search refused. Analysis failed.";
308             LOGGER.error(message, tre);
309             throw new AnalysisException(message, tre);
310         } catch (IllegalArgumentException iae) {
311             LOGGER.info("invalid sha1-hash on {}", dependency.getFileName());
312         } catch (FileNotFoundException fnfe) {
313             LOGGER.debug("Artifact not found in repository: '{}", dependency.getFileName());
314         } catch (IOException ioe) {
315             final String message = "Could not connect to Central search. Analysis failed.";
316             LOGGER.error(message, ioe);
317             throw new AnalysisException(message, ioe);
318         }
319     }

321     /**
322      * Downloads the corresponding list of MavenArtifacts of the given
323      * dependency from MavenCentral.
324      * <p>
325      * As the connection to MavenCentral is known to be unreliable, we implement
326      * a simple retry logic in order to compensate for 99% of the issues.
327      *
328      * @param dependency the dependency to analyze
329      * @return the downloaded list of MavenArtifacts
330      * @throws FileNotFoundException if the specified artifact is not found
331      * @throws IOException if connecting to MavenCentral finally failed
332      * @throws TooManyRequestsException if Central has received too many
333      * requests.
334      */
335     protected List<MavenArtifact> fetchMavenArtifacts(Dependency dependency) throws IOException, TooManyRequestsException {
336         IOException lastException = null;
337         long sleepingTimeBetweenRetriesInMillis = BASE_RETRY_WAIT;
338         int triesLeft = numberOfRetries;
339         while (triesLeft-- > 0) {
340             try {
341                 return searcher.searchSha1(dependency.getSha1sum());
342             } catch (FileNotFoundException fnfe) {
343                 // retry does not make sense, just throw the exception
344                 throw fnfe;
345             } catch (IOException ioe) {
346                 LOGGER.debug("Could not connect to Central search (tries left: {}): {}",
347                         triesLeft, ioe.getMessage());
348                 lastException = ioe;
349 
350                 if (triesLeft > 0) {
351                     try {
352                         Thread.sleep(sleepingTimeBetweenRetriesInMillis);
353                     } catch (InterruptedException e) {
354                         Thread.currentThread().interrupt();
355                         throw new UnexpectedAnalysisException(e);
356                     }
357                     sleepingTimeBetweenRetriesInMillis *= 2;
358                 }
359             }
360         }
361 
362         final String message = "Finally failed connecting to Central search."
363                 + " Giving up after " + numberOfRetries + " tries.";
364         throw new IOException(message, lastException);
365     }
