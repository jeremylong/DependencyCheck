52  /**
53   * Analyzer which will attempt to locate a dependency on a Nexus service by
54   * SHA-1 digest of the dependency.
55   *
56   * There are two settings which govern this behavior:
57   *
58   * <ul>
59   * <li>{@link org.owasp.dependencycheck.utils.Settings.KEYS#ANALYZER_NEXUS_ENABLED}
60   * determines whether this analyzer is even enabled. This can be overridden by
61   * setting the system property.</li>
62   * <li>{@link org.owasp.dependencycheck.utils.Settings.KEYS#ANALYZER_NEXUS_URL}
63   * the URL to a Nexus service to search by SHA-1. There is an expected
64   * <code>%s</code> in this where the SHA-1 will get entered.</li>
65   * </ul>
66   *
67   * @author colezlaw
68   */
69  @ThreadSafe
70  public class NexusAnalyzer extends AbstractFileTypeAnalyzer {
71  
72      /**
73       * The default URL - this will be used by the CentralAnalyzer to determine
74       * whether to enable this.
75       */
76      public static final String DEFAULT_URL = "https://repository.sonatype.org/service/local/";
77  
78      /**
79       * The logger.
80       */
81      private static final Logger LOGGER = LoggerFactory.getLogger(NexusAnalyzer.class);
82  
83      /**
84       * The name of the analyzer.
85       */
86      private static final String ANALYZER_NAME = "Nexus Analyzer";
87  
88      /**
89       * The phase in which the analyzer runs.
90       */
91      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
92  
93      /**
94       * The types of files on which this will work.
95       */
96      private static final String SUPPORTED_EXTENSIONS = "jar";
97  
98      /**
99       * The file filter used to determine which files this analyzer supports.
100      */
101     private static final FileFilter FILTER = FileFilterBuilder.newInstance().addExtensions(SUPPORTED_EXTENSIONS).build();
102 
103     /**
104      * The Nexus Search to be set up for this analyzer.
105      */
106     private NexusSearch searcher;
107 
108     /**
109      * Field indicating if the analyzer is enabled.
110      */
111     private boolean enabled = true;
112 
113     /**
114      * Initializes the analyzer with the configured settings.
115      *
116      * @param settings the configured settings to use
117      */
118     @Override
119     public void initialize(Settings settings) {
120         super.initialize(settings);
121         enabled = checkEnabled();
122     }
123 
124     /**
125      * Determines if this analyzer is enabled
126      *
127      * @return <code>true</code> if the analyzer is enabled; otherwise
128      * <code>false</code>
129      */
130     private boolean checkEnabled() {
131         /* Enable this analyzer ONLY if the Nexus URL has been set to something
132          other than the default one (if it's the default one, we'll use the
133          central analyzer) and it's enabled by the user.
134          */
135         boolean retval = false;
136         try {
137             if (getSettings().getBoolean(Settings.KEYS.ANALYZER_NEXUS_ENABLED)) {
138                 if (getSettings().getString(Settings.KEYS.ANALYZER_NEXUS_URL) != null
139                         && !DEFAULT_URL.equals(getSettings().getString(Settings.KEYS.ANALYZER_NEXUS_URL))) {
140                     retval = true;
141                 } else {
142                     LOGGER.warn("Disabling Nexus analyzer - please specify the URL to a Nexus Server");
143                 }
144             }
145         } catch (InvalidSettingException ise) {
146             LOGGER.warn("Invalid setting. Disabling Nexus analyzer");
147         }
148 
149         return retval;
150     }
151 
152     /**
153      * Determine whether to enable this analyzer or not.
154      *
155      * @return whether the analyzer should be enabled
156      */
157     @Override
158     public boolean isEnabled() {
159         return enabled;
160     }
161 
162     /**
163      * Initializes the analyzer once before any analysis is performed.
164      *
165      * @param engine a reference to the dependency-check engine
166      * @throws InitializationException if there's an error during initialization
167      */
168     @Override
169     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
170         LOGGER.debug("Initializing Nexus Analyzer");
171         LOGGER.debug("Nexus Analyzer enabled: {}", isEnabled());
172         if (isEnabled()) {
173             final boolean useProxy = useProxy();
174             LOGGER.debug("Using proxy: {}", useProxy);
175             searcher = createNexusSearchOrDisable(useProxy);
176         }
177     }
178 
179     /**
180      * Creates a NexusSearch for the appropriate Nexus version (Nexus V2 and V3 supported).
181      * <p>
182      * If errors are encountered creating or validating the NexusSearch it disables this analyzer.
183      *
184      * @param useProxy Whether a proxy is to be used
185      * @return A NexusSearch appropriate for the configured ANALYZER_NEXUS_URL
186      * @throws InitializationException Upon errors creating of validating the ANALYZER_NEXUS_URL
187      */
188     private NexusSearch createNexusSearchOrDisable(boolean useProxy) throws InitializationException {
189         final Settings settings = getSettings();
190         final String nexusRootURL = settings.getString(Settings.KEYS.ANALYZER_NEXUS_URL);
191         final NexusSearch result;
192         try {
193             if (nexusRootURL.toLowerCase(Locale.ROOT).contains("service/local/")) {
194                 result = new NexusV2Search(settings, useProxy);
195             } else {
196                 result = new NexusV3Search(settings, useProxy);
197             }
198             if (!result.preflightRequest()) {
199                 setEnabled(false);
200                 throw new InitializationException("There was an error getting Nexus status. Disabling NexusAnalyzer.");
201             }
202         } catch (MalformedURLException mue) {
203             setEnabled(false);
204             throw new InitializationException("Malformed URL to Nexus. Disabling NexusAnalyzer", mue);
205         }
206         return result;
207     }

250     /**
251      * Performs the analysis.
252      *
253      * @param dependency the dependency to analyze
254      * @param engine the engine
255      * @throws AnalysisException when there's an exception during analysis
256      */
257     @Override
258     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
259         if (!isEnabled()) {
260             return;
261         }
262         try {
263             final MavenArtifact ma = searcher.searchSha1(dependency.getSha1sum());
264             dependency.addAsEvidence("nexus", ma, Confidence.HIGH);
265             boolean pomAnalyzed = false;
266             LOGGER.debug("POM URL {}", ma.getPomUrl());
267             for (Evidence e : dependency.getEvidence(EvidenceType.VENDOR)) {
268                 if ("pom".equals(e.getSource())) {
269                     pomAnalyzed = true;
270                     break;
271                 }
272             }
273             if (!pomAnalyzed && ma.getPomUrl() != null) {
274                 File pomFile = null;
275                 try {
276                     final File baseDir = getSettings().getTempDirectory();
277                     pomFile = File.createTempFile("pom", ".xml", baseDir);
278                     if (!pomFile.delete()) {
279                         LOGGER.warn("Unable to fetch pom.xml for {} from Nexus repository; "
280                                 + "this could result in undetected CPE/CVEs.", dependency.getFileName());
281                         LOGGER.debug("Unable to delete temp file");
282                     }
283                     LOGGER.debug("Downloading {}", ma.getPomUrl());
284                     final Downloader downloader = new Downloader(getSettings());
285                     downloader.fetchFile(new URL(ma.getPomUrl()), pomFile);
286                     PomUtils.analyzePOM(dependency, pomFile);
287                 } catch (DownloadFailedException ex) {
288                     LOGGER.warn("Unable to download pom.xml for {} from Nexus repository; "
289                             + "this could result in undetected CPE/CVEs.", dependency.getFileName());
290                 } catch (TooManyRequestsException ex) {
291                     this.setEnabled(false);
292                     throw new AnalysisException("Received a 429 - too many requests from nexus; "
293                             + "the nexus analyzer is being disabled.", ex);
294                 } catch (ResourceNotFoundException ex) {
295                     LOGGER.warn("pom.xml not found for {} from nexus; "
296                             + "this could result in undetected CPE/CVEs.", dependency.getFileName());
297                 } finally {
298                     if (pomFile != null && pomFile.exists() && !FileUtils.deleteQuietly(pomFile)) {
299                         LOGGER.debug("Failed to delete temporary pom file {}", pomFile);
300                         pomFile.deleteOnExit();
301                     }
302                 }
303             }
304         } catch (IllegalArgumentException iae) {
305             //dependency.addAnalysisException(new AnalysisException("Invalid SHA-1"));
306             LOGGER.info("invalid sha-1 hash on {}", dependency.getFileName());
307         } catch (FileNotFoundException fnfe) {
308             //dependency.addAnalysisException(new AnalysisException("Artifact not found on repository"));
309             LOGGER.debug("Artifact not found in repository '{}'", dependency.getFileName());
310             LOGGER.debug(fnfe.getMessage(), fnfe);
311         } catch (IOException ioe) {
312             //dependency.addAnalysisException(new AnalysisException("Could not connect to repository", ioe));
313             LOGGER.debug("Could not connect to nexus repository", ioe);
314         }
315     }

317     /**
318      * Determine if a proxy should be used for the Nexus Analyzer.
319      *
320      * @return {@code true} if a proxy should be used
321      */
322     public boolean useProxy() {
323         try {
324             return getSettings().getString(Settings.KEYS.PROXY_SERVER) != null
325                     && getSettings().getBoolean(Settings.KEYS.ANALYZER_NEXUS_USES_PROXY);
326         } catch (InvalidSettingException ise) {
327             LOGGER.warn("Failed to parse proxy settings.", ise);
328             return false;
329         }
330     }
