75       * An expected error from `yarn audit --offline --verbose --json` that will
76       * be ignored.
77       */
78      private static final String EXPECTED_ERROR = "{\"type\":\"error\",\"data\":\"Can't make a request in "
79              + "offline mode (\\\"https://registry.yarnpkg.com/-/npm/v1/security/audits\\\")\"}\n";

86      /**
87       * Analyzes the yarn lock file to determine vulnerable dependencies. Uses
88       * yarn audit --offline to generate the payload to be sent to the NPM API.
89       *
90       * @param dependency the yarn lock file
91       * @param engine the analysis engine
92       * @throws AnalysisException thrown if there is an error analyzing the file
93       */
94      @Override
95      protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
96          if (dependency.getDisplayFileName().equals(dependency.getFileName())) {
97              engine.removeDependency(dependency);
98          }
99          final File packageLock = dependency.getActualFile();
100         if (!packageLock.isFile() || packageLock.length() == 0 || !shouldProcess(packageLock)) {
101             return;
102         }
103         final File packageJson = new File(packageLock.getParentFile(), "package.json");
104         final List<Advisory> advisories;
105         final MultiValuedMap<String, String> dependencyMap = new HashSetValuedHashMap<>();
106         advisories = analyzePackage(packageLock, packageJson, dependency, dependencyMap);
107         try {
108             processResults(advisories, engine, dependency, dependencyMap);
109         } catch (CpeValidationException ex) {
110             throw new UnexpectedAnalysisException(ex);
111         }
112     }

134     /**
135      * Initializes the analyzer once before any analysis is performed.
136      *
137      * @param engine a reference to the dependency-check engine
138      * @throws InitializationException if there's an error during initialization
139      */
140     @Override
141     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
142         super.prepareFileTypeAnalyzer(engine);
143         if (!isEnabled()) {
144             LOGGER.debug("{} Analyzer is disabled skipping yarn executable check", getName());
145             return;
146         }
147         final List<String> args = new ArrayList<>();
148         args.add(getYarn());
149         args.add("--help");
150         final ProcessBuilder builder = new ProcessBuilder(args);
151         LOGGER.debug("Launching: {}", args);
152         try {
153             final Process process = builder.start();
154             try (ProcessReader processReader = new ProcessReader(process)) {
155                 processReader.readAll();
156                 final int exitValue = process.waitFor();
157                 final int expectedExitValue = 0;
158                 final int yarnExecutableNotFoundExitValue = 127;
159                 switch (exitValue) {
160                     case expectedExitValue:
161                         LOGGER.debug("{} is enabled.", getName());
162                         break;
163                     case yarnExecutableNotFoundExitValue:
164                     default:
165                         this.setEnabled(false);
166                         LOGGER.warn("The {} has been disabled. Yarn executable was not found.", getName());
167                 }
168             }
169         } catch (Exception ex) {
170             this.setEnabled(false);
171             LOGGER.warn("The {} has been disabled. Yarn executable was not found.", getName());
172             throw new InitializationException("Unable to read yarn audit output.", ex);
173         }
174     }

176     /**
177      * Attempts to determine the path to `yarn`.
178      *
179      * @return the path to `yarn`
180      */
181     private String getYarn() {
182         final String value;
183         synchronized (this) {
184             if (yarnPath == null) {
185                 final String path = getSettings().getString(Settings.KEYS.ANALYZER_YARN_PATH);
186                 if (path == null) {
187                     yarnPath = "yarn";
188                 } else {
189                     final File yarnFile = new File(path);
190                     if (yarnFile.isFile()) {
191                         yarnPath = yarnFile.getAbsolutePath();
192                     } else {
193                         LOGGER.warn("Provided path to `yarn` executable is invalid.");
194                         yarnPath = "yarn";
195                     }
196                 }
197             }
198             value = yarnPath;
199         }
200         return value;
201     }

203     private JsonObject fetchYarnAuditJson(Dependency dependency, boolean skipDevDependencies) throws AnalysisException {
204         final File folder = dependency.getActualFile().getParentFile();
205         if (!folder.isDirectory()) {
206             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
207         }
208         try {
209             final List<String> args = new ArrayList<>();
210 
211             args.add(getYarn());
212             args.add("audit");
213             //offline audit is not supported - but the audit request is generated in the verbose output
214             args.add("--offline");
215             if (skipDevDependencies) {
216                 args.add("--groups");
217                 args.add("dependencies");
218             }
219             args.add("--json");
220             args.add("--verbose");
221             final ProcessBuilder builder = new ProcessBuilder(args);
222             builder.directory(folder);
223             LOGGER.debug("Launching: {}", args);
224             // Workaround 64k limitation of InputStream, redirect stdout to a file that we will read later
225             // instead of reading directly stdout from Process's InputStream which is topped at 64k
226 
227             final File tmpFile = getSettings().getTempFile("yarn_audit", "json");
228             builder.redirectOutput(tmpFile);
229             final Process process = builder.start();
230             try (ProcessReader processReader = new ProcessReader(process)) {
231                 processReader.readAll();
232                 final String errOutput = processReader.getError();
233 
234                 if (!StringUtils.isBlank(errOutput) && !EXPECTED_ERROR.equals(errOutput)) {
235                     LOGGER.debug("Process Error Out: {}", errOutput);
236                     LOGGER.debug("Process Out: {}", processReader.getOutput());
237                 }
238                 final String verboseJson = FileUtils.readFileToString(tmpFile, StandardCharsets.UTF_8);
239                 final String auditRequestJson = Arrays.stream(verboseJson.split("\n"))
240                         .filter(line -> line.contains("Audit Request"))
241                         .findFirst().get();
242                 String auditRequest;
243                 try (JsonReader reader = Json.createReader(IOUtils.toInputStream(auditRequestJson, StandardCharsets.UTF_8))) {
244                     final JsonObject jsonObject = reader.readObject();
245                     auditRequest = jsonObject.getString("data");
246                     auditRequest = auditRequest.substring(15);
247                 }
248                 LOGGER.debug("Audit Request: {}", auditRequest);
249 
250                 return Json.createReader(IOUtils.toInputStream(auditRequest, StandardCharsets.UTF_8)).readObject();
251             } catch (InterruptedException ex) {
252                 Thread.currentThread().interrupt();
253                 throw new AnalysisException("Yarn audit process was interrupted.", ex);
254             }
255         } catch (IOException ioe) {
256             throw new AnalysisException("yarn audit failure; this error can be ignored if you are not analyzing projects with a yarn lockfile.", ioe);
257         }
258     }

260     /**
261      * Analyzes the package and yarn lock files by extracting dependency
262      * information, creating a payload to submit to the npm audit API,
263      * submitting the payload, and returning the identified advisories.
264      *
265      * @param lockFile a reference to the package-lock.json
266      * @param packageFile a reference to the package.json
267      * @param dependency a reference to the dependency-object for the yarn.lock
268      * @param dependencyMap a collection of module/version pairs; during
269      * creation of the payload the dependency map is populated with the
270      * module/version information.
271      * @return a list of advisories
272      * @throws AnalysisException thrown when there is an error creating or
273      * submitting the npm audit API payload
274      */
275     private List<Advisory> analyzePackage(final File lockFile, final File packageFile,
276             Dependency dependency, MultiValuedMap<String, String> dependencyMap)
277             throws AnalysisException {
278         try {
279             final boolean skipDevDependencies = getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_SKIPDEV, false);
280             // Retrieves the contents of package-lock.json from the Dependency
281             final JsonObject lockJson = fetchYarnAuditJson(dependency, skipDevDependencies);
282             // Retrieves the contents of package-lock.json from the Dependency
283             final JsonReader packageReader = Json.createReader(FileUtils.openInputStream(packageFile));
284             final JsonObject packageJson = packageReader.readObject();
285 
286             // Modify the payload to meet the NPM Audit API requirements
287             final JsonObject payload = NpmPayloadBuilder.build(lockJson, packageJson, dependencyMap, skipDevDependencies);
288 
289             // Submits the package payload to the nsp check service
290             return getSearcher().submitPackage(payload);
291 
292         } catch (URLConnectionFailureException e) {
293             this.setEnabled(false);
294             throw new AnalysisException("Failed to connect to the NPM Audit API (YarnAuditAnalyzer); the analyzer "
295                     + "is being disabled and may result in false negatives.", e);
296         } catch (IOException e) {
297             LOGGER.debug("Error reading dependency or connecting to NPM Audit API", e);
298             this.setEnabled(false);
299             throw new AnalysisException("Failed to read results from the NPM Audit API (YarnAuditAnalyzer); "
300                     + "the analyzer is being disabled and may result in false negatives.", e);
301         } catch (JsonException e) {
302             throw new AnalysisException(String.format("Failed to parse %s file from the NPM Audit API "
303                     + "(YarnAuditAnalyzer).", lockFile.getPath()), e);
304         } catch (SearchException ex) {
305             LOGGER.error("YarnAuditAnalyzer failed on {}", dependency.getActualFilePath());
306             throw ex;
307         }
308     }
