115     @Override
116     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
117         if (engine != null) {
118             this.cvedb = engine.getDatabase();
119         }
120 
121         // Here we check if mix_audit actually runs from this location. We do this by running the
122         // `mix_audit --version` command and seeing whether or not it succeeds (if it returns with an exit value of 0)
123         final Process process;
124         try {
125             final List<String> mixAuditArgs = ImmutableList.of("--version");
126             process = launchMixAudit(getSettings().getTempDirectory(), mixAuditArgs);
127         } catch (AnalysisException ae) {
128             setEnabled(false);
129             final String msg = String.format("Exception from mix_audit process: %s. Disabling %s", ae.getCause(), ANALYZER_NAME);
130             throw new InitializationException(msg, ae);
131         } catch (IOException ex) {
132             setEnabled(false);
133             throw new InitializationException("Unable to create temporary file, the Mix Audit Analyzer will be disabled", ex);
134         }
135 
136         final int exitValue;
137         try {
138             exitValue = process.waitFor();
139         } catch (InterruptedException ex) {
140             setEnabled(false);
141             final String msg = String.format("mix_audit process was interrupted. Disabling %s", ANALYZER_NAME);
142             Thread.currentThread().interrupt();
143             throw new InitializationException(msg);
144         }
145 
146         final String mixAuditVersionDetails;
147         if (exitValue != 0) {
148             try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
149                 if (!reader.ready()) {
150                     LOGGER.warn("Unexpected exit value from mix_audit process and error stream unexpectedly not ready to capture error details. "
151                             + "Disabling {}. Exit value was: {}", ANALYZER_NAME, exitValue);
152                     setEnabled(false);
153                     throw new InitializationException("mix_audit error stream unexpectedly not ready.");
154                 } else {
155                     final String line = reader.readLine();
156                     setEnabled(false);
157                     LOGGER.warn("Unexpected exit value from mix_audit process. Disabling {}. Exit value was: {}. "
158                             + "error stream output from mix_audit process was: {}", ANALYZER_NAME, exitValue, line);
159                     throw new InitializationException("Unexpected exit value from bundle-audit process.");
160                 }
161             } catch (UnsupportedEncodingException ex) {
162                 setEnabled(false);
163                 throw new InitializationException("Unexpected mix_audit encoding when reading error stream.", ex);
164             } catch (IOException ex) {
165                 setEnabled(false);
166                 throw new InitializationException("Unable to read mix_audit output from error stream.", ex);
167             }
168         } else {
169             try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
170                 if (!reader.ready()) {
171                     LOGGER.warn("mix_audit input stream unexpectedly not ready to capture version details. Disabling {}", ANALYZER_NAME);
172                     setEnabled(false);
173                     throw new InitializationException("mix_audit input stream unexpectedly not ready to capture version details.");
174                 } else {
175                     mixAuditVersionDetails = reader.readLine();
176                 }
177             } catch (UnsupportedEncodingException ex) {
178                 setEnabled(false);
179                 throw new InitializationException("Unexpected mix_audit encoding when reading input stream.", ex);
180             } catch (IOException ex) {
181                 setEnabled(false);
182                 throw new InitializationException("Unable to read mix_audit output from input stream.", ex);
183             }
184         }
185 
186         if (isEnabled()) {
187             LOGGER.info("{} is enabled and is using mix_audit with version: {}.", ANALYZER_NAME, mixAuditVersionDetails);
188         }
189     }

222     /**
223      * Launch mix audit.
224      *
225      * @param folder directory that contains the mix.lock file
226      * @param mixAuditArgs the arguments to pass to mix audit
227      * @return a handle to the process
228      * @throws AnalysisException thrown when there is an issue launching mix
229      * audit
230      */
231     private Process launchMixAudit(File folder, List<String> mixAuditArgs) throws AnalysisException {
232         if (!folder.isDirectory()) {
233             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
234         }
235 
236         final List<String> args = new ArrayList<>();
237         final String mixAuditPath = getSettings().getString(Settings.KEYS.ANALYZER_MIX_AUDIT_PATH);
238         File mixAudit = null;
239 
240         if (mixAuditPath != null) {
241             mixAudit = new File(mixAuditPath);
242             if (!mixAudit.isFile()) {
243                 LOGGER.warn("Supplied `mixAudit` path is incorrect: {}", mixAuditPath);
244                 mixAudit = null;
245             }
246         } else {
247             final Path homePath = Paths.get(System.getProperty("user.home"));
248             final Path escriptPath = Paths.get(homePath.toString(), ".mix", "escripts", "mix_audit");
249             mixAudit = escriptPath.toFile();
250         }
251 
252         args.add(mixAudit != null ? mixAudit.getAbsolutePath() : "mix_audit");
253         args.addAll(mixAuditArgs);
254         final ProcessBuilder builder = new ProcessBuilder(args);
255 
256         builder.directory(folder);
257         try {
258             LOGGER.info("Launching: {} from {}", args, folder);
259             return builder.start();
260         } catch (IOException ioe) {
261             throw new AnalysisException("mix_audit initialization failure; this error can be ignored if you are not analyzing Elixir. "
262                     + "Otherwise ensure that mix_audit is installed and the path to mix_audit is correctly specified", ioe);
263         }
264     }

266     /**
267      * Determines if the analyzer can analyze the given file type.
268      *
269      * @param dependency the dependency to determine if it can analyze
270      * @param engine the dependency-check engine
271      * @throws AnalysisException thrown if there is an analysis exception.
272      */
273     @Override
274     protected void analyzeDependency(Dependency dependency, Engine engine)
275             throws AnalysisException {
276         final File parentFile = dependency.getActualFile().getParentFile();
277         final List<String> mixAuditArgs = ImmutableList.of("--format", "json");
278 
279         final Process process = launchMixAudit(parentFile, mixAuditArgs);
280         final int exitValue;
281         try {
282             exitValue = process.waitFor();
283         } catch (InterruptedException ie) {
284             Thread.currentThread().interrupt();
285             throw new AnalysisException("mix_audit process interrupted", ie);
286         }
287         if (exitValue < 0 || exitValue > 1) {
288             final String msg = String.format("Unexpected exit code from mix_audit process; exit code: %s", exitValue);
289             throw new AnalysisException(msg);
290         }
291         try {
292             try (BufferedReader errReader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
293                 while (errReader.ready()) {
294                     final String error = errReader.readLine();
295                     LOGGER.warn(error);
296                 }
297             }
298             try (BufferedReader rdr = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
299                 processMixAuditOutput(dependency, engine, rdr);
300             }
301         } catch (IOException | CpeValidationException ioe) {
302             LOGGER.warn("mix_audit failure", ioe);
303         }
304     }
305 
306     /**
307      * Processes the mix audit output.
308      *
309      * @param original the dependency
310      * @param engine the dependency-check engine
311      * @param rdr the reader of the report
312      * @throws IOException thrown if the report cannot be read
313      * @throws CpeValidationException if there is an error building the
314      * CPE/VulnerableSoftware object
315      */
316     private void processMixAuditOutput(Dependency original, Engine engine, BufferedReader rdr) throws AnalysisException, CpeValidationException {
317         final MixAuditJsonParser parser = new MixAuditJsonParser(rdr);
318         parser.process();
319 
320         for (MixAuditResult result : parser.getResults()) {
321             final Dependency dependency = createDependency(original, result.getDependencyPackage(), result.getDependencyVersion());
322             Vulnerability vulnerability = cvedb.getVulnerability(result.getCve());
323 
324             if (vulnerability == null) {
325                 vulnerability = createVulnerability(result);
326             }
327 
328             dependency.addVulnerability(vulnerability);
329             engine.addDependency(dependency);
330         }
331     }

333     private Dependency createDependency(Dependency parentDependency, String packageName, String version) {
334         final Dependency dep = new Dependency(parentDependency.getActualFile(), true);
335 
336         final String identifier = String.format("%s:%s", packageName, version);
337 
338         dep.setEcosystem(DEPENDENCY_ECOSYSTEM);
339         dep.setDisplayFileName(identifier);
340         dep.setName(packageName);
341         dep.setVersion(version);
342         dep.setPackagePath(identifier);
343         dep.setMd5sum(Checksum.getMD5Checksum(identifier));
344         dep.setSha1sum(Checksum.getSHA1Checksum(identifier));
345         dep.setSha256sum(Checksum.getSHA256Checksum(identifier));
346 
347         dep.addEvidence(EvidenceType.VERSION, "mix_audit", "Version", version, Confidence.HIGHEST);
348         dep.addEvidence(EvidenceType.PRODUCT, "mix_audit", "Package", packageName, Confidence.HIGHEST);
349 
350         try {
351             final PackageURL purl = PackageURLBuilder.aPackageURL().withType("hex").withName(packageName)
352                     .withVersion(version).build();
353             dep.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
354         } catch (MalformedPackageURLException ex) {
355             LOGGER.debug("Unable to build package url for hex", ex);
356             final GenericIdentifier id = new GenericIdentifier("hex:" + packageName + "@" + version,
357                     Confidence.HIGHEST);
358             dep.addSoftwareIdentifier(id);
359         }
360 
361         return dep;
362     }
363 
364     private Vulnerability createVulnerability(MixAuditResult result) throws CpeValidationException {
365         final String product = result.getDependencyPackage();
366         final String version = result.getDependencyVersion();
367 
368         final Vulnerability vulnerability = new Vulnerability();
369         vulnerability.setSource(Vulnerability.Source.MIXAUDIT);
370 
371         final VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
372         final VulnerableSoftware vs = builder.part(Part.APPLICATION)
373                 .vendor(String.format("%s_project", product))
374                 .product(product)
375                 .version(version).build();
376 
377         vulnerability.addVulnerableSoftware(vs);
378         vulnerability.setMatchedVulnerableSoftware(vs);
379 
380         vulnerability.setCvssV2(new CvssV2(-1, "-", "-", "-", "-", "-", "-", "unknown"));
381         vulnerability.setDescription(result.getDescription());
382         vulnerability.setName(result.getCve());
383 
384         return vulnerability;
385     }
