46  /**
47   * Analyzer which will parse a Nuspec file to gather module information.
48   *
49   * @author colezlaw
50   */
51  @ThreadSafe
52  public class NuspecAnalyzer extends AbstractFileTypeAnalyzer {
53  
54      /**
55       * A descriptor for the type of dependencies processed or added by this
56       * analyzer.
57       */
58      public static final String DEPENDENCY_ECOSYSTEM = Ecosystem.DOTNET;
59  
60      /**
61       * The logger.
62       */
63      private static final Logger LOGGER = LoggerFactory.getLogger(NuspecAnalyzer.class);
64  
65      /**
66       * The name of the analyzer.
67       */
68      private static final String ANALYZER_NAME = "Nuspec Analyzer";
69  
70      /**
71       * The phase in which the analyzer runs.
72       */
73      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
74  
75      /**
76       * The types of files on which this will work.
77       */
78      private static final String SUPPORTED_EXTENSIONS = "nuspec";
79      /**
80       * The file filter used to determine which files this analyzer supports.
81       */
82      private static final FileFilter FILTER = FileFilterBuilder.newInstance().addExtensions(SUPPORTED_EXTENSIONS).build();
83  
84      /**
85       * Initializes the analyzer once before any analysis is performed.
86       *
87       * @param engine a reference to the dependency-check engine
88       * @throws InitializationException if there's an error during initialization
89       */
90      @Override
91      public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
92          //nothing to initialize
93      }
94  
95      /**
96       * Returns the analyzer's name.
97       *
98       * @return the name of the analyzer
99       */
100     @Override
101     public String getName() {
102         return ANALYZER_NAME;
103     }
104 
105     /**
106      * Returns the key used in the properties file to reference the analyzer's
107      * enabled property.
108      *
109      * @return the analyzer's enabled property setting key
110      */
111     @Override
112     protected String getAnalyzerEnabledSettingKey() {
113         return Settings.KEYS.ANALYZER_NUSPEC_ENABLED;
114     }
115 
116     /**
117      * Returns the analysis phase under which the analyzer runs.
118      *
119      * @return the phase under which this analyzer runs
120      */
121     @Override
122     public AnalysisPhase getAnalysisPhase() {
123         return ANALYSIS_PHASE;
124     }
125 
126     /**
127      * Returns the FileFilter
128      *
129      * @return the FileFilter
130      */
131     @Override
132     protected FileFilter getFileFilter() {
133         return FILTER;
134     }
135 
136     /**
137      * Performs the analysis.
138      *
139      * @param dependency the dependency to analyze
140      * @param engine the engine
141      * @throws AnalysisException when there's an exception during analysis
142      */
143     @Override
144     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
145         LOGGER.debug("Checking Nuspec file {}", dependency);
146         try {
147             final NuspecParser parser = new XPathNuspecParser();
148             final NugetPackage np;
149             try (FileInputStream fis = new FileInputStream(dependency.getActualFilePath())) {
150                 np = parser.parse(fis);
151             } catch (NuspecParseException | FileNotFoundException ex) {
152                 throw new AnalysisException(ex);
153             }
154 
155             dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
156             if (np.getOwners() != null) {
157                 dependency.addEvidence(EvidenceType.VENDOR, "nuspec", "owners", np.getOwners(), Confidence.HIGHEST);
158             }
159             dependency.addEvidence(EvidenceType.VENDOR, "nuspec", "authors", np.getAuthors(), Confidence.HIGH);
160             dependency.addEvidence(EvidenceType.VERSION, "nuspec", "version", np.getVersion(), Confidence.HIGHEST);
161             dependency.addEvidence(EvidenceType.PRODUCT, "nuspec", "id", np.getId(), Confidence.HIGHEST);
162             dependency.setName(np.getId());
163             dependency.setVersion(np.getVersion());
164             try {
165                 final PackageURL purl = PackageURLBuilder.aPackageURL().withType("nuget").withName(np.getId()).withVersion(np.getVersion()).build();
166                 dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
167             } catch (MalformedPackageURLException ex) {
168                 LOGGER.debug("Unable to build package url for nuspec", ex);
169                 final GenericIdentifier gid = new GenericIdentifier("nuspec:" + np.getId() + "@" + np.getVersion(), Confidence.HIGHEST);
170                 dependency.addSoftwareIdentifier(gid);
171             }
172             final String packagePath = String.format("%s:%s", np.getId(), np.getVersion());
173             dependency.setPackagePath(packagePath);
174             dependency.setDisplayFileName(packagePath);
175             if (np.getLicenseUrl() != null && !np.getLicenseUrl().isEmpty()) {
176                 dependency.setLicense(np.getLicenseUrl());
177             }
178             if (np.getTitle() != null) {
179                 dependency.addEvidence(EvidenceType.PRODUCT, "nuspec", "title", np.getTitle(), Confidence.MEDIUM);
180             }
181         } catch (Throwable e) {
182             throw new AnalysisException(e);
183         }
184     }
185 }
