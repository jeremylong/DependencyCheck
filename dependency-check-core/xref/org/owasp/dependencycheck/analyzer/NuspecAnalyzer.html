45  /**
46   * Analyzer which will parse a Nuspec file to gather module information.
47   *
48   * @author colezlaw
49   */
50  @ThreadSafe
51  public class NuspecAnalyzer extends AbstractFileTypeAnalyzer {
52  
53      /**
54       * A descriptor for the type of dependencies processed or added by this
55       * analyzer.
56       */
57      public static final String DEPENDENCY_ECOSYSTEM = Ecosystem.DOTNET;
58  
59      /**
60       * The logger.
61       */
62      private static final Logger LOGGER = LoggerFactory.getLogger(NuspecAnalyzer.class);
63  
64      /**
65       * The name of the analyzer.
66       */
67      private static final String ANALYZER_NAME = "Nuspec Analyzer";
68  
69      /**
70       * The phase in which the analyzer runs.
71       */
72      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
73  
74      /**
75       * The types of files on which this will work.
76       */
77      private static final String SUPPORTED_EXTENSIONS = "nuspec";
78      /**
79       * The file filter used to determine which files this analyzer supports.
80       */
81      private static final FileFilter FILTER = FileFilterBuilder.newInstance().addExtensions(SUPPORTED_EXTENSIONS).build();
82  
83      /**
84       * Initializes the analyzer once before any analysis is performed.
85       *
86       * @param engine a reference to the dependency-check engine
87       * @throws InitializationException if there's an error during initialization
88       */
89      @Override
90      public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
91          //nothing to initialize
92      }
93  
94      /**
95       * Returns the analyzer's name.
96       *
97       * @return the name of the analyzer
98       */
99      @Override
100     public String getName() {
101         return ANALYZER_NAME;
102     }
103 
104     /**
105      * Returns the key used in the properties file to reference the analyzer's
106      * enabled property.
107      *
108      * @return the analyzer's enabled property setting key
109      */
110     @Override
111     protected String getAnalyzerEnabledSettingKey() {
112         return Settings.KEYS.ANALYZER_NUSPEC_ENABLED;
113     }
114 
115     /**
116      * Returns the analysis phase under which the analyzer runs.
117      *
118      * @return the phase under which this analyzer runs
119      */
120     @Override
121     public AnalysisPhase getAnalysisPhase() {
122         return ANALYSIS_PHASE;
123     }
124 
125     /**
126      * Returns the FileFilter
127      *
128      * @return the FileFilter
129      */
130     @Override
131     protected FileFilter getFileFilter() {
132         return FILTER;
133     }
134 
135     /**
136      * Performs the analysis.
137      *
138      * @param dependency the dependency to analyze
139      * @param engine the engine
140      * @throws AnalysisException when there's an exception during analysis
141      */
142     @Override
143     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
144         LOGGER.debug("Checking Nuspec file {}", dependency);
145         try {
146             final XPathNuspecParser parser = new XPathNuspecParser();
147             final NugetPackage np;
148             try (FileInputStream fis = new FileInputStream(dependency.getActualFilePath())) {
149                 np = parser.parse(fis);
150             } catch (NuspecParseException | FileNotFoundException ex) {
151                 throw new AnalysisException(ex);
152             }
153 
154             dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
155             if (np.getOwners() != null) {
156                 dependency.addEvidence(EvidenceType.VENDOR, "nuspec", "owners", np.getOwners(), Confidence.HIGHEST);
157             }
158             dependency.addEvidence(EvidenceType.VENDOR, "nuspec", "authors", np.getAuthors(), Confidence.HIGH);
159             dependency.addEvidence(EvidenceType.VERSION, "nuspec", "version", np.getVersion(), Confidence.HIGHEST);
160             dependency.addEvidence(EvidenceType.PRODUCT, "nuspec", "id", np.getId(), Confidence.HIGHEST);
161             dependency.addEvidence(EvidenceType.VENDOR, "nuspec", "description", np.getDescription(), Confidence.LOW);
162             dependency.addEvidence(EvidenceType.PRODUCT, "nuspec", "description", np.getDescription(), Confidence.LOW);
163             dependency.setName(np.getId());
164             dependency.setVersion(np.getVersion());
165             try {
166                 final PackageURL purl = PackageURLBuilder.aPackageURL().withType("nuget").withName(np.getId()).withVersion(np.getVersion()).build();
167                 dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
168             } catch (MalformedPackageURLException ex) {
169                 LOGGER.debug("Unable to build package url for nuspec", ex);
170                 final GenericIdentifier gid = new GenericIdentifier("nuspec:" + np.getId() + "@" + np.getVersion(), Confidence.HIGHEST);
171                 dependency.addSoftwareIdentifier(gid);
172             }
173             final String packagePath = String.format("%s:%s", np.getId(), np.getVersion());
174             dependency.setPackagePath(packagePath);
175             dependency.setDisplayFileName(packagePath);
176             if (np.getLicenseUrl() != null && !np.getLicenseUrl().isEmpty()) {
177                 dependency.setLicense(np.getLicenseUrl());
178             }
179             if (np.getTitle() != null) {
180                 dependency.addEvidence(EvidenceType.PRODUCT, "nuspec", "title", np.getTitle(), Confidence.MEDIUM);
181             }
182         } catch (Throwable e) {
183             throw new AnalysisException(e);
184         }
185     }
186 }
