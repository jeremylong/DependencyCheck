47  /**
48   * Abstract base suppression analyzer that contains methods for parsing the
49   * suppression XML file.
50   *
51   * @author Jeremy Long
52   */
53  @ThreadSafe
54  public abstract class AbstractSuppressionAnalyzer extends AbstractAnalyzer {
55  
56      /**
57       * The Logger for use throughout the class.
58       */
59      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractSuppressionAnalyzer.class);
60      /**
61       * The file name of the base suppression XML file.
62       */
63      private static final String BASE_SUPPRESSION_FILE = "dependencycheck-base-suppression.xml";
64      /**
65       * The list of suppression rules.
66       */
67      private final List<SuppressionRule> rules = new ArrayList<>();
68  
69      /**
70       * Get the number of suppression rules.
71       *
72       * @return the number of suppression rules
73       */
74      protected int getRuleCount() {
75          return rules.size();
76      }
77  
78      /**
79       * Returns a list of file EXTENSIONS supported by this analyzer.
80       *
81       * @return a list of file EXTENSIONS supported by this analyzer.
82       */
83      public Set<String> getSupportedExtensions() {
84          return null;
85      }
86  
87      /**
88       * The prepare method loads the suppression XML file.
89       *
90       * @param engine a reference the dependency-check engine
91       * @throws InitializationException thrown if there is an exception
92       */
93      @Override
94      public synchronized void prepareAnalyzer(Engine engine) throws InitializationException {
95          if (rules.isEmpty()) {
96              try {
97                  loadSuppressionBaseData();
98              } catch (SuppressionParseException ex) {
99                  throw new InitializationException("Error initializing the suppression analyzer: " + ex.getLocalizedMessage(), ex, true);
100             }
101 
102             try {
103                 loadSuppressionData();
104             } catch (SuppressionParseException ex) {
105                 throw new InitializationException("Warn initializing the suppression analyzer: " + ex.getLocalizedMessage(), ex, false);
106             }
107         }
108     }
109 
110     @Override
111     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
112         if (rules.isEmpty()) {
113             return;
114         }
115         rules.forEach((rule) -> rule.process(dependency));
116     }
117 
118     /**
119      * Loads all the suppression rules files configured in the {@link Settings}.
120      *
121      * @throws SuppressionParseException thrown if the XML cannot be parsed.
122      */
123     private void loadSuppressionData() throws SuppressionParseException {
124         final List<SuppressionRule> ruleList = new ArrayList<>();
125         final SuppressionParser parser = new SuppressionParser();
126         final String[] suppressionFilePaths = getSettings().getArray(Settings.KEYS.SUPPRESSION_FILE);
127         final List<String> failedLoadingFiles = new ArrayList<>();
128         if (suppressionFilePaths != null && suppressionFilePaths.length > 0) {
129             // Load all the suppression file paths
130             for (final String suppressionFilePath : suppressionFilePaths) {
131                 try {
132                     ruleList.addAll(loadSuppressionFile(parser, suppressionFilePath));
133                 } catch (SuppressionParseException ex) {
134                     final String msg = String.format("Failed to load %s, caused by %s. ", suppressionFilePath, ex.getMessage());
135                     failedLoadingFiles.add(msg);
136                 }
137             }
138         }
139         LOGGER.debug("{} suppression rules were loaded.", ruleList.size());
140         rules.addAll(ruleList);
141         if (!failedLoadingFiles.isEmpty()) {
142             LOGGER.debug("{} suppression files failed to load.", failedLoadingFiles.size());
143             final StringBuilder sb = new StringBuilder();
144             failedLoadingFiles.forEach((item) -> sb.append(item));
145             throw new SuppressionParseException(sb.toString());
146         }
147     }
148 
149     /**
150      * Loads all the base suppression rules files.
151      *
152      * @throws SuppressionParseException thrown if the XML cannot be parsed.
153      */
154     private void loadSuppressionBaseData() throws SuppressionParseException {
155         final SuppressionParser parser = new SuppressionParser();
156         final List<SuppressionRule> ruleList;
157         try (InputStream in = FileUtils.getResourceAsStream(BASE_SUPPRESSION_FILE)) {
158             if (in == null) {
159                 throw new SuppressionParseException("Suppression rules `" + BASE_SUPPRESSION_FILE + "` could not be found");
160             }
161             ruleList = parser.parseSuppressionRules(in);
162         } catch (SAXException | IOException ex) {
163             throw new SuppressionParseException("Unable to parse the base suppression data file", ex);
164         }
165         rules.addAll(ruleList);
166     }
167 
168     /**
169      * Load a single suppression rules file from the path provided using the
170      * parser provided.
171      *
172      * @param parser the parser to use for loading the file
173      * @param suppressionFilePath the path to load
174      * @return the list of loaded suppression rules
175      * @throws SuppressionParseException thrown if the suppression file cannot
176      * be loaded and parsed.
177      */
178     private List<SuppressionRule> loadSuppressionFile(final SuppressionParser parser,
179             final String suppressionFilePath) throws SuppressionParseException {
180         LOGGER.debug("Loading suppression rules from '{}'", suppressionFilePath);
181         final List<SuppressionRule> list = new ArrayList<>();
182         File file = null;
183         boolean deleteTempFile = false;
184         try {
185             final Pattern uriRx = Pattern.compile("^(https?|file):.*", Pattern.CASE_INSENSITIVE);
186             if (uriRx.matcher(suppressionFilePath).matches()) {
187                 deleteTempFile = true;
188                 file = getSettings().getTempFile("suppression", "xml");
189                 final URL url = new URL(suppressionFilePath);
190                 final Downloader downloader = new Downloader(getSettings());
191                 try {
192                     downloader.fetchFile(url, file, false);
193                 } catch (DownloadFailedException ex) {
194                     LOGGER.trace("Failed download suppression file - first attempt", ex);
195                     try {
196                         Thread.sleep(500);
197                         downloader.fetchFile(url, file, true);
198                     } catch (TooManyRequestsException ex1) {
199                         throw new SuppressionParseException("Unable to download supression file `" + file
200                                 + "`; received 429 - too many requests", ex1);
201                     } catch (ResourceNotFoundException ex1) {
202                         throw new SuppressionParseException("Unable to download supression file `" + file
203                                 + "`; received 404 - resource not found", ex1);
204                     } catch (InterruptedException ex1) {
205                         Thread.currentThread().interrupt();
206                         throw new SuppressionParseException("Unable to download supression file `" + file + "`", ex1);
207                     }
208                 } catch (TooManyRequestsException ex) {
209                     throw new SuppressionParseException("Unable to download supression file `" + file
210                             + "`; received 429 - too many requests", ex);
211                 } catch (ResourceNotFoundException ex) {
212                     throw new SuppressionParseException("Unable to download supression file `" + file + "`; received 404 - resource not found", ex);
213                 }
214             } else {
215                 file = new File(suppressionFilePath);
216 
217                 if (!file.exists()) {
218                     try (InputStream suppressionFromClasspath = FileUtils.getResourceAsStream(suppressionFilePath)) {
219                         if (suppressionFromClasspath != null) {
220                             deleteTempFile = true;
221                             file = getSettings().getTempFile("suppression", "xml");
222                             try {
223                                 org.apache.commons.io.FileUtils.copyInputStreamToFile(suppressionFromClasspath, file);
224                             } catch (IOException ex) {
225                                 throwSuppressionParseException("Unable to locate suppression file in classpath", ex, suppressionFilePath);
226                             }
227                         }
228                     }
229                 }
230             }
231             if (file != null) {
232                 if (!file.exists()) {
233                     final String msg = String.format("Suppression file '%s' does not exist", file.getPath());
234                     LOGGER.warn(msg);
235                     throw new SuppressionParseException(msg);
236                 }
237                 try {
238                     list.addAll(parser.parseSuppressionRules(file));
239                 } catch (SuppressionParseException ex) {
240                     LOGGER.warn("Unable to parse suppression xml file '{}'", file.getPath());
241                     LOGGER.warn(ex.getMessage());
242                     throw ex;
243                 }
244             }
245         } catch (DownloadFailedException ex) {
246             throwSuppressionParseException("Unable to fetch the configured suppression file", ex, suppressionFilePath);
247         } catch (MalformedURLException ex) {
248             throwSuppressionParseException("Configured suppression file has an invalid URL", ex, suppressionFilePath);
249         } catch (SuppressionParseException ex) {
250             throw ex;
251         } catch (IOException ex) {
252             throwSuppressionParseException("Unable to create temp file for suppression file", ex, suppressionFilePath);
253         } finally {
254             if (deleteTempFile && file != null) {
255                 FileUtils.delete(file);
256             }
257         }
258         return list;
259     }
260 
261     /**
262      * Utility method to throw parse exceptions.
263      *
264      * @param message the exception message
265      * @param exception the cause of the exception
266      * @param suppressionFilePath the path file
267      * @throws SuppressionParseException throws the generated
268      * SuppressionParseException
269      */
270     private void throwSuppressionParseException(String message, Exception exception, String suppressionFilePath) throws SuppressionParseException {
271         LOGGER.warn(String.format(message + "'%s'", suppressionFilePath));
272         LOGGER.debug("", exception);
273         throw new SuppressionParseException(message, exception);
274     }
