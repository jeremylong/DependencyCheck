178     @Override
179     protected void analyzeDependency(Dependency dependency, Engine engine)
180             throws AnalysisException {
181 
182         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
183         final File actualFile = dependency.getActualFile();
184         if (WHL_FILTER.accept(actualFile)) {
185             collectMetadataFromArchiveFormat(dependency, DIST_INFO_FILTER,
186                     METADATA_FILTER);
187         } else if (EGG_OR_ZIP.accept(actualFile)) {
188             collectMetadataFromArchiveFormat(dependency, EGG_INFO_FILTER,
189                     PKG_INFO_FILTER);
190         } else {
191             final String name = actualFile.getName();
192             final boolean metadata = METADATA.equals(name);
193             if (metadata || PKG_INFO.equals(name)) {
194                 final File parent = actualFile.getParentFile();
195                 final String parentName = parent.getName();
196                 if (parent.isDirectory()
197                         && ((metadata && parentName.endsWith(".dist-info"))
198                         || parentName.endsWith(".egg-info") || "EGG-INFO"
199                         .equals(parentName))) {
200                     collectWheelMetadata(dependency, actualFile);
201                 }
202             }
203         }
204     }
205 
206     /**
207      * Collects the meta data from an archive.
208      *
209      * @param dependency the archive being scanned
210      * @param folderFilter the filter to apply to the folder
211      * @param metadataFilter the filter to apply to the meta data
212      * @throws AnalysisException thrown when there is a problem analyzing the
213      * dependency
214      */
215     private void collectMetadataFromArchiveFormat(Dependency dependency,
216             FilenameFilter folderFilter, FilenameFilter metadataFilter)
217             throws AnalysisException {
218         final File temp = getNextTempDirectory();
219         LOGGER.debug("{} exists? {}", temp, temp.exists());
220         try {
221             ExtractionUtil.extractFilesUsingFilter(
222                     new File(dependency.getActualFilePath()), temp,
223                     metadataFilter);
224         } catch (ExtractionException ex) {
225             throw new AnalysisException(ex);
226         }
227 
228         File matchingFile = getMatchingFile(temp, folderFilter);
229         if (matchingFile != null) {
230             matchingFile = getMatchingFile(matchingFile, metadataFilter);
231             if (matchingFile != null) {
232                 collectWheelMetadata(dependency, matchingFile);
233             }
234         }
235     }

237     /**
238      * Makes sure a usable temporary directory is available.
239      *
240      * @param engine a reference to the dependency-check engine
241      * @throws InitializationException an AnalyzeException is thrown when the
242      * temp directory cannot be created
243      */
244     @Override
245     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
246         try {
247             final File baseDir = getSettings().getTempDirectory();
248             tempFileLocation = File.createTempFile("check", "tmp", baseDir);
249             if (!tempFileLocation.delete()) {
250                 setEnabled(false);
251                 final String msg = String.format(
252                         "Unable to delete temporary file '%s'.",
253                         tempFileLocation.getAbsolutePath());
254                 throw new InitializationException(msg);
255             }
256             if (!tempFileLocation.mkdirs()) {
257                 setEnabled(false);
258                 final String msg = String.format(
259                         "Unable to create directory '%s'.",
260                         tempFileLocation.getAbsolutePath());
261                 throw new InitializationException(msg);
262             }
263         } catch (IOException ex) {
264             setEnabled(false);
265             throw new InitializationException("Unable to create a temporary file", ex);
266         }
267     }

269     /**
270      * Deletes any files extracted from the Wheel during analysis.
271      */
272     @Override
273     public void closeAnalyzer() {
274         if (tempFileLocation != null && tempFileLocation.exists()) {
275             LOGGER.debug("Attempting to delete temporary files");
276             final boolean success = FileUtils.delete(tempFileLocation);
277             if (!success && tempFileLocation.exists()) {
278                 final String[] l = tempFileLocation.list();
279                 if (l != null && l.length > 0) {
280                     LOGGER.warn("Failed to delete some temporary files, see the log for more details");
281                 }
282             }
283         }
284     }

286     /**
287      * Gathers evidence from the METADATA file.
288      *
289      * @param dependency the dependency being analyzed
290      * @param file a reference to the manifest/properties file
291      */
292     private static void collectWheelMetadata(Dependency dependency, File file) throws AnalysisException {
293         final Properties headers = PyPACoreMetadataParser.getProperties(file);
294         final String version = addPropertyToEvidence(dependency, EvidenceType.VERSION, Confidence.HIGHEST, headers, "Version");
295         final String name = addPropertyToEvidence(dependency, EvidenceType.VENDOR, Confidence.HIGHEST, headers, "Name");
296         addPropertyToEvidence(dependency, EvidenceType.PRODUCT, Confidence.HIGHEST, headers, "Name");
297 
298         final String packagePath = String.format("%s:%s", name, version);
299         dependency.setName(name);
300         dependency.setVersion(version);
301         dependency.setPackagePath(packagePath);
302         dependency.setDisplayFileName(packagePath);
303         final String url = headers.getProperty("Home-page", null);
304         if (StringUtils.isNotBlank(url)) {
305             if (UrlStringUtils.isUrl(url)) {
306                 dependency.addEvidence(EvidenceType.VENDOR, METADATA, "vendor", url, Confidence.MEDIUM);
307             }
308         }
309         addPropertyToEvidence(dependency, EvidenceType.VENDOR, Confidence.LOW, headers, "Author");
310         final String summary = headers.getProperty("Summary", null);
311         if (StringUtils.isNotBlank(summary)) {
312             JarAnalyzer.addDescription(dependency, summary, METADATA, "summary");
313         }
314 
315         try {
316             final PackageURL purl = PackageURLBuilder.aPackageURL().withType("pypi")
317                     .withName(name).withVersion(version).build();
318             dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
319         } catch (MalformedPackageURLException ex) {
320             LOGGER.debug("Unable to build package url for python", ex);
321             final GenericIdentifier id = new GenericIdentifier("generic:" + name + "@" + version, Confidence.HIGHEST);
322             dependency.addSoftwareIdentifier(id);
323         }
324     }

326     /**
327      * Adds a value to the evidence collection.
328      *
329      * @param dependency the dependency being analyzed
330      * @param type the type of evidence to add
331      * @param confidence the confidence in the evidence being added
332      * @param headers the properties collection
333      * @param property the property name
334      * @return returns the value of the property if found; otherwise
335      * <code>null</code>
336      */
337     private static String addPropertyToEvidence(Dependency dependency, EvidenceType type, Confidence confidence,
338             Properties headers, String property) {
339         final String value = headers.getProperty(property, null);
340         LOGGER.debug("Property: {}, Value: {}", property, value);
341         if (StringUtils.isNotBlank(value)) {
342             dependency.addEvidence(type, METADATA, property, value, confidence);
343         }
344         return value;
345     }

347     /**
348      * Returns a list of files that match the given filter, this does not
349      * recursively scan the directory.
350      *
351      * @param folder the folder to filter
352      * @param filter the filter to apply to the files in the directory
353      * @return the list of Files in the directory that match the provided filter
354      */
355     private static File getMatchingFile(File folder, FilenameFilter filter) {
356         File result = null;
357         final File[] matches = folder.listFiles(filter);
358         if (null != matches && 1 == matches.length) {
359             result = matches[0];
360         }
361         return result;
362     }

364     /**
365      * Reads the manifest entries from the provided file.
366      *
367      * @param manifest the manifest
368      * @return the manifest entries
369      */
370     private static Properties getManifestProperties(File manifest) {
371         final Properties prop = new Properties();
372         if (null == manifest) {
373             LOGGER.debug("Manifest file not found.");
374         } else {
375             try (InputStream in = new BufferedInputStream(new FileInputStream(manifest))) {
376                 prop.load(in);
377             } catch (IOException e) {
378                 LOGGER.warn(e.getMessage(), e);
379             }
380         }
381         return prop;
382     }

384     /**
385      * Retrieves the next temporary destination directory for extracting an
386      * archive.
387      *
388      * @return a directory
389      * @throws AnalysisException thrown if unable to create temporary directory
390      */
391     private File getNextTempDirectory() throws AnalysisException {
392         File directory;
393 
394         // getting an exception for some directories not being able to be
395         // created; might be because the directory already exists?
396         do {
397             final int dirCount = DIR_COUNT.incrementAndGet();
398             directory = new File(tempFileLocation, String.valueOf(dirCount));
399         } while (directory.exists());
400         if (!directory.mkdirs()) {
401             throw new AnalysisException(String.format(
402                     "Unable to create temp directory '%s'.",
403                     directory.getAbsolutePath()));
404         }
405         return directory;
406     }
