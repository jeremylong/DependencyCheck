48  /**
49   * Used to analyze Maven pinned dependency files named {@code *install*.json}, a
50   * Java Maven dependency lockfile like Python's {@code requirements.txt}.
51   *
52   * @author dhalperi
53   * @see
54   * <a href="https://github.com/bazelbuild/rules_jvm_external#pinning-artifacts-and-integration-with-bazels-downloader">rules_jvm_external</a>
55   */
56  @Experimental
57  @ThreadSafe
58  public class PinnedMavenInstallAnalyzer extends AbstractFileTypeAnalyzer {
59  
60      /**
61       * The logger.
62       */
63      private static final Logger LOGGER = LoggerFactory.getLogger(PinnedMavenInstallAnalyzer.class);
64  
65      /**
66       * The name of the analyzer.
67       */
68      private static final String ANALYZER_NAME = "Pinned Maven install Analyzer";
69  
70      /**
71       * The phase that this analyzer is intended to run in.
72       */
73      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
74  
75      /**
76       * Pattern matching files with "install" in the basename and extension
77       * "json".
78       *
79       * <p>
80       * This regex is designed to explicitly skip files named
81       * {@code install.json} since those are used for Cloudflare installations
82       * and this will save on work.
83       */
84      private static final Pattern MAVEN_INSTALL_JSON_PATTERN = Pattern.compile("(.+install.*|.*install.+)\\.json");
85  
86      /**
87       * Match any files that look like *install*.json.
88       */
89      private static final FileFilter FILTER = (File file) -> MAVEN_INSTALL_JSON_PATTERN.matcher(file.getName()).matches();
90  
91      @Override
92      protected FileFilter getFileFilter() {
93          return FILTER;
94      }
95  
96      @Override
97      public String getName() {
98          return ANALYZER_NAME;
99      }
100 
101     @Override
102     public AnalysisPhase getAnalysisPhase() {
103         return ANALYSIS_PHASE;
104     }
105 
106     @Override
107     protected String getAnalyzerEnabledSettingKey() {
108         return Settings.KEYS.ANALYZER_MAVEN_INSTALL_ENABLED;
109     }
110 
111     @Override
112     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
113         LOGGER.debug("Checking file {}", dependency.getActualFilePath());
114 
115         final File dependencyFile = dependency.getActualFile();
116         if (!dependencyFile.isFile() || dependencyFile.length() == 0) {
117             return;
118         }
119 
120         final DependencyTree tree;
121         try {
122             final InstallFile installFile = INSTALL_FILE_READER.readValue(dependencyFile);
123             tree = installFile.getDependencyTree();
124         } catch (IOException e) {
125             return;
126         }
127 
128         if (tree == null) {
129             return;
130         } else if (!Objects.equals(tree.getAutogeneratedSentinel(), "THERE_IS_NO_DATA_ONLY_ZUUL")) {
131             return;
132         }
133 
134         engine.removeDependency(dependency);
135 
136         if (!Objects.equals(tree.getVersion(), "0.1.0")) {
137             LOGGER.warn("Unsupported pinned maven_install.json version {}. Continuing optimistically.", tree.getVersion());
138         }
139 
140         List<MavenDependency> deps = tree.getDependencies();
141         if (deps == null) {
142             deps = Collections.emptyList();
143         }
144 
145         for (MavenDependency dep : deps) {
146             if (dep.getCoord() == null) {
147                 LOGGER.warn("Unexpected null coordinate in {}", dependency.getActualFilePath());
148                 continue;
149             }
150 
151             LOGGER.debug("Analyzing {}", dep.getCoord());
152             final String[] pieces = dep.getCoord().split(":");
153             if (pieces.length < 3 || pieces.length > 5) {
154                 LOGGER.warn("Invalid maven coordinate {}", dep.getCoord());
155                 continue;
156             }
157 
158             final String group = pieces[0];
159             final String artifact = pieces[1];
160             final String version;
161             String classifier = null;
162             switch (pieces.length) {
163                 case 3:
164                     version = pieces[2];
165                     break;
166                 case 4:
167                     classifier = pieces[2];
168                     version = pieces[3];
169                     break;
170                 default:
171                     // length == 5 as guaranteed above.
172                     classifier = pieces[3];
173                     version = pieces[4];
174                     break;
175             }
176 
177             if ("sources".equals(classifier) || "javadoc".equals(classifier)) {
178                 LOGGER.debug("Skipping sources jar {}", dep.getCoord());
179                 continue;
180             }
181 
182             final Dependency d = new Dependency(dependency.getActualFile(), true);
183             d.setEcosystem(Ecosystem.JAVA);
184             d.addEvidence(EvidenceType.VENDOR, "project", "groupid", group, Confidence.HIGHEST);
185             d.addEvidence(EvidenceType.PRODUCT, "project", "artifactid", artifact, Confidence.HIGHEST);
186             d.addEvidence(EvidenceType.VERSION, "project", "version", version, Confidence.HIGHEST);
187             d.setName(String.format("%s:%s", group, artifact));
188             d.setFilePath(String.format("%s>>%s", dependency.getActualFile(), dep.getCoord()));
189             d.setFileName(dep.getCoord());
190             try {
191                 final PackageURLBuilder purl = PackageURLBuilder.aPackageURL()
192                         .withType(PackageURL.StandardTypes.MAVEN)
193                         .withNamespace(group)
194                         .withName(artifact)
195                         .withVersion(version);
196                 if (classifier != null) {
197                     purl.withQualifier("classifier", classifier);
198                 }
199                 d.addSoftwareIdentifier(new PurlIdentifier(purl.build(), Confidence.HIGHEST));
200             } catch (MalformedPackageURLException e) {
201                 d.addSoftwareIdentifier(new GenericIdentifier("maven_install JSON coord " + dep.getCoord(), Confidence.HIGH));
202             }
203             d.setVersion(version);
204             engine.addDependency(d);
205         }
206     }

210         // No initialization needed.
211     }

213     /**
214      * Represents the entire pinned Maven dependency set in an install.json
215      * file.
216      *
217      * <p>
218      * At the time of writing, the latest version is 0.1.0, and the dependencies
219      * are stored in {@code .dependency_tree.dependencies[].coord}.
220      *
221      * <p>
222      * The only top-level key we care about is {@code .dependency_tree}.
223      */
224     private static class InstallFile {
225 
226         /**
227          * The dependency tree.
228          */
229         @JsonProperty("dependency_tree")
230         private DependencyTree dependencyTree;
231 
232         /**
233          * Returns dependencyTree.
234          *
235          * @return dependencyTree
236          */
237         public DependencyTree getDependencyTree() {
238             return dependencyTree;
239         }
240     }
241 
242     /**
243      * Represents the values at {@code .dependency_tree} in the
244      * {@link InstallFile install file}.
245      */
246     private static class DependencyTree {
247 
248         /**
249          * A sentinel value placed in the file to indicate that it is an
250          * auto-generated pinned maven install file.
251          */
252         @JsonProperty("__AUTOGENERATED_FILE_DO_NOT_MODIFY_THIS_FILE_MANUALLY")
253         private String autogeneratedSentinel;
254 
255         /**
256          * A list of Maven dependencies made available. Note that this list is
257          * transitively closed and pinned to a specific version of each
258          * artifact.
259          */
260         @JsonProperty("dependencies")
261         private List<MavenDependency> dependencies;
262 
263         /**
264          * The file format version.
265          */
266         @JsonProperty("version")
267         private String version;
268 
269         /**
270          * Returns autogeneratedSentinel.
271          *
272          * @return autogeneratedSentinel
273          */
274         public String getAutogeneratedSentinel() {
275             return autogeneratedSentinel;
276         }
277 
278         /**
279          * Returns dependencies.
280          *
281          * @return dependencies
282          */
283         public List<MavenDependency> getDependencies() {
284             return dependencies;
285         }
286 
287         /**
288          * Returns version.
289          *
290          * @return version
291          */
292         public String getVersion() {
293             return version;
294         }
295 
296     }
297 
298     /**
299      * Represents a single dependency in the list at
300      * {@code .dependency_tree.dependencies}.
301      */
302     private static class MavenDependency {
303 
304         /**
305          * The standard Maven coordinate string
306          * {@code group:artifact[:optional classifier][:optional packaging]:version}.
307          */
308         @JsonProperty("coord")
309         private String coord;
310 
311         /**
312          * Returns the value of coord.
313          *
314          * @return the value of coord
315          */
316         public String getCoord() {
317             return coord;
318         }
319     }
320 
321     /**
322      * A reusable reader for {@link InstallFile}.
323      */
324     private static final ObjectReader INSTALL_FILE_READER;
325 
326     static {
327         final ObjectMapper mapper = new ObjectMapper();
328         mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
329         INSTALL_FILE_READER = mapper.readerFor(InstallFile.class);
330     }
