191      *
192      * @param engine a reference to the dependency-check engine
193      * @throws InitializationException is thrown if there is an exception
194      * deleting or creating temporary files
195      */
196     @Override
197     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
198         try {
199             final File baseDir = getSettings().getTempDirectory();
200             tempFileLocation = File.createTempFile("check", "tmp", baseDir);
201             if (!tempFileLocation.delete()) {
202                 setEnabled(false);
203                 final String msg = String.format("Unable to delete temporary file '%s'.", tempFileLocation.getAbsolutePath());
204                 throw new InitializationException(msg);
205             }
206             if (!tempFileLocation.mkdirs()) {
207                 setEnabled(false);
208                 final String msg = String.format("Unable to create directory '%s'.", tempFileLocation.getAbsolutePath());
209                 throw new InitializationException(msg);
210             }
211         } catch (IOException ex) {
212             setEnabled(false);
213             throw new InitializationException("Unable to create a temporary file", ex);
214         }
215     }

219      * during analysis.
220      *
221      * @throws Exception thrown if there is an exception deleting temporary

224     @Override
225     public void closeAnalyzer() throws Exception {
226         if (tempFileLocation != null && tempFileLocation.exists()) {
227             LOGGER.debug("Attempting to delete temporary files from `{}`", tempFileLocation.toString());
228             final boolean success = FileUtils.delete(tempFileLocation);
229             if (!success && tempFileLocation.exists()) {
230                 final String[] l = tempFileLocation.list();
231                 if (l != null && l.length > 0) {
232                     LOGGER.warn("Failed to delete the Archive Analyzer's temporary files from `{}`, "
233                             + "see the log for more details", tempFileLocation.toString());
234                 }
235             }
236         }

239     /**
240      * Determines if the file can be analyzed by the analyzer. If the npm
241      * analyzer are enabled the archive analyzer will skip the node_modules and
242      * bower_modules directories.
243      *
244      * @param pathname the path to the file
245      * @return true if the file can be analyzed by the given analyzer; otherwise
246      * false
247      */
248     @Override
249     public boolean accept(File pathname) {
250         boolean accept = super.accept(pathname);
251         final boolean npmEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED, false);
252         final boolean yarnEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_YARN_AUDIT_ENABLED, false);
253         final boolean pnpmEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_PNPM_AUDIT_ENABLED, false);
254         if (accept && (npmEnabled || yarnEnabled || pnpmEnabled)) {
255             try {
256                 accept = shouldProcess(pathname);
257             } catch (AnalysisException ex) {
258                 throw new UnexpectedAnalysisException(ex.getMessage(), ex.getCause());
259             }
260         }
261         return accept;
262     }

279     /**
280      * Extracts the contents of the archive dependency and scans for additional
281      * dependencies.
282      *
283      * @param dependency the dependency being analyzed
284      * @param engine the engine doing the analysis
285      * @param scanDepth the current scan depth; extracctAndAnalyze is recursive
286      * and will, be default, only go 3 levels deep
287      * @throws AnalysisException thrown if there is a problem analyzing the
288      * dependencies
289      */
290     private void extractAndAnalyze(Dependency dependency, Engine engine, int scanDepth) throws AnalysisException {
291         final File f = new File(dependency.getActualFilePath());
292         final File tmpDir = getNextTempDirectory();
293         extractFiles(f, tmpDir, engine);
294 
295         //make a copy
296         final List<Dependency> dependencySet = findMoreDependencies(engine, tmpDir);
297 
298         if (dependencySet != null && !dependencySet.isEmpty()) {
299             for (Dependency d : dependencySet) {
300                 if (d.getFilePath().startsWith(tmpDir.getAbsolutePath())) {
301                     //fix the dependency's display name and path
302                     final String displayPath = String.format("%s%s",
303                             dependency.getFilePath(),
304                             d.getActualFilePath().substring(tmpDir.getAbsolutePath().length()));
305                     final String displayName = String.format("%s: %s",
306                             dependency.getFileName(),
307                             d.getFileName());
308                     d.setFilePath(displayPath);
309                     d.setFileName(displayName);
310                     d.addAllProjectReferences(dependency.getProjectReferences());
311 
312                     //TODO - can we get more evidence from the parent? EAR contains module name, etc.
313                     //analyze the dependency (i.e. extract files) if it is a supported type.
314                     if (this.accept(d.getActualFile()) && scanDepth < maxScanDepth) {
315                         extractAndAnalyze(d, engine, scanDepth + 1);
316                     }
317                 } else {
318                     dependencySet.stream().filter((sub) -> sub.getFilePath().startsWith(tmpDir.getAbsolutePath())).forEach((sub) -> {
319                         final String displayPath = String.format("%s%s",
320                                 dependency.getFilePath(),
321                                 sub.getActualFilePath().substring(tmpDir.getAbsolutePath().length()));
322                         final String displayName = String.format("%s: %s",
323                                 dependency.getFileName(),
324                                 sub.getFileName());
325                         sub.setFilePath(displayPath);
326                         sub.setFileName(displayName);
327                     });
328                 }
329             }
330         }
331         if (REMOVE_FROM_ANALYSIS.accept(dependency.getActualFile())) {
332             addDisguisedJarsToDependencies(dependency, engine);
333             engine.removeDependency(dependency);
334         }
335     }

337     /**
338      * If a zip file was identified as a possible JAR, this method will add the
339      * zip to the list of dependencies.
340      *
341      * @param dependency the zip file
342      * @param engine the engine
343      * @throws AnalysisException thrown if there is an issue
344      */
345     private void addDisguisedJarsToDependencies(Dependency dependency, Engine engine) throws AnalysisException {
346         if (ZIP_FILTER.accept(dependency.getActualFile()) && isZipFileActuallyJarFile(dependency)) {
347             final File tempDir = getNextTempDirectory();
348             final String fileName = dependency.getFileName();
349 
350             LOGGER.info("The zip file '{}' appears to be a JAR file, making a copy and analyzing it as a JAR.", fileName);
351             final File tmpLoc = new File(tempDir, fileName.substring(0, fileName.length() - 3) + "jar");
352             //store the archives sha1 and change it so that the engine doesn't think the zip and jar file are the same
353             // and add it is a related dependency.
354             final String archiveMd5 = dependency.getMd5sum();
355             final String archiveSha1 = dependency.getSha1sum();
356             final String archiveSha256 = dependency.getSha256sum();
357             try {
358                 dependency.setMd5sum("");
359                 dependency.setSha1sum("");
360                 dependency.setSha256sum("");
361                 org.apache.commons.io.FileUtils.copyFile(dependency.getActualFile(), tmpLoc);
362                 final List<Dependency> dependencySet = findMoreDependencies(engine, tmpLoc);
363                 if (dependencySet != null && !dependencySet.isEmpty()) {
364                     dependencySet.forEach((d) -> {
365                         //fix the dependency's display name and path
366                         if (d.getActualFile().equals(tmpLoc)) {
367                             d.setFilePath(dependency.getFilePath());
368                             d.setDisplayFileName(dependency.getFileName());
369                         } else {
370                             d.getRelatedDependencies().stream().filter((rel) -> rel.getActualFile().equals(tmpLoc)).forEach((rel) -> {
371                                 rel.setFilePath(dependency.getFilePath());
372                                 rel.setDisplayFileName(dependency.getFileName());
373                             });
374                         }
375                     });
376                 }
377             } catch (IOException ex) {
378                 LOGGER.debug("Unable to perform deep copy on '{}'", dependency.getActualFile().getPath(), ex);
379             } finally {
380                 dependency.setMd5sum(archiveMd5);
381                 dependency.setSha1sum(archiveSha1);
382                 dependency.setSha256sum(archiveSha256);
383             }
384         }
385     }

398     /**
399      * Retrieves the next temporary directory to extract an archive too.
400      *
401      * @return a directory
402      * @throws AnalysisException thrown if unable to create temporary directory
403      */
404     private File getNextTempDirectory() throws AnalysisException {
405         final File directory = new File(tempFileLocation, String.valueOf(DIRECTORY_COUNT.incrementAndGet()));
406         //getting an exception for some directories not being able to be created; might be because the directory already exists?
407         if (directory.exists()) {
408             return getNextTempDirectory();
409         }
410         if (!directory.mkdirs()) {
411             final String msg = String.format("Unable to create temp directory '%s'.", directory.getAbsolutePath());
412             throw new AnalysisException(msg);
413         }
414         return directory;
415     }

417     /**
418      * Extracts the contents of an archive into the specified directory.
419      *
420      * @param archive an archive file such as a WAR or EAR
421      * @param destination a directory to extract the contents to
422      * @param engine the scanning engine
423      * @throws AnalysisException thrown if the archive is not found
424      */
425     private void extractFiles(File archive, File destination, Engine engine) throws AnalysisException {
426         if (archive != null && destination != null) {
427             String archiveExt = FileUtils.getFileExtension(archive.getName());
428             if (archiveExt == null) {
429                 return;
430             }
431             archiveExt = archiveExt.toLowerCase();
432 
433             final FileInputStream fis;
434             try {
435                 fis = new FileInputStream(archive);
436             } catch (FileNotFoundException ex) {
437                 final String msg = String.format("Error extracting file `%s`: %s", archive.getAbsolutePath(), ex.getMessage());
438                 LOGGER.debug(msg, ex);
439                 throw new AnalysisException(msg);
440             }
441             BufferedInputStream in = null;
442             ZipArchiveInputStream zin = null;
443             TarArchiveInputStream tin = null;
444             GzipCompressorInputStream gin = null;
445             BZip2CompressorInputStream bzin = null;
446             RpmInputStream rin = null;
447             CpioArchiveInputStream cain = null;
448             try {
449                 if (KNOWN_ZIP_EXT.contains(archiveExt) || ADDITIONAL_ZIP_EXT.contains(archiveExt)) {
450                     in = new BufferedInputStream(fis);
451                     ensureReadableJar(archiveExt, in);
452                     zin = new ZipArchiveInputStream(in);
453                     extractArchive(zin, destination, engine);
454                 } else if ("tar".equals(archiveExt)) {
455                     in = new BufferedInputStream(fis);
456                     tin = new TarArchiveInputStream(in);
457                     extractArchive(tin, destination, engine);
458                 } else if ("gz".equals(archiveExt) || "tgz".equals(archiveExt)) {
459                     final String uncompressedName = GzipUtils.getUncompressedFilename(archive.getName());
460                     final File f = new File(destination, uncompressedName);
461                     if (engine.accept(f)) {
462                         final String destPath = destination.getCanonicalPath();
463                         if (!f.getCanonicalPath().startsWith(destPath)) {
464                             final String msg = String.format(
465                                     "Archive (%s) contains a file that would be written outside of the destination directory",
466                                     archive.getPath());
467                             throw new AnalysisException(msg);
468                         }
469                         in = new BufferedInputStream(fis);
470                         gin = new GzipCompressorInputStream(in);
471                         decompressFile(gin, f);
472                     }
473                 } else if ("bz2".equals(archiveExt) || "tbz2".equals(archiveExt)) {
474                     final String uncompressedName = BZip2Utils.getUncompressedFilename(archive.getName());
475                     final File f = new File(destination, uncompressedName);
476                     if (engine.accept(f)) {
477                         final String destPath = destination.getCanonicalPath();
478                         if (!f.getCanonicalPath().startsWith(destPath)) {
479                             final String msg = String.format(
480                                     "Archive (%s) contains a file that would be written outside of the destination directory",
481                                     archive.getPath());
482                             throw new AnalysisException(msg);
483                         }
484                         in = new BufferedInputStream(fis);
485                         bzin = new BZip2CompressorInputStream(in);
486                         decompressFile(bzin, f);
487                     }
488                 } else if ("rpm".equals(archiveExt)) {
489                     rin = new RpmInputStream(fis);
490                     //return of getTag is not used - but the call is a
491                     //necassary step in reading from the stream
492                     rin.getPayloadHeader().getTag(RpmTag.NAME);
493                     cain = new CpioArchiveInputStream(rin);
494                     extractArchive(cain, destination, engine);
495                 }
496             } catch (ArchiveExtractionException ex) {
497                 LOGGER.warn("Exception extracting archive '{}'.", archive.getName());
498                 LOGGER.debug("", ex);
499             } catch (IOException ex) {
500                 LOGGER.warn("Exception reading archive '{}'.", archive.getName());
501                 LOGGER.debug("", ex);
502             } finally {
503                 //overly verbose and not needed... but keeping it anyway due to
504                 //having issue with file handles being left open
505                 FileUtils.close(fis);
506                 FileUtils.close(in);
507                 FileUtils.close(zin);
508                 FileUtils.close(tin);
509                 FileUtils.close(gin);
510                 FileUtils.close(bzin);
511             }
512         }
513     }

515     /**
516      * Checks if the file being scanned is a JAR or WAR that begins with
517      * '#!/bin' which indicates it is a fully executable jar. If a fully
518      * executable JAR is identified the input stream will be advanced to the
519      * start of the actual JAR file ( skipping the script).
520      *
521      * @see
522      * <a href="http://docs.spring.io/spring-boot/docs/1.3.0.BUILD-SNAPSHOT/reference/htmlsingle/#deployment-install">Installing
523      * Spring Boot Applications</a>
524      * @param archiveExt the file extension
525      * @param in the input stream
526      * @throws IOException thrown if there is an error reading the stream
527      */
528     private void ensureReadableJar(final String archiveExt, BufferedInputStream in) throws IOException {
529         if (("war".equals(archiveExt) || "jar".equals(archiveExt)) && in.markSupported()) {
530             in.mark(7);
531             final byte[] b = new byte[7];
532             final int read = in.read(b);
533             if (read == 7
534                     && b[0] == '#'
535                     && b[1] == '!'
536                     && b[2] == '/'
537                     && b[3] == 'b'
538                     && b[4] == 'i'
539                     && b[5] == 'n'
540                     && b[6] == '/') {
541                 boolean stillLooking = true;
542                 int chr;
543                 int nxtChr;
544                 //CSOFF: InnerAssignment
545                 //CSOFF: NestedIfDepth
546                 while (stillLooking && (chr = in.read()) != -1) {
547                     if (chr == '\n' || chr == '\r') {
548                         in.mark(4);
549                         if ((chr = in.read()) != -1) {
550                             if (chr == 'P' && (chr = in.read()) != -1) {
551                                 if (chr == 'K' && (chr = in.read()) != -1) {
552                                     if ((chr == 3 || chr == 5 || chr == 7) && (nxtChr = in.read()) != -1) {
553                                         if (nxtChr == chr + 1) {
554                                             stillLooking = false;
555                                             in.reset();
556                                         }
557                                     }
558                                 }
559                             }
560                         }
561                     }
562                 }
563                 //CSON: InnerAssignment
564                 //CSON: NestedIfDepth
565             } else {
566                 in.reset();
567             }
568         }
569     }

571     /**
572      * Extracts files from an archive.
573      *
574      * @param input the archive to extract files from
575      * @param destination the location to write the files too
576      * @param engine the dependency-check engine
577      * @throws ArchiveExtractionException thrown if there is an exception
578      * extracting files from the archive
579      */
580     private void extractArchive(ArchiveInputStream input, File destination, Engine engine) throws ArchiveExtractionException {
581         ArchiveEntry entry;
582         try {
583             //final String destPath = destination.getCanonicalPath();
584             final Path d = destination.toPath();
585             while ((entry = input.getNextEntry()) != null) {
586                 //final File file = new File(destination, entry.getName());
587                 final Path f = d.resolve(entry.getName()).normalize();
588                 if (!f.startsWith(d)) {
589                     LOGGER.debug("ZipSlip detected\n-Destination: " + d + "\n-Path: " + f);
590                     final String msg = String.format(
591                             "Archive contains a file (%s) that would be extracted outside of the target directory.",
592                             entry.getName());
593                     throw new ArchiveExtractionException(msg);
594                 }
595                 final File file = f.toFile();
596                 if (entry.isDirectory()) {
597                     if (!file.exists() && !file.mkdirs()) {
598                         final String msg = String.format("Unable to create directory '%s'.", file.getAbsolutePath());
599                         throw new AnalysisException(msg);
600                     }
601                 } else if (engine.accept(file)) {
602                     extractAcceptedFile(input, file);
603                 }
604             }
605         } catch (IOException | AnalysisException ex) {
606             throw new ArchiveExtractionException(ex);
607         } finally {
608             FileUtils.close(input);
609         }
610     }

612     /**
613      * Extracts a file from an archive.
614      *
615      * @param input the archives input stream
616      * @param file the file to extract
617      * @throws AnalysisException thrown if there is an error
618      */
619     private static void extractAcceptedFile(ArchiveInputStream input, File file) throws AnalysisException {
620         LOGGER.debug("Extracting '{}'", file.getPath());
621         final File parent = file.getParentFile();
622         if (!parent.isDirectory() && !parent.mkdirs()) {
623             final String msg = String.format("Unable to build directory '%s'.", parent.getAbsolutePath());
624             throw new AnalysisException(msg);
625         }
626         try (FileOutputStream fos = new FileOutputStream(file)) {
627             IOUtils.copy(input, fos);
628         } catch (FileNotFoundException ex) {
629             LOGGER.debug("", ex);
630             final String msg = String.format("Unable to find file '%s'.", file.getName());
631             throw new AnalysisException(msg, ex);
632         } catch (IOException ex) {
633             LOGGER.debug("", ex);
634             final String msg = String.format("IO Exception while parsing file '%s'.", file.getName());
635             throw new AnalysisException(msg, ex);
636         }
637     }

641      *
642      * @param inputStream the compressed file
643      * @param outputFile the location to write the decompressed file
644      * @throws ArchiveExtractionException thrown if there is an exception
645      * decompressing the file
646      */
647     private void decompressFile(CompressorInputStream inputStream, File outputFile) throws ArchiveExtractionException {
648         LOGGER.debug("Decompressing '{}'", outputFile.getPath());
649         try (FileOutputStream out = new FileOutputStream(outputFile)) {
650             IOUtils.copy(inputStream, out);
651         } catch (IOException ex) {
652             LOGGER.debug("", ex);
653             throw new ArchiveExtractionException(ex);
654         }
655     }

657     /**
658      * Attempts to determine if a zip file is actually a JAR file.
659      *
660      * @param dependency the dependency to check
661      * @return true if the dependency appears to be a JAR file; otherwise false
662      */
663     private boolean isZipFileActuallyJarFile(Dependency dependency) {
664         boolean isJar = false;
665         ZipFile zip = null;
666         try {
667             zip = new ZipFile(dependency.getActualFilePath());
668             if (zip.getEntry("META-INF/MANIFEST.MF") != null
669                     || zip.getEntry("META-INF/maven") != null) {
670                 final Enumeration<ZipArchiveEntry> entries = zip.getEntries();
671                 while (entries.hasMoreElements()) {
672                     final ZipArchiveEntry entry = entries.nextElement();
673                     if (!entry.isDirectory()) {
674                         final String name = entry.getName().toLowerCase();
675                         if (name.endsWith(".class")) {
676                             isJar = true;
677                             break;
678                         }
679                     }
680                 }
681             }
682         } catch (IOException ex) {
683             LOGGER.debug("Unable to unzip zip file '{}'", dependency.getFilePath(), ex);
684         } finally {
685             ZipFile.closeQuietly(zip);
686         }
687         return isJar;
688     }

690     /**
691      * Initializes settings used by the scanning functions of the archive
692      * analyzer.
693      */
694     private void initializeSettings() {
695         maxScanDepth = getSettings().getInt("archive.scan.depth", 3);
696         final Set<String> extensions = new HashSet<>(EXTENSIONS);
697         extensions.addAll(KNOWN_ZIP_EXT);
698         final String additionalZipExt = getSettings().getString(Settings.KEYS.ADDITIONAL_ZIP_EXTENSIONS);
699         if (additionalZipExt != null) {
700             final String[] ext = additionalZipExt.split("\\s*,\\s*");
701             Collections.addAll(extensions, ext);
702             Collections.addAll(ADDITIONAL_ZIP_EXT, ext);
703         }
704         fileFilter = FileFilterBuilder.newInstance().addExtensions(extensions).build();
705     }
