142     /**
143      * Collects information about the file name.
144      *
145      * @param dependency the dependency to analyze.
146      * @param engine the engine that is scanning the dependencies
147      * @throws AnalysisException is thrown if there is an error analyzing the PE
148      * file.
149      */
150     @Override
151     protected void analyzeDependency(final Dependency dependency, final Engine engine) throws AnalysisException {
152         for (Evidence e : dependency.getEvidence()) {
153             if ("grokassembly".equals(e.getSource())) {
154                 LOGGER.debug("Skipping {} because it was already analyzed by the Assembly Analyzer", dependency.getFileName());
155                 return;
156             }
157         }
158         try {
159             final File fileToCheck = dependency.getActualFile();
160             final PE pe = PEParser.parse(fileToCheck.getPath());
161             final ResourceDirectory rd = pe.getImageData().getResourceTable();
162             final ResourceEntry[] entries = ResourceHelper.findResources(rd, ResourceType.VERSION_INFO);
163             for (ResourceEntry entrie : entries) {
164                 final byte[] data = entrie.getData();
165                 final VersionInfo version = ResourceParser.readVersionInfo(data);
166                 final StringFileInfo strings = version.getStringFileInfo();
167                 final StringTable table = strings.getTable(0);
168                 String pVersion = null;
169                 String fVersion = null;
170 
171                 for (int j = 0; j < table.getCount(); j++) {
172                     final String key = table.getString(j).getKey();
173                     final String value = table.getString(j).getValue();
174                     switch (key) {
175                         case "ProductVersion":
176                             dependency.addEvidence(EvidenceType.VERSION, "PE Header", "ProductVersion", value, Confidence.HIGHEST);
177                             pVersion = value;
178                             break;
179                         case "CompanyName":
180                             dependency.addEvidence(EvidenceType.VENDOR, "PE Header", "CompanyName", value, Confidence.HIGHEST);
181                             break;
182                         case "FileVersion":
183                             dependency.addEvidence(EvidenceType.VERSION, "PE Header", "FileVersion", value, Confidence.HIGH);
184                             fVersion = value;
185                             break;
186                         case "InternalName":
187                             dependency.addEvidence(EvidenceType.PRODUCT, "PE Header", "InternalName", value, Confidence.MEDIUM);
188                             determineDependencyName(dependency, value);
189                             break;
190                         case "LegalCopyright":
191                             dependency.addEvidence(EvidenceType.VENDOR, "PE Header", "LegalCopyright", value, Confidence.HIGHEST);
192                             if (dependency.getLicense() != null && dependency.getLicense().length() > 0) {
193                                 dependency.setLicense(dependency.getLicense() + "/n/nLegal Copyright: " + value);
194                             } else {
195                                 dependency.setLicense("Legal Copyright: " + value);
196                             }
197                             break;
198                         case "OriginalFilename":
199                             dependency.addEvidence(EvidenceType.VERSION, "PE Header", "OriginalFilename", value, Confidence.MEDIUM);
200                             determineDependencyName(dependency, value);
201                             break;
202                         case "ProductName":
203                             dependency.addEvidence(EvidenceType.PRODUCT, "PE Header", "ProductName", value, Confidence.HIGHEST);
204                             determineDependencyName(dependency, value);
205                             break;
206                         default:
207                             LOGGER.debug("PE Analyzer found `" + key + "` with a value:" + value);
208                     }
209                     if (fVersion != null && pVersion != null) {
210                         final int max = fVersion.length() > pVersion.length() ? pVersion.length() : fVersion.length();
211                         int pos;
212                         for (pos = 0; pos < max; pos++) {
213                             if (fVersion.charAt(pos) != pVersion.charAt(pos)) {
214                                 break;
215                             }
216                         }
217                         final DependencyVersion fileVersion = DependencyVersionUtil.parseVersion(fVersion, true);
218                         final DependencyVersion productVersion = DependencyVersionUtil.parseVersion(pVersion, true);
219                         if (pos > 0) {
220                             final DependencyVersion matchingVersion = DependencyVersionUtil.parseVersion(fVersion.substring(0, pos), true);
221                             if (fileVersion != null && fileVersion.toString().length() == fVersion.length()) {
222                                 if (matchingVersion != null && matchingVersion.getVersionParts().size() > 2) {
223                                     dependency.addEvidence(EvidenceType.VERSION, "PE Header", "FilteredVersion",
224                                             matchingVersion.toString(), Confidence.HIGHEST);
225                                     dependency.setVersion(matchingVersion.toString());
226                                 }
227                             }
228                         }
229                         if (dependency.getVersion() == null) {
230                             if (fVersion.length() >= pVersion.length()) {
231                                 if (fileVersion != null && fileVersion.toString().length() == fVersion.length()) {
232                                     dependency.setVersion(fileVersion.toString());
233                                 } else if (productVersion != null && productVersion.toString().length() == pVersion.length()) {
234                                     dependency.setVersion(productVersion.toString());
235                                 }
236                             } else {
237                                 if (productVersion != null && productVersion.toString().length() == pVersion.length()) {
238                                     dependency.setVersion(productVersion.toString());
239                                 } else if (fileVersion != null && fileVersion.toString().length() == fVersion.length()) {
240                                     dependency.setVersion(fileVersion.toString());
241                                 }
242                             }
243                         }
244                     } else if (pVersion != null) {
245                         final DependencyVersion productVersion = DependencyVersionUtil.parseVersion(pVersion, true);
246                         if (productVersion != null && dependency.getActualFile().getName().contains(productVersion.toString())) {
247                             dependency.setVersion(productVersion.toString());
248                         }
249                     } else if (fVersion != null) {
250                         final DependencyVersion fileVersion = DependencyVersionUtil.parseVersion(fVersion, true);
251                         if (fileVersion != null && dependency.getActualFile().getName().contains(fileVersion.toString())) {
252                             dependency.setVersion(fileVersion.toString());
253                         }
254                     }
255                     if (dependency.getName() != null && dependency.getVersion() != null) {
256                         try {
257                             dependency.addSoftwareIdentifier(new PurlIdentifier("generic", dependency.getName(),
258                                     dependency.getVersion(), Confidence.MEDIUM));
259                         } catch (MalformedPackageURLException ex) {
260                             LOGGER.debug("Unable to create Package URL Identifier for " + dependency.getName(), ex);
261                             dependency.addSoftwareIdentifier(new GenericIdentifier(
262                                     String.format("%s@%s", dependency.getName(), dependency.getVersion()),
263                                     Confidence.MEDIUM));
264                         }
265                     }
266                     if (dependency.getEcosystem() == null) {
267                         //this could be an assembly
268                         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
269                     }
270                 }
271             }
272         } catch (IOException ex) {
273             throw new AnalysisException(ex);
274         }
275     }
276 
277     private void determineDependencyName(final Dependency dependency, final String value) {
278         if (dependency.getName() == null && StringUtils.containsIgnoreCase(dependency.getActualFile().getName(), value)) {
279             final String ext = FileUtils.getFileExtension(value);
280             if (ext != null) {
281                 dependency.setName(value.substring(0, value.length() - ext.length() - 1));
282             } else {
283                 dependency.setName(value);
284             }
285         }
