44  /**
45   * Used to analyze a composer.lock file for a composer PHP app.
46   *
47   * @author colezlaw
48   */
49  @Experimental
50  public class ComposerLockAnalyzer extends AbstractFileTypeAnalyzer {
51  
52      /**
53       * A descriptor for the type of dependencies processed or added by this
54       * analyzer.
55       */
56      public static final String DEPENDENCY_ECOSYSTEM = Ecosystem.PYTHON;
57  
58      /**
59       * The logger.
60       */
61      private static final Logger LOGGER = LoggerFactory.getLogger(ComposerLockAnalyzer.class);
62  
63      /**
64       * The analyzer name.
65       */
66      private static final String ANALYZER_NAME = "Composer.lock analyzer";
67  
68      /**
69       * composer.json.
70       */
71      private static final String COMPOSER_LOCK = "composer.lock";
72  
73      /**
74       * The FileFilter.
75       */
76      private static final FileFilter FILE_FILTER = FileFilterBuilder.newInstance().addFilenames(COMPOSER_LOCK).build();
77  
78      /**
79       * Returns the FileFilter.
80       *
81       * @return the FileFilter
82       */
83      @Override
84      protected FileFilter getFileFilter() {
85          return FILE_FILTER;
86      }
87  
88      /**
89       * Initializes the analyzer.
90       *
91       * @param engine a reference to the dependency-check engine
92       * @throws InitializationException thrown if an exception occurs getting an
93       * instance of SHA1
94       */
95      @Override
96      protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
97          // do nothing
98      }
99  
100     /**
101      * Entry point for the analyzer.
102      *
103      * @param dependency the dependency to analyze
104      * @param engine the engine scanning
105      * @throws AnalysisException if there's a failure during analysis
106      */
107     @Override
108     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
109         engine.removeDependency(dependency);
110         try (FileInputStream fis = new FileInputStream(dependency.getActualFile())) {
111             final ComposerLockParser clp = new ComposerLockParser(fis);
112             LOGGER.debug("Checking composer.lock file {}", dependency.getActualFilePath());
113             clp.process();
114             clp.getDependencies().stream().map((dep) -> {
115                 final Dependency d = new Dependency(dependency.getActualFile(), true);
116                 final String filePath = String.format("%s:%s/%s/%s", dependency.getFilePath(), dep.getGroup(), dep.getProject(), dep.getVersion());
117                 d.setName(dep.getProject());
118                 d.setVersion(dep.getVersion());
119                 try {
120                     final PackageURL purl = PackageURLBuilder.aPackageURL().withType("composer").withNamespace(dep.getGroup())
121                             .withName(dep.getProject()).withVersion(dep.getVersion()).build();
122                     d.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
123                 } catch (MalformedPackageURLException ex) {
124                     LOGGER.debug("Unable to build package url for composer", ex);
125                     d.addSoftwareIdentifier(new GenericIdentifier("cocoapods:" + dep.getGroup() + "/" + dep.getProject()
126                             + "@" + dep.getVersion(), Confidence.HIGHEST));
127                 }
128                 d.setPackagePath(String.format("%s:%s", dep.getProject(), dep.getVersion()));
129                 d.setEcosystem(DEPENDENCY_ECOSYSTEM);
130                 d.setFilePath(filePath);
131                 d.setSha1sum(Checksum.getSHA1Checksum(filePath));
132                 d.setSha256sum(Checksum.getSHA256Checksum(filePath));
133                 d.setMd5sum(Checksum.getMD5Checksum(filePath));
134                 d.addEvidence(EvidenceType.VENDOR, COMPOSER_LOCK, "vendor", dep.getGroup(), Confidence.HIGHEST);
135                 d.addEvidence(EvidenceType.PRODUCT, COMPOSER_LOCK, "product", dep.getProject(), Confidence.HIGHEST);
136                 d.addEvidence(EvidenceType.VERSION, COMPOSER_LOCK, "version", dep.getVersion(), Confidence.HIGHEST);
137                 return d;
138             }).forEach((d) -> {
139                 LOGGER.debug("Adding dependency {}", d.getDisplayFileName());
140                 engine.addDependency(d);
141             });
142         } catch (IOException ex) {
143             LOGGER.warn("Error opening dependency {}", dependency.getActualFilePath());
144         } catch (ComposerException ce) {
145             LOGGER.warn("Error parsing composer.json {}", dependency.getActualFilePath(), ce);
146         }
147     }
148 
149     /**
150      * Gets the key to determine whether the analyzer is enabled.
151      *
152      * @return the key specifying whether the analyzer is enabled
153      */
154     @Override
155     protected String getAnalyzerEnabledSettingKey() {
156         return Settings.KEYS.ANALYZER_COMPOSER_LOCK_ENABLED;
157     }
158 
159     /**
160      * Returns the analyzer's name.
161      *
162      * @return the analyzer's name
163      */
164     @Override
165     public String getName() {
166         return ANALYZER_NAME;
167     }
168 
169     /**
170      * Returns the phase this analyzer should run under.
171      *
172      * @return the analysis phase
173      */
174     @Override
175     public AnalysisPhase getAnalysisPhase() {
176         return AnalysisPhase.INFORMATION_COLLECTION;
177     }
178 }
