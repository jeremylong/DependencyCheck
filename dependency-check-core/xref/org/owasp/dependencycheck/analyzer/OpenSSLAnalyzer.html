45  /**
46   * Used to analyze OpenSSL source code present in the file system.
47   *
48   * @author Dale Visser
49   */
50  @ThreadSafe
51  public class OpenSSLAnalyzer extends AbstractFileTypeAnalyzer {
52  
53      /**
54       * The logger.
55       */
56      private static final Logger LOGGER = LoggerFactory.getLogger(OpenSSLAnalyzer.class);
57      /**
58       * Hexadecimal.
59       */
60      private static final int HEXADECIMAL = 16;
61      /**
62       * Filename to analyze. All other .h files get removed from consideration.
63       */
64      private static final String OPENSSLV_H = "opensslv.h";
65  
66      /**
67       * Filter that detects files named "__init__.py".
68       */
69      private static final FileFilter OPENSSLV_FILTER = FileFilterBuilder.newInstance().addFilenames(OPENSSLV_H).build();
70      /**
71       * Open SSL Version number pattern.
72       */
73      private static final Pattern VERSION_PATTERN = Pattern.compile(
74              "define\\s+OPENSSL_VERSION_NUMBER\\s+0x([0-9a-zA-Z]{8})L", Pattern.DOTALL
75              | Pattern.CASE_INSENSITIVE);
76      /**
77       * The offset of the major version number.
78       */
79      private static final int MAJOR_OFFSET = 28;
80      /**
81       * The mask for the minor version number.
82       */
83      private static final long MINOR_MASK = 0x0ff00000L;
84      /**
85       * The offset of the minor version number.
86       */
87      private static final int MINOR_OFFSET = 20;
88      /**
89       * The max for the fix version.
90       */
91      private static final long FIX_MASK = 0x000ff000L;
92      /**
93       * The offset for the fix version.
94       */
95      private static final int FIX_OFFSET = 12;
96      /**
97       * The mask for the patch version.
98       */
99      private static final long PATCH_MASK = 0x00000ff0L;
100     /**
101      * The offset for the patch version.
102      */
103     private static final int PATCH_OFFSET = 4;
104     /**
105      * Number of letters.
106      */
107     private static final int NUM_LETTERS = 26;
108     /**
109      * The status mask.
110      */
111     private static final int STATUS_MASK = 0x0000000f;
112 
113     /**
114      * Returns the open SSL version as a string.
115      *
116      * @param openSSLVersionConstant The open SSL version
117      * @return the version of openssl
118      */
119     protected static String getOpenSSLVersion(long openSSLVersionConstant) {
120         final long major = openSSLVersionConstant >>> MAJOR_OFFSET;
121         final long minor = (openSSLVersionConstant & MINOR_MASK) >>> MINOR_OFFSET;
122         final long fix = (openSSLVersionConstant & FIX_MASK) >>> FIX_OFFSET;
123         final long patchLevel = (openSSLVersionConstant & PATCH_MASK) >>> PATCH_OFFSET;
124         final String patch = 0 == patchLevel || patchLevel > NUM_LETTERS ? "" : String.valueOf((char) (patchLevel + 'a' - 1));
125         final int statusCode = (int) (openSSLVersionConstant & STATUS_MASK);
126         final String status = 0xf == statusCode ? "" : (0 == statusCode ? "-dev" : "-beta" + statusCode);
127         return String.format("%d.%d.%d%s%s", major, minor, fix, patch, status);
128     }

181     /**
182      * Analyzes python packages and adds evidence to the dependency.
183      *
184      * @param dependency the dependency being analyzed
185      * @param engine the engine being used to perform the scan
186      * @throws AnalysisException thrown if there is an unrecoverable error
187      * analyzing the dependency
188      */
189     @Override
190     protected void analyzeDependency(Dependency dependency, Engine engine)
191             throws AnalysisException {
192         final File file = dependency.getActualFile();
193         final String parentName = file.getParentFile().getName();
194         boolean found = false;
195         final String contents = getFileContents(file);
196         if (!contents.isEmpty()) {
197             final Matcher matcher = VERSION_PATTERN.matcher(contents);
198             if (matcher.find()) {
199                 found = true;
200                 final String version = getOpenSSLVersion(Long.parseLong(matcher.group(1), HEXADECIMAL));
201                 dependency.addEvidence(EvidenceType.VERSION, OPENSSLV_H, "Version Constant",
202                         version, Confidence.HIGH);
203                 try {
204                     final PackageURL purl = PackageURLBuilder.aPackageURL().withType("generic")
205                             .withName("openssl").withVersion(version).build();
206                     dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
207                 } catch (MalformedPackageURLException ex) {
208                     LOGGER.debug("Unable to build package url for openssl", ex);
209                     final GenericIdentifier id = new GenericIdentifier("generic:openssl@" + version, Confidence.HIGHEST);
210                     dependency.addSoftwareIdentifier(id);
211                 }
212 
213             }
214         }
215         if (found) {
216             dependency.setDisplayFileName(parentName + File.separatorChar + OPENSSLV_H);
217             dependency.addEvidence(EvidenceType.VENDOR, OPENSSLV_H, "Vendor", "OpenSSL", Confidence.HIGHEST);
218             dependency.addEvidence(EvidenceType.PRODUCT, OPENSSLV_H, "Product", "OpenSSL", Confidence.HIGHEST);
219         } else {
220             engine.removeDependency(dependency);
221         }
222     }
223 
224     /**
225      * Retrieves the contents of a given file.
226      *
227      * @param actualFile the file to read
228      * @return the contents of the file
229      * @throws AnalysisException thrown if there is an IO Exception
230      */
231     private String getFileContents(final File actualFile)
232             throws AnalysisException {
233         try {
234             return FileUtils.readFileToString(actualFile, Charset.defaultCharset()).trim();
235         } catch (IOException e) {
236             throw new AnalysisException(
237                     "Problem occurred while reading dependency file.", e);
238         }
239     }
