34      /**
35       * Test of equals method, of class Evidence.
36       */
37      @Test
38      public void testEquals() {
39          Evidence that0 = new Evidence("file", "name", "guice-3.0", Confidence.HIGHEST);
40          Evidence that1 = new Evidence("jar", "package name", "dependency", Confidence.HIGHEST);
41          Evidence that2 = new Evidence("jar", "package name", "google", Confidence.HIGHEST);
42          Evidence that3 = new Evidence("jar", "package name", "guice", Confidence.HIGHEST);
43          Evidence that4 = new Evidence("jar", "package name", "inject", Confidence.HIGHEST);
44          Evidence that5 = new Evidence("jar", "package name", "inject", Confidence.LOW);
45          Evidence that6 = new Evidence("jar", "package name", "internal", Confidence.LOW);
46          Evidence that7 = new Evidence("manifest", "Bundle-Description", "Guice is a lightweight dependency injection framework for Java 5 and above", Confidence.MEDIUM);
47          Evidence that8 = new Evidence("Manifest", "Implementation-Title", "Spring Framework", Confidence.HIGH);
48  
49          Evidence instance = new Evidence("Manifest", "Implementation-Title", "Spring Framework", Confidence.HIGH);
50          assertFalse(instance.equals(that0));
51          assertFalse(instance.equals(that1));
52          assertFalse(instance.equals(that2));
53          assertFalse(instance.equals(that3));
54          assertFalse(instance.equals(that4));
55          assertFalse(instance.equals(that5));
56          assertFalse(instance.equals(that6));
57          assertFalse(instance.equals(that7));
58          assertTrue(instance.equals(that8));
59      }
60  
61      @Test
62      public void testHashcodeContract() throws Exception {
63          final Evidence titleCase = new Evidence("Manifest", "Implementation-Title", "Spring Framework", Confidence.HIGH);
64          final Evidence lowerCase = new Evidence("manifest", "implementation-title", "spring framework", Confidence.HIGH);
65          assertThat(titleCase, is(equalTo(lowerCase)));
66          assertThat(titleCase.hashCode(), is(equalTo(lowerCase.hashCode())));
67      }

71       */
72      @Test
73      public void testCompareTo() {
74          Evidence that0 = new Evidence("file", "name", "guice-3.0", Confidence.HIGHEST);
75          Evidence that1 = new Evidence("jar", "package name", "dependency", Confidence.HIGHEST);
76          Evidence that2 = new Evidence("jar", "package name", "google", Confidence.HIGHEST);
77          Evidence that3 = new Evidence("jar", "package name", "guice", Confidence.HIGHEST);
78          Evidence that4 = new Evidence("jar", "package name", "inject", Confidence.HIGHEST);
79          Evidence that5 = new Evidence("jar", "package name", "inject", Confidence.LOW);
80          Evidence that6 = new Evidence("jar", "package name", "internal", Confidence.LOW);
81          Evidence that7 = new Evidence("manifest", "Bundle-Description", "Guice is a lightweight dependency injection framework for Java 5 and above", Confidence.MEDIUM);
82          Evidence that8 = new Evidence("Manifest", "Implementation-Title", "Spring Framework", Confidence.HIGH);
83  
84          Evidence that9 = new Evidence("manifest", "implementation-title", "zippy", Confidence.HIGH);
85  
86          Evidence instance = new Evidence("Manifest", "Implementation-Title", "Spring Framework", Confidence.HIGH);
87  
88          int result = instance.compareTo(that0);

91          result = instance.compareTo(that1);

94          result = instance.compareTo(that2);
95          assertTrue(result > 0);

97          result = instance.compareTo(that3);
98          assertTrue(result > 0);
99  
100         result = instance.compareTo(that4);
101         assertTrue(result > 0);

107         assertTrue(result > 0);
