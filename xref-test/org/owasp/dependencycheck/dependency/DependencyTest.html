213     /**
214      * Test of addVulnerableSoftwareIdentifier method, of class Dependency.
215      */
216     @Test
217     public void testAddVulnerableSoftwareIdentifier() throws Exception {
218         CpeBuilder builder = new CpeBuilder();
219         Cpe cpe = builder.part(Part.APPLICATION).vendor("apache").product("struts").version("2.1.2").build();
220         CpeIdentifier id = new CpeIdentifier(cpe, Confidence.HIGHEST);
221 
222         cpe = builder.part(Part.APPLICATION).vendor("apache").product("struts").version("2.1.2").build();
223         CpeIdentifier expResult = new CpeIdentifier(cpe, Confidence.HIGHEST);
224 
225         Dependency instance = new Dependency();
226         instance.addVulnerableSoftwareIdentifier(id);
227         assertEquals(1, instance.getVulnerableSoftwareIdentifiers().size());
228         assertTrue("Identifier doesn't contain expected result.", instance.getVulnerableSoftwareIdentifiers().contains(expResult));
229     }

262     public void testAddAsEvidenceWithEmptyArtifact() {
263         Dependency instance = new Dependency();
264         MavenArtifact mavenArtifact = new MavenArtifact(null, null, null, null);
265         instance.addAsEvidence("pom", mavenArtifact, Confidence.HIGH);
266         assertFalse(instance.getEvidence(EvidenceType.VENDOR).stream().anyMatch(e -> e.getConfidence() == Confidence.HIGH));
267         assertTrue(instance.size() == 0);
268         assertTrue(instance.getSoftwareIdentifiers().isEmpty());
269     }

271     /**
272      * Test of addAsEvidence method, of class Dependency.
273      */
274     @Test
275     public void testAddAsEvidenceWithExisting() {
276         Dependency instance = new Dependency();
277         MavenArtifact mavenArtifact = new MavenArtifact("group", "artifact", "version", null);
278         instance.addAsEvidence("pom", mavenArtifact, Confidence.HIGH);
279         assertTrue(instance.getEvidence(EvidenceType.VENDOR).stream().anyMatch(e -> e.getConfidence() == Confidence.HIGH));
280         assertTrue(instance.size() == 4);
281         assertFalse(instance.getSoftwareIdentifiers().isEmpty());
282 
283         instance.getSoftwareIdentifiers().forEach((i) -> assertNull(i.getUrl()));
284 
285         mavenArtifact = new MavenArtifact("group", "artifact", "version", "url");
286         instance.addAsEvidence("pom", mavenArtifact, Confidence.HIGH);
287         assertTrue(instance.getEvidence(EvidenceType.VENDOR).stream().anyMatch(e -> e.getConfidence() == Confidence.HIGH));
288         assertTrue(instance.size() == 4);
289         assertFalse(instance.getSoftwareIdentifiers().isEmpty());
290 
291         instance.getSoftwareIdentifiers().forEach((i) -> assertNotNull(i.getUrl()));
292     }
