20  import java.io.File;

24  import java.io.Reader;

26  import javax.xml.parsers.SAXParser;
27  import static org.junit.Assert.assertEquals;
28  import org.junit.Test;
29  import org.owasp.dependencycheck.BaseTest;
30  import org.owasp.dependencycheck.utils.XmlUtils;
31  import org.xml.sax.InputSource;
32  import org.xml.sax.XMLReader;
33  
34  /**
35   *
36   * @author Jeremy Long
37   */
38  public class GrokHandlerTest extends BaseTest {
39  
40      /**
41       * Test of getSuppressionRules method, of class SuppressionHandler.
42       *
43       * @throws Exception thrown if there is an exception....
44       */
45      @Test
46      public void testHandler() throws Exception {
47          File file = BaseTest.getResourceAsFile(this, "assembly/sample-grok.xml");
48          InputStream schemaStream = BaseTest.getResourceAsStream(this, "schema/grok-assembly.1.0.xsd");
49  
50          GrokHandler handler = new GrokHandler();
51          SAXParser saxParser = XmlUtils.buildSecureSaxParser(schemaStream);
52          XMLReader xmlReader = saxParser.getXMLReader();
53          xmlReader.setErrorHandler(new GrokErrorHandler());
54          xmlReader.setContentHandler(handler);
55  
56          InputStream inputStream = new FileInputStream(file);
57          Reader reader = new InputStreamReader(inputStream, StandardCharsets.UTF_8);
58          InputSource in = new InputSource(reader);
59  
60          xmlReader.parse(in);
61  
62          AssemblyData result = handler.getAssemblyData();
63          assertEquals("OWASP Contributors", result.getCompanyName());
64          assertEquals("GrokAssembly", result.getProductName());
65          assertEquals("Inspects a .NET Assembly to determine Company, Product, and Version information", result.getComments());
66          assertEquals(1, result.getNamespaces().size());
67      }
68  }
