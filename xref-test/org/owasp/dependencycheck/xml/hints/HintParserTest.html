34      /**
35       * Test of parseHints method, of class HintParser.
36       */
37      @Test
38      public void testParseHints_File() throws Exception {
39          File file = BaseTest.getResourceAsFile(this, "hints.xml");
40          HintParser instance = new HintParser();
41          instance.parseHints(file);
42          List<HintRule> hintRules = instance.getHintRules();
43          List<VendorDuplicatingHintRule> vendorRules = instance.getVendorDuplicatingHintRules();
44          assertEquals("Two duplicating hints should have been read", 2, vendorRules.size());
45          assertEquals("Two hint rules should have been read", 2, hintRules.size());
46  
47          assertEquals("One add product should have been read", 1, hintRules.get(0).getAddProduct().size());
48          assertEquals("One add vendor should have been read", 1, hintRules.get(0).getAddVendor().size());
49          assertEquals("Two file name should have been read", 2, hintRules.get(1).getFileNames().size());
50  
51          assertEquals("add product name not found", "add product name", hintRules.get(0).getAddProduct().get(0).getName());
52          assertEquals("add vendor name not found", "add vendor name", hintRules.get(0).getAddVendor().get(0).getName());
53          assertEquals("given product name not found", "given product name", hintRules.get(0).getGivenProduct().get(0).getName());
54          assertEquals("given vendor name not found", "given vendor name", hintRules.get(0).getGivenVendor().get(0).getName());
55  
56          assertEquals("spring file name not found", "spring", hintRules.get(1).getFileNames().get(0).getValue());
57          assertEquals("file name 1 should not be case sensitive", false, hintRules.get(1).getFileNames().get(0).isCaseSensitive());
58          assertEquals("file name 1 should not be a regex", false, hintRules.get(1).getFileNames().get(0).isRegex());
59          assertEquals("file name 2 should be case sensitive", true, hintRules.get(1).getFileNames().get(1).isCaseSensitive());
60          assertEquals("file name 2 should be a regex", true, hintRules.get(1).getFileNames().get(1).isRegex());
61  
62          assertEquals("sun duplicating vendor", "sun", vendorRules.get(0).getValue());
63          assertEquals("sun duplicates vendor oracle", "oracle", vendorRules.get(0).getDuplicate());
64      }

66      /**
67       * Test the application of the correct XSD by the parser by using a
68       * hints-file with namespace
69       * {@code https://jeremylong.github.io/DependencyCheck/dependency-hint.1.1.xsd}
70       * that is using the version evidence for {@code<given>} that was introduced
71       * with namespace
72       * {@code https://jeremylong.github.io/DependencyCheck/dependency-hint.1.2.xsd}.
73       * This should yield a specific SAXParseException that gets wrapped into a
74       * HintParseException. We check for the correct error by searching for the
75       * error-message of the SAXParser in the exception's message.
76       */
77      @Test
78      public void testParseHintsXSDSelection() throws Exception {
79          File file = BaseTest.getResourceAsFile(this, "hints_invalid.xml");
80          HintParser instance = new HintParser();
81          Exception exception = Assert.assertThrows(org.owasp.dependencycheck.xml.hints.HintParseException.class, () -> instance.parseHints(file));
82          Assert.assertTrue(exception.getMessage().contains("Line=7, Column=133: cvc-enumeration-valid: Value 'version' is not facet-valid with respect to enumeration '[vendor, product]'. It must be a value from the enumeration."));
83  
84      }
