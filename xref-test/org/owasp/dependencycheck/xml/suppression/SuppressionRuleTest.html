336     @Test
337     public void testCpeMatches() throws CpeValidationException, MalformedPackageURLException {
338         CpeIdentifier identifier = new CpeIdentifier("microsoft", ".net_framework", "4.5", Confidence.HIGHEST);

340         PropertyType cpe = new PropertyType();
341         cpe.setValue("cpe:/a:microsoft:.net_framework:4.5");
342 
343         SuppressionRule instance = new SuppressionRule();

345         boolean result = instance.identifierMatches(cpe, identifier);
346         assertEquals(expResult, result);
347 
348         cpe.setValue("cpe:/a:microsoft:.net_framework:4.0");
349         expResult = false;
350         result = instance.identifierMatches(cpe, identifier);
351         assertEquals(expResult, result);
352 
353         cpe.setValue("CPE:/a:microsoft:.net_framework:4.5");

356         result = instance.identifierMatches(cpe, identifier);
357         assertEquals(expResult, result);
358 
359         cpe.setValue("cpe:/a:microsoft:.net_framework");
360         cpe.setCaseSensitive(false);
361         expResult = true;
362         result = instance.identifierMatches(cpe, identifier);
363         assertEquals(expResult, result);
364 
365         cpe.setValue("cpe:/a:microsoft:.*");
366         cpe.setRegex(true);
367         expResult = true;
368         result = instance.identifierMatches(cpe, identifier);
369         assertEquals(expResult, result);
370 
371         cpe.setValue("CPE:/a:microsoft:.*");
372         cpe.setRegex(true);

409     /**
410      * Test of process method, of class SuppressionRule.
411      */
412     @Test
413     public void testProcess() throws CpeValidationException {
414         //File struts = new File(this.getClass().getClassLoader().getResource("struts2-core-2.1.2.jar").getPath());
415         File struts = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
416         Dependency dependency = new Dependency(struts);
417         CpeIdentifier cpeId = new CpeIdentifier("microsoft", ".net_framework", "4.5", Confidence.HIGH);
418         dependency.addVulnerableSoftwareIdentifier(cpeId);
419         String sha1 = dependency.getSha1sum();
420         dependency.setSha1sum("384FAA82E193D4E4B0546059CA09572654BC3970");
421         Vulnerability v = createVulnerability();
422         dependency.addVulnerability(v);
423 
424         //cwe
425         SuppressionRule instance = new SuppressionRule();
426         instance.setSha1(sha1);
427         instance.addCwe("287");
428         instance.process(dependency);
429         assertEquals(1, dependency.getVulnerabilities().size());
430         dependency.setSha1sum(sha1);
431         instance.process(dependency);
432         assertTrue(dependency.getVulnerabilities().isEmpty());
433         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
434 
435         //cvss
436         dependency.addVulnerability(v);
437         instance = new SuppressionRule();
438         instance.addCvssBelow(5f);
439         instance.process(dependency);
440         assertEquals(1, dependency.getVulnerabilities().size());
441         instance.addCvssBelow(8f);
442         instance.process(dependency);
443         assertTrue(dependency.getVulnerabilities().isEmpty());
444         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
445 
446         //cve
447         dependency.addVulnerability(v);
448         instance = new SuppressionRule();
449         instance.addCve("CVE-2012-1337");
450         instance.process(dependency);
451         assertEquals(1, dependency.getVulnerabilities().size());
452         instance.addCve("CVE-2013-1337");
453         instance.process(dependency);
454         assertTrue(dependency.getVulnerabilities().isEmpty());
455         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
456 
457         //cpe
458         instance = new SuppressionRule();
459         PropertyType pt = new PropertyType();
460         pt.setValue("cpe:/a:microsoft:.net_framework:4.0");
461         instance.addCpe(pt);
462         instance.process(dependency);
463         assertTrue(dependency.getVulnerableSoftwareIdentifiers().size() == 1);
464         pt = new PropertyType();
465         pt.setValue("cpe:/a:microsoft:.net_framework:4.5");
466         instance.addCpe(pt);
467         pt = new PropertyType();
468         pt.setValue(".*");
469         pt.setRegex(true);
470         instance.setFilePath(pt);
471         instance.process(dependency);
472         assertTrue(dependency.getVulnerableSoftwareIdentifiers().isEmpty());
473         assertEquals(1, dependency.getSuppressedIdentifiers().size());
474 
475         instance = new SuppressionRule();
476 
477         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "4.0", Confidence.HIGH));
478         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "4.5", Confidence.HIGH));
479         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("microsoft", ".net_framework", "5.0", Confidence.HIGH));
480         pt = new PropertyType();
481         pt.setValue("cpe:/a:microsoft:.net_framework");
482         instance.addCpe(pt);
483         instance.setBase(true);
484         assertEquals(3, dependency.getVulnerableSoftwareIdentifiers().size());
485         assertEquals(1, dependency.getSuppressedIdentifiers().size());
486         instance.process(dependency);
487         assertTrue(dependency.getVulnerableSoftwareIdentifiers().isEmpty());
488         assertEquals(1, dependency.getSuppressedIdentifiers().size());
489     }

529     @Test
530     public void testProcessVulnerabilityNames() throws CpeValidationException, MalformedPackageURLException {
531         File spring = BaseTest.getResourceAsFile(this, "spring-security-web-3.0.0.RELEASE.jar");
532         Dependency dependency = new Dependency(spring);
533         dependency.addVulnerableSoftwareIdentifier(new CpeIdentifier("vmware", "springsource_spring_security", "3.0.0", Confidence.HIGH));
534         dependency.addSoftwareIdentifier(new PurlIdentifier("maven", "org.springframework.security", "spring-security-web", "3.0.0.RELEASE", Confidence.HIGH));
535 
536         dependency.addVulnerability(createVulnerability());
537         SuppressionRule instance = new SuppressionRule();
538         //gav
539         PropertyType pt = new PropertyType();
540         pt.setValue("pkg:maven/org.springframework.security/spring-security-web@3.0.0.RELEASE");
541         pt.setRegex(false);
542         pt.setCaseSensitive(false);
543         instance.setPackageUrl(pt);
544 
545         pt = new PropertyType();
546         pt.setValue("CVE-2013-1338");
547         instance.addVulnerabilityName(pt);
548 
549         instance.process(dependency);
550         assertEquals(1, dependency.getVulnerabilities().size());
551         assertEquals(0, dependency.getSuppressedVulnerabilities().size());
552 
553         
554         pt = new PropertyType();
555         pt.setValue("CVE-2013-1337");
556         instance.addVulnerabilityName(pt);
557         
558         instance.process(dependency);
559         assertEquals(0, dependency.getVulnerabilities().size());
560         assertEquals(1, dependency.getSuppressedVulnerabilities().size());
561     }

564         Vulnerability v = new Vulnerability();
565         v.addCwe("CWE-287 Improper Authentication");
566         v.setName("CVE-2013-1337");
567         v.setCvssV2(new CvssV2(7.5f, "Network", "Low", "None", "Partial", "Partial", "Partial", "High"));
568         return v;
