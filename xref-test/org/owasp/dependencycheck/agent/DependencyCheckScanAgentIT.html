18  package org.owasp.dependencycheck.agent;
19  
20  import org.junit.Assert;
21  import org.junit.BeforeClass;
22  import org.junit.Test;
23  import org.owasp.dependencycheck.dependency.Confidence;
24  import org.owasp.dependencycheck.dependency.Dependency;
25  import org.owasp.dependencycheck.dependency.EvidenceType;
26  import org.owasp.dependencycheck.reporting.ReportGenerator;
27  import org.owasp.dependencycheck.utils.FileUtils;
28  import java.io.File;
29  import java.util.ArrayList;
30  import java.util.List;
31  import org.owasp.dependencycheck.BaseDBTestCase;
32  
33  public class DependencyCheckScanAgentIT extends BaseDBTestCase {
34  
35      private static final File REPORT_DIR = new File("target/test-scan-agent/report");
36  
37      @BeforeClass
38      public static void beforeClass() {
39          if (!REPORT_DIR.exists()) {
40              REPORT_DIR.mkdirs();
41          }
42      }
43  
44      @Test
45      public void testComponentMetadata() throws Exception {
46          List<Dependency> dependencies = new ArrayList<>();
47          dependencies.add(createDependency("apache", "tomcat", "5.0.5"));
48          DependencyCheckScanAgent scanAgent = createScanAgent();
49          scanAgent.setDependencies(dependencies);
50          scanAgent.execute();
51  
52          Dependency tomcat = scanAgent.getDependencies().get(0);
53          Assert.assertTrue(tomcat.getVulnerableSoftwareIdentifiers().size() >= 1);
54  
55          // This will change over time
56          Assert.assertTrue(tomcat.getVulnerabilities().size() > 5);
57      }
58  
59      private DependencyCheckScanAgent createScanAgent() {
60          final DependencyCheckScanAgent scanAgent = new DependencyCheckScanAgent();
61          scanAgent.setApplicationName("Dependency-Track");
62          //the following does not work because it will be over-ridden by the system
63          //  properties configured during surefire/failsafe
64          //scanAgent.setDataDirectory(DATA_DIR.getAbsolutePath());
65          scanAgent.setCentralAnalyzerEnabled(false);
66          scanAgent.setReportOutputDirectory(REPORT_DIR.getAbsolutePath());
67          scanAgent.setReportFormat(ReportGenerator.Format.XML);
68          scanAgent.setAutoUpdate(false);
69          scanAgent.setUpdateOnly(false);
70          return scanAgent;
71      }
72  
73      private Dependency createDependency(final String vendor, final String name, final String version) {
74          final Dependency dependency = new Dependency(new File(FileUtils.getBitBucket()), true);
75          dependency.setName(name);
76          dependency.setVersion(version);
77          if (vendor != null) {
78              dependency.addEvidence(EvidenceType.VENDOR, "dependency-track", "vendor", vendor, Confidence.HIGHEST);
79              dependency.addVendorWeighting(vendor);
80          }
81          if (name != null) {
82              dependency.addEvidence(EvidenceType.PRODUCT, "dependency-track", "name", name, Confidence.HIGHEST);
83              dependency.addProductWeighting(name);
84          }
85          if (version != null) {
86              dependency.addEvidence(EvidenceType.VERSION, "dependency-track", "version", version, Confidence.HIGHEST);
87          }
88          return dependency;
89      }
90  }
