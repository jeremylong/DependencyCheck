42      @Before
43      @Override
44      public void setUp() throws Exception {
45          super.setUp();
46          // Have to use System properties as the Settings object pulls from the 
47          // system properties before configured properties
48          originalDataDir = getSettings().getString(Settings.KEYS.DATA_DIRECTORY);
49          System.setProperty(Settings.KEYS.DATA_DIRECTORY, tempDir.newFolder().getAbsolutePath());
50          int year = Calendar.getInstance().get(Calendar.YEAR);
51          if (Calendar.getInstance().get(Calendar.MONTH) == 0) {
52              year -= 1;
53          }
54          System.setProperty(Settings.KEYS.CVE_START_YEAR, Integer.toString(year));
55      }
56  
57      @After
58      @Override
59      public void tearDown() throws Exception {
60          try {
61              //delete temp files
62              FileUtils.delete(getSettings().getDataDirectory());
63              //Reset system property to original value just to be safe for other tests.
64              System.setProperty(Settings.KEYS.DATA_DIRECTORY, originalDataDir);
65              System.clearProperty(Settings.KEYS.H2_DATA_DIRECTORY);
66              System.clearProperty(Settings.KEYS.CVE_START_YEAR);
67          } catch (IOException ex) {
68              throw new UnexpectedAnalysisException(ex);
69          } finally {
70              super.tearDown();
71          }
72      }

74      @Test
75      public void testEvidenceCollectionAndEvidenceProcessingModes() throws Exception {
76          Dependency[] dependencies;
77          try (Engine engine = new Engine(Engine.Mode.EVIDENCE_COLLECTION, getSettings())) {
78              engine.openDatabase(); //does nothing in the current mode
79              assertDatabase(false);
80              Engine.Mode.EVIDENCE_COLLECTION.getPhases().forEach((phase) -> {
81                  assertThat(engine.getAnalyzers(phase), is(notNullValue()));
82              });
83              Engine.Mode.EVIDENCE_PROCESSING.getPhases().forEach((phase) -> {
84                  assertThat(engine.getAnalyzers(phase), is(nullValue()));
85              });
86              File file = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
87              engine.scan(file);
88              engine.analyzeDependencies();
89              dependencies = engine.getDependencies();
90              assertThat(dependencies.length, is(1));
91              Dependency dependency = dependencies[0];
92              assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("apache"));
93              assertTrue(dependency.getVendorWeightings().contains("apache"));
94              assertTrue(dependency.getVulnerabilities().isEmpty());
95          }
96  
97          try (Engine engine = new Engine(Engine.Mode.EVIDENCE_PROCESSING, getSettings())) {
98              assertDatabase(false);
99              engine.openDatabase();
100             
101             Engine.Mode.EVIDENCE_PROCESSING.getPhases().forEach((phase) -> {
102                 assertThat(engine.getAnalyzers(phase), is(notNullValue()));
103             });
104             Engine.Mode.EVIDENCE_COLLECTION.getPhases().forEach((phase) -> {
105                 assertThat(engine.getAnalyzers(phase), is(nullValue()));
106             });
107             engine.addDependency(dependencies[0]);
108             engine.analyzeDependencies();
109             Dependency dependency = dependencies[0];
110             assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("apache"));
111             assertTrue(dependency.getVendorWeightings().contains("apache"));
112             assertDatabase(true);
113         }
114     }

116     @Test
117     public void testStandaloneMode() throws Exception {
118         try (Engine engine = new Engine(Engine.Mode.STANDALONE, getSettings())) {
119             assertDatabase(false);
120             engine.openDatabase();
121             
122             for (AnalysisPhase phase : Engine.Mode.STANDALONE.getPhases()) {
123                 assertThat(engine.getAnalyzers(phase), is(notNullValue()));
124             }
125             File file = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
126             engine.scan(file);
127             engine.analyzeDependencies();
128             Dependency[] dependencies = engine.getDependencies();
129             //8 because there is JS being caught by the retireJS analyzer
130             assertThat(dependencies.length, is(8));
131             Dependency dependency = dependencies[0];
132             assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("apache"));
133             assertTrue(dependency.getVendorWeightings().contains("apache"));
134             assertDatabase(true);
135         }
136     }

138     private void assertDatabase(boolean exists) throws Exception {
139         Assume.assumeThat(getSettings().getString(Settings.KEYS.DB_DRIVER_NAME), is("org.h2.Driver"));
140         Path directory = getSettings().getDataDirectory().toPath();
141         assertThat(Files.exists(directory), is(true));
142         assertThat(Files.isDirectory(directory), is(true));
143         Path database = directory.resolve(getSettings().getString(Settings.KEYS.DB_FILE_NAME));
144         assertThat(Files.exists(database), is(exists));
145     }
