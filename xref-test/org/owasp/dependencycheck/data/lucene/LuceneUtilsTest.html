18  package org.owasp.dependencycheck.data.lucene;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.junit.Assert.assertFalse;
22  import static org.junit.Assert.assertTrue;
23  
24  import org.junit.Test;
25  import org.owasp.dependencycheck.BaseTest;
26  
27  /**
28   *
29   * @author Jeremy Long
30   */
31  public class LuceneUtilsTest extends BaseTest {
32  
33      /**
34       * Test of appendEscapedLuceneQuery method, of class LuceneUtils.
35       */
36      @Test
37      public void testAppendEscapedLuceneQuery() {
38          StringBuilder buf = new StringBuilder();
39          CharSequence text = "test encoding + - & | ! ( ) { } [ ] ^ \" ~ * ? : \\";
40          String expResult = "test encoding \\+ \\- \\& \\| \\! \\( \\) \\{ \\} \\[ \\] \\^ \\\" \\~ \\* \\? \\: \\\\";
41          LuceneUtils.appendEscapedLuceneQuery(buf, text);
42          assertEquals(expResult, buf.toString());
43      }
44  
45      /**
46       * Test of appendEscapedLuceneQuery method, of class LuceneUtils.
47       */
48      @Test
49      public void testAppendEscapedLuceneQuery_null() {
50          StringBuilder buf = new StringBuilder();
51          CharSequence text = null;
52          LuceneUtils.appendEscapedLuceneQuery(buf, text);
53          assertEquals(0, buf.length());
54      }
55  
56      /**
57       * Test of escapeLuceneQuery method, of class LuceneUtils.
58       */
59      @Test
60      public void testEscapeLuceneQuery() {
61          CharSequence text = "test encoding + - & | ! ( ) { } [ ] ^ \" ~ * ? : \\";
62          String expResult = "test encoding \\+ \\- \\& \\| \\! \\( \\) \\{ \\} \\[ \\] \\^ \\\" \\~ \\* \\? \\: \\\\";
63          String result = LuceneUtils.escapeLuceneQuery(text);
64          assertEquals(expResult, result);
65      }
66  
67      /**
68       * Test of escapeLuceneQuery method, of class LuceneUtils.
69       */
70      @Test
71      public void testEscapeLuceneQuery_null() {
72          CharSequence text = null;
73          String expResult = null;
74          String result = LuceneUtils.escapeLuceneQuery(text);
75          assertEquals(expResult, result);
76      }
77  
78      @Test
79      public void testIsKeyword() {
80          assertTrue("'AND' is a keyword and should return true", LuceneUtils.isKeyword("And"));
81          assertTrue("'OR' is a keyword and should return true", LuceneUtils.isKeyword("OR"));
82          assertTrue("'NOT' is a keyword and should return true", LuceneUtils.isKeyword("nOT"));
83          assertTrue("'TO' is being considered a keyword and should return true", LuceneUtils.isKeyword("TO"));
84          assertTrue("'+' is being considered a keyword and should return true", LuceneUtils.isKeyword("+"));
85          assertTrue("'-' is being considered a keyword and should return true", LuceneUtils.isKeyword("-"));
86          assertFalse("'the' is not a keyword and should return false", LuceneUtils.isKeyword("test"));
87      }
88  }
