18  package org.owasp.dependencycheck.data.nvd.json;
19  
20  import java.util.ArrayList;
21  import java.util.List;
22  import java.util.stream.Collectors;
23  import org.junit.Before;
24  import org.junit.Test;
25  import static org.junit.Assert.*;
26  
27  /**
28   *
29   * @author jeremy long
30   */
31  public class NodeFlatteningCollectorTest {
32  
33      private List<DefNode> nodes;
34  
35      @Before
36      public void setUp() {
37          nodes = new ArrayList<>();
38          DefNode node = new DefNode();
39          node.setOperator("top");
40          nodes.add(node);
41  
42          DefNode parent = node;
43          for (int x = 0; x < 5; x++) {
44              DefNode child = new DefNode();
45              child.setOperator("Child " + x);
46              List<DefNode> l = new ArrayList<>(1);
47              l.add(child);
48              parent.setChildren(l);
49              parent = child;
50          }
51      }
52  
53      /**
54       * Test of supplier method, of class NodeFlatteningCollector.
55       */
56      @Test
57      public void testCollector() {
58          assertFalse(nodes.stream().anyMatch((node) -> "Child 4".equals(node.getOperator())));
59          assertTrue(nodes.stream().collect(NodeFlatteningCollector.getInstance()).anyMatch((node) -> "Child 4".equals(node.getOperator())));
60          List<String> operators = nodes.stream().collect(NodeFlatteningCollector.getInstance()).map(mapper -> mapper.getOperator()).collect(Collectors.toList());
61          assertEquals(6, operators.size());
62      }
63  }
