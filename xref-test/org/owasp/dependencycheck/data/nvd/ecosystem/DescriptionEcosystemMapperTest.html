27      private static final String POSTFIX = ".ecosystem.txt";
28  
29      protected static final File directory = new File("./src/test/resources/ecosystem");
30  
31      protected static Map<String, File> getEcosystemFiles() throws IOException {
32          if (!directory.exists()) {
33              throw new RuntimeException(directory.getCanonicalPath());
34          }
35  
36          File[] listFiles = directory.listFiles((d, name) -> name.endsWith(POSTFIX));
37  
38          Map<String, File> map = new HashMap<>();
39          for (File file : listFiles) {
40              String name = file.getName();
41              map.put(name.substring(0, name.length() - POSTFIX.length()), file);
42          }
43          return map;
44      }
45  
46      @Test
47      public void testDescriptionEcosystemMapper() throws IOException {
48          DescriptionEcosystemMapper mapper = new DescriptionEcosystemMapper();
49          Map<String, File> ecosystemFiles = getEcosystemFiles();
50          for (Entry<String, File> entry : ecosystemFiles.entrySet()) {
51              try (BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(new FileInputStream(entry.getValue()), StandardCharsets.UTF_8))) {
52                  String description;
53                  while ((description = bufferedReader.readLine()) != null) {
54                      if (description.length() > 0 && !description.startsWith("#")) {
55                          String ecosystem = mapper.getEcosystem(asCve(description));
56                          if (entry.getKey().equals("null")) {
57                              assertNull(description, ecosystem);
58                          } else {
59                              assertEquals(description, entry.getKey(), ecosystem);
60                          }
61                      }
62                  }
63              }
64          }
65      }

103         
104         List<LangString> descriptions = new ArrayList<>();

106         descriptions.add(desc);
107         CveItem cveItem = new CveItem(null, null, null, null, null, null, null, null, null, null, null, null, descriptions, null, null, null, null, null);
108         DefCveItem defCveItem = new DefCveItem(cveItem);
