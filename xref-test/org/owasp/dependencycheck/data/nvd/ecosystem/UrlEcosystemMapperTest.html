25      private DefCveItem asCve(String url) {
26          
27          List<Reference> references  = new ArrayList<>();
28          Reference ref = new Reference(url, null, null);
29          references.add(ref);
30          CveItem cveItem = new CveItem(null, null, null, null, null, null, null, null, null, null, null, null, null, references, null, null, null, null);
31          DefCveItem defCveItem = new DefCveItem(cveItem);
32          
33          return defCveItem;
34      }
