22      private DefCveItem asCve(String url) {

24          
25          References references = new References();
26          
27          Reference reference = new Reference();
28          reference.setUrl(url);
29          
30          references.getReferenceData().add(reference);
31          
32          CVEJSON40Min11 cve = new CVEJSON40Min11();

34          
35          defCveItem.setCve(cve);
36          
37          return defCveItem;
