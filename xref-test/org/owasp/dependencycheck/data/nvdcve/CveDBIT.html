89      @Test
90      public void testGetVulnerabilities() throws Exception {
91  
92          CpeBuilder builder = new CpeBuilder();
93          Cpe cpe = builder.part(Part.APPLICATION).vendor("apache").product("struts").version("2.1.2").build();
94          List<Vulnerability> results;
95  
96          results = instance.getVulnerabilities(cpe);
97          assertTrue(results.size() > 5);
98          cpe = builder.part(Part.APPLICATION).vendor("apache").product("tomcat").version("6.0.1").build();
99          results = instance.getVulnerabilities(cpe);
100         assertTrue(results.size() > 1);
101 
102         boolean found = false;
103         String expected = "CVE-2014-0075";
104         for (Vulnerability v : results) {
105             if (expected.equals(v.getName())) {
106                 found = true;
107                 break;
108             }
109         }
110         assertTrue("Expected " + expected + ", but was not identified", found);
111 
112         found = false;
113         expected = "CVE-2014-0096";
114         for (Vulnerability v : results) {
115             if (expected.equals(v.getName())) {
116                 found = true;
117                 break;
118             }
119         }
120         assertTrue("Expected " + expected + ", but was not identified", found);
121 
122         cpe = builder.part(Part.APPLICATION).vendor("jenkins").product("mailer").version("1.13").build();
123         results = instance.getVulnerabilities(cpe);
124         assertTrue(results.size() >= 1);
125 
126         found = false;
127         expected = "CVE-2017-2651";
128         for (Vulnerability v : results) {
129             if (expected.equals(v.getName())) {
130                 found = true;
131                 break;
132             }
133         }
134         assertTrue("Expected " + expected + ", but was not identified", found);
135 
136         cpe = builder.part(Part.APPLICATION).vendor("fasterxml").product("jackson-databind").version("2.8.1").build();
137         results = instance.getVulnerabilities(cpe);
138         assertTrue(results.size() >= 1);

151     /**
152      * Test of getMatchingSoftware method, of class CveDB.
153      */
154     @Test
155     public void testGetMatchingSoftware() throws Exception {
156 
157         VulnerableSoftwareBuilder vsBuilder = new VulnerableSoftwareBuilder();
158         Set<VulnerableSoftware> software = new HashSet<>();
159         software.add(vsBuilder.part(Part.APPLICATION).vendor("openssl").product("openssl").version("1.0.1e").build());
160 
161         CpeBuilder cpeBuilder = new CpeBuilder();
162         Cpe identified = cpeBuilder.part(Part.APPLICATION).vendor("openssl").product("openssl").version("1.0.1o").build();
163         VulnerableSoftware results = instance.getMatchingSoftware(identified, software);
164         assertNull(results);
165         software.add(vsBuilder.part(Part.APPLICATION).vendor("openssl").product("openssl").version("1.0.1p").build());
166         results = instance.getMatchingSoftware(identified, software);
167         assertNull(results);
168 
169         software.add(vsBuilder.part(Part.APPLICATION).vendor("openssl").product("openssl").version("1.0.1p")
170                 .versionStartIncluding("1.0.0").versionEndExcluding("1.0.1p").build());
171         results = instance.getMatchingSoftware(identified, software);
172         assertNotNull(results);
173         assertEquals("cpe:/a:openssl:openssl:1.0.1p", results.toCpe22Uri());
174 
175         software.clear();
176 
177         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("3.2.5").build());
178         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("3.2.6").build());
179         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("3.2.7")
180                 .versionStartIncluding("3.0.0").versionEndIncluding("3.2.7").build());
181 
182         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.0.1")
183                 .versionStartIncluding("4.0.0").versionEndIncluding("4.0.1").build());
184         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.0.0").update("m1").build());
185         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.0.0").update("m2").build());
186         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.0.0").update("rc1").build());
187 
188         identified = cpeBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("3.2.2").build();
189         results = instance.getMatchingSoftware(identified, software);
190         assertEquals("cpe:/a:springsource:spring_framework:3.2.7", results.toCpe22Uri());
191 
192         identified = cpeBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("3.2.12").build();
193         results = instance.getMatchingSoftware(identified, software);
194         assertNull(results);
195 
196         identified = cpeBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.0.0").build();
197         results = instance.getMatchingSoftware(identified, software);
198         assertEquals("cpe:/a:springsource:spring_framework:4.0.1", results.toCpe22Uri());
199 
200         identified = cpeBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("4.1.0").build();
201         results = instance.getMatchingSoftware(identified, software);
202         assertNull(results);
203 
204         software.clear();
205 
206         software.add(vsBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version(LogicalValue.NA).build());
207         identified = cpeBuilder.part(Part.APPLICATION).vendor("springsource").product("spring_framework").version("1.6.3").build();
208         results = instance.getMatchingSoftware(identified, software);
209         assertNull(results);
210 
211         software.clear();
212         software.add(vsBuilder.part(Part.APPLICATION).vendor("eclipse").product("jetty").update("20170531").versionEndIncluding("9.5.6").build());
213         identified = cpeBuilder.part(Part.APPLICATION).vendor("eclipse").product("jetty").version("9.0.0").update("20170532").build();
214         results = instance.getMatchingSoftware(identified, software);
215         assertNull(results);
216         identified = cpeBuilder.part(Part.APPLICATION).vendor("eclipse").product("jetty").version("9.0.0").update("20170531").build();
217         results = instance.getMatchingSoftware(identified, software);
218         assertEquals("cpe:/a:eclipse:jetty::20170531", results.toCpe22Uri());
219 
220         software.clear();
221         software.add(vsBuilder.part(Part.APPLICATION).vendor("jenkins").product("mailer").versionEndExcluding("1.20").targetSw("jenkins").build());
222         identified = cpeBuilder.part(Part.APPLICATION).vendor("jenkins").product("mailer").version("1.13").build();
223         results = instance.getMatchingSoftware(identified, software);
224         assertEquals("cpe:/a:jenkins:mailer:::~~~jenkins~~", results.toCpe22Uri());
225     }
