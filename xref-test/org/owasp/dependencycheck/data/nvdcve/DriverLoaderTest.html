38      /**
39       * Test of load method, of class DriverLoader.
40       *
41       * @throws java.sql.SQLException thrown if there is an error de-registering
42       * the driver
43       */
44      @Test
45      public void testLoad_String() throws SQLException {
46          String className = "org.h2.Driver";
47          Driver d = null;
48          try {
49              d = DriverLoader.load(className);
50          } catch (DriverLoadException ex) {
51              fail(ex.getMessage());
52          } finally {
53              if (d != null) {
54                  DriverManager.deregisterDriver(d);
55              }
56          }
57      }

69      /**
70       * Test of load method, of class DriverLoader.
71       */
72      @Test
73      public void testLoad_String_String() throws Exception {
74          String className = "com.mysql.jdbc.Driver";
75          //we know this is in target/test-classes
76          //File testClassPath = (new File(this.getClass().getClassLoader().getResource("org.mortbay.jetty.jar").getPath())).getParentFile();
77          File testClassPath = BaseTest.getResourceAsFile(this, "org.mortbay.jetty.jar").getParentFile();
78          File driver = new File(testClassPath, "../../src/test/resources/mysql-connector-java-5.1.27-bin.jar");
79          assertTrue("MySQL Driver JAR file not found in src/test/resources?", driver.isFile());
80  
81          Driver d = null;
82          try {
83              d = DriverLoader.load(className, driver.getAbsolutePath());
84              d = DriverManager.getDriver("jdbc:mysql://localhost:3306/dependencycheck");
85              assertNotNull(d);
86          } finally {
87              if (d != null) {
88                  DriverManager.deregisterDriver(d);
89              }
90          }
91      }

93      /**
94       * Test of load method, of class DriverLoader.
95       */
96      @Test
97      public void testLoad_String_String_multiple_paths()  {
98          final String className = "com.mysql.jdbc.Driver";
99          //we know this is in target/test-classes
100         //final File testClassPath = (new File(this.getClass().getClassLoader().getResource("org.mortbay.jetty.jar").getPath())).getParentFile();
101         final File testClassPath = BaseTest.getResourceAsFile(this, "org.mortbay.jetty.jar").getParentFile();
102         final File dir1 = new File(testClassPath, "../../src/test/");
103         final File dir2 = new File(testClassPath, "../../src/test/resources/");
104         final String paths = String.format("%s" + File.pathSeparator + "%s", dir1.getAbsolutePath(), dir2.getAbsolutePath());
105 
106         Driver d = null;
107         try {
108             d = DriverLoader.load(className, paths);
109         } catch (DriverLoadException ex) {
110             fail(ex.getMessage());
111         } finally {
112             if (d != null) {
113                 try {
114                     DriverManager.deregisterDriver(d);
115                 } catch (SQLException ex) {
116                     fail(ex.getMessage());
117                 }
118             }
119         }
120     }
