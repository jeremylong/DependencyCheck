67      @Test
68      public void testMSBuildProjectAnalysis() throws Exception {
69  
70          try (Engine engine = new Engine(getSettings())) {
71              File file = BaseTest.getResourceAsFile(this, "msbuild/test.csproj");
72              Dependency toScan = new Dependency(file);
73              MSBuildProjectAnalyzer analyzer = new MSBuildProjectAnalyzer();
74              analyzer.setFilesMatched(true);
75              analyzer.initialize(getSettings());
76              analyzer.prepare(engine);
77              analyzer.setEnabled(true);
78              analyzer.analyze(toScan, engine);
79  
80              assertEquals("5 dependencies should be found", 5, engine.getDependencies().length);
81  
82              int foundCount = 0;
83  
84              for (Dependency result : engine.getDependencies()) {
85                  assertEquals(DEPENDENCY_ECOSYSTEM, result.getEcosystem());
86                  assertTrue(result.isVirtual());
87  
88                  if (null != result.getName()) {
89                      switch (result.getName()) {
90                          case "Humanizer":
91                              foundCount++;
92                              assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("Humanizer"));
93                              assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Humanizer"));
94                              assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("2.2.0"));
95                              break;
96                          case "JetBrains.Annotations":
97                              foundCount++;
98                              assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("JetBrains"));
99                              assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("JetBrains.Annotations"));
100                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Annotations"));
101                             assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("11.1.0"));
102                             break;
103                         case "Microsoft.AspNetCore.All":
104                             foundCount++;
105                             assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("Microsoft"));
106                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Microsoft.AspNetCore.All"));
107                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("AspNetCore"));
108                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("AspNetCore.All"));
109                             assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("2.0.5"));
110                             break;
111                         case "Microsoft.Extensions.Logging":
112                             foundCount++;
113                             assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("Microsoft"));
114                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Microsoft.Extensions.Logging"));
115                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Extensions"));
116                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Extensions.Logging"));
117                             assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("6.0.0"));
118                             break;
119                         case "NodaTime":
120                             foundCount++;
121                             assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("NodaTime"));
122                             assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("NodaTime"));
123                             assertTrue("Expected 3.0.0; contained: " + result.getEvidence(EvidenceType.VERSION).stream().map(e -> e.toString()).collect(Collectors.joining(",", "{", "}")), result.getEvidence(EvidenceType.VERSION).toString().contains("3.0.0"));
124                             break;
125                         default:
126                             break;
127                     }
128                 }
129             }
130 
131             assertEquals("5 expected dependencies should be found", 5, foundCount);
132         }
133     }
