62      /**
63       * Test Elixir MixAudit analysis.
64       *
65       * @throws AnalysisException is thrown when an exception occurs.
66       */
67      @Test
68      public void testAnalysis() throws AnalysisException, DatabaseException {
69          try (Engine engine = new Engine(getSettings())) {
70              engine.openDatabase();
71              analyzer.prepare(engine);
72              final String resource = "elixir/vulnerable/mix.lock";
73              final Dependency result = new Dependency(BaseTest.getResourceAsFile(this, resource));
74              analyzer.analyze(result, engine);
75  
76              final Dependency[] dependencies = engine.getDependencies();
77              assertEquals("should be one result exactly", 1, dependencies.length);
78  
79              Dependency d = dependencies[0];
80              assertTrue(d.isVirtual());
81              assertEquals("plug:1.3.4", d.getPackagePath());
82              assertEquals("1.3.4", d.getVersion());
83              assertEquals("plug", d.getName());
84  
85              Evidence packageEvidence = d.getEvidence(EvidenceType.PRODUCT).iterator().next();
86              assertEquals("Package", packageEvidence.getName());
87              assertEquals("plug", packageEvidence.getValue());
88  
89              Evidence versionEvidence = d.getEvidence(EvidenceType.VERSION).iterator().next();
90              assertEquals("Version", versionEvidence.getName());
91              assertEquals("1.3.4", versionEvidence.getValue());
92  
93              assertTrue(d.getFilePath().endsWith(resource));
94              assertTrue(d.getFileName().equals("mix.lock"));
95  
96              Vulnerability v = d.getVulnerabilities().iterator().next();
97              assertEquals("2018-1000883", v.getName());
98              assertEquals("Cookie headers were not validated\n", v.getDescription());
99              assertEquals(-1.0f, v.getCvssV2().getCvssData().getBaseScore(), 0.0);
100 
101             VulnerableSoftware s = v.getVulnerableSoftware().iterator().next();
102             assertEquals("cpe:2.3:a:plug_project:plug:1.3.4:*:*:*:*:*:*:*", s.toString());
103 
104         } catch (InitializationException | DatabaseException | AnalysisException e) {
105             LOGGER.warn("Exception setting up ElixirAuditAnalyzer. Make sure Elixir and the mix_audit escript is installed. You may also need to set property \"analyzer.mix.audit.path\".");
106             Assume.assumeNoException("Exception setting up ElixirMixAuditAnalyzer; mix_audit may not be installed, or property \"analyzer.mix.audit.path\" may not be set.", e);
107         }
108     }

146                 LOGGER.error("NPE", ex);

148             } catch (ExceptionCollection ex) {
149                 Assume.assumeNoException("Exception setting up ElixirMixAuditAnalyzer; mix_audit may not be installed, or property \"analyzer.mix.audit.path\" may not be set.", ex);

151             }
