41      @Test
42      public void testSupportsExtensions() {
43          ArchiveAnalyzer instance = new ArchiveAnalyzer();
44          instance.initialize(getSettings());
45          Set<String> expResult = new HashSet<>();
46          expResult.add("zip");
47          expResult.add("war");
48          expResult.add("ear");
49          expResult.add("jar");
50          expResult.add("sar");
51          expResult.add("apk");
52          expResult.add("nupkg");
53          expResult.add("tar");
54          expResult.add("gz");
55          expResult.add("tgz");
56          expResult.add("bz2");
57          expResult.add("tbz2");
58          expResult.add("rpm");
59          for (String ext : expResult) {
60              assertTrue(ext, instance.accept(new File("test." + ext)));
61          }
62      }

125      */
126     @Test
127     public void testAnalyze() throws Exception {
128         Settings settings = getSettings();
129         settings.setBoolean(Settings.KEYS.AUTO_UPDATE, false);
130         settings.setBoolean(Settings.KEYS.ANALYZER_NEXUS_ENABLED, false);
131         settings.setBoolean(Settings.KEYS.ANALYZER_CENTRAL_ENABLED, false);
132         settings.setBoolean(Settings.KEYS.ANALYZER_OSSINDEX_ENABLED, false);
133 
134         ArchiveAnalyzer instance = new ArchiveAnalyzer();
135         instance.initialize(settings);
136         //trick the analyzer into thinking it is active.
137         instance.accept(new File("test.ear"));
138         try (Engine engine = new Engine(settings)) {
139 
140             instance.prepare(engine);
141             File file = BaseTest.getResourceAsFile(this, "daytrader-ear-2.1.7.ear");
142             Dependency dependency = new Dependency(file);
143 
144             int initial_size = engine.getDependencies().length;
145             instance.analyze(dependency, engine);
146             int ending_size = engine.getDependencies().length;
147             assertTrue(initial_size < ending_size);
148         } finally {
149             instance.close();
150         }
151     }
