41      @Test
42      public void testSupportsExtensions() {
43          ArchiveAnalyzer instance = new ArchiveAnalyzer();
44          instance.initialize(getSettings());
45          Set<String> expResult = new HashSet<>();
46          expResult.add("zip");
47          expResult.add("war");
48          expResult.add("ear");
49          expResult.add("jar");
50          expResult.add("sar");
51          expResult.add("apk");
52          expResult.add("nupkg");
53          expResult.add("tar");
54          expResult.add("gz");
55          expResult.add("tgz");
56          expResult.add("bz2");
57          expResult.add("tbz2");
58          for (String ext : expResult) {
59              assertTrue(ext, instance.accept(new File("test." + ext)));
60          }
61      }

125     @Test
126     public void testAnalyze() throws Exception {
127         Settings settings = getSettings();
128         settings.setBoolean(Settings.KEYS.AUTO_UPDATE, false);
129         settings.setBoolean(Settings.KEYS.ANALYZER_NEXUS_ENABLED, false);
130         settings.setBoolean(Settings.KEYS.ANALYZER_CENTRAL_ENABLED, false);
131         settings.setBoolean(Settings.KEYS.ANALYZER_OSSINDEX_ENABLED, false);
132 
133         ArchiveAnalyzer instance = new ArchiveAnalyzer();
134         instance.initialize(settings);
135         //trick the analyzer into thinking it is active.
136         instance.accept(new File("test.ear"));
137         try (Engine engine = new Engine(settings)) {
138 
139             instance.prepare(engine);
140             File file = BaseTest.getResourceAsFile(this, "daytrader-ear-2.1.7.ear");
141             Dependency dependency = new Dependency(file);
142 
143             int initial_size = engine.getDependencies().length;
144             instance.analyze(dependency, engine);
145             int ending_size = engine.getDependencies().length;
146             assertTrue(initial_size < ending_size);
147         } finally {
148             instance.close();
149         }
150     }

185         settings.setBoolean(Settings.KEYS.AUTO_UPDATE, false);
186         settings.setBoolean(Settings.KEYS.ANALYZER_NEXUS_ENABLED, false);
187         settings.setBoolean(Settings.KEYS.ANALYZER_CENTRAL_ENABLED, false);
188         settings.setBoolean(Settings.KEYS.ANALYZER_OSSINDEX_ENABLED, false);
189 
190         ArchiveAnalyzer instance = new ArchiveAnalyzer();
191         instance.initialize(settings);
192         //trick the analyzer into thinking it is active.
193         instance.accept(new File("test.ear"));
194         try (Engine engine = new Engine(settings)) {
195             instance.prepare(null);
196             File filea = BaseTest.getResourceAsFile(this, "archive/handle-a.jar");
197             Dependency dependencya = new Dependency(filea);
198 
199             File fileb = BaseTest.getResourceAsFile(this, "archive/handle-b.jar");

206

209 
210         } finally {
