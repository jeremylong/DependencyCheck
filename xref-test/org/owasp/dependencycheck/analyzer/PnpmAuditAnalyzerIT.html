17      @Test
18      @Ignore("unfortunately pnpm and brew are somewhat broken on my machine atm...")
19      public void testAnalyzePackagePnpm() throws AnalysisException {
20  
21          try (Engine engine = new Engine(getSettings())) {
22              PnpmAuditAnalyzer analyzer = new PnpmAuditAnalyzer();
23              analyzer.setFilesMatched(true);
24              analyzer.initialize(getSettings());
25              analyzer.prepare(engine);
26              analyzer.setEnabled(true);
27              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, "pnpmaudit/pnpm-lock.yaml"));
28              analyzer.analyze(toScan, engine);
29              boolean found = false;
30              assertTrue("More than 1 dependency should be identified", 1 < engine.getDependencies().length);
31              for (Dependency result : engine.getDependencies()) {
32                  if ("pnpm-lock.yaml?dns-sync".equals(result.getFileName())) {
33                      found = true;
34                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("dns-sync"));
35                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("dns-sync"));
36                      assertTrue(result.isVirtual());
37                  }
38              }
39              assertTrue("dns-sync was not found", found);
40          } catch (InitializationException ex) {
41              //yarn is not installed - skip the test case.
42              Assume.assumeNoException(ex);
43          }
44      }
