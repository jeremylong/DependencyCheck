38  /**
39   * Unit tests for PipAnalyzerTest.
40   */
41  public class PipfileAnalyzerTest extends BaseDBTestCase {
42  
43      /**
44       * The analyzer to test.
45       */
46      private PipfileAnalyzer analyzer;
47  
48      /**
49       * Correctly setup the analyzer for testing.
50       *
51       * @throws Exception thrown if there is a problem
52       */
53      @Before
54      @Override
55      public void setUp() throws Exception {
56          super.setUp();
57          analyzer = new PipfileAnalyzer();
58          analyzer.initialize(getSettings());
59          analyzer.setFilesMatched(true);
60          analyzer.prepare(null);
61      }
62  
63      /**
64       * Cleanup the analyzer's temp files, etc.
65       *
66       * @throws Exception thrown if there is a problem
67       */
68      @After
69      @Override
70      public void tearDown() throws Exception {
71          analyzer.close();
72          super.tearDown();
73      }
74  
75      /**
76       * Test of getName method, of class PipAnalyzer.
77       */
78      @Test
79      public void testGetName() {
80          assertEquals("Pipfile Analyzer", analyzer.getName());
81      }
82  
83      /**
84       * Test of supportsExtension method, of class PipAnalyzer.
85       */
86      @Test
87      public void testSupportsFiles() {
88          assertTrue(analyzer.accept(new File("Pipfile")));
89          assertFalse(analyzer.accept(new File("Pipfile.lock")));
90      }
91  
92      /**
93       * Test of inspect method, of class PipfileAnalyzer.
94       *
95       * @throws AnalysisException is thrown when an exception occurs.
96       */
97      @Test
98      public void testAnalyzePackageJson() throws Exception {
99          try (Engine engine = new Engine(getSettings())) {
100             final Dependency result = new Dependency(BaseTest.getResourceAsFile(this, "Pipfile"));
101             engine.addDependency(result);
102             analyzer.analyze(result, engine);
103             assertFalse(ArrayUtils.contains(engine.getDependencies(), result));
104             assertEquals(40, engine.getDependencies().length);
105             boolean foundUrllib3 = false;
106             boolean foundCryptography = false;
107             for (Dependency d : engine.getDependencies()) {
108                 if ("urllib3".equals(d.getName())) {
109                     foundUrllib3 = true;
110                     assertEquals("1.25.9", d.getVersion());
111                     assertThat(d.getDisplayFileName(), equalTo("urllib3:1.25.9"));
112                     assertEquals(PythonDistributionAnalyzer.DEPENDENCY_ECOSYSTEM, d.getEcosystem());
113                 }
114                 if ("cryptography".equals(d.getName())) {
115                     foundCryptography = true;
116                     assertEquals("1.8.2", d.getVersion());
117                     assertThat(d.getDisplayFileName(), equalTo("cryptography:1.8.2"));
118                     assertEquals(PythonDistributionAnalyzer.DEPENDENCY_ECOSYSTEM, d.getEcosystem());
119                 }
120             }
121             assertTrue("Expeced to find urllib3", foundUrllib3);
122             assertTrue("Expeced to find cryptography", foundCryptography);
123         }
124     }
