86      @Test
87      public void testDartPubspecLockAnalyzer() throws AnalysisException {
88          final Engine engine = new Engine(getSettings());
89          final Dependency result = new Dependency(BaseTest.getResourceAsFile(this,
90                  "dart/pubspec.lock"));
91          dartAnalyzer.analyze(result, engine);
92  
93          assertThat(engine.getDependencies().length, equalTo(5));
94  
95          Dependency dependency1 = engine.getDependencies()[0];
96          Dependency dependency2 = engine.getDependencies()[1];
97          Dependency dependency3 = engine.getDependencies()[2];
98          Dependency dependency4 = engine.getDependencies()[3];
99          Dependency dependency5 = engine.getDependencies()[4];
100 
101         assertThat(dependency1.getName(), equalTo("_fe_analyzer_shared"));
102         assertThat(dependency1.getVersion(), equalTo("40.0.0"));
103         for (Identifier identifier : dependency1.getSoftwareIdentifiers()) {
104             if (identifier instanceof GenericIdentifier) {
105                 assertThat(identifier.getValue(), equalTo("cpe:2.3:a:*:_fe_analyzer_shared:40.0.0:*:*:*:*:*:*:*"));
106             }
107             if (identifier instanceof PurlIdentifier) {
108                 assertThat(identifier.getValue(), equalTo("pkg:pub/_fe_analyzer_shared@40.0.0"));
109             }
110         }
