18  package org.owasp.dependencycheck.analyzer;
19  
20  import org.junit.After;
21  import org.junit.Before;
22  import org.junit.Test;
23  import org.owasp.dependencycheck.BaseTest;
24  import org.owasp.dependencycheck.Engine;
25  import org.owasp.dependencycheck.dependency.Dependency;
26  import org.owasp.dependencycheck.utils.Settings;
27  import java.io.File;
28  import java.util.List;
29  import static org.junit.Assert.*;
30  import org.owasp.dependencycheck.BaseDBTestCase;
31  import org.owasp.dependencycheck.data.update.RetireJSDataSource;
32  
33  public class RetireJsAnalyzerFiltersTest extends BaseDBTestCase {
34  
35      /**
36       * Test of filters method.
37       *
38       * @throws Exception is thrown when an exception occurs.
39       */
40      @Test
41      public void testFilters() throws Exception {
42  
43          String[] filter = {"jQuery JavaScript Library"};
44          getSettings().setArrayIfNotEmpty(Settings.KEYS.ANALYZER_RETIREJS_FILTERS, filter);
45          getSettings().setBoolean(Settings.KEYS.ANALYZER_RETIREJS_FILTER_NON_VULNERABLE, true);
46  
47          RetireJsAnalyzer analyzer = null;
48          try (Engine engine = new Engine(getSettings())) {
49              engine.openDatabase(true, true);
50              RetireJSDataSource ds = new RetireJSDataSource();
51              ds.update(engine);
52              analyzer = new RetireJsAnalyzer();
53              analyzer.setFilesMatched(true);
54  
55              analyzer.initialize(getSettings());
56              analyzer.prepare(engine);
57  
58              //removed by filter (see setup above)
59              File file = BaseTest.getResourceAsFile(this, "javascript/jquery-1.6.2.js");
60              List<Dependency> scanned = engine.scan(file);
61              assertTrue(scanned == null || scanned.isEmpty());
62  
63              //remove non-vulnerable
64              file = BaseTest.getResourceAsFile(this, "javascript/custom.js");
65              scanned = engine.scan(file);
66              assertTrue(scanned.size() == 1);
67              assertEquals(1, engine.getDependencies().length);
68              analyzer.analyze(scanned.get(0), engine);
69              assertEquals(0, engine.getDependencies().length);
70  
71              //kept because it is does not match the filter and is vulnerable
72              file = BaseTest.getResourceAsFile(this, "javascript/ember.js");
73              scanned = engine.scan(file);
74              assertTrue(scanned.size() == 1);
75              assertEquals(1, engine.getDependencies().length);
76              analyzer.analyze(scanned.get(0), engine);
77              assertEquals(1, engine.getDependencies().length);
78          } finally {
79              if (analyzer != null) {
80                  analyzer.close();
81              }
82          }
83      }
84  }
