18  package org.owasp.dependencycheck.analyzer;
19  
20  import java.io.File;
21  import org.junit.After;
22  
23  import static org.junit.Assert.assertEquals;
24  import static org.junit.Assert.assertTrue;
25  import static org.junit.Assert.fail;
26  import org.junit.Before;
27  import org.junit.Test;
28  import org.owasp.dependencycheck.BaseTest;
29  import org.owasp.dependencycheck.analyzer.exception.AnalysisException;
30  import org.owasp.dependencycheck.analyzer.exception.UnexpectedAnalysisException;
31  import org.owasp.dependencycheck.dependency.Confidence;
32  import org.owasp.dependencycheck.dependency.Dependency;
33  import org.owasp.dependencycheck.dependency.Evidence;
34  import org.owasp.dependencycheck.dependency.EvidenceType;
35  import org.slf4j.Logger;
36  import org.slf4j.LoggerFactory;
37  
38  /**
39   * Tests for the PEAnalyzer.
40   *
41   * @author Jeremy Long
42   *
43   */
44  public class PEAnalyzerTest extends BaseTest {
45  
46      private static final Logger LOGGER = LoggerFactory.getLogger(PEAnalyzerTest.class);
47  
48      private static final String LOG_KEY = "org.slf4j.simpleLogger.org.owasp.dependencycheck.analyzer.PEAnalyzer";
49  
50      private PEAnalyzer analyzer;
51  
52      /**
53       * Sets up the analyzer.
54       *
55       * @throws Exception if anything goes sideways
56       */
57      @Before
58      @Override
59      public void setUp() throws Exception {
60          super.setUp();
61          analyzer = new PEAnalyzer();
62          analyzer.initialize(getSettings());
63          analyzer.accept(new File("test.dll")); // trick into "thinking it is active"
64          analyzer.prepare(null);
65      }
66  
67      /**
68       * Tests to make sure the name is correct.
69       */
70      @Test
71      public void testGetName() {
72          assertEquals("PE Analyzer", analyzer.getName());
73      }
74  
75      @Test
76      public void testAnalysis() throws Exception {
77          File f = BaseTest.getResourceAsFile(this, "log4net.dll");
78  
79          Dependency d = new Dependency(f);
80          analyzer.analyze(d, null);
81          assertTrue(d.contains(EvidenceType.VERSION, new Evidence("PE Header", "FileVersion", "1.2.13.0", Confidence.HIGH)));
82          assertEquals("1.2.13.0", d.getVersion());
83          assertTrue(d.contains(EvidenceType.VENDOR, new Evidence("PE Header", "CompanyName", "The Apache Software Foundation", Confidence.HIGHEST)));
84          assertTrue(d.contains(EvidenceType.PRODUCT, new Evidence("PE Header", "ProductName", "log4net", Confidence.HIGHEST)));
85          assertEquals("log4net", d.getName());
86      }
87  
88      @After
89      @Override
90      public void tearDown() throws Exception {
91          try {
92              analyzer.closeAnalyzer();
93          } catch (Exception ex) {
94              throw new UnexpectedAnalysisException(ex);
95          } finally {
96              super.tearDown();
97          }
98      }
99  }
