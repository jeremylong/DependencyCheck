87      /**
88       * Test of mergeDependencies method, of class DependencyMergingAnalyzer.
89       */
90      @Test
91      public void testMergeDependencies() {
92          Dependency dependency = new Dependency(true);
93          dependency.setName("main");
94          dependency.addEvidence(EvidenceType.VENDOR, "test", "vendor", "main", Confidence.HIGHEST);
95          dependency.addEvidence(EvidenceType.PRODUCT, "test", "product", "main", Confidence.HIGHEST);
96          dependency.addEvidence(EvidenceType.VERSION, "test", "version", "1.0", Confidence.HIGHEST);
97          Dependency relatedDependency = new Dependency(true);
98          relatedDependency.addEvidence(EvidenceType.VENDOR, "test", "vendor", "related", Confidence.HIGHEST);
99          relatedDependency.addEvidence(EvidenceType.PRODUCT, "test", "product", "related", Confidence.HIGHEST);
100         relatedDependency.addEvidence(EvidenceType.VERSION, "test", "version", "1.0", Confidence.HIGHEST);
101         relatedDependency.setName("related");
102         relatedDependency.addProjectReference("related");
103         Set<Dependency> dependenciesToRemove = new HashSet<>();
104         DependencyMergingAnalyzer.mergeDependencies(dependency, relatedDependency, dependenciesToRemove);
105         assertTrue(dependenciesToRemove.contains(relatedDependency));
106         assertEquals(2, dependency.getEvidence(EvidenceType.VENDOR).size());
107         assertEquals(2, dependency.getEvidence(EvidenceType.PRODUCT).size());
108         assertEquals(1, dependency.getEvidence(EvidenceType.VERSION).size());
109         assertEquals(1, dependency.getRelatedDependencies().size());
110         assertEquals(0, relatedDependency.getRelatedDependencies().size());
111     }

222     /**
223      * Test of getMainAndroidDependency method, of class
224      * DependencyMergingAnalyzer.
225      *
226      * @throws Exception thrown if there is an analysis exception
227      */
228     @Test
229     public void testGetMainAndroidDependency() throws Exception {
230         ArchiveAnalyzer aa = null;
231         try (Engine engine = new Engine(Engine.Mode.EVIDENCE_COLLECTION, getSettings())) {
232             Dependency dependency1 = new Dependency(BaseTest.getResourceAsFile(this, "aar-1.0.0.aar"));
233             dependency1.setEcosystem(Ecosystem.JAVA);
234             aa = new ArchiveAnalyzer();
235 
236             aa.initialize(getSettings());
237             aa.accept(dependency1.getActualFile());
238             aa.prepareAnalyzer(engine);
239             aa.analyze(dependency1, engine);
240 
241             Dependency dependency2 = null;
242             for (Dependency d : engine.getDependencies()) {
243                 if ("classes.jar".equals(d.getActualFile().getName())) {
244                     dependency2 = d;
245                     break;
246                 }
247             }
248             assertNotNull("classes.jar was not found", dependency2);
249             dependency2.setEcosystem(Ecosystem.JAVA);
250             DependencyMergingAnalyzer instance = new DependencyMergingAnalyzer();
251             Dependency expResult = dependency1;
252             Dependency result = instance.getMainAndroidDependency(dependency1, dependency2);
253             assertEquals(expResult, result);
254         } finally {
255             if (aa != null) {
256                 aa.closeAnalyzer();
257             }
258         }
259     }
