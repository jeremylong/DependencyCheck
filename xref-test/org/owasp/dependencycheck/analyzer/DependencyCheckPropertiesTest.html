27      @Test
28      public void should_each_analyzer_have_default_enabled_property()
29              throws IOException, InstantiationException, IllegalAccessException {
30          String packageName = "org.owasp.dependencycheck.analyzer";
31          Set<Class<AbstractAnalyzer>> analyzerImplementations = findAllAnalyzerImplementations(packageName);
32  
33          Set<String> analyzerEnabledSettingKeys = new HashSet<>();
34  
35          for (Class<AbstractAnalyzer> analyzerClass : analyzerImplementations) {
36              AbstractAnalyzer analyzer = analyzerClass.newInstance();
37              String enabledKey = analyzer.getAnalyzerEnabledSettingKey();
38              analyzerEnabledSettingKeys.add(enabledKey);
39          }
40  
41  
42          Properties properties = new Properties();
43          Path propertiesPath = Paths.get("src", "main", "resources", "dependencycheck.properties");
44          try (FileInputStream fis = new FileInputStream(propertiesPath.toFile())) {
45              properties.load(fis);
46          }
47  
48          Assert.assertFalse(analyzerEnabledSettingKeys.isEmpty());
49  
50          Set<String> absentKeys = analyzerEnabledSettingKeys.stream()
51                  .filter(key -> !properties.containsKey(key))
52                  .collect(Collectors.toSet());
53  
54          Assert.assertTrue(absentKeys.isEmpty());
55      }

57      public Set<Class<AbstractAnalyzer>> findAllAnalyzerImplementations(String packageName)
58              throws IOException {
59  
60          Set<Class<?>> packageClasses = findAllClasses(packageName);
61  
62          Set<Class<AbstractAnalyzer>> analyzers = new HashSet<>();
63          for (Class<?> clazz : packageClasses) {
64              if (isAnalyzerImplementation(clazz)) {
65                  // We can safely cast due to call to isAnalyzerImplementation()
66                  @SuppressWarnings("unchecked")
67                  Class<AbstractAnalyzer> abstractAnalyzer = (Class<AbstractAnalyzer>) clazz;
68                  analyzers.add(abstractAnalyzer);
69              }
70          }
71  
72          return analyzers;
73      }
74  
75      private boolean isAnalyzerImplementation(Class<?> clazz) {
76          if (isAnAbstractClass(clazz) || isATestAnalyzer(clazz)) {
77              return false;
78          }
79  
80          return AbstractAnalyzer.class.isAssignableFrom(clazz);
81      }
82  
83      private boolean isAnAbstractClass(Class<?> clazz) {
84          return Modifier.isAbstract(clazz.getModifiers());
85      }
86  
87      private boolean isATestAnalyzer(Class<?> clazz) {
88          return clazz == AbstractSuppressionAnalyzerTest.AbstractSuppressionAnalyzerImpl.class;
89      }

91      public Set<Class<?>> findAllClasses(String packageName) throws IOException {
92          String parsedPackageName = packageName.replaceAll("[.]", "/");
93  
94          Set<Class<?>> classes = new HashSet<>();
95          Enumeration<URL> resources = ClassLoader.getSystemClassLoader().getResources(parsedPackageName);
96          while (resources.hasMoreElements()) {
97              URL resource = resources.nextElement();
98              classes.addAll(getClasses(resource, packageName));
99          }
100 
101         return classes;
102     }
103 
104     private Set<Class<?>> getClasses(URL resource, String packageName) throws IOException {
105         if (Objects.nonNull(resource)) {
106             try (InputStream is = resource.openStream();
107                  InputStreamReader isr = new InputStreamReader(is, StandardCharsets.UTF_8);
108                  BufferedReader reader = new BufferedReader(isr)) {
109 
110                 return tryGetClasses(packageName, reader);
111             }
112         }
113 
114         return Collections.emptySet();
115     }
116 
117     private Set<Class<?>> tryGetClasses(String packageName, BufferedReader reader) {
118         return reader.lines()
119                 .filter(line -> line.endsWith(".class"))
120                 .map(line -> getClass(line, packageName))
121                 .collect(Collectors.toSet());
122     }
123 
124     private Class<?> getClass(String className, String packageName) {
125         try {
126             return tryGetClass(className, packageName);
127         } catch (ClassNotFoundException e) {
128             throw new IllegalArgumentException(e);
129         }
130     }
131 
132     private Class<?> tryGetClass(String className, String packageName) throws ClassNotFoundException {
133         return Class.forName(packageName + "." + className.substring(0, className.lastIndexOf('.')));
134     }
