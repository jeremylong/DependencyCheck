52      @Test
53      public void testBuildSearch() throws Exception {
54          Set<String> productWeightings = new HashSet<>();//Collections.singleton("struts2");
55          Set<String> vendorWeightings = new HashSet<>();//Collections.singleton("apache");        
56          Map<String, MutableInt> vendor = new HashMap<>();
57          Map<String, MutableInt> product = new HashMap<>();
58          vendor.put("apache software foundation", new MutableInt(1));
59          product.put("struts 2 core", new MutableInt(1));
60  
61          CPEAnalyzer instance = new CPEAnalyzer();
62          instance.initialize(getSettings());
63          String queryText = instance.buildSearch(vendor, product, vendorWeightings, productWeightings);
64          String expResult = "product:(struts 2 core) AND vendor:(apache software foundation)";
65          assertTrue(expResult.equals(queryText));
66  
67          vendorWeightings.add("apache");
68          productWeightings.add("struts2");
69  
70          queryText = instance.buildSearch(vendor, product, vendorWeightings, productWeightings);
71          expResult = "product:(struts^2 2 core struts2^2) AND vendor:(apache^2 software foundation)";
72          assertTrue(expResult.equals(queryText));
73          instance.close();
74      }

120     /**
121      * Test of determineCPE method, of class CPEAnalyzer.
122      *
123      * @param depName dependency name
124      * @param expResult expected results
125      * @param cpeAnalyzer the CPE analyzer
126      * @param fnAnalyzer the file name analyzer
127      * @param jarAnalyzer the jar analyzer
128      * @param hAnalyzer the hint analyzer
129      * @param fp the false positive analyzer
130      * @param cpeSuppression the CPE suppression analyzer
131      * @throws Exception is thrown when an exception occurs
132      */
133     public void callDetermineCPE_full(String depName, String expResult, CPEAnalyzer cpeAnalyzer, FileNameAnalyzer fnAnalyzer,
134             JarAnalyzer jarAnalyzer, HintAnalyzer hAnalyzer, FalsePositiveAnalyzer fp, CpeSuppressionAnalyzer cpeSuppression) throws Exception {
135 
136         //File file = new File(this.getClass().getClassLoader().getResource(depName).getPath());
137         File file = BaseTest.getResourceAsFile(this, depName);
138 
139         Dependency dep = new Dependency(file);
140 
141         fnAnalyzer.analyze(dep, null);
142         jarAnalyzer.analyze(dep, null);
143         hAnalyzer.analyze(dep, null);
144         cpeAnalyzer.analyze(dep, null);
145         fp.analyze(dep, null);
146         cpeSuppression.analyze(dep, null);
147 
148         if (expResult != null) {
149             boolean found = false;
150             for (Identifier i : dep.getVulnerableSoftwareIdentifiers()) {
151                 if (expResult.equals(i.getValue())) {
152                     found = true;
153                     break;
154                 }
155             }
156             assertTrue("Match not found: { dep:'" + dep.getFileName() + "', exp:'" + expResult + "' }", found);
157         } else {
158             dep.getVulnerableSoftwareIdentifiers().forEach((id) -> fail("Unexpected match found: { dep:'" + dep.getFileName() + "', found:'" + id + "' }"));
159         }
160     }

176             suppressionAnalyzer.prepare(engine);
177 
178             FileNameAnalyzer fnAnalyzer = new FileNameAnalyzer();
179             fnAnalyzer.analyze(struts, engine);
180 
181             HintAnalyzer hintAnalyzer = new HintAnalyzer();
182             hintAnalyzer.initialize(getSettings());
183             hintAnalyzer.prepare(engine);
184             JarAnalyzer jarAnalyzer = new JarAnalyzer();
185             jarAnalyzer.initialize(getSettings());
186             jarAnalyzer.accept(new File("test.jar"));//trick analyzer into "thinking it is active"
187             jarAnalyzer.prepare(engine);
188 
189             jarAnalyzer.analyze(struts, engine);
190             hintAnalyzer.analyze(struts, engine);
191             //File fileCommonValidator = new File(this.getClass().getClassLoader().getResource("commons-validator-1.4.0.jar").getPath());
192             File fileCommonValidator = BaseTest.getResourceAsFile(this, "commons-validator-1.4.0.jar");

194             jarAnalyzer.analyze(commonValidator, engine);
195             hintAnalyzer.analyze(commonValidator, engine);

198             File fileSpring = BaseTest.getResourceAsFile(this, "spring-core-2.5.5.jar");

201             hintAnalyzer.analyze(spring, engine);

203             //File fileSpring3 = new File(this.getClass().getClassLoader().getResource("spring-core-3.0.0.RELEASE.jar").getPath());
204             File fileSpring3 = BaseTest.getResourceAsFile(this, "spring-core-3.0.0.RELEASE.jar");
205             Dependency spring3 = new Dependency(fileSpring3);
206             jarAnalyzer.analyze(spring3, engine);
207             hintAnalyzer.analyze(spring3, engine);
208 
209             CPEAnalyzer instance = new CPEAnalyzer();
210 
211             engine.openDatabase(true, true);
212             instance.initialize(getSettings());
213             instance.prepare(engine);
214             instance.determineCPE(commonValidator);
215             instance.determineCPE(struts);
216             instance.determineCPE(spring);
217             instance.determineCPE(spring3);
218             instance.close();
219 
220             suppressionAnalyzer.analyze(commonValidator, engine);
221             commonValidator.getVulnerableSoftwareIdentifiers().forEach((i) -> fail("Apache Common Validator found an unexpected CPE identifier - " + i.getValue()));

223             String expResult = "cpe:2.3:a:apache:struts:2.1.2:*:*:*:*:*:*:*";
224             assertTrue("Incorrect match size - struts", struts.getVulnerableSoftwareIdentifiers().size() >= 1);
225             boolean found = false;
226             for (Identifier i : struts.getVulnerableSoftwareIdentifiers()) {
227                 if (expResult.equals(i.getValue())) {
228                     found = true;
229                     break;
230                 }
231             }
232             assertTrue("Incorrect match - struts", found);
233             assertTrue("Incorrect match size - spring3 - " + spring3.getVulnerableSoftwareIdentifiers().size(), spring3.getVulnerableSoftwareIdentifiers().size() >= 1);

235             jarAnalyzer.close();

269      * @param product
270      * @param version
271      * @param expectedCpe
272      * @param cpeAnalyzer
273      * @throws Exception
274      */
275     private void callDetermieIdentifiers(String vendor, String product, String version, String expectedCpe, CPEAnalyzer cpeAnalyzer) throws Exception {
276         Dependency dep = new Dependency();
277         dep.addEvidence(EvidenceType.VENDOR, "test", "vendor", vendor, Confidence.HIGHEST);
278         dep.addEvidence(EvidenceType.PRODUCT, "test", "product", product, Confidence.HIGHEST);
279         dep.addEvidence(EvidenceType.VERSION, "test", "version", version, Confidence.HIGHEST);
280 
281         cpeAnalyzer.determineIdentifiers(dep, vendor, product, Confidence.HIGHEST);
282 
283         boolean found = dep.getVulnerableSoftwareIdentifiers().stream().anyMatch(id -> {
284             System.out.println(id.getValue());
285             return expectedCpe.equals(id.getValue());
286         });
287         assertTrue(String.format("%s:%s:%s identifier not found", vendor, product, version), found);

332     private void callAnalyzeDependency(String vendor, String product, String version, String expectedCpe, CPEAnalyzer cpeAnalyzer, Engine engine) throws Exception {
333         Dependency dep = new Dependency(true);
334         dep.addEvidence(EvidenceType.VENDOR, "test", "vendor", vendor, Confidence.HIGHEST);
335         dep.addEvidence(EvidenceType.PRODUCT, "test", "product", product, Confidence.HIGHEST);
336         dep.addEvidence(EvidenceType.VERSION, "test", "version", version, Confidence.HIGHEST);
337         dep.setMd5sum("");
338         dep.setSha1sum("");
339         dep.setSha256sum("");
340 
341         cpeAnalyzer.analyzeDependency(dep, engine);
342 
343         boolean found = dep.getVulnerableSoftwareIdentifiers().stream().anyMatch(id -> {
344             System.out.println(id.getValue());
345             return expectedCpe.equals(id.getValue());
346         });
347         assertTrue(String.format("%s:%s:%s identifier not found", vendor, product, version), found);
348     }

352      *
353      * @throws Exception is thrown when an exception occurs
354      */
355     @Test
356     public void testSearchCPE() throws Exception {
357         Map<String, MutableInt> vendor = new HashMap<>();
358         Map<String, MutableInt> product = new HashMap<>();
359         vendor.put("apache software foundation", new MutableInt(1));
360         product.put("struts 2 core", new MutableInt(1));
361 
362         String expVendor = "apache";
363         String expProduct = "struts";
364 
365         CPEAnalyzer instance = new CPEAnalyzer();
366         try (Engine engine = new Engine(getSettings())) {
367             engine.openDatabase(true, true);
368             instance.initialize(getSettings());
369             instance.prepare(engine);
370 
371             Set<String> productWeightings = Collections.singleton("struts2");
372             Set<String> vendorWeightings = Collections.singleton("apache");
373             List<IndexEntry> result = instance.searchCPE(vendor, product, vendorWeightings, productWeightings, Ecosystem.JAVA);
374 
375             boolean found = false;
376             for (IndexEntry entry : result) {
377                 if (expVendor.equals(entry.getVendor()) && expProduct.equals(entry.getProduct())) {
378                     found = true;
379                     break;
380                 }
381             }
382             assertTrue("apache:struts was not identified", found);
383         }
384         instance.close();
385     }
