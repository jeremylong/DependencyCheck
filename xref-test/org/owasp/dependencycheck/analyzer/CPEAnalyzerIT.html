52      @Test
53      public void testBuildSearch() throws Exception {
54          Set<String> productWeightings = new HashSet<>();//Collections.singleton("struts2");
55          Set<String> vendorWeightings = new HashSet<>();//Collections.singleton("apache");        
56          Map<String, MutableInt> vendor = new HashMap<>();
57          Map<String, MutableInt> product = new HashMap<>();
58          vendor.put("apache software foundation", new MutableInt(1));
59          product.put("struts 2 core", new MutableInt(1));
60  
61          CPEAnalyzer instance = new CPEAnalyzer();
62          instance.initialize(getSettings());
63          String queryText = instance.buildSearch(vendor, product, vendorWeightings, productWeightings);
64          String expResult = "product:(struts 2 core) AND vendor:(apache software foundation)";
65          assertTrue(expResult.equals(queryText));
66  
67          vendorWeightings.add("apache");
68          productWeightings.add("struts2");
69  
70          queryText = instance.buildSearch(vendor, product, vendorWeightings, productWeightings);
71          expResult = "product:(struts^2 2 core struts2^2) AND vendor:(apache^2 software foundation)";
72          assertTrue(expResult.equals(queryText));
73          instance.close();
74      }

122      *
123      * @param depName dependency name
124      * @param expResult expected results
125      * @param cpeAnalyzer the CPE analyzer
126      * @param fnAnalyzer the file name analyzer
127      * @param jarAnalyzer the jar analyzer
128      * @param hAnalyzer the hint analyzer
129      * @param fp the false positive analyzer
130      * @param cpeSuppression the CPE suppression analyzer
131      * @throws Exception is thrown when an exception occurs
132      */
133     public void callDetermineCPE_full(String depName, String expResult, CPEAnalyzer cpeAnalyzer, FileNameAnalyzer fnAnalyzer,
134             JarAnalyzer jarAnalyzer, HintAnalyzer hAnalyzer, FalsePositiveAnalyzer fp, CpeSuppressionAnalyzer cpeSuppression) throws Exception {
135 
136         //File file = new File(this.getClass().getClassLoader().getResource(depName).getPath());
137         File file = BaseTest.getResourceAsFile(this, depName);
138 
139         Dependency dep = new Dependency(file);
140 
141         fnAnalyzer.analyze(dep, null);
142         jarAnalyzer.analyze(dep, null);
143         hAnalyzer.analyze(dep, null);
144         cpeAnalyzer.analyze(dep, null);
145         fp.analyze(dep, null);
146         cpeSuppression.analyze(dep, null);
147 
148         if (expResult != null) {
149             boolean found = false;
150             for (Identifier i : dep.getVulnerableSoftwareIdentifiers()) {
151                 if (expResult.equals(i.getValue())) {
152                     found = true;
153                     break;
154                 }
155             }
156             assertTrue("Match not found: { dep:'" + dep.getFileName() + "', exp:'" + expResult + "' }", found);
157         } else {
158             dep.getVulnerableSoftwareIdentifiers().forEach((id) -> {
159                 fail("Unexpected match found: { dep:'" + dep.getFileName() + "', found:'" + id + "' }");
160             });
161         }
162     }

175

177         suppressionAnalyzer.initialize(getSettings());
178         suppressionAnalyzer.prepare(null);
179 
180         FileNameAnalyzer fnAnalyzer = new FileNameAnalyzer();
181         fnAnalyzer.analyze(struts, null);
182 
183         HintAnalyzer hintAnalyzer = new HintAnalyzer();
184         hintAnalyzer.initialize(getSettings());
185         hintAnalyzer.prepare(null);
186         JarAnalyzer jarAnalyzer = new JarAnalyzer();
187         jarAnalyzer.initialize(getSettings());
188         jarAnalyzer.accept(new File("test.jar"));//trick analyzer into "thinking it is active"
189         jarAnalyzer.prepare(null);
190 
191         jarAnalyzer.analyze(struts, null);
192         hintAnalyzer.analyze(struts, null);
193         //File fileCommonValidator = new File(this.getClass().getClassLoader().getResource("commons-validator-1.4.0.jar").getPath());
194         File fileCommonValidator = BaseTest.getResourceAsFile(this, "commons-validator-1.4.0.jar");
195         Dependency commonValidator = new Dependency(fileCommonValidator);
196         jarAnalyzer.analyze(commonValidator, null);
197         hintAnalyzer.analyze(commonValidator, null);
198 
199         //File fileSpring = new File(this.getClass().getClassLoader().getResource("spring-core-2.5.5.jar").getPath());
200         File fileSpring = BaseTest.getResourceAsFile(this, "spring-core-2.5.5.jar");
201         Dependency spring = new Dependency(fileSpring);
202         jarAnalyzer.analyze(spring, null);
203         hintAnalyzer.analyze(spring, null);
204 
205         //File fileSpring3 = new File(this.getClass().getClassLoader().getResource("spring-core-3.0.0.RELEASE.jar").getPath());
206         File fileSpring3 = BaseTest.getResourceAsFile(this, "spring-core-3.0.0.RELEASE.jar");
207         Dependency spring3 = new Dependency(fileSpring3);
208         jarAnalyzer.analyze(spring3, null);
209         hintAnalyzer.analyze(spring3, null);
210 
211         CPEAnalyzer instance = new CPEAnalyzer();
212         try (Engine engine = new Engine(getSettings())) {
213             engine.openDatabase(true, true);
214             instance.initialize(getSettings());
215             instance.prepare(engine);
216             instance.determineCPE(commonValidator);
217             instance.determineCPE(struts);
218             instance.determineCPE(spring);
219             instance.determineCPE(spring3);
220             instance.close();
221 
222             suppressionAnalyzer.analyze(commonValidator, engine);
223             commonValidator.getVulnerableSoftwareIdentifiers().forEach((i) -> {
224                 fail("Apache Common Validator found an unexpected CPE identifier - " + i.getValue());
225             });
226 
227             String expResult = "cpe:2.3:a:apache:struts:2.1.2:*:*:*:*:*:*:*";
228             assertTrue("Incorrect match size - struts", struts.getVulnerableSoftwareIdentifiers().size() >= 1);
229             boolean found = false;
230             for (Identifier i : struts.getVulnerableSoftwareIdentifiers()) {
231                 if (expResult.equals(i.getValue())) {
232                     found = true;
233                     break;
234                 }
235             }
236             assertTrue("Incorrect match - struts", found);
237             assertTrue("Incorrect match size - spring3 - " + spring3.getVulnerableSoftwareIdentifiers().size(), spring3.getVulnerableSoftwareIdentifiers().size() >= 1);
238 
239             jarAnalyzer.close();
240             suppressionAnalyzer.close();
241         }

271      *

273      * @param product
274      * @param version
275      * @param expectedCpe
276      * @param cpeAnalyzer

278      */
279     private void callDetermieIdentifiers(String vendor, String product, String version, String expectedCpe, CPEAnalyzer cpeAnalyzer) throws Exception {
280         Dependency dep = new Dependency();
281         dep.addEvidence(EvidenceType.VENDOR, "test", "vendor", vendor, Confidence.HIGHEST);
282         dep.addEvidence(EvidenceType.PRODUCT, "test", "product", product, Confidence.HIGHEST);
283         dep.addEvidence(EvidenceType.VERSION, "test", "version", version, Confidence.HIGHEST);
284 
285         cpeAnalyzer.determineIdentifiers(dep, vendor, product, Confidence.HIGHEST);
286 
287         boolean found = dep.getVulnerableSoftwareIdentifiers().stream().anyMatch(id -> {
288             System.out.println(id.getValue());
289             return expectedCpe.equals(id.getValue());
290         });
291         assertTrue(String.format("%s:%s:%s identifier not found", vendor, product, version), found);

342         dep.setSha1sum("");
343         dep.setSha256sum("");
344 
345         cpeAnalyzer.analyzeDependency(dep, engine);
346 
347         boolean found = dep.getVulnerableSoftwareIdentifiers().stream().anyMatch(id -> {
348             System.out.println(id.getValue());
349             return expectedCpe.equals(id.getValue());
350         });
351         assertTrue(String.format("%s:%s:%s identifier not found", vendor, product, version), found);

356      *
357      * @throws Exception is thrown when an exception occurs

359     @Test
360     public void testSearchCPE() throws Exception {
361         Map<String, MutableInt> vendor = new HashMap<>();
362         Map<String, MutableInt> product = new HashMap<>();
363         vendor.put("apache software foundation", new MutableInt(1));
364         product.put("struts 2 core", new MutableInt(1));
365 
366         String expVendor = "apache";
367         String expProduct = "struts";
368 
369         CPEAnalyzer instance = new CPEAnalyzer();
370         try (Engine engine = new Engine(getSettings())) {
371             engine.openDatabase(true, true);
372             instance.initialize(getSettings());
373             instance.prepare(engine);
374 
375             Set<String> productWeightings = Collections.singleton("struts2");
376             Set<String> vendorWeightings = Collections.singleton("apache");
377             List<IndexEntry> result = instance.searchCPE(vendor, product, vendorWeightings, productWeightings, Ecosystem.JAVA);
378 
379             boolean found = false;
380             for (IndexEntry entry : result) {
381                 if (expVendor.equals(entry.getVendor()) && expProduct.equals(entry.getProduct())) {
382                     found = true;
383                     break;
384                 }
385             }
386             assertTrue("apache:struts was not identified", found);
387         }
388         instance.close();
389     }
