109     /**
110      * Test Ruby BundlerAudit analysis.
111      *
112      * @throws AnalysisException is thrown when an exception occurs.
113      */
114     @Test
115     public void testAnalysis() throws AnalysisException, DatabaseException {
116         try (Engine engine = new Engine(getSettings())) {
117             engine.openDatabase();
118             analyzer.prepare(engine);
119             final String resource = "ruby/vulnerable/gems/rails-4.1.15/Gemfile.lock";
120             final Dependency result = new Dependency(BaseTest.getResourceAsFile(this, resource));
121             analyzer.analyze(result, engine);
122             final Dependency[] dependencies = engine.getDependencies();
123             final int size = dependencies.length;
124             assertTrue(size >= 1);
125             boolean found = false;
126             for (Dependency dependency : dependencies) {
127                 found = dependency.getEvidence(EvidenceType.PRODUCT).toString().toLowerCase().contains("redcarpet");
128                 found &= dependency.getEvidence(EvidenceType.VERSION).toString().toLowerCase().contains("2.2.2");
129                 found &= dependency.getFilePath().endsWith(resource);
130                 found &= dependency.getFileName().equals("Gemfile.lock");
131                 if (found) {
132                     break;
133                 }
134             }
135             assertTrue("redcarpet was not identified", found);
136 
137         } catch (InitializationException | DatabaseException | AnalysisException e) {
138             LOGGER.warn("Exception setting up RubyBundleAuditAnalyzer. Make sure Ruby gem bundle-audit is installed. You may also need to set property \"analyzer.bundle.audit.path\".");
139             Assume.assumeNoException("Exception setting up RubyBundleAuditAnalyzer; bundle audit may not be installed, or property \"analyzer.bundle.audit.path\" may not be set.", e);
140         }
141     }

143     /**
144      * Test Ruby addCriticalityToVulnerability
145      */
146     @Test
147     public void testAddCriticalityToVulnerability() throws AnalysisException, DatabaseException {
148         try (Engine engine = new Engine(getSettings())) {
149             engine.doUpdates(true);
150             analyzer.prepare(engine);
151 
152             final Dependency result = new Dependency(BaseTest.getResourceAsFile(this,
153                     "ruby/vulnerable/gems/sinatra/Gemfile.lock"));
154             analyzer.analyze(result, engine);
155             Dependency dependency = engine.getDependencies()[0];
156             boolean found =false;
157             for (Vulnerability vulnerability : dependency.getVulnerabilities()) {
158                 if ("CVE-2015-3225".equals(vulnerability.getName())) {
159                     found = true;
160                     // validate that the score is from NVD rather than translated from the Bundle Audit severity text
161                     assertEquals(5.0, vulnerability.getCvssV2().getCvssData().getBaseScore(), 0.0);
162                     break;
163                 }
164             }
165             assertTrue("CVE-2015-3225 was not found among the vulnerabilities",found);
166         } catch (InitializationException | DatabaseException | AnalysisException | UpdateException e) {
167             LOGGER.warn("Exception setting up RubyBundleAuditAnalyzer. Make sure Ruby gem bundle-audit is installed. You may also need to set property \"analyzer.bundle.audit.path\".");
168             Assume.assumeNoException("Exception setting up RubyBundleAuditAnalyzer; bundle audit may not be installed, or property \"analyzer.bundle.audit.path\" may not be set.", e);
169         }
170     }

172     /**
173      * Test when Ruby bundle-audit is not available on the system.
174      *
175      * @throws AnalysisException is thrown when an exception occurs.
176      */
177     @Test
178     public void testInvalidBundleAudit() throws AnalysisException, DatabaseException {
179 
180         String path = BaseTest.getResourceAsFile(this, "ruby/invalid-bundle-audit").getAbsolutePath();
181         getSettings().setString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_PATH, path);
182         analyzer.initialize(getSettings());
183         try {
184             //initialize should fail.
185             analyzer.prepare(null);
186         } catch (InitializationException e) {
187             //expected, so ignore.
188             assertNotNull(e);
189         } finally {
190             assertThat("`invalid-bundle-audit` is not a valid executable. Ruby Bundle Audit Analyzer is disabled as expected.",
191                     analyzer.isEnabled(), is(false));
192         }
193     }

195     /**
196      * Test Ruby dependencies and their paths.
197      *
198      * @throws AnalysisException is thrown when an exception occurs.
199      * @throws DatabaseException thrown when an exception occurs
200      */
201     @Test
202     public void testDependenciesPath() throws AnalysisException, DatabaseException {
203         try (Engine engine = new Engine(getSettings())) {
204             try {
205                 engine.scan(BaseTest.getResourceAsFile(this, "ruby/vulnerable/gems/rails-4.1.15/"));
206                 engine.analyzeDependencies();
207             } catch (NullPointerException ex) {
208                 LOGGER.error("NPE", ex);
209                 fail(ex.getMessage());
210             } catch (ExceptionCollection ex) {
211                 Assume.assumeNoException("Exception setting up RubyBundleAuditAnalyzer; bundle audit may not be installed, or property \"analyzer.bundle.audit.path\" may not be set.", ex);
212                 return;
213             }
214             Dependency[] dependencies = engine.getDependencies();
215             LOGGER.info("{} dependencies found.", dependencies.length);
216         }
