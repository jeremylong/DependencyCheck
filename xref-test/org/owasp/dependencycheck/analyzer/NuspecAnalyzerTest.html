18  package org.owasp.dependencycheck.analyzer;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.junit.Assert.assertFalse;
22  import static org.junit.Assert.assertTrue;
23  import org.junit.Before;
24  import org.junit.Test;
25  import org.owasp.dependencycheck.BaseTest;
26  import org.owasp.dependencycheck.analyzer.exception.AnalysisException;
27  import org.owasp.dependencycheck.dependency.Dependency;
28  import org.owasp.dependencycheck.dependency.Evidence;
29  
30  import java.io.File;
31  import org.owasp.dependencycheck.dependency.EvidenceType;
32  
33  public class NuspecAnalyzerTest extends BaseTest {
34  
35      private NuspecAnalyzer instance;
36  
37      @Before
38      @Override
39      public void setUp() throws Exception {
40          super.setUp();
41          instance = new NuspecAnalyzer();
42          instance.initialize(getSettings());
43          instance.prepare(null);
44          instance.setEnabled(true);
45      }
46  
47      @Test
48      public void testGetAnalyzerName() {
49          assertEquals("Nuspec Analyzer", instance.getName());
50      }
51  
52      @Test
53      public void testSupportsFileExtensions() {
54          assertTrue(instance.accept(new File("test.nuspec")));
55          assertFalse(instance.accept(new File("test.nupkg")));
56      }
57  
58      @Test
59      public void testGetAnalysisPhaze() {
60          assertEquals(AnalysisPhase.INFORMATION_COLLECTION, instance.getAnalysisPhase());
61      }
62  
63      @Test
64      public void testNuspecAnalysis() throws Exception {
65  
66          File file = BaseTest.getResourceAsFile(this, "nuspec/test.nuspec");
67          Dependency result = new Dependency(file);
68          instance.analyze(result, null);
69  
70          assertEquals(NuspecAnalyzer.DEPENDENCY_ECOSYSTEM, result.getEcosystem());
71  
72          //checking the owner field
73          assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("bobsmack"));
74  
75          //checking the author field
76          assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("brianfox"));
77  
78          //checking the id field
79          assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("TestDepCheck"));
80  
81          //checking the title field
82          assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("Test Package"));
83  
84          assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("1.0.0"));
85          assertEquals("1.0.0", result.getVersion());
86          assertEquals("TestDepCheck", result.getName());
87          assertEquals("TestDepCheck:1.0.0", result.getDisplayFileName());
88      }
89  }
