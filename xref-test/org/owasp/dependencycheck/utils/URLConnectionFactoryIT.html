18  package org.owasp.dependencycheck.utils;

20  import io.netty.handler.codec.http.HttpResponseStatus;
21  import java.io.IOException;
22  import java.net.HttpURLConnection;
23  import java.net.URL;
24  import org.junit.Test;
25  import static org.junit.Assert.*;
26  import org.junit.Before;
27  import org.junit.Rule;
28  import org.mockserver.client.MockServerClient;
29  import org.mockserver.junit.MockServerRule;
30  import org.mockserver.matchers.Times;
31  import org.mockserver.model.HttpRequest;
32  import org.mockserver.model.HttpResponse;
33  
34  /**
35   *
36   * @author Jeremy Long
37   */
38  public class URLConnectionFactoryIT extends BaseTest {
39  
40      @Rule
41      public MockServerRule mockServerRule = new MockServerRule(this);
42  
43      private MockServerClient mockServerClient;
44  
45      @Before
46      public void reset() {
47          mockServerClient.reset();
48      }
49  
50      /**
51       * Test of createHttpURLConnection method, of class URLConnectionFactory to
52       * validate if a basic authorization header is added.
53       */
54      @Test
55      public void testCreateHttpURLConnection_Authorization_unauthorized() throws Exception {
56          mockServerClient.when(HttpRequest.request().withMethod("GET")
57                  .withHeader("Authorization", "Basic dXNlcm5hbWU6cGFzc3dvcmQ=")
58                  .withPath("/secure/resource.xml"), Times.once())
59                  .respond(HttpResponse.response().withBody("ok").withStatusCode(200));
60          mockServerClient.when(HttpRequest.request().withMethod("GET")
61                  .withPath("/secure/resource.xml"), Times.once())
62                  .respond(HttpResponse.response().withBody("Unauthorized").withStatusCode(401));
63  
64          URL url = new URL("http://localhost:"
65                  + mockServerClient.remoteAddress().getPort()
66                  + "/secure/resource.xml");
67          URLConnectionFactory instance = new URLConnectionFactory(getSettings());
68          HttpURLConnection conn = instance.createHttpURLConnection(url);
69          try {
70              conn.connect();
71          } catch (IOException ex) {
72  
73          }
74          assertEquals(HttpResponseStatus.UNAUTHORIZED.code(), conn.getResponseCode());
75          conn.disconnect();
76      }
77  
78      /**
79       * Test of createHttpURLConnection method, of class URLConnectionFactory to
80       * validate if a basic authorization header is added.
81       */
82      @Test
83      public void testCreateHttpURLConnection_Authorization() throws Exception {
84          mockServerClient.when(HttpRequest.request().withMethod("GET")
85                  .withHeader("Authorization", "Basic dXNlcm5hbWU6cGFzc3dvcmQ=")
86                  .withPath("/secure/resource.xml"), Times.once())
87                  .respond(HttpResponse.response().withBody("ok").withStatusCode(200));
88          mockServerClient.when(HttpRequest.request().withMethod("GET")
89                  .withPath("/secure/resource.xml"), Times.once())
90                  .respond(HttpResponse.response().withBody("Unauthorized").withStatusCode(401));
91  
92          URL url = new URL("http://username:password@localhost:"
93                  + mockServerClient.remoteAddress().getPort()
94                  + "/secure/resource.xml");
95          URLConnectionFactory instance = new URLConnectionFactory(getSettings());
96          HttpURLConnection conn = instance.createHttpURLConnection(url);
97          try {
98              conn.connect();
99          } catch (IOException ex) {
100 
101         }
102         assertEquals(HttpResponseStatus.OK.code(), conn.getResponseCode());
103         conn.disconnect();
104     }
105 }
