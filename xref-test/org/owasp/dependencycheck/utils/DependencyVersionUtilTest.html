33       * Test of parseVersion method, of class DependencyVersionUtil.
34       */
35      @Test
36      public void testParseVersion_String() {
37          final String[] fileName = {"openssl1.0.1c", "something-0.9.5.jar", "lib2-1.1.jar", "lib1.5r4-someflag-R26.jar",
38              "lib-1.2.5-dev-20050313.jar", "testlib_V4.4.0.jar", "lib-core-2.0.0-RC1-SNAPSHOT.jar",
39              "lib-jsp-2.0.1_R114940.jar", "dev-api-2.3.11_R121413.jar", "lib-api-3.7-SNAPSHOT.jar",
40              "-", "", "1.3-beta", "6", "jsf-impl-2.2.8-02.jar",
41              "plone.rfc822-1.1.1-py2-none-any.whl"};
42          final String[] expResult = {"1.0.1c", "0.9.5", "1.1", "1.5.r4", "1.2.5.dev-20050313", "4.4.0", "2.0.0.rc1",
43              "2.0.1.r114940", "2.3.11.r121413", "3.7.snapshot", "-", null, "1.3.beta", "6",
44              "2.2.8.02", "1.1.1"};
45  
46          for (int i = 0; i < fileName.length; i++) {
47              final DependencyVersion version = DependencyVersionUtil.parseVersion(fileName[i]);
48              String result = null;
49              if (version != null) {
50                  result = version.toString();
51              }
52              assertEquals("Failed extraction on \"" + fileName[i] + "\".", expResult[i], result);
53          }
54  
55          String[] failingNames = {"no-version-identified.jar", "somelib-04aug2000r7-dev.jar", /*"no.version15.jar",*/
56              "lib_1.0_spec-1.1.jar", "lib-api_1.0_spec-1.0.1.jar"};
57          for (String failingName : failingNames) {
58              final DependencyVersion version = DependencyVersionUtil.parseVersion(failingName);
59              assertNull("Found version in name that should have failed \"" + failingName + "\".", version);
60          }
61      }

63      /**
64       * Test of parseVersion method, of class DependencyVersionUtil.
65       */
66      @Test
67      public void testParseVersion_String_boolean() {
68          //cpe:/a:playframework:play_framework:2.1.1:rc1-2.9.x-backport
69          String text = "2.1.1.rc1.2.9.x-backport";
70          boolean firstMatchOnly = false;
71          DependencyVersion expResult;
72          DependencyVersion result = DependencyVersionUtil.parseVersion(text, firstMatchOnly);
73          assertNull(result);
74          firstMatchOnly = true;
75          expResult = DependencyVersionUtil.parseVersion("2.1.1.rc1");
76          result = DependencyVersionUtil.parseVersion(text, firstMatchOnly);
77          assertEquals(expResult, result);
78  
79          result = DependencyVersionUtil.parseVersion("1.0.0-RC", firstMatchOnly);
80          assertEquals(4, result.getVersionParts().size());
81          assertEquals("rc", result.getVersionParts().get(3));
82  
83          result = DependencyVersionUtil.parseVersion("1.0.0-RC2", firstMatchOnly);
84          assertEquals(4, result.getVersionParts().size());
85          assertEquals("rc2", result.getVersionParts().get(3));
86      }
