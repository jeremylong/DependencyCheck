197     /**
198      * Test boundary conditions of the buffer window
199      *
200      * @throws Exception because one might happen
201      */
202     @Test()
203     public void testRead_3args() throws Exception {
204         byte[] input = new byte[2048];
205         Arrays.fill(input, (byte) ' ');
206         input[0] = '{';
207         input[2047] = '}';
208         byte[] results = new byte[2050];
209         byte[] expected = new byte[2050];
210         Arrays.fill(expected, (byte) ' ');
211         expected[0] = '[';
212         expected[1] = '{';
213         expected[2048] = '}';
214         expected[2049] = ']';
215         try (InputStream sample = new ByteArrayInputStream(input);
216                 JsonArrayFixingInputStream instance = new JsonArrayFixingInputStream(sample)) {
217             int read = 0;
218             int pos = 0;
219             while (read >= 0) {
220                 read = instance.read(results, pos, 2050 - pos);
221                 pos += read;
222             }
223             Assert.assertArrayEquals(expected, results);
224         }
225     }
