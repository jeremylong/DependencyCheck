18  package org.owasp.dependencycheck.utils;
19  
20  import java.io.File;
21  import java.net.URL;
22  import org.junit.Test;
23  import static org.junit.Assert.assertTrue;
24  import org.junit.Before;
25  
26  /**
27   *
28   * @author Jeremy Long
29   */
30  public class DownloaderIT extends BaseTest {
31  
32      /**
33       * Initialize the {@link Settings}.
34       */
35      @Before
36      @Override
37      public void setUp() {
38          super.setUp();
39          applyHackFix();
40      }
41  
42      /**
43       * Test of fetchFile method, of class Downloader.
44       *
45       * @throws Exception thrown when an exception occurs.
46       */
47      @Test
48      public void testFetchFile() throws Exception {
49          final String str = getSettings().getString(Settings.KEYS.ENGINE_VERSION_CHECK_URL, "https://jeremylong.github.io/DependencyCheck/current.txt");
50          URL url = new URL(str);
51          File outputPath = new File("target/current.txt");
52          Downloader downloader = new Downloader(getSettings());
53          downloader.fetchFile(url, outputPath);
54          assertTrue(outputPath.isFile());
55      }
56  
57      /**
58       * Upgrading to org.mock-server:mockserver-netty:5.8.0 caused this test case
59       * to fail as netty does not allow TLSv1.3 to be "used" in Java 1.8. Under
60       * 1.8 for some reason `https.protocols` includes TLSv1.3 even though it is
61       * not supported in most implementations. Thus, we need to explicitly remove
62       * it for this test case to work.
63       */
64      private void applyHackFix() {
65          String httpProtocols = System.getProperty("https.protocols");
66          if (httpProtocols != null && httpProtocols.contains(",TLSv1.3")) {
67              httpProtocols = httpProtocols.replace(",TLSv1.3", "");
68              System.setProperty("https.protocols", httpProtocols);
69          } else if (httpProtocols != null && httpProtocols.contains("TLSv1.3,")) {
70              httpProtocols = httpProtocols.replace("TLSv1.3,", "");
71              System.setProperty("https.protocols", httpProtocols);
72          }
73      }
74  }
