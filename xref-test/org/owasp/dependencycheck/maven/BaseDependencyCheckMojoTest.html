58      /**
59       * Checks if the test can be run. The test in this class fail, presumable
60       * due to jmockit, if the JDK is 1.8+.
61       *
62       * @return true if the JDK is below 1.8.
63       */
64      public boolean canRun() {
65          String version = System.getProperty("java.version");
66          int firstDot = version.indexOf('.');
67          if (firstDot < 0) {
68              // new java.version format, so Java 9 or above
69              return false;
70          }
71          int secondDot = version.indexOf('.', firstDot+1);
72          if (secondDot < 0) {
73              // new java.version format, so Java 9 or above
74              return false;
75          }
76          version = version.substring(0, secondDot);
77  
78          double v = Double.parseDouble(version);
79          return v == 1.7;
80      }

82      /**
83       * Test of scanArtifacts method, of class BaseDependencyCheckMojo.
84       */
85      @Test
86      public void testScanArtifacts() throws DatabaseException, InvalidSettingException {
87          new MockUp<MavenProject>() {
88              @Mock
89              public Set<Artifact> getArtifacts() {
90                  Set<Artifact> artifacts = new HashSet<>();
91                  Artifact a = new ArtifactStub();
92                  try {
93                      File file = new File(Test.class.getProtectionDomain().getCodeSource().getLocation().toURI());
94                      a.setFile(file);
95                      artifacts.add(a);
96                  } catch (URISyntaxException ex) {
97                      Logger.getLogger(BaseDependencyCheckMojoTest.class.getName()).log(Level.SEVERE, null, ex);
98                  }
99                  //File file = new File(this.getClass().getClassLoader().getResource("daytrader-ear-2.1.7.ear").getPath());
100 
101                 return artifacts;
102             }
103 
104             @SuppressWarnings("SameReturnValue")
105             @Mock
106             public String getName() {
107                 return "test-project";
108             }
109         };
110 
111         if (canRun()) {
112             boolean autoUpdate = getSettings().getBoolean(Settings.KEYS.AUTO_UPDATE);
113             getSettings().setBoolean(Settings.KEYS.AUTO_UPDATE, false);
114             try (Engine engine = new Engine(getSettings())) {
115                 getSettings().setBoolean(Settings.KEYS.AUTO_UPDATE, autoUpdate);
116 
117                 assertTrue(engine.getDependencies().length == 0);
118                 BaseDependencyCheckMojoImpl instance = new BaseDependencyCheckMojoImpl();
119                 ExceptionCollection exCol = null;
120                 try { //the mock above fails under some JDKs
121                     exCol = instance.scanArtifacts(project, engine);
122                 } catch (NullPointerException ex) {
123                     Assume.assumeNoException(ex);
124                 }
125                 assertNull(exCol);
126                 assertFalse(engine.getDependencies().length == 0);
127             }
128         }
129     }
