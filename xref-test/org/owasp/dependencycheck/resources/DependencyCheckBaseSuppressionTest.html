19  public class DependencyCheckBaseSuppressionTest {
20  
21      @Test
22      public void testAllSuppressionsHaveBaseAttribute() throws ParserConfigurationException, SAXException, IOException {
23          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
24          factory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
25  
26          Path path = Paths.get("src", "main", "resources", "dependencycheck-base-suppression.xml");
27          Document document = factory.newDocumentBuilder().parse(path.toFile());
28          document.getDocumentElement().normalize();
29  
30          NodeList nodes = document.getElementsByTagName("suppress");
31  
32          int numberOfSuppressTagsWithoutBaseTrueAttribute = 0;
33          for (int i = 0; i < nodes.getLength(); i++) {
34              Node node = nodes.item(i);
35  
36              if (node.getNodeType() == Node.ELEMENT_NODE) {
37                  Element element = (Element) node;
38  
39                  if (!"true".equalsIgnoreCase(element.getAttribute("base"))) {
40                      numberOfSuppressTagsWithoutBaseTrueAttribute++;
41                  }
42              }
43          }
44  
45          Assert.assertEquals(0, numberOfSuppressTagsWithoutBaseTrueAttribute);
46      }
47  }
