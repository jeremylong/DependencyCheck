20  public class DependencyCheckBaseSuppressionTest {
21  
22      @Test
23      public void testAllSuppressionsHaveBaseAttribute() throws ParserConfigurationException, SAXException, IOException {
24          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
25          factory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
26  
27          Path path = Paths.get("src", "main", "resources", "dependencycheck-base-suppression.xml");
28          File file = path.toFile();
29          boolean e = file.exists();
30          Document document = factory.newDocumentBuilder().parse(file);
31          document.getDocumentElement().normalize();
32  
33          NodeList nodes = document.getElementsByTagName("suppress");
34  
35          int numberOfSuppressTagsWithoutBaseTrueAttribute = 0;
36          for (int i = 0; i < nodes.getLength(); i++) {
37              Node node = nodes.item(i);
38  
39              if (node.getNodeType() == Node.ELEMENT_NODE) {
40                  Element element = (Element) node;
41  
42                  if (!"true".equalsIgnoreCase(element.getAttribute("base"))) {
43                      numberOfSuppressTagsWithoutBaseTrueAttribute++;
44                  }
45              }
46          }
47  
48          Assert.assertEquals(0, numberOfSuppressTagsWithoutBaseTrueAttribute);
49      }
