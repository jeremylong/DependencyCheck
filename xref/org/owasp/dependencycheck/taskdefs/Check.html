482     /**
483      * If this is a reference, this method will add the referenced resource
484      * collection to the collection of paths.
485      *
486      * @throws BuildException if the reference is not to a resource collection
487      */
488     //declaring a throw that extends runtime exception may be a bad practice
489     //but seems to be an ingrained practice within Ant as even the base `Task`
490     //contains an `execute() throws BuildExecption`.
491     @SuppressWarnings("squid:RedundantThrowsDeclarationCheck")
492     private void dealWithReferences() throws BuildException {
493         if (isReference()) {
494             final Object o = refId.getReferencedObject(getProject());
495             if (!(o instanceof ResourceCollection)) {
496                 throw new BuildException("refId '" + refId.getRefId()
497                         + "' does not refer to a resource collection.");
498             }
499             getPath().add((ResourceCollection) o);
500         }
501     }

1892     //see note on `dealWithReferences()` for information on this suppression
1893     @SuppressWarnings("squid:RedundantThrowsDeclarationCheck")
1894     @Override
1895     protected void executeWithContextClassloader() throws BuildException {
1896         dealWithReferences();
1897         validateConfiguration();
1898         populateSettings();
1899         try (Engine engine = new Engine(Check.class.getClassLoader(), getSettings())) {
1900             for (Resource resource : getPath()) {
1901                 final FileProvider provider = resource.as(FileProvider.class);
1902                 if (provider != null) {
1903                     final File file = provider.getFile();
1904                     if (file != null && file.exists()) {
1905                         engine.scan(file);
1906                     }
1907                 }
1908             }
1909             final ExceptionCollection exceptions = callExecuteAnalysis(engine);
1910             if (exceptions == null || !exceptions.isFatal()) {
1911                 for (String format : getReportFormats()) {
1912                     engine.writeReports(getProjectName(), new File(reportOutputDirectory), format, exceptions);
1913                 }
1914                 if (this.failBuildOnCVSS <= 10) {
1915                     checkForFailure(engine.getDependencies());
1916                 }
1917                 if (this.showSummary) {
1918                     DependencyCheckScanAgent.showSummary(engine.getDependencies());
1919                 }
1920             }
1921         } catch (DatabaseException ex) {
1922             final String msg = "Unable to connect to the dependency-check database; analysis has stopped";
1923             if (this.isFailOnError()) {
1924                 throw new BuildException(msg, ex);
1925             }
1926             log(msg, ex, Project.MSG_ERR);
1927         } catch (ReportException ex) {
1928             final String msg = "Unable to generate the dependency-check report";
1929             if (this.isFailOnError()) {
1930                 throw new BuildException(msg, ex);
1931             }
1932             log(msg, ex, Project.MSG_ERR);
1933         } finally {
1934             getSettings().cleanup();
1935         }
1936     }

2067     /**
2068      * Checks to see if a vulnerability has been identified with a CVSS score
2069      * that is above the threshold set in the configuration.
2070      *
2071      * @param dependencies the list of dependency objects
2072      * @throws BuildException thrown if a CVSS score is found that is higher
2073      * than the threshold set
2074      */
2075     //see note on `dealWithReferences()` for information on this suppression
2076     @SuppressWarnings("squid:RedundantThrowsDeclarationCheck")
2077     private void checkForFailure(Dependency[] dependencies) throws BuildException {
2078         final StringBuilder ids = new StringBuilder();
2079         for (Dependency d : dependencies) {
2080             boolean addName = true;
2081             for (Vulnerability v : d.getVulnerabilities()) {
2082                 if ((v.getCvssV2() != null && v.getCvssV2().getScore() >= failBuildOnCVSS)
2083                         || (v.getCvssV3() != null && v.getCvssV3().getBaseScore() >= failBuildOnCVSS)
2084                         || (v.getUnscoredSeverity() != null && SeverityUtil.estimateCvssV2(v.getUnscoredSeverity()) >= failBuildOnCVSS)
2085                         //safety net to fail on any if for some reason the above misses on 0
2086                         || (failBuildOnCVSS <= 0.0f)) {
2087                     if (addName) {
2088                         addName = false;
2089                         ids.append(NEW_LINE).append(d.getFileName()).append(": ");
2090                         ids.append(v.getName());
2091                     } else {
2092                         ids.append(", ").append(v.getName());
2093                     }
2094                 }
2095             }
2096         }
2097         if (ids.length() > 0) {
2098             final String msg;
2099             if (showSummary) {
2100                 msg = String.format("%n%nDependency-Check Failure:%n"
2101                         + "One or more dependencies were identified with vulnerabilities that have a CVSS score greater than or equal to '%.1f': %s%n"
2102                         + "See the dependency-check report for more details.%n%n", failBuildOnCVSS, ids);
2103             } else {
2104                 msg = String.format("%n%nDependency-Check Failure:%n"
2105                         + "One or more dependencies were identified with vulnerabilities.%n%n"
2106                         + "See the dependency-check report for more details.%n%n");
2107             }
2108             throw new BuildException(msg);
2109         }
2110     }
