20  import java.io.File;
21  import java.io.IOException;
22  import java.nio.charset.Charset;
23  import java.nio.charset.StandardCharsets;
24  import java.nio.charset.UnsupportedCharsetException;
25  import java.security.MessageDigest;
26  import java.security.NoSuchAlgorithmException;
27  import java.util.HashMap;
28  import java.util.Map;
29  import java.util.concurrent.ConcurrentHashMap;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  import com.google.common.hash.HashCode;
35  import com.google.common.hash.HashFunction;
36  import com.google.common.hash.Hashing;
37  import com.google.common.io.Files;

39  /**
40   * Includes methods to generate the MD5 and SHA1 checksum.
41   *
42   * @author Jeremy Long
43   * @version $Id: $Id
44   */
45  public final class Checksum {
46  
47      /**
48       * Hex code characters used in getHex.
49       */
50      private static final String HEXES = "0123456789abcdef";
51  
52      /**
53       * The logger.
54       */
55      private static final Logger LOGGER = LoggerFactory.getLogger(Checksum.class);
56      /**
57       * MD5 constant.
58       */
59      private static final String MD5 = "MD5";
60      /**
61       * SHA1 constant.
62       */
63      private static final String SHA1 = "SHA1";
64      /**
65       * SHA256 constant.
66       */
67      private static final String SHA256 = "SHA-256";
68      /**
69       * Cached file checksums for each supported algorithm.
70       */
71      private static final Map<String, Map<File, byte[]>> CHECKSUM_CACHES = new HashMap<>(3);
72  
73      static {
74          CHECKSUM_CACHES.put(MD5, new ConcurrentHashMap<>());
75          CHECKSUM_CACHES.put(SHA256, new ConcurrentHashMap<>());
76          CHECKSUM_CACHES.put(SHA1, new ConcurrentHashMap<>());
77      }
78  
79      /**
80       * Private constructor for a utility class.
81       */
82      private Checksum() {
83      }
84  
85      /**
86       * <p>
87       * Creates the cryptographic checksum of a given file using the specified
88       * algorithm.</p>
89       *
90       * @param algorithm the algorithm to use to calculate the checksum
91       * @param file the file to calculate the checksum for
92       * @return the checksum
93       * @throws java.io.IOException when the file does not exist
94       * @throws java.security.NoSuchAlgorithmException when an algorithm is
95       * specified that does not exist
96       */
97      public static byte[] getChecksum(String algorithm, File file) throws NoSuchAlgorithmException, IOException {
98          final Map<File, byte[]> checksumCache = CHECKSUM_CACHES.get(algorithm.toUpperCase());
99          if (checksumCache == null) {
100             throw new NoSuchAlgorithmException(algorithm);
101         }
102         byte[] checksum = checksumCache.get(file);
103         if (checksum == null) {
104             HashFunction hashFunction = null;
105             switch (algorithm.toUpperCase()) {
106                 case MD5:
107                     hashFunction = Hashing.md5();
108                     break;
109                 case SHA1:
110                     hashFunction = Hashing.sha1();
111                     break;
112                 case SHA256:
113                     hashFunction = Hashing.sha256();
114                     break;
115                 default:
116                     throw new NoSuchAlgorithmException(algorithm);
117             }
118             final HashCode hash = Files.asByteSource(file).hash(hashFunction);
119             checksum = hash.asBytes();
120             checksumCache.put(file, checksum);
121         }
122         return checksum;
123     }
124 
125     /**
126      * Calculates the MD5 checksum of a specified file.
127      *
128      * @param file the file to generate the MD5 checksum
129      * @return the hex representation of the MD5 hash
130      * @throws java.io.IOException when the file passed in does not exist
131      * @throws java.security.NoSuchAlgorithmException when the MD5 algorithm is
132      * not available
133      */
134     public static String getMD5Checksum(File file) throws IOException, NoSuchAlgorithmException {
135         final byte[] b = getChecksum(MD5, file);
136         return getHex(b);
137     }
138 
139     /**
140      * Calculates the SHA1 checksum of a specified file.
141      *
142      * @param file the file to generate the MD5 checksum
143      * @return the hex representation of the SHA1 hash
144      * @throws java.io.IOException when the file passed in does not exist
145      * @throws java.security.NoSuchAlgorithmException when the SHA1 algorithm is
146      * not available
147      */
148     public static String getSHA1Checksum(File file) throws IOException, NoSuchAlgorithmException {
149         final byte[] b = getChecksum(SHA1, file);
150         return getHex(b);
151     }
152 
153     /**
154      * Calculates the SH256 checksum of a specified file.
155      *
156      * @param file the file to generate the MD5 checksum
157      * @return the hex representation of the SHA1 hash
158      * @throws java.io.IOException when the file passed in does not exist
159      * @throws java.security.NoSuchAlgorithmException when the SHA1 algorithm is
160      * not available
161      */
162     public static String getSHA256Checksum(File file) throws IOException, NoSuchAlgorithmException {
163         final byte[] b = getChecksum(SHA256, file);
164         return getHex(b);
165     }
166 
167     /**
168      * Calculates the MD5 checksum of a specified bytes.
169      *
170      * @param algorithm the algorithm to use (md5, sha1, etc.) to calculate the
171      * message digest
172      * @param bytes the bytes to generate the MD5 checksum
173      * @return the hex representation of the MD5 hash
174      */
175     public static String getChecksum(String algorithm, byte[] bytes) {
176         final MessageDigest digest = getMessageDigest(algorithm);
177         final byte[] b = digest.digest(bytes);
178         return getHex(b);
179     }
180 
181     /**
182      * Calculates the MD5 checksum of the specified text.
183      *
184      * @param text the text to generate the MD5 checksum
185      * @return the hex representation of the MD5
186      */
187     public static String getMD5Checksum(String text) {
188         final byte[] data = stringToBytes(text);
189         return getChecksum(MD5, data);
190     }
191 
192     /**
193      * Calculates the SHA1 checksum of the specified text.
194      *
195      * @param text the text to generate the SHA1 checksum
196      * @return the hex representation of the SHA1
197      */
198     public static String getSHA1Checksum(String text) {
199         final byte[] data = stringToBytes(text);
200         return getChecksum(SHA1, data);
201     }
202 
203     /**
204      * Calculates the SHA1 checksum of the specified text.
205      *
206      * @param text the text to generate the SHA1 checksum
207      * @return the hex representation of the SHA1
208      */
209     public static String getSHA256Checksum(String text) {
210         final byte[] data = stringToBytes(text);
211         return getChecksum(SHA256, data);
212     }
213 
214     /**
215      * Converts the given text into bytes.
216      *
217      * @param text the text to convert
218      * @return the bytes
219      */
220     private static byte[] stringToBytes(String text) {
221         byte[] data;
222         try {
223             data = text.getBytes(Charset.forName(StandardCharsets.UTF_8.name()));
224         } catch (UnsupportedCharsetException ex) {
225             data = text.getBytes(Charset.defaultCharset());
226         }
227         return data;
228     }
229 
230     /**
231      * <p>
232      * Converts a byte array into a hex string.</p>
233      *
234      * <p>
235      * This method was copied from <a
236      * href="http://www.rgagnon.com/javadetails/java-0596.html">http://www.rgagnon.com/javadetails/java-0596.html</a></p>
237      *
238      * @param raw a byte array
239      * @return the hex representation of the byte array
240      */
241     public static String getHex(byte[] raw) {
242         if (raw == null) {
243             return null;
244         }
245         final StringBuilder hex = new StringBuilder(2 * raw.length);
246         for (final byte b : raw) {
247             hex.append(HEXES.charAt((b & 0xF0) >> 4)).append(HEXES.charAt(b & 0x0F));
248         }
249         return hex.toString();
250     }
251 
252     /**
253      * Returns the message digest.
254      *
255      * @param algorithm the algorithm for the message digest
256      * @return the message digest
257      */
258     private static MessageDigest getMessageDigest(String algorithm) {
259         try {
260             return MessageDigest.getInstance(algorithm);
261         } catch (NoSuchAlgorithmException e) {
262             LOGGER.error(e.getMessage(), e);
263             final String msg = String.format("Failed to obtain the %s message digest.", algorithm);
264             throw new IllegalStateException(msg, e);
265         }
266     }
