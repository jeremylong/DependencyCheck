18  package org.owasp.dependencycheck.utils;
19  
20  import java.sql.PreparedStatement;
21  import java.sql.ResultSet;
22  import java.sql.SQLException;
23  import java.sql.Statement;
24  import javax.annotation.concurrent.ThreadSafe;
25  import org.owasp.dependencycheck.data.nvdcve.DatabaseException;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  /**
30   * Collection of utility methods for working with database objects.
31   *
32   * @author Jeremy Long
33   */
34  @ThreadSafe
35  public final class DBUtils {
36  
37      /**
38       * The logger.
39       */
40      private static final Logger LOGGER = LoggerFactory.getLogger(DBUtils.class);
41  
42      /**
43       * Private constructor for a utility class.
44       */
45      private DBUtils() {
46      }
47  
48      /**
49       * Returns the generated integer primary key for a newly inserted row.
50       *
51       * @param statement a prepared statement that just executed an insert
52       * @return a primary key
53       * @throws DatabaseException thrown if there is an exception obtaining the
54       * key
55       */
56      public static int getGeneratedKey(PreparedStatement statement) throws DatabaseException {
57          ResultSet rs = null;
58          final int id;
59          try {
60              rs = statement.getGeneratedKeys();
61              if (!rs.next()) {
62                  throw new DatabaseException("Unable to get primary key for inserted row");
63              }
64              id = rs.getInt(1);
65          } catch (SQLException ex) {
66              throw new DatabaseException("Unable to get primary key for inserted row");
67          } finally {
68              closeResultSet(rs);
69          }
70          return id;
71      }
72  
73      /**
74       * Closes the given statement object ignoring any exceptions that occur.
75       *
76       * @param statement a Statement object
77       */
78      public static void closeStatement(Statement statement) {
79          try {
80              if (statement != null && !statement.isClosed()) {
81  
82                  statement.close();
83              }
84          } catch (SQLException ex) {
85              LOGGER.trace(statement.toString(), ex);
86          }
87      }
88  
89      /**
90       * Closes the result set capturing and ignoring any SQLExceptions that
91       * occur.
92       *
93       * @param rs a ResultSet to close
94       */
95      public static void closeResultSet(ResultSet rs) {
96          try {
97              if (rs != null && !rs.isClosed()) {
98                  rs.close();
99              }
100         } catch (SQLException ex) {
101             LOGGER.trace(rs.toString(), ex);
102         }
103     }
104 }
