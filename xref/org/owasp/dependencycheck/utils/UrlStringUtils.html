18  package org.owasp.dependencycheck.utils;
19  
20  import java.net.MalformedURLException;
21  import java.net.URL;
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.List;
25  import java.util.regex.Pattern;
26  import javax.annotation.concurrent.ThreadSafe;
27  
28  /**
29   *
30   * @author Jeremy Long
31   */
32  @ThreadSafe
33  public final class UrlStringUtils {
34  
35      /**
36       * A regular expression to test if a string contains a URL.
37       */
38      private static final Pattern CONTAINS_URL_TEST = Pattern.compile("^.*(ht|f)tps?://.*$", Pattern.MULTILINE | Pattern.CASE_INSENSITIVE);
39      /**
40       * A regular expression to test if a string is a URL.
41       */
42      private static final Pattern IS_URL_TEST = Pattern.compile("^(ht|f)tps?://.*", Pattern.CASE_INSENSITIVE);
43      /**
44       * A listing of domain parts that should not be used as evidence. Yes, this
45       * is an incomplete list.
46       */
47      private static final String[] IGNORE_LIST = {"www", "com", "org", "gov", "info", "name", "net", "pro", "tel", "mobi", "xxx", "github", "gitlab"};
48  
49      static {
50          Arrays.sort(IGNORE_LIST);
51      }
52  
53      /**
54       * Private constructor for a utility class.
55       */
56      private UrlStringUtils() {
57      }
58  
59      /**
60       * Tests if the text provided contains a URL. This is somewhat limited
61       * search in that it only looks for (ftp|http|https)://
62       *
63       * @param text the text to search
64       * @return true if the text contains a url, otherwise false
65       */
66      public static boolean containsUrl(String text) {
67          return CONTAINS_URL_TEST.matcher(text).matches();
68      }
69  
70      /**
71       * Tests if the given text is url.
72       *
73       * @param text the string to test
74       * @return returns true if the text is a url, otherwise false
75       */
76      public static boolean isUrl(String text) {
77          return IS_URL_TEST.matcher(text).matches();
78      }
79  
80      /**
81       * <p>
82       * Takes a URL, in String format, and adds the important parts of the URL to
83       * a list of strings.</p>
84       * <p>
85       * Example, given the following input:</p>
86       * <code>"https://www.somedomain.com/path1/path2/file.php?id=439"</code>
87       * <p>
88       * The function would return:</p>
89       * <code>{"some.domain", "path1", "path2", "file"}</code>
90       *
91       * @param text a URL
92       * @return importantParts a list of the important parts of the URL
93       * @throws MalformedURLException thrown if the URL is malformed
94       */
95      @SuppressWarnings("StringSplitter")
96      public static List<String> extractImportantUrlData(String text) throws MalformedURLException {
97          final List<String> importantParts = new ArrayList<>();
98          final URL url = new URL(text);
99          final String[] domain = url.getHost().split("\\.");
100         //add the domain except www and the tld.
101         for (int i = 0; i < domain.length - 1; i++) {
102             final String sub = domain[i];
103             if (Arrays.binarySearch(IGNORE_LIST, sub.toLowerCase()) < 0) {
104                 importantParts.add(sub);
105             }
106         }
107         final String document = url.getPath();
108         final String[] pathParts = document.split("[\\\\//]");
109         for (int i = 0; i < pathParts.length - 1; i++) {
110             if (!pathParts[i].isEmpty()) {
111                 importantParts.add(pathParts[i]);
112             }
113         }
114         if (pathParts.length > 0 && !pathParts[pathParts.length - 1].isEmpty()) {
115             final String tmp = pathParts[pathParts.length - 1];
116             final int pos = tmp.lastIndexOf('.');
117             if (pos > 1) {
118                 importantParts.add(tmp.substring(0, pos));
119             } else if (pos == 0 && tmp.length() > 1) {
120                 importantParts.add(tmp.substring(1));
121             } else {
122                 importantParts.add(tmp);
123             }
124         }
125         return importantParts;
126     }
127 }
