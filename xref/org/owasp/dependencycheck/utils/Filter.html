1   package org.owasp.dependencycheck.utils;
2   
3   import org.jetbrains.annotations.NotNull;
4   
5   import java.util.Iterator;
6   import java.util.NoSuchElementException;
7   import javax.annotation.concurrent.NotThreadSafe;
8   
9   /**
10   * This is an abstract filter that can be used to filter iterable list.
11   *
12   * This Filter class was copied from:
13   * http://erikras.com/2008/01/18/the-filter-pattern-java-conditional-abstraction-with-iterables/
14   *
15   * Erik Rasmussen - Â© 2006 - 2012 All Rights Reserved. @author Erik Rasmussen
16   * https://plus.google.com/115403795880834599019/?rel=author
17   *
18   * @param <T> the type to filter
19   */
20  @NotThreadSafe
21  public abstract class Filter<T> {
22  
23      /**
24       * Determines whether the object passes the filter.
25       *
26       * @param object the object to test
27       * @return whether or not the object passes the filter
28       */
29      public abstract boolean passes(T object);
30  
31      /**
32       * Filters a given iterator.
33       *
34       * @param iterator the iterator to filter
35       * @return the filtered iterator
36       */
37      public Iterator<T> filter(Iterator<T> iterator) {
38          return new FilterIterator(iterator);
39      }
40  
41      /**
42       * Filters a given iterable.
43       *
44       * @param iterable the iterable to filter
45       * @return the filtered iterable
46       */
47      public Iterable<T> filter(final Iterable<T> iterable) {
48          return new Iterable<T>() {
49  
50              @NotNull
51              @Override
52              public Iterator<T> iterator() {
53                  return filter(iterable.iterator());
54              }
55          };
56      }
57  
58      private class FilterIterator implements Iterator<T> {
59  
60          private final Iterator<T> iterator;
61          private T next;
62  
63          private FilterIterator(Iterator<T> iterator) {
64              this.iterator = iterator;
65              toNext();
66          }
67  
68          @Override
69          public boolean hasNext() {
70              return next != null;
71          }
72  
73          @Override
74          public T next() {
75              if (next == null) {
76                  throw new NoSuchElementException();
77              }
78              final T returnValue = next;
79              toNext();
80              return returnValue;
81          }
82  
83          @Override
84          public void remove() {
85              throw new UnsupportedOperationException();
86          }
87  
88          private void toNext() {
89              next = null;
90              while (iterator.hasNext()) {
91                  final T item = iterator.next();
92                  if (item != null && passes(item)) {
93                      next = item;
94                      break;
95                  }
96              }
97          }
98      }
99  }
