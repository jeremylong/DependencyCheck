18  package org.owasp.dependencycheck.utils;
19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * Utility to estimate severity level scores.
25   *
26   * @author Jeremy Long
27   */
28  public final class SeverityUtil {
29  
30      /**
31       * The cutoff for high severity.
32       */
33      private static final Double HIGH = 10.0;
34      /**
35       * The cutoff for medium severity.
36       */
37      private static final Double MEDIUM = 6.9;
38      /**
39       * The cutoff for low severity.
40       */
41      private static final Double LOW = 3.9;
42      /**
43       * The cutoff for none severity.
44       */
45      private static final Double NONE = 0.0;
46      /**
47       * The logger.
48       */
49      private static final Logger LOGGER = LoggerFactory.getLogger(SeverityUtil.class);
50  
51      /**
52       * Private constructor for a utility class.
53       */
54      private SeverityUtil() {
55          //noop
56      }
57  
58      /**
59       * Estimates the CVSS V2 Score based on a given severity. The implementation
60       * will default to 3.9 if no recognized "severity" level is given (critical,
61       * high, low).
62       *
63       * @param severity the severity text (e.g. "medium")
64       * @return a score from 0 to 10
65       */
66      public static Double estimateCvssV2(String severity) {
67          switch (severity == null ? "none" : severity.toLowerCase()) {
68              case "critical":
69              case "high":
70                  return HIGH;
71              case "moderate":
72              case "medium":
73                  return MEDIUM;
74              case "info":
75              case "none":
76              case "informational":
77                  return NONE;
78              case "low":
79              case "unknown":
80              default:
81                  return LOW;
82          }
83      }
84  
85      /**
86       * Converts a textual severity to the text that should be used to signal it
87       * in a report.
88       *
89       * @param severity The textual unscored severity
90       * @return The severity when properly recognized, otherwise the severity
91       * extended with a remark that it was not recognized and assumed to
92       * represent a critical severity.
93       */
94      public static String unscoredToSeveritytext(final String severity) {
95          switch (Severity.forUnscored(severity)) {
96              case CRITICAL:
97              case HIGH:
98              case MEDIUM:
99              case LOW:
100             case INFO:
101                 return severity;
102             case ASSUMED_CRITICAL:
103             default:
104                 final String sevText;
105                 if ("0.0".equals(severity)) {
106                     sevText = "Unknown";
107                 } else {
108                     sevText = severity;
109                 }
110                 return sevText + " (not recognized; assumed to be critical)";
111         }
112     }
113 
114     /**
115      * Creates an estimated sort-adjusted CVSSv3 score for an unscored textual
116      * severity. For recognized severities below critical it returns a value at
117      * the lower bound of the CVSSv3 baseScore for that severity. For recognized
118      * critical severities it returns a score in-between the upper bound of the
119      * HIGH CVSSv2 score and the lowest sort-adjusted CVSSv3 critical score, so
120      * that unscored critical vulnerabilties are ordered in between CRITICAL
121      * scored CVSSv3 rated vulnerabilities and HIGH-scored CVSSv2 rated
122      * vulnerabilities. For unrecognized severities it returns a score
123      * in-between the top HIGH CVSSv2 score and the estimatedSortAdjustedCVSSv3
124      * score for an unscored severity recognized as critical, so that recognized
125      * critical will win over unrecognized severities while unrecognized
126      * severities are assumed to be of a critical nature.
127      *
128      * @param severity The textual severity, may be null
129      * @return A float that can be used to numerically sort vulnerabilities in
130      * approximated severity (highest float represents highest severity).
131      * @see #sortAdjustedCVSSv3BaseScore(float)
132      */
133     public static Double estimatedSortAdjustedCVSSv3(final String severity) {
134         switch (Severity.forUnscored(severity)) {
135             case CRITICAL:
136                 return 10.2;
137             case HIGH:
138                 return 7.0;
139             case MEDIUM:
140                 return 4.0;
141             case LOW:
142                 return 0.1;
143             case INFO:
144                 return 0.0;
145             case ASSUMED_CRITICAL:
146             default:
147                 SeverityUtil.LOGGER.debug("Unrecognized unscored textual severity: {}, assuming critical score as worst-case "
148                         + "estimate for sorting",
149                         severity);
150                 return 10.1;
151         }
152     }
153 
154     /**
155      * Compute an adjusted CVSSv3 baseScore that ensures that CRITICAL CVSSv3
156      * scores will win over HIGH CVSSv2 and CRITICAL unscored severities to
157      * allow for a best-effort sorting that enables the report to list a
158      * reliable 'highest severity' in the report.
159      *
160      * @param cvssV3BaseScore The cvssV3 baseScore severity of a vulnerability
161      * @return The cvssV3 baseScore, adjusted if necessary in order to guarantee
162      * that CVSSv3 CRITICAL scores will rate higher than CVSSv2 HIGH, unscored
163      * critical severities and unscored unrecognized severities (which are
164      * assumed for sorting to be of a critical nature)
165      * @see #estimatedSortAdjustedCVSSv3(String)
166      */
167     public static Double sortAdjustedCVSSv3BaseScore(final Double cvssV3BaseScore) {
168         if (cvssV3BaseScore.floatValue() >= 9.0f) {
169             return cvssV3BaseScore + 1.3;
170         }
171         return cvssV3BaseScore;
172     }
173 
174     /**
175      * An enum to translate unscored severity texts to a severity level of a
176      * defined set of severities. Allows for re-use of the text-to-severity
177      * mapping in multiple helper methods.
178      */
179     private enum Severity {
180         /**
181          * A severity level for textual values that should be regarded as
182          * accompanying a critical severity vulnerability
183          */
184         CRITICAL,
185         /**
186          * A severity level for textual values that are not recognized and
187          * therefor assumed to be accompanying a critical severity vulnerability
188          */
189         ASSUMED_CRITICAL,
190         /**
191          * A severity level for textual values that should be regarded as
192          * accompanying a high severity vulnerability
193          */
194         HIGH,
195         /**
196          * A severity level for textual values that should be regarded as
197          * accompanying a medium severity vulnerability
198          */
199         MEDIUM,
200         /**
201          * A severity level for textual values that should be regarded as
202          * accompanying a low severity vulnerability
203          */
204         LOW,
205         /**
206          * A severity level for textual values that should be regarded as
207          * accompanying a vulnerability of informational nature
208          */
209         INFO;
210 
211         public static Severity forUnscored(String value) {
212             switch (value == null ? "none" : value.toLowerCase()) {
213                 case "critical":
214                     return CRITICAL;
215                 case "high":
216                     return HIGH;
217                 case "moderate":
218                 case "medium":
219                     return MEDIUM;
220                 case "info":
221                 case "informational":
222                     return INFO;
223                 case "low":
224                 case "unknown":
225                 case "none":
226                     return LOW;
227                 default:
228                     return ASSUMED_CRITICAL;
229             }
230         }
231     }
