40  /**
41   * A URLConnection Factory to create new connections. This encapsulates several
42   * configuration checks to ensure that the connection uses the correct proxy
43   * settings.
44   *
45   * @author Jeremy Long
46   */
47  public final class URLConnectionFactory {
48  
49      /**
50       * The logger.
51       */
52      private static final Logger LOGGER = LoggerFactory.getLogger(URLConnectionFactory.class);
53      /**
54       * The configured settings.
55       */
56      private final Settings settings;
57  
58      /**
59       * Private constructor for this factory.
60       *
61       * @param settings reference to the configured settings
62       */
63      public URLConnectionFactory(Settings settings) {
64          this.settings = settings;
65      }
66  
67      /**
68       * Utility method to create an HttpURLConnection. If the application is
69       * configured to use a proxy this method will retrieve the proxy settings
70       * and use them when setting up the connection.
71       *
72       * @param url the URL to connect to
73       * @return an HttpURLConnection
74       * @throws org.owasp.dependencycheck.utils.URLConnectionFailureException
75       * thrown if there is an exception
76       */
77      @SuppressWarnings("squid:S2583")
78      @SuppressFBWarnings(justification = "yes, there is a redundant null check in the catch - to suppress warnings we are leaving the null check",
79              value = {"RCN_REDUNDANT_NULLCHECK_OF_NULL_VALUE"})
80      public HttpURLConnection createHttpURLConnection(URL url) throws URLConnectionFailureException {
81          HttpURLConnection conn = null;
82          final String proxyHost = settings.getString(Settings.KEYS.PROXY_SERVER);
83  
84          try {
85              if (proxyHost != null && !matchNonProxy(url)) {
86                  final int proxyPort = settings.getInt(Settings.KEYS.PROXY_PORT);
87                  final SocketAddress address = new InetSocketAddress(proxyHost, proxyPort);
88  
89                  final String username = settings.getString(Settings.KEYS.PROXY_USERNAME);
90                  final String password = settings.getString(Settings.KEYS.PROXY_PASSWORD);
91  
92                  if (username != null && password != null) {
93                      final Authenticator auth = new Authenticator() {
94                          @Override
95                          public PasswordAuthentication getPasswordAuthentication() {
96                              if (proxyHost.equals(getRequestingHost()) || getRequestorType().equals(Authenticator.RequestorType.PROXY)) {
97                                  LOGGER.debug("Using the configured proxy username and password");
98                                  if (settings.getBoolean(Settings.KEYS.PROXY_DISABLE_SCHEMAS, true)) {
99                                      System.setProperty("jdk.http.auth.tunneling.disabledSchemes", "");
100                                 }
101                                 return new PasswordAuthentication(username, password.toCharArray());
102                             }
103                             return super.getPasswordAuthentication();
104                         }
105                     };
106                     Authenticator.setDefault(auth);
107                 }
108 
109                 final Proxy proxy = new Proxy(Proxy.Type.HTTP, address);
110                 conn = (HttpURLConnection) url.openConnection(proxy);
111             } else {
112                 conn = (HttpURLConnection) url.openConnection();
113             }
114             final int connectionTimeout = settings.getInt(Settings.KEYS.CONNECTION_TIMEOUT, 10000);
115             conn.setConnectTimeout(connectionTimeout);
116             conn.setInstanceFollowRedirects(true);
117         } catch (IOException ex) {
118             if (conn != null) {
119                 try {
120                     conn.disconnect();
121                 } finally {
122                     conn = null;
123                 }
124             }
125             throw new URLConnectionFailureException("Error getting connection.", ex);
126         }
127         //conn.setRequestProperty("user-agent",
128         //  "Mozilla/5.0 (Macintosh; Intel Mac OS X 10_13_4) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/65.0.3325.181 Safari/537.36");
129         configureTLS(url, conn);
130         addAuthenticationIfPresent(conn);
131         return conn;
132     }
133 
134     /**
135      * Adds the basic authorization header if the URL contains a username and
136      * password. Example URL that will have the basic authorization header
137      * added:
138      * <code>http://username:password@passwordprotectednvdsite.internal/feeds/json/cve/1.1/nvdcve-1.1-modified.json.gz</code>;
139      *
140      * @param conn the connection
141      */
142     private void addAuthenticationIfPresent(HttpURLConnection conn) {
143         final String userInfo = conn.getURL().getUserInfo();
144         if (userInfo != null) {
145             final String basicAuth = "Basic " + Base64.getEncoder().encodeToString(userInfo.getBytes(UTF_8));
146             if (LOGGER.isDebugEnabled()) {
147                 LOGGER.debug("Adding user info as basic authorization");
148             }
149             conn.addRequestProperty("Authorization", basicAuth);
150         }
151     }
152 
153     /**
154      * Adds a basic authentication header if the values in the settings are not
155      * null.
156      *
157      * @param conn the connection to add the basic auth header
158      * @param userKey the settings key for the username
159      * @param passwordKey the settings key for the password
160      */
161     public void addBasicAuthentication(HttpURLConnection conn, String userKey, String passwordKey) {
162         if (StringUtils.isNotEmpty(settings.getString(userKey))
163                 && StringUtils.isNotEmpty(settings.getString(passwordKey))) {
164             final String user = settings.getString(userKey);
165             final String password = settings.getString(passwordKey);
166             final String userColonPassword = user + ":" + password;
167             final String basicAuth = "Basic " + Base64.getEncoder().encodeToString(userColonPassword.getBytes(UTF_8));
168             if (LOGGER.isDebugEnabled()) {
169                 LOGGER.debug("Adding user/pw from settings.xml as basic authorization");
170             }
171             conn.addRequestProperty("Authorization", basicAuth);
172         }
173     }
174 
175     /**
176      * Check if host name matches nonProxy settings
177      *
178      * @param url the URL to connect to
179      * @return matching result. true: match nonProxy
180      */
181     @SuppressWarnings("StringSplitter")
182     private boolean matchNonProxy(final URL url) {
183         final String host = url.getHost();
184 
185         // code partially from org.apache.maven.plugins.site.AbstractDeployMojo#getProxyInfo
186         final String nonProxyHosts = settings.getString(Settings.KEYS.PROXY_NON_PROXY_HOSTS);
187         if (null != nonProxyHosts) {
188             final String[] nonProxies = nonProxyHosts.split("(,)|(;)|(\\|)");
189             for (final String nonProxyHost : nonProxies) {
190                 //if ( StringUtils.contains( nonProxyHost, "*" ) )
191                 if (null != nonProxyHost && nonProxyHost.contains("*")) {
192                     // Handle wildcard at the end, beginning or middle of the nonProxyHost
193                     final int pos = nonProxyHost.indexOf('*');
194                     final String nonProxyHostPrefix = nonProxyHost.substring(0, pos);
195                     final String nonProxyHostSuffix = nonProxyHost.substring(pos + 1);
196                     // prefix*
197                     if (!StringUtils.isBlank(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix) && StringUtils.isBlank(nonProxyHostSuffix)) {
198                         return true;
199                     }
200                     // *suffix
201                     if (StringUtils.isBlank(nonProxyHostPrefix) && !StringUtils.isBlank(nonProxyHostSuffix) && host.endsWith(nonProxyHostSuffix)) {
202                         return true;
203                     }
204                     // prefix*suffix
205                     if (!StringUtils.isBlank(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix) && !StringUtils.isBlank(nonProxyHostSuffix)
206                             && host.endsWith(nonProxyHostSuffix)) {
207                         return true;
208                     }
209                 } else if (host.equals(nonProxyHost)) {
210                     return true;
211                 }
212             }
213         }
214         return false;
215     }

217     /**
218      * Utility method to create an HttpURLConnection. The use of a proxy here is
219      * optional as there may be cases where a proxy is configured but we don't
220      * want to use it (for example, if there's an internal repository
221      * configured)
222      *
223      * @param url the URL to connect to
224      * @param proxy whether to use the proxy (if configured)
225      * @return a newly constructed HttpURLConnection
226      * @throws org.owasp.dependencycheck.utils.URLConnectionFailureException
227      * thrown if there is an exception
228      */
229     public HttpURLConnection createHttpURLConnection(URL url, boolean proxy) throws URLConnectionFailureException {
230         if (proxy) {
231             return createHttpURLConnection(url);
232         }
233         final HttpURLConnection conn;
234         try {
235             conn = (HttpURLConnection) url.openConnection();
236             final int timeout = settings.getInt(Settings.KEYS.CONNECTION_TIMEOUT, 10000);
237             conn.setConnectTimeout(timeout);
238             conn.setInstanceFollowRedirects(true);
239         } catch (IOException ioe) {
240             throw new URLConnectionFailureException("Error getting connection.", ioe);
241         }
242         configureTLS(url, conn);
243         addAuthenticationIfPresent(conn);
244         return conn;
245     }
246 
247     /**
248      * If the protocol is HTTPS, this will configure the cipher suites so that
249      * connections can be made to the NVD, and others, using older versions of
250      * Java.
251      *
252      * @param url the URL
253      * @param conn the connection
254      */
255     private void configureTLS(URL url, URLConnection conn) {
256         if ("https".equals(url.getProtocol())) {
257             try {
258                 final HttpsURLConnection secCon = (HttpsURLConnection) conn;
259                 final SSLSocketFactoryEx factory = new SSLSocketFactoryEx(settings);
260                 secCon.setSSLSocketFactory(factory);
261             } catch (NoSuchAlgorithmException ex) {
262                 LOGGER.debug("Unsupported algorithm in SSLSocketFactoryEx", ex);
263             } catch (KeyManagementException ex) {
264                 LOGGER.debug("Key management exception in SSLSocketFactoryEx", ex);
265             }
266         }
267     }
