40  /**
41   * A URLConnection Factory to create new connections. This encapsulates several
42   * configuration checks to ensure that the connection uses the correct proxy
43   * settings.
44   *
45   * @author Jeremy Long
46   */
47  public final class URLConnectionFactory {
48  
49      /**
50       * The logger.
51       */
52      private static final Logger LOGGER = LoggerFactory.getLogger(URLConnectionFactory.class);
53      /**
54       * The configured settings.
55       */
56      private final Settings settings;
57  
58      /**
59       * Private constructor for this factory.
60       *
61       * @param settings reference to the configured settings
62       */
63      public URLConnectionFactory(Settings settings) {
64          this.settings = settings;
65      }
66  
67      /**
68       * Utility method to create an HttpURLConnection. If the application is
69       * configured to use a proxy this method will retrieve the proxy settings
70       * and use them when setting up the connection.
71       *
72       * @param url the URL to connect to
73       * @return an HttpURLConnection
74       * @throws org.owasp.dependencycheck.utils.URLConnectionFailureException
75       * thrown if there is an exception
76       */
77      @SuppressWarnings("squid:S2583")
78      @SuppressFBWarnings(justification = "yes, there is a redundant null check in the catch - to suppress warnings we are leaving the null check",
79              value = {"RCN_REDUNDANT_NULLCHECK_OF_NULL_VALUE"})
80      public HttpURLConnection createHttpURLConnection(URL url) throws URLConnectionFailureException {
81          HttpURLConnection conn = null;
82          final String proxyHost = settings.getString(Settings.KEYS.PROXY_SERVER);
83  
84          try {
85              if (proxyHost != null && !matchNonProxy(url)) {
86                  final int proxyPort = settings.getInt(Settings.KEYS.PROXY_PORT);
87                  final SocketAddress address = new InetSocketAddress(proxyHost, proxyPort);
88  
89                  final String username = settings.getString(Settings.KEYS.PROXY_USERNAME);
90                  final String password = settings.getString(Settings.KEYS.PROXY_PASSWORD);
91  
92                  if (username != null && password != null) {
93                      final Authenticator auth = new Authenticator() {
94                          @Override
95                          public PasswordAuthentication getPasswordAuthentication() {
96                              if (proxyHost.equals(getRequestingHost()) || getRequestorType().equals(Authenticator.RequestorType.PROXY)) {
97                                  LOGGER.debug("Using the configured proxy username and password");
98                                  if (settings.getBoolean(Settings.KEYS.PROXY_DISABLE_SCHEMAS, true)) {
99                                      System.setProperty("jdk.http.auth.tunneling.disabledSchemes", "");
100                                 }
101                                 return new PasswordAuthentication(username, password.toCharArray());
102                             }
103                             return super.getPasswordAuthentication();
104                         }
105                     };
106                     Authenticator.setDefault(auth);
107                 }
108 
109                 final Proxy proxy = new Proxy(Proxy.Type.HTTP, address);
110                 conn = (HttpURLConnection) url.openConnection(proxy);
111             } else {
112                 conn = (HttpURLConnection) url.openConnection();
113             }
114             final int connectionTimeout = settings.getInt(Settings.KEYS.CONNECTION_TIMEOUT, 10000);
115             // set a conservative long default timeout to compensate for MITM-proxies that return the (final) bytes only
116             // after all security checks passed
117             final int readTimeout = settings.getInt(Settings.KEYS.CONNECTION_READ_TIMEOUT, 60_000);
118             conn.setConnectTimeout(connectionTimeout);
119             conn.setReadTimeout(readTimeout);
120             conn.setInstanceFollowRedirects(true);
121         } catch (IOException ex) {
122             if (conn != null) {
123                 try {
124                     conn.disconnect();
125                 } finally {
126                     conn = null;
127                 }
128             }
129             throw new URLConnectionFailureException("Error getting connection.", ex);
130         }
131         //conn.setRequestProperty("user-agent",
132         //  "Mozilla/5.0 (Macintosh; Intel Mac OS X 10_13_4) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/65.0.3325.181 Safari/537.36");
133         configureTLS(url, conn);
134         addAuthenticationIfPresent(conn);
135         return conn;
136     }
137 
138     /**
139      * Adds the basic authorization header if the URL contains a username and
140      * password. Example URL that will have the basic authorization header
141      * added:
142      * <code>http://username:password@passwordprotectednvdsite.internal/feeds/json/cve/1.1/nvdcve-1.1-modified.json.gz</code>;
143      *
144      * @param conn the connection
145      */
146     private void addAuthenticationIfPresent(HttpURLConnection conn) {
147         final String userInfo = conn.getURL().getUserInfo();
148         if (userInfo != null) {
149             final String basicAuth = "Basic " + Base64.getEncoder().encodeToString(userInfo.getBytes(UTF_8));
150             if (LOGGER.isDebugEnabled()) {
151                 LOGGER.debug("Adding user info as basic authorization");
152             }
153             conn.addRequestProperty("Authorization", basicAuth);
154         }
155     }
156 
157     /**
158      * Adds a basic authentication header if the values in the settings are not
159      * null.
160      *
161      * @param conn the connection to add the basic auth header
162      * @param userKey the settings key for the username
163      * @param passwordKey the settings key for the password
164      */
165     public void addBasicAuthentication(HttpURLConnection conn, String userKey, String passwordKey) {
166         if (StringUtils.isNotEmpty(settings.getString(userKey))
167                 && StringUtils.isNotEmpty(settings.getString(passwordKey))) {
168             final String user = settings.getString(userKey);
169             final String password = settings.getString(passwordKey);
170 
171             if (user.isEmpty() || password.isEmpty()) {
172                 if (LOGGER.isDebugEnabled()) {
173                     LOGGER.debug("Skip authentication as user and/or password is empty");
174                 }
175             } else {
176                 final String userColonPassword = user + ":" + password;
177                 final String basicAuth = "Basic " + Base64.getEncoder().encodeToString(userColonPassword.getBytes(UTF_8));
178                 if (LOGGER.isDebugEnabled()) {
179                     LOGGER.debug("Adding user/password from settings.xml as basic authorization");
180                 }
181                 conn.addRequestProperty("Authorization", basicAuth);
182             }
183         }
184     }

186     /**
187      * Check if host name matches nonProxy settings
188      *
189      * @param url the URL to connect to
190      * @return matching result. true: match nonProxy
191      */
192     @SuppressWarnings("StringSplitter")
193     private boolean matchNonProxy(final URL url) {
194         final String host = url.getHost();
195 
196         // code partially from org.apache.maven.plugins.site.AbstractDeployMojo#getProxyInfo
197         final String nonProxyHosts = settings.getString(Settings.KEYS.PROXY_NON_PROXY_HOSTS);
198         if (null != nonProxyHosts) {
199             final String[] nonProxies = nonProxyHosts.split("(,)|(;)|(\\|)");
200             for (final String nonProxyHost : nonProxies) {
201                 //if ( StringUtils.contains( nonProxyHost, "*" ) )
202                 if (null != nonProxyHost && nonProxyHost.contains("*")) {
203                     // Handle wildcard at the end, beginning or middle of the nonProxyHost
204                     final int pos = nonProxyHost.indexOf('*');
205                     final String nonProxyHostPrefix = nonProxyHost.substring(0, pos);
206                     final String nonProxyHostSuffix = nonProxyHost.substring(pos + 1);
207                     // prefix*
208                     if (!StringUtils.isBlank(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix) && StringUtils.isBlank(nonProxyHostSuffix)) {
209                         return true;
210                     }
211                     // *suffix
212                     if (StringUtils.isBlank(nonProxyHostPrefix) && !StringUtils.isBlank(nonProxyHostSuffix) && host.endsWith(nonProxyHostSuffix)) {
213                         return true;
214                     }
215                     // prefix*suffix
216                     if (!StringUtils.isBlank(nonProxyHostPrefix) && host.startsWith(nonProxyHostPrefix) && !StringUtils.isBlank(nonProxyHostSuffix)
217                             && host.endsWith(nonProxyHostSuffix)) {
218                         return true;
219                     }
220                 } else if (host.equals(nonProxyHost)) {
221                     return true;
222                 }
223             }
224         }
225         return false;
226     }

228     /**
229      * Utility method to create an HttpURLConnection. The use of a proxy here is
230      * optional as there may be cases where a proxy is configured but we don't
231      * want to use it (for example, if there's an internal repository
232      * configured)
233      *
234      * @param url the URL to connect to
235      * @param proxy whether to use the proxy (if configured)
236      * @return a newly constructed HttpURLConnection
237      * @throws org.owasp.dependencycheck.utils.URLConnectionFailureException
238      * thrown if there is an exception
239      */
240     public HttpURLConnection createHttpURLConnection(URL url, boolean proxy) throws URLConnectionFailureException {
241         if (proxy) {
242             return createHttpURLConnection(url);
243         }
244         final HttpURLConnection conn;
245         try {
246             conn = (HttpURLConnection) url.openConnection();
247             final int timeout = settings.getInt(Settings.KEYS.CONNECTION_TIMEOUT, 10000);
248             conn.setConnectTimeout(timeout);
249             conn.setInstanceFollowRedirects(true);
250         } catch (IOException ioe) {
251             throw new URLConnectionFailureException("Error getting connection.", ioe);
252         }
253         configureTLS(url, conn);
254         addAuthenticationIfPresent(conn);
255         return conn;
256     }
257 
258     /**
259      * If the protocol is HTTPS, this will configure the cipher suites so that
260      * connections can be made to the NVD, and others, using older versions of
261      * Java.
262      *
263      * @param url the URL
264      * @param conn the connection
265      */
266     private void configureTLS(URL url, URLConnection conn) {
267         if ("https".equals(url.getProtocol())) {
268             try {
269                 final HttpsURLConnection secCon = (HttpsURLConnection) conn;
270                 final SSLSocketFactoryEx factory = new SSLSocketFactoryEx(settings);
271                 secCon.setSSLSocketFactory(factory);
272             } catch (NoSuchAlgorithmException ex) {
273                 LOGGER.debug("Unsupported algorithm in SSLSocketFactoryEx", ex);
274             } catch (KeyManagementException ex) {
275                 LOGGER.debug("Key management exception in SSLSocketFactoryEx", ex);
276             }
277         }
278     }
