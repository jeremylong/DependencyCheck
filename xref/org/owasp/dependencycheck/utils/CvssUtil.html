29  /**
30   * Utility class to create CVSS Objects.
31   *
32   * @author Jeremy Long
33   */
34  public final class CvssUtil {
35  
36      private CvssUtil() {
37          //empty constructor for utility class.
38      }
39      /**
40       * The CVSS v3 Base Metrics (that are required by the spec for any CVSS v3
41       * Vector String)
42       */
43      private static final List<String> BASE_METRICS_V3 = Arrays.asList("AV", "AC", "PR", "UI", "S", "C", "I", "A");
44  
45      /**
46       * The CVSS V2 Metrics required for the vector string to be complete.
47       */
48      private static final List<String> BASE_METRICS_V2 = Arrays.asList("AV", "AC", "Au", "C", "I", "A");
49      /**
50       * ZERO.
51       */
52      private static final Double ZERO = 0.0;
53      /**
54       * ONE.
55       */
56      private static final Double ONE = 1.0;
57      /**
58       * FOUR.
59       */
60      private static final Double FOUR = 4.0;
61      /**
62       * SEVEN.
63       */
64      private static final Double SEVEN = 7.0;
65      /**
66       * NINE.
67       */
68      private static final Double NINE = 9.0;
69      /**
70       * TEN.
71       */
72      private static final Double TEN = 10.0;
73      /**
74       * UNKNOWN.
75       */
76      private static final String UNKNOWN = "UNKNOWN";
77      /**
78       * HIGH.
79       */
80      private static final String HIGH = "HIGH";
81      /**
82       * MEDIUM.
83       */
84      private static final String MEDIUM = "MEDIUM";
85      /**
86       * LOW.
87       */
88      private static final String LOW = "LOW";
89  
90      /**
91       * Convert a CVSSv2 vector String into a CvssV3 Object.
92       *
93       * @param vectorString the vector string
94       * @param baseScore the base score
95       * @return the CVSSv2 object
96       */
97      public static CvssV2 vectorToCvssV2(String vectorString, Double baseScore) {
98          if (vectorString.startsWith("CVSS:")) {
99              throw new IllegalArgumentException("Not a valid CVSSv2 vector string: " + vectorString);
100         }
101         final String[] metricStrings = vectorString.substring(vectorString.indexOf('/') + 1).split("/");
102         final HashMap<String, String> metrics = new HashMap<>();
103         for (int i = 0; i < metricStrings.length; i++) {
104             final String[] metricKeyVal = metricStrings[i].split(":");
105             if (metricKeyVal.length != 2) {
106                 throw new IllegalArgumentException(
107                         String.format("Not a valid CVSSv2 vector string '%s', invalid metric component '%s'",
108                                 vectorString, metricStrings[i]));
109             }
110             metrics.put(metricKeyVal[0], metricKeyVal[1]);
111         }
112         if (!metrics.keySet().containsAll(BASE_METRICS_V2)) {
113             throw new IllegalArgumentException(
114                     String.format("Not a valid CVSSv2 vector string '%s'; missing one or more required Metrics;",
115                             vectorString));
116         }
117 
118         final String version = CvssV2Data.Version._2_0.value();
119         //"AV:L/AC:L/Au:N/C:N/I:N/A:C"
120         final CvssV2Data.AccessVectorType accessVector = CvssV2Data.AccessVectorType.fromValue(metrics.get("AV"));
121         final CvssV2Data.AccessComplexityType attackComplexity = CvssV2Data.AccessComplexityType.fromValue(metrics.get("AC"));
122         final CvssV2Data.AuthenticationType authentication = CvssV2Data.AuthenticationType.fromValue(metrics.get("Au"));
123         final CvssV2Data.CiaType confidentialityImpact = CvssV2Data.CiaType.fromValue(metrics.get("C"));
124         final CvssV2Data.CiaType integrityImpact = CvssV2Data.CiaType.fromValue(metrics.get("I"));
125         final CvssV2Data.CiaType availabilityImpact = CvssV2Data.CiaType.fromValue(metrics.get("A"));
126 
127         final String baseSeverity = cvssV2ScoreToSeverity(baseScore);
128         final CvssV2Data data = new CvssV2Data(version, vectorString, accessVector, attackComplexity,
129                 authentication, confidentialityImpact, integrityImpact, availabilityImpact, baseScore, baseSeverity,
130                 null, null, null, null, null, null, null, null, null, null);
131         final CvssV2 cvss = new CvssV2(null, null, data, baseSeverity, null, null, null, null, null, null, null);
132         return cvss;
133 
134     }
135 
136     /**
137      * Determines the severity from the score.
138      *
139      * @param score the score
140      * @return the severity
141      */
142     public static String cvssV2ScoreToSeverity(Double score) {
143         if (score != null) {
144             if (ZERO.compareTo(score) <= 0 && FOUR.compareTo(score) > 0) {
145                 return LOW;
146             } else if (FOUR.compareTo(score) <= 0 && SEVEN.compareTo(score) > 0) {
147                 return MEDIUM;
148             } else if (SEVEN.compareTo(score) <= 0 && TEN.compareTo(score) >= 0) {
149                 return HIGH;
150             }
151         }
152         return UNKNOWN;
153     }
154 
155     /**
156      * Determines the severity from the score.
157      *
158      * @param score the score
159      * @return the severity
160      */
161     public static CvssV3Data.SeverityType cvssV3ScoreToSeverity(Double score) {
162         if (score != null) {
163             if (ZERO.compareTo(score) == 0) {
164                 return CvssV3Data.SeverityType.NONE;
165             } else if (ZERO.compareTo(score) <= 0 && FOUR.compareTo(score) > 0) {
166                 return CvssV3Data.SeverityType.LOW;
167             } else if (FOUR.compareTo(score) <= 0 && SEVEN.compareTo(score) > 0) {
168                 return CvssV3Data.SeverityType.MEDIUM;
169             } else if (SEVEN.compareTo(score) <= 0 && NINE.compareTo(score) > 0) {
170                 return CvssV3Data.SeverityType.HIGH;
171             } else if (NINE.compareTo(score) <= 0 && TEN.compareTo(score) >= 0) {
172                 return CvssV3Data.SeverityType.CRITICAL;
173             }
174         }
175         return null;
176     }

178     /**
179      * Convert a CVSSv3 vector String into a CvssV3 Object.
180      *
181      * @param vectorString the vector string
182      * @param baseScore the base score
183      * @return the CVSSv3 object
184      */
185     public static CvssV3 vectorToCvssV3(String vectorString, Double baseScore) {
186         if (!vectorString.startsWith("CVSS:3")) {
187             throw new IllegalArgumentException("Not a valid CVSSv3 vector string: " + vectorString);
188         }
189         final String versionString = vectorString.substring(5, vectorString.indexOf('/'));
190         final String[] metricStrings = vectorString.substring(vectorString.indexOf('/') + 1).split("/");
191         final HashMap<String, String> metrics = new HashMap<>();
192         for (int i = 0; i < metricStrings.length; i++) {
193             final String[] metricKeyVal = metricStrings[i].split(":");
194             if (metricKeyVal.length != 2) {
195                 throw new IllegalArgumentException(
196                         String.format("Not a valid CVSSv3 vector string '%s', invalid metric component '%s'",
197                                 vectorString, metricStrings[i]));
198             }
199             metrics.put(metricKeyVal[0], metricKeyVal[1]);
200         }
201         if (!metrics.keySet().containsAll(BASE_METRICS_V3)) {
202             throw new IllegalArgumentException(
203                     String.format("Not a valid CVSSv3 vector string '%s'; missing one or more required Base Metrics;",
204                             vectorString));
205         }
206 
207         final CvssV3Data.Version version = CvssV3Data.Version.fromValue(versionString);
208         //"CVSS:3.1\/AV:L\/AC:L\/PR:L\/UI:N\/S:U\/C:N\/I:N\/A:H"
209         final CvssV3Data.AttackVectorType attackVector = CvssV3Data.AttackVectorType.fromValue(metrics.get("AV"));
210         final CvssV3Data.AttackComplexityType attackComplexity = CvssV3Data.AttackComplexityType.fromValue(metrics.get("AC"));
211         final CvssV3Data.PrivilegesRequiredType privilegesRequired = CvssV3Data.PrivilegesRequiredType.fromValue(metrics.get("PR"));
212         final CvssV3Data.UserInteractionType userInteraction = CvssV3Data.UserInteractionType.fromValue(metrics.get("UI"));
213         final CvssV3Data.ScopeType scope = CvssV3Data.ScopeType.fromValue(metrics.get("S"));
214         final CvssV3Data.CiaType confidentialityImpact = CvssV3Data.CiaType.fromValue(metrics.get("C"));
215         final CvssV3Data.CiaType integrityImpact = CvssV3Data.CiaType.fromValue(metrics.get("I"));
216         final CvssV3Data.CiaType availabilityImpact = CvssV3Data.CiaType.fromValue(metrics.get("A"));
217 
218         final String baseSeverityString = Cvss3Severity.of(baseScore.floatValue()).name();
219         final CvssV3Data.SeverityType baseSeverity = CvssV3Data.SeverityType.fromValue(baseSeverityString);
220         final CvssV3Data data = new CvssV3Data(version, vectorString, attackVector, attackComplexity,
221                 privilegesRequired, userInteraction, scope, confidentialityImpact, integrityImpact, availabilityImpact, baseScore,
222                 baseSeverity, null, null, null, null, null, null, null, null, null, null, null, null, null, null, null, null, null, null);
223         final CvssV3 cvss = new CvssV3(null, null, data, null, null);
224         return cvss;
225 
226     }
