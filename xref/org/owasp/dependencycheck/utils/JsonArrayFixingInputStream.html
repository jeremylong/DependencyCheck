18  package org.owasp.dependencycheck.utils;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  /**
26   * Fixes a poorly formatted jSON array such as those returned by
27   * <code>go list -json -m all</code>.
28   * <p>
29   * Example:</p>
30   * <code>
31   * { "name1": "value" }
32   * { "name2" : "value" }
33   * </code> Would be transformed into:  <code>
34   * [{ "name1": "value" },
35   * { "name2" : "value" }]
36   * </code>
37   * <p>
38   * Note that this is a naive implementation and will incorrectly transform a
39   * stream if there are '}' contained within the names of values.</p>
40   *
41   * @author Jeremy Long
42   */
43  public class JsonArrayFixingInputStream extends InputStream {
44  
45      /**
46       * The logger.
47       */
48      private static final Logger LOGGER = LoggerFactory.getLogger(JsonArrayFixingInputStream.class);
49      /**
50       * The buffer length.
51       */
52      private static final int BUFFER_SIZE = 2048;
53      /**
54       * The input stream to be filtered.
55       */
56      private final InputStream in;
57      /**
58       * The read buffer.
59       */
60      private final byte[] buffer = new byte[BUFFER_SIZE];
61      /**
62       * The read offset for the buffer.
63       */
64      private int bufferStart = 0;
65      /**
66       * The number characters available to read.
67       */
68      private int bufferAvailable = 0;
69      /**
70       * A flag indicating if the output still requires a trailing curly brace.
71       */
72      private boolean needsTrailingBrace = true;
73      /**
74       * A flag indicating if the stream is just starting.
75       */
76      private boolean firstRead = true;
77  
78      /**
79       * Constructs a new filtering input stream used to fix a poorly formatted
80       * JSON array.
81       *
82       * @param in the InputStream containing the poorly formed JSON array
83       */
84      public JsonArrayFixingInputStream(InputStream in) {
85          this.in = in;
86          this.bufferAvailable = 1;
87          buffer[bufferStart] = '[';
88      }
89  
90      /**
91       * Advances the buffer to the next block if required.
92       *
93       * @return the number of characters read
94       * @throws IOException thrown if there is an error reading from the stream
95       */
96      private boolean advanceStream() throws IOException {
97          boolean chomp = false;
98          if (firstRead) {
99              chomp = true;
100             firstRead = false;
101             bufferStart = 0;
102             bufferAvailable = in.read(buffer, 1, BUFFER_SIZE - 1) + 1;
103         } else if (bufferAvailable == 0) {
104             chomp = true;
105             bufferStart = 0;
106             bufferAvailable = in.read(buffer, 0, BUFFER_SIZE);
107         }
108         if (chomp) {
109             //chomp new lines
110             while (bufferAvailable > 0
111                     && (buffer[bufferStart + bufferAvailable - 1] == '\n'
112                     || buffer[bufferStart + bufferAvailable - 1] == '\r')) {
113                 bufferAvailable -= 1;
114             }
115             if (bufferAvailable == 0) {
116                 return advanceStream();
117             }
118         }
119         return bufferAvailable > 0;
120     }
121 
122     //CSOFF: NestedIfDepth
123     /**
124      * Increments the buffer start and appropriately inserts any necessary
125      * needed curly braces or commas.
126      *
127      * @throws IOException thrown if there is an error reading from the stream
128      */
129     private void incrementRead() throws IOException {
130         // buffer[bufferStart] is the value that was just read.
131         if (buffer[bufferStart] == '}') {
132             if (bufferAvailable > 1) {
133                 if (hasTrailingComma(bufferStart)) {
134                     //increment normally
135                     bufferStart += 1;
136                     bufferAvailable -= 1;
137                 } else {
138                     //replace '}' with ',' and don't increment counters
139                     buffer[bufferStart] = ',';
140                 }
141             } else {
142                 bufferAvailable = 0;
143                 if (advanceStream() || needsTrailingBrace) {
144                     if (bufferAvailable >= 1) {
145                         //the stream was advanced - so check if buffer[0] has a comma
146                         if (!hasTrailingComma(-1)) {
147                             //this only works because the output always
148                             // has a \n following a closing/open brace...
149                             if (buffer[bufferStart] == 10 || buffer[bufferStart] == 13) {
150                                 buffer[bufferStart] = ',';
151                             } else if (buffer[bufferStart] == '{'
152                                     && (buffer[bufferStart + 1] == 10 || buffer[bufferStart + 1] == 13)) {
153                                 buffer[bufferStart] = ',';
154                                 buffer[bufferStart + 1] = '{';
155                             }
156                         }
157                     } else if (needsTrailingBrace) {
158                         needsTrailingBrace = false;
159                         buffer[bufferStart] = ']';
160                         bufferAvailable = 1;
161                     }
162                 }
163             }
164         } else {
165             bufferStart += 1;
166             bufferAvailable -= 1;
167         }
168     }
169     //CSON: NestedIfDepth
170 
171     /**
172      * Searches for the offset from the buffer start for the next closing curly
173      * brace.
174      *
175      * @return the offset if found; otherwise <code>-1</code>
176      */
177     private int getClosingBraceOffset() {
178         for (int pos = bufferStart; pos < BUFFER_SIZE && pos < (bufferStart + bufferAvailable); pos++) {
179             if (buffer[pos] == '}') {
180                 return pos - bufferStart;
181             }
182         }
183         return -1;
184     }
185 
186     @Override
187     public int read() throws IOException {
188         if (advanceStream()) {
189             final int value = buffer[bufferStart];
190             incrementRead();
191             return value;
192         }
193         return -1;
194     }
195 
196     @Override
197     public int read(byte[] b) throws IOException {
198         return this.read(b, 0, b.length);
199     }
200 
201     @Override
202     public int read(byte[] b, int off, int len) throws IOException {
203         if (advanceStream()) {
204             final int brace = getClosingBraceOffset();
205             if (brace == 0) {
206                 b[off] = buffer[bufferStart];
207                 incrementRead();
208                 return 1;
209             }
210             int copyLength = Math.min(bufferAvailable, len);
211             if (brace > 0) {
212                 copyLength = Math.min(copyLength, brace);
213             }
214             final int copyEnd = copyLength + off;
215             for (int pos = off; pos < copyEnd; pos++) {
216                 b[pos] = buffer[bufferStart];
217                 bufferStart += 1;
218             }
219             bufferAvailable -= copyLength;
220             return copyLength;
221         }
222         return -1;
223     }
224 
225     @Override
226     public long skip(long n) throws IOException {
227         throw new UnsupportedOperationException("Unable to skip using the JsonArrayFixingInputStream");
228     }
229 
230     @Override
231     public int available() throws IOException {
232         return in.available() + bufferAvailable;
233     }
234 
235     @Override
236     public void close() throws IOException {
237         in.close();
238     }
239 
240     @Override
241     public boolean markSupported() {
242         return false;
243     }
244 
245     /**
246      * Tests if the buffer has a trailing comma after having just output a
247      * closing curly brace.
248      *
249      * @param start the position to start checking the buffer from
250      * @return <code>true</code> if there is a trailing comma; otherwise
251      * <code>false</code>
252      */
253     private boolean hasTrailingComma(int start) {
254         return (bufferAvailable >= 1 && buffer[start + 1] == ',')
255                 || bufferAvailable >= 2 && isWhiteSpace(buffer[start + 1]) && buffer[start + 2] == ',';
256     }
257 
258     /**
259      * Tests if the byte passed in is a white space character.
260      *
261      * @param c the byte representing the character to test
262      * @return <code>true</code> if the byte is white space; otherwise
263      * <code>false</code>
264      */
265     protected boolean isWhiteSpace(byte c) {
266         switch (c) {
267             case '\n':
268             case '\r':
269             case '\t':
270             case ' ':
271                 return true;
272             default:
273                 return false;
274         }
275     }
