827     /**
828      * Check if a given key is considered to have a value with sensitive data.
829      *
830      * @param key the key to determine if the property should be masked
831      * @return <code>true</code> if the key is for a sensitive property value;
832      * otherwise <code>false</code>
833      */
834     private boolean isKeyMasked(@NotNull String key) {
835         if (maskedKeys == null || maskedKeys.isEmpty()) {
836             initMaskedKeys();
837         }
838         return maskedKeys.stream().anyMatch(maskExp -> maskExp.test(key));
839     }

841     /**
842      * Obtains the printable/loggable value for a given key/value pair. This
843      * will mask some values so as to not leak sensitive information.
844      *
845      * @param key the property key
846      * @param value the property value
847      * @return the printable value
848      */
849     protected String getPrintableValue(@NotNull String key, String value) {
850         String printableValue = null;
851         if (value != null) {
852             printableValue = isKeyMasked(key) ? "********" : value;
853         }
854         return printableValue;
855     }

857     /**
858      * Initializes the masked keys collection. This is done outside of the
859      * {@link #initialize(java.lang.String)} method because a caller may use the
860      * {@link #mergeProperties(java.io.File)} to add additional properties after
861      * the call to initialize.
862      */
863     protected void initMaskedKeys() {
864         final String[] masked = getArray(Settings.KEYS.MASKED_PROPERTIES);
865         if (masked == null) {
866             maskedKeys = new ArrayList<>();
867         } else {
868             maskedKeys = Arrays.asList(masked)
869                     .stream()
870                     .map(v -> Pattern.compile(v).asPredicate())
871                     .collect(Collectors.toList());
872         }
873     }

875     /**
876      * Logs the properties. This will not log any properties that contain
877      * 'password' in the key.
878      *
879      * @param header the header to print with the log message
880      * @param properties the properties to log
881      */
882     private void logProperties(@NotNull final String header, @NotNull final Properties properties) {
883         if (LOGGER.isDebugEnabled()) {
884             initMaskedKeys();
885             final StringWriter sw = new StringWriter();
886             try (PrintWriter pw = new PrintWriter(sw)) {
887                 pw.format("%s:%n%n", header);
888                 final Enumeration<?> e = properties.propertyNames();
889                 while (e.hasMoreElements()) {
890                     final String key = (String) e.nextElement();
891                     final String value = getPrintableValue(key, properties.getProperty(key));
892                     if (value != null) {
893                         pw.format("%s='%s'%n", key, value);
894                     }
895                 }
896                 pw.flush();
897                 LOGGER.debug(sw.toString());
898             }
899         }
900     }

1126     /**
1127      * Attempts to retrieve the folder containing the Jar file containing the
1128      * Settings class.
1129      *
1130      * @return a File object
1131      */
1132     private File getJarPath() {
1133         String decodedPath = ".";
1134         String jarPath = "";
1135         final ProtectionDomain domain = Settings.class.getProtectionDomain();
1136         if (domain != null && domain.getCodeSource() != null && domain.getCodeSource().getLocation() != null) {
1137             jarPath = Settings.class.getProtectionDomain().getCodeSource().getLocation().getPath();
1138         }
1139         try {
1140             decodedPath = URLDecoder.decode(jarPath, StandardCharsets.UTF_8.name());
1141         } catch (UnsupportedEncodingException ex) {
1142             LOGGER.trace("", ex);
1143         }
1144 
1145         final File path = new File(decodedPath);
1146         if (path.getName().toLowerCase().endsWith(".jar")) {
1147             return path.getParentFile();
1148         } else {
1149             return new File(".");
1150         }

1346     /**
1347      * Returns a connection string from the configured properties. If the
1348      * connection string contains a %s, this method will determine the 'data'
1349      * directory and replace the %s with the path to the data directory. If the
1350      * data directory does not exist it will be created.
1351      *
1352      * @param connectionStringKey the property file key for the connection
1353      * string
1354      * @param dbFileNameKey the settings key for the db filename
1355      * @return the connection string
1356      * @throws IOException thrown the data directory cannot be created
1357      * @throws InvalidSettingException thrown if there is an invalid setting
1358      */
1359     public String getConnectionString(String connectionStringKey, String dbFileNameKey)
1360             throws IOException, InvalidSettingException {
1361         final String connStr = getString(connectionStringKey);
1362         if (connStr == null) {
1363             final String msg = String.format("Invalid properties file; %s is missing.", connectionStringKey);
1364             throw new InvalidSettingException(msg);
1365         }
1366         if (connStr.contains("%s")) {
1367             final File directory = getH2DataDirectory();
1368             LOGGER.debug("Data directory: {}", directory);
1369             String fileName = null;
1370             if (dbFileNameKey != null) {
1371                 fileName = getString(dbFileNameKey);
1372             }
1373             if (fileName == null) {
1374                 final String msg = String.format("Invalid properties file to get a file based connection string; '%s' must be defined.",
1375                         dbFileNameKey);
1376                 throw new InvalidSettingException(msg);
1377             }
1378             if (connStr.startsWith("jdbc:h2:file:") && fileName.endsWith(".mv.db")) {
1379                 fileName = fileName.substring(0, fileName.length() - 6);
1380             }
1381             // yes, for H2 this path won't actually exists - but this is sufficient to get the value needed
1382             final File dbFile = new File(directory, fileName);
1383             final String cString = String.format(connStr, dbFile.getCanonicalPath());
1384             LOGGER.debug("Connection String: '{}'", cString);
1385             return cString;
1386         }
1387         return connStr;
1388     }

1431      * Generates a new temporary file name that is guaranteed to be unique.
1432      *
1433      * @param prefix the prefix for the file name to generate
1434      * @param extension the extension of the generated file name
1435      * @return a temporary File
1436      * @throws java.io.IOException if any.
1437      */
1438     public File getTempFile(@NotNull final String prefix, @NotNull final String extension) throws IOException {
1439         final File dir = getTempDirectory();
1440         final String tempFileName = String.format("%s%s.%s", prefix, UUID.randomUUID().toString(), extension);
1441         final File tempFile = new File(dir, tempFileName);
1442         if (tempFile.exists()) {
1443             return getTempFile(prefix, extension);
1444         }
1445         return tempFile;
