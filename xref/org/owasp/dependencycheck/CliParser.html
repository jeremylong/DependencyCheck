78      /**
79       * Parses the arguments passed in and captures the results for later use.
80       *
81       * @param args the command line arguments
82       * @throws FileNotFoundException is thrown when a 'file' argument does not
83       * point to a file that exists.
84       * @throws ParseException is thrown when a Parse Exception occurs.
85       */
86      public void parse(String[] args) throws FileNotFoundException, ParseException {
87          line = parseArgs(args);
88  
89          if (line != null) {
90              validateArgs();
91          }
92      }

94      /**
95       * Parses the command line arguments.
96       *
97       * @param args the command line arguments
98       * @return the results of parsing the command line arguments
99       * @throws ParseException if the arguments are invalid
100      */
101     private CommandLine parseArgs(String[] args) throws ParseException {
102         final CommandLineParser parser = new DefaultParser();
103         final Options options = createCommandLineOptions();
104         return parser.parse(options, args);
105     }

107     /**
108      * Validates that the command line arguments are valid.
109      *
110      * @throws FileNotFoundException if there is a file specified by either the
111      * SCAN or CPE command line arguments that does not exist.
112      * @throws ParseException is thrown if there is an exception parsing the
113      * command line.
114      */
115     private void validateArgs() throws FileNotFoundException, ParseException {
116         if (isUpdateOnly() || isRunScan()) {
117             String value = line.getOptionValue(ARGUMENT.CVE_VALID_FOR_HOURS);
118             if (value != null) {
119                 try {
120                     final int i = Integer.parseInt(value);
121                     if (i < 0) {
122                         throw new ParseException("Invalid Setting: cveValidForHours must be a number greater than or equal to 0.");
123                     }
124                 } catch (NumberFormatException ex) {
125                     throw new ParseException("Invalid Setting: cveValidForHours must be a number greater than or equal to 0.");
126                 }
127             }
128             value = line.getOptionValue(ARGUMENT.CVE_START_YEAR);
129             if (value != null) {
130                 try {
131                     final int i = Integer.parseInt(value);
132                     if (i < 2002) {
133                         throw new ParseException("Invalid Setting: cveStartYear must be a number greater than or equal to 2002.");
134                     }
135                 } catch (NumberFormatException ex) {
136                     throw new ParseException("Invalid Setting: cveStartYear must be a number greater than or equal to 2002.");
137                 }
138             }
139 
140         }
141         if (isRunScan()) {
142             validatePathExists(getScanFiles(), ARGUMENT.SCAN);
143             validatePathExists(getReportDirectory(), ARGUMENT.OUT);
144             final String pathToCore = getStringArgument(ARGUMENT.PATH_TO_CORE);
145             if (pathToCore != null) {
146                 validatePathExists(pathToCore, ARGUMENT.PATH_TO_CORE);
147             }
148             if (line.hasOption(ARGUMENT.OUTPUT_FORMAT)) {
149                 String validating = null;
150                 try {
151                     for (String format : getReportFormat()) {
152                         validating = format;
153                         Format.valueOf(format);
154                     }
155                 } catch (IllegalArgumentException ex) {
156                     final String msg = String.format("An invalid 'format' of '%s' was specified. "
157                             + "Supported output formats are " + SUPPORTED_FORMATS, validating);
158                     throw new ParseException(msg);
159                 }
160             }
161             final String base = getStringArgument(ARGUMENT.CVE_BASE_URL);
162             final String modified = getStringArgument(ARGUMENT.CVE_MODIFIED_URL);
163             if ((base != null && modified == null) || (base == null && modified != null)) {
164                 final String msg = "If one of the CVE URLs is specified they must all be specified; please add the missing CVE URL.";
165                 throw new ParseException(msg);
166             }
167             if (line.hasOption(ARGUMENT.SYM_LINK_DEPTH)) {
168                 try {
169                     final int i = Integer.parseInt(line.getOptionValue(ARGUMENT.SYM_LINK_DEPTH));
170                     if (i < 0) {
171                         throw new ParseException("Symbolic Link Depth (symLink) must be greater than zero.");
172                     }
173                 } catch (NumberFormatException ex) {
174                     throw new ParseException("Symbolic Link Depth (symLink) is not a number.");
175                 }
176             }
177         }
178     }

180     /**
181      * Validates whether or not the path(s) points at a file that exists; if the
182      * path(s) does not point to an existing file a FileNotFoundException is
183      * thrown.
184      *
185      * @param paths the paths to validate if they exists
186      * @param optType the option being validated (e.g. scan, out, etc.)
187      * @throws FileNotFoundException is thrown if one of the paths being
188      * validated does not exist.
189      */
190     private void validatePathExists(String[] paths, String optType) throws FileNotFoundException {
191         for (String path : paths) {
192             validatePathExists(path, optType);
193         }
194     }

196     /**
197      * Validates whether or not the path points at a file that exists; if the
198      * path does not point to an existing file a FileNotFoundException is
199      * thrown.
200      *
201      * @param path the paths to validate if they exists
202      * @param argumentName the argument being validated (e.g. scan, out, etc.)
203      * @throws FileNotFoundException is thrown if the path being validated does
204      * not exist.
205      */
206     private void validatePathExists(String path, String argumentName) throws FileNotFoundException {
207         if (path == null) {
208             isValid = false;
209             final String msg = String.format("Invalid '%s' argument: null", argumentName);
210             throw new FileNotFoundException(msg);
211         } else if (!path.contains("*") && !path.contains("?")) {
212             File f = new File(path);
213             final String[] formats = this.getReportFormat();
214             if ("o".equalsIgnoreCase(argumentName.substring(0, 1)) && formats.length == 1 && !"ALL".equalsIgnoreCase(formats[0])) {
215                 final String checkPath = path.toLowerCase();
216                 if (checkPath.endsWith(".html") || checkPath.endsWith(".xml") || checkPath.endsWith(".htm")
217                         || checkPath.endsWith(".csv") || checkPath.endsWith(".json")) {
218                     if (f.getParentFile() == null) {
219                         f = new File(".", path);
220                     }
221                     if (!f.getParentFile().isDirectory()) {
222                         isValid = false;
223                         final String msg = String.format("Invalid '%s' argument: '%s' - directory path does not exist", argumentName, path);
224                         throw new FileNotFoundException(msg);
225                     }
226                 }
227             } else if ("o".equalsIgnoreCase(argumentName.substring(0, 1)) && !f.isDirectory()) {
228                 if (f.getParentFile() != null && f.getParentFile().isDirectory() && !f.mkdir()) {
229                     isValid = false;
230                     final String msg = String.format("Invalid '%s' argument: '%s' - unable to create the output directory", argumentName, path);
231                     throw new FileNotFoundException(msg);
232                 }
233                 if (!f.isDirectory()) {
234                     isValid = false;
235                     final String msg = String.format("Invalid '%s' argument: '%s' - path does not exist", argumentName, path);
236                     throw new FileNotFoundException(msg);
237                 }
238             } else if (!f.exists()) {
239                 isValid = false;
240                 final String msg = String.format("Invalid '%s' argument: '%s' - path does not exist", argumentName, path);
241                 throw new FileNotFoundException(msg);
242             }
243 //        } else if (path.startsWith("//") || path.startsWith("\\\\")) {
244 //            isValid = false;
245 //            final String msg = String.format("Invalid '%s' argument: '%s'%nUnable to scan paths that start with '//'.", argumentName, path);
246 //            throw new FileNotFoundException(msg);
247         } else if ((path.endsWith("/*") && !path.endsWith("**/*")) || (path.endsWith("\\*") && path.endsWith("**\\*"))) {
248             LOGGER.warn("Possibly incorrect path '{}' from argument '{}' because it ends with a slash star; "
249                     + "dependency-check uses ant-style paths", path, argumentName);
250         }
251     }
