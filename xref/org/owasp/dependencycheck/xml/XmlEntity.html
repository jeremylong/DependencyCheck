295     /**
296      * Converts a named XML entity into its HTML encoded Unicode code point.
297      *
298      * @param s the named entity (note, this should not include the leading
299      * '&amp;' or trailing ';'
300      * @return the HTML encoded Unicode code point representation of the named
301      * entity
302      */
303     public static String fromNamedReference(CharSequence s) {
304         if (s == null) {
305             return null;
306         }
307         final Integer code = SPECIALS.get(s.toString());
308         if (code != null) {
309             return "&#" + code + ";";
310         }
311         return null;
312     }
