420     /**
421      * Processes a given dependency to determine if any CPE, CVE, CWE, or CVSS
422      * scores should be suppressed. If any should be, they are removed from the
423      * dependency.
424      *
425      * @param dependency a project dependency to analyze
426      */
427     public void process(Dependency dependency) {
428         if (filePath != null && !filePath.matches(dependency.getFilePath())) {
429             return;
430         }
431         if (sha1 != null && !sha1.equalsIgnoreCase(dependency.getSha1sum())) {
432             return;
433         }
434         if (hasGav()) {
435             final Iterator<Identifier> itr = dependency.getSoftwareIdentifiers().iterator();
436             boolean found = false;
437             while (itr.hasNext()) {
438                 final Identifier i = itr.next();
439                 if (identifierMatches(this.gav, i)) {
440                     found = true;
441                     break;
442                 }
443             }
444             if (!found) {
445                 return;
446             }
447         }
448         if (hasPackageUrl()) {
449             final Iterator<Identifier> itr = dependency.getSoftwareIdentifiers().iterator();
450             boolean found = false;
451             while (itr.hasNext()) {
452                 final Identifier i = itr.next();
453                 if (purlMatches(this.packageUrl, i)) {
454                     found = true;
455                     break;
456                 }
457             }
458             if (!found) {
459                 return;
460             }
461         }
462 
463         if (this.hasCpe()) {
464             final Set<Identifier> removalList = new HashSet<>();
465             for (Identifier i : dependency.getVulnerableSoftwareIdentifiers()) {
466                 for (PropertyType c : this.cpe) {
467                     if (identifierMatches(c, i)) {
468                         if (!isBase()) {
469                             if (this.notes != null) {
470                                 i.setNotes(this.notes);
471                             }
472                             dependency.addSuppressedIdentifier(i);
473                         }
474                         removalList.add(i);
475                         break;
476                     }
477                 }
478             }
479             removalList.forEach((i) -> dependency.removeVulnerableSoftwareIdentifier(i));
480         }
481         if (hasCve() || hasVulnerabilityName() || hasCwe() || hasCvssBelow()) {
482             final Set<Vulnerability> removeVulns = new HashSet<>();
483             for (Vulnerability v : dependency.getVulnerabilities()) {
484                 boolean remove = false;
485                 for (String entry : this.cve) {
486                     if (entry.equalsIgnoreCase(v.getName())) {
487                         removeVulns.add(v);
488                         remove = true;
489                         break;
490                     }
491                 }
492                 if (!remove && this.cwe != null && !v.getCwes().isEmpty()) {
493                     for (String entry : this.cwe) {
494                         final String toMatch = String.format("CWE-%s", entry);
495                         if (v.getCwes().stream().anyMatch(toTest -> toMatch.regionMatches(0, toTest, 0, toMatch.length()))) {
496                             remove = true;
497                             removeVulns.add(v);
498                             break;
499                         }
500                     }
501                 }
502                 if (!remove && v.getName() != null) {
503                     for (PropertyType entry : this.vulnerabilityNames) {
504                         if (entry.matches(v.getName())) {
505                             remove = true;
506                             removeVulns.add(v);
507                             break;
508                         }
509 
510                     }
511                 }
512                 if (!remove) {
513                     for (float cvss : this.cvssBelow) {
514                         if (v.getCvssV2() != null && v.getCvssV2().getScore() < cvss) {
515                             remove = true;
516                             removeVulns.add(v);
517                             break;
518                         }
519                         if (v.getCvssV3() != null && v.getCvssV3().getBaseScore() < cvss) {
520                             remove = true;
521                             removeVulns.add(v);
522                             break;
523                         }
524                     }
525                 }
526                 if (remove) {
527                     if (!isBase()) {
528                         if (this.notes != null) {
529                             v.setNotes(this.notes);
530                         }
531                         dependency.addSuppressedVulnerability(v);
532                     }
533                 }
534             }
535             removeVulns.forEach((v) -> {
536                 dependency.removeVulnerability(v);
537             });
538         }
539     }

541     /**
542      * Identifies if the cpe specified by the cpe suppression rule does not
543      * specify a version.
544      *
545      * @param c a suppression rule identifier
546      * @return true if the property type does not specify a version; otherwise
547      * false
548      */
549     protected boolean cpeHasNoVersion(PropertyType c) {
550         return !c.isRegex() && countCharacter(c.getValue(), ':') <= 3;
551     }
552 
553     /**
554      * Counts the number of occurrences of the character found within the
555      * string.
556      *
557      * @param str the string to check
558      * @param c the character to count
559      * @return the number of times the character is found in the string
560      */
561     private int countCharacter(String str, char c) {
562         int count = 0;
563         int pos = str.indexOf(c) + 1;
564         while (pos > 0) {
565             count += 1;
566             pos = str.indexOf(c, pos) + 1;
567         }
568         return count;
569     }

571     /**
572      * Determines if the cpeEntry specified as a PropertyType matches the given
573      * Identifier.
574      *
575      * @param suppressionEntry a suppression rule entry
576      * @param identifier a CPE identifier to check
577      * @return true if the entry matches; otherwise false
578      */
579     protected boolean purlMatches(PropertyType suppressionEntry, Identifier identifier) {
580         if (identifier instanceof PurlIdentifier) {
581             final PurlIdentifier purl = (PurlIdentifier) identifier;
582             return suppressionEntry.matches(purl.toString());
583         }
584         return false;
585     }

587     /**
588      * Determines if the cpeEntry specified as a PropertyType matches the given
589      * Identifier.
590      *
591      * @param suppressionEntry a suppression rule entry
592      * @param identifier a CPE identifier to check
593      * @return true if the entry matches; otherwise false
594      */
595     protected boolean identifierMatches(PropertyType suppressionEntry, Identifier identifier) {
596         if (identifier instanceof PurlIdentifier) {
597             final PurlIdentifier purl = (PurlIdentifier) identifier;
598             return suppressionEntry.matches(purl.toGav());
599         } else if (identifier instanceof CpeIdentifier) {
600             //TODO check for regex - not just type
601             final Cpe cpeId = ((CpeIdentifier) identifier).getCpe();
602             if (suppressionEntry.isRegex()) {
603                 try {
604                     return suppressionEntry.matches(cpeId.toCpe22Uri());
605                 } catch (CpeEncodingException ex) {
606                     LOGGER.debug("Unable to convert CPE to 22 URI?" + cpeId.toString());
607                 }
608             } else if (suppressionEntry.isCaseSensitive()) {
609                 try {
610                     return cpeId.toCpe22Uri().startsWith(suppressionEntry.getValue());
611                 } catch (CpeEncodingException ex) {
612                     LOGGER.debug("Unable to convert CPE to 22 URI?" + cpeId.toString());
613                 }
614             } else {
615                 final String id;
616                 try {
617                     id = cpeId.toCpe22Uri().toLowerCase();
618                 } catch (CpeEncodingException ex) {
619                     LOGGER.debug("Unable to convert CPE to 22 URI?" + cpeId.toString());
620                     return false;
621                 }
622                 final String check = suppressionEntry.getValue().toLowerCase();
623                 return id.startsWith(check);
624             }
625         }
626         return suppressionEntry.matches(identifier.getValue());
627     }

629     /**
630      * Standard toString implementation.
631      *
632      * @return a string representation of this object
633      */
634     @Override
635     public String toString() {
636         final StringBuilder sb = new StringBuilder(64);
637         sb.append("SuppressionRule{");
638         if (until != null) {
639             sb.append("until=").append(until).append(',');
640         }
641         if (filePath != null) {
642             sb.append("filePath=").append(filePath).append(',');
643         }
644         if (sha1 != null) {
645             sb.append("sha1=").append(sha1).append(',');
646         }
647         if (gav != null) {
648             sb.append("gav=").append(gav).append(',');
649         }
650         if (cpe != null && !cpe.isEmpty()) {
651             sb.append("cpe={");
652             cpe.forEach((pt) -> {
653                 sb.append(pt).append(',');
654             });
655             sb.append('}');
656         }
657         if (cwe != null && !cwe.isEmpty()) {
658             sb.append("cwe={");
659             cwe.forEach((s) -> {
660                 sb.append(s).append(',');
661             });
662             sb.append('}');
663         }
664         if (cve != null && !cve.isEmpty()) {
665             sb.append("cve={");
666             cve.forEach((s) -> {
667                 sb.append(s).append(',');
668             });
669             sb.append('}');
670         }
671         if (vulnerabilityNames != null && !vulnerabilityNames.isEmpty()) {
672             sb.append("vulnerabilityName={");
673             vulnerabilityNames.forEach((pt) -> {
674                 sb.append(pt).append(',');
675             });
676             sb.append('}');
677         }
678         if (cvssBelow != null && !cvssBelow.isEmpty()) {
679             sb.append("cvssBelow={");
680             cvssBelow.forEach((s) -> {
681                 sb.append(s).append(',');
682             });
683             sb.append('}');
684         }
685         sb.append('}');
686         return sb.toString();
687     }
