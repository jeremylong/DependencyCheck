40      /**
41       * Adds a CPE to a vulnerability; if the CPE is not contained in the
42       * database it is first added.
43       *
44       * @param conn the database connection
45       * @param vulnerabilityId the vulnerability id
46       * @param part the CPE part
47       * @param vendor the CPE vendor
48       * @param product the CPE product
49       * @param version the CPE version
50       * @param update the CPE update version
51       * @param edition the CPE edition
52       * @param language the CPE language
53       * @param swEdition the CPE software edition
54       * @param targetSw the CPE target software
55       * @param targetHw the CPE target hardware
56       * @param other the CPE other
57       * @param ecosystem the ecosystem
58       * @param versionEndExcluding a version range to identify the software
59       * @param versionEndIncluding a version range to identify the software
60       * @param versionStartExcluding a version range to identify the software
61       * @param versionStartIncluding a version range to identify the software
62       * @param vulnerable a flag indicating whether or not the software is
63       * vulnerable
64       * @throws SQLException thrown if there is an error adding the CPE or
65       * software reference
66       */
67      public static void insertSoftware(final Connection conn, int vulnerabilityId, String part, String vendor,
68              String product, String version, String update, String edition, String language, String swEdition,
69              String targetSw, String targetHw, String other, String ecosystem, String versionEndExcluding,
70              String versionEndIncluding, String versionStartExcluding, String versionStartIncluding, Boolean vulnerable) throws SQLException {
71          int cpeID = 0;
72          try (PreparedStatement selectCpeId = conn.prepareStatement("SELECT id, ecosystem FROM cpeEntry WHERE part=? AND vendor=? AND product=? "
73                  + "AND version=? AND update_version=? AND edition=? AND lang=? AND sw_edition=? AND target_sw=? AND target_hw=? AND other=?")) {
74              selectCpeId.setString(1, part);
75              selectCpeId.setString(2, vendor);
76              selectCpeId.setString(3, product);
77              selectCpeId.setString(4, version);
78              selectCpeId.setString(5, update);
79              selectCpeId.setString(6, edition);
80              selectCpeId.setString(7, language);
81              selectCpeId.setString(8, swEdition);
82              selectCpeId.setString(9, targetSw);
83              selectCpeId.setString(10, targetHw);
84              selectCpeId.setString(11, other);
85  
86              try (ResultSet rs = selectCpeId.executeQuery()) {
87                  if (rs.next()) {
88                      cpeID = rs.getInt(1);
89                      final String e = rs.getString(2);
90                      if (e == null && ecosystem != null) {
91                          try (PreparedStatement updateEcosystem = conn.prepareStatement("UPDATE cpeEntry SET ecosystem=? WHERE id=?")) {
92                              updateEcosystem.setString(1, ecosystem);
93                              updateEcosystem.setInt(2, cpeID);
94                              updateEcosystem.execute();
95                          }
96                      }
97                  }
98              }
99          }
100         if (cpeID == 0) {
101             final String[] returnedColumns = {"id"};
102             try (PreparedStatement insertCpe = conn.prepareStatement("INSERT INTO cpeEntry (part, vendor, product, version, update_version, "
103                     + "edition, lang, sw_edition, target_sw, target_hw, other, ecosystem) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
104                     returnedColumns)) {
105                 insertCpe.setString(1, part);
106                 insertCpe.setString(2, vendor);
107                 insertCpe.setString(3, product);
108                 insertCpe.setString(4, version);
109                 insertCpe.setString(5, update);
110                 insertCpe.setString(6, edition);
111                 insertCpe.setString(7, language);
112                 insertCpe.setString(8, swEdition);
113                 insertCpe.setString(9, targetSw);
114                 insertCpe.setString(10, targetHw);
115                 insertCpe.setString(11, other);
116                 setStringOrNull(insertCpe, 12, ecosystem);
117                 insertCpe.executeUpdate();
118                 try (ResultSet rs = insertCpe.getGeneratedKeys()) {
119                     if (rs.next()) {
120                         cpeID = rs.getInt(1);
121                     }
122                 }
123             }
124         }
125         //CSON: ParameterNumber
126 
127         try (PreparedStatement insertSoftware = conn.prepareStatement("INSERT INTO software (cveid, cpeEntryId, "
128                 + "versionEndExcluding, versionEndIncluding, versionStartExcluding, versionStartIncluding, "
129                 + "vulnerable) VALUES (?, ?, ?, ?, ?, ?, ?)")) {
130             insertSoftware.setInt(1, vulnerabilityId);
131             insertSoftware.setInt(2, cpeID);
132 
133             setStringOrNull(insertSoftware, 3, versionEndExcluding);
134             setStringOrNull(insertSoftware, 4, versionEndIncluding);
135             setStringOrNull(insertSoftware, 5, versionStartExcluding);
136             setStringOrNull(insertSoftware, 6, versionStartIncluding);
137             setBooleanOrNull(insertSoftware, 7, vulnerable);
138             insertSoftware.execute();
139         }
140     }

143     /**
144      * Updates or inserts the vulnerability into the database. If updating a
145      * vulnerability the method will delete all software, CWE, and references
146      * and new entries will be added later.
147      *
148      * @param conn the database connection
149      * @param cve the CVE identifier
150      * @param description the vulnerability description
151      * @param v2Severity the CVSS v2 severity
152      * @param v2ExploitabilityScore the CVSS v2 exploitability score
153      * @param v2ImpactScore the CVSS v2 impact score
154      * @param v2AcInsufInfo the CVSS v2 AcInsufInfo
155      * @param v2ObtainAllPrivilege the CVSS v2 obtain all privilege flag
156      * @param v2ObtainUserPrivilege the CVSS v2 obtain user privilege flag
157      * @param v2ObtainOtherPrivilege the CVSS v2 obtain other privilege flag
158      * @param v2UserInteractionRequired the CVSS v2 user interaction required
159      * flag
160      * @param v2Score the CVSS v2 score
161      * @param v2AccessVector the CVSS v2 access vector
162      * @param v2AccessComplexity the CVSS v2 access complexity
163      * @param v2Authentication the CVSS v2 authentication
164      * @param v2ConfidentialityImpact the CVSS v2 confidentiality impact
165      * @param v2IntegrityImpact the CVSS v2 integrity impact
166      * @param v2AvailabilityImpact the CVSS v2 availability impact
167      * @param v2Version the CVSS v2 version
168      * @param v3ExploitabilityScore the CVSS v3 exploitability score
169      * @param v3ImpactScore the CVSS v3 impact score
170      * @param v3AttackVector the CVSS v3 attack vector
171      * @param v3AttackComplexity the CVSS v3 attack complexity
172      * @param v3PrivilegesRequired the CVSS v3 privilege required flag
173      * @param v3UserInteraction the CVSS v3 user interaction required flag
174      * @param v3Scope the CVSS v3 scope
175      * @param v3ConfidentialityImpact the CVSS v3 confidentiality impact
176      * @param v3IntegrityImpact the CVSS v3 integrity impact
177      * @param v3AvailabilityImpact the CVSS v3 availability impact
178      * @param v3BaseScore the CVSS v3 base score
179      * @param v3BaseSeverity the CVSS v3 base severity
180      * @param v3Version the CVSS v3 version
181      * @return a result set containing the vulnerability id
182      * @throws SQLException thrown if there is an error updating or inserting
183      * the vulnerability
184      */
185     public static ResultSet updateVulnerability(final Connection conn, String cve,
186             String description, String v2Severity, Float v2ExploitabilityScore,
187             Float v2ImpactScore, Boolean v2AcInsufInfo, Boolean v2ObtainAllPrivilege,
188             Boolean v2ObtainUserPrivilege, Boolean v2ObtainOtherPrivilege, Boolean v2UserInteractionRequired,
189             Float v2Score, String v2AccessVector, String v2AccessComplexity,
190             String v2Authentication, String v2ConfidentialityImpact, String v2IntegrityImpact,
191             String v2AvailabilityImpact, String v2Version, Float v3ExploitabilityScore,
192             Float v3ImpactScore, String v3AttackVector, String v3AttackComplexity,
193             String v3PrivilegesRequired, String v3UserInteraction, String v3Scope,
194             String v3ConfidentialityImpact, String v3IntegrityImpact, String v3AvailabilityImpact,
195             Float v3BaseScore, String v3BaseSeverity, String v3Version) throws SQLException {
196 
197         final SimpleResultSet ret = new SimpleResultSet();
198         ret.addColumn("id", Types.INTEGER, 10, 0);
199         final String url = conn.getMetaData().getURL();
200         if ("jdbc:columnlist:connection".equals(url)) {
201             // Virtual Table Functions get called multiple times by H2
202             // JDBC URL jdbc:columnlist:connection indicates that H2 only wants to discover
203             // the metadata (list of result columns) of the result and is not interested in the actual
204             // execution of the function, so we should exit early with an empty resultset.
205             return ret;
206         }
207 
208         int vulnerabilityId = 0;
209         try (PreparedStatement selectVulnerabilityId = conn.prepareStatement("SELECT id FROM VULNERABILITY CVE WHERE cve=?")) {
210             selectVulnerabilityId.setString(1, cve);
211             try (ResultSet rs = selectVulnerabilityId.executeQuery()) {
212                 if (rs.next()) {
213                     vulnerabilityId = rs.getInt(1);
214                 }
215             }
216         }
217         PreparedStatement merge = null;
218         try {
219             if (vulnerabilityId > 0) {
220                 //do deletes and updates
221                 try (PreparedStatement refs = conn.prepareStatement("DELETE FROM reference WHERE cveid = ?")) {
222                     refs.setInt(1, vulnerabilityId);
223                     refs.executeUpdate();
224                 }
225                 try (PreparedStatement software = conn.prepareStatement("DELETE FROM software WHERE cveid = ?")) {
226                     software.setInt(1, vulnerabilityId);
227                     software.executeUpdate();
228                 }
229                 try (PreparedStatement cwe = conn.prepareStatement("DELETE FROM cweEntry WHERE cveid = ?")) {
230                     cwe.setInt(1, vulnerabilityId);
231                     cwe.executeUpdate();
232                 }
233                 merge = conn.prepareStatement("UPDATE VULNERABILITY SET description=?, "
234                         + "v2Severity=?, v2ExploitabilityScore=?, "
235                         + "v2ImpactScore=?, v2AcInsufInfo=?, v2ObtainAllPrivilege=?, "
236                         + "v2ObtainUserPrivilege=?, v2ObtainOtherPrivilege=?, v2UserInteractionRequired=?, "
237                         + "v2Score=?, v2AccessVector=?, v2AccessComplexity=?, "
238                         + "v2Authentication=?, v2ConfidentialityImpact=?, v2IntegrityImpact=?, "
239                         + "v2AvailabilityImpact=?, v2Version=?, v3ExploitabilityScore=?, "
240                         + "v3ImpactScore=?, v3AttackVector=?, v3AttackComplexity=?, "
241                         + "v3PrivilegesRequired=?, v3UserInteraction=?, v3Scope=?, "
242                         + "v3ConfidentialityImpact=?, v3IntegrityImpact=?, v3AvailabilityImpact=?, "
243                         + "v3BaseScore=?, v3BaseSeverity=?, v3Version=? "
244                         + "WHERE id=?");
245             } else {
246                 //just do insert
247                 final String[] returnedColumns = {"id"};
248                 merge = conn.prepareStatement("INSERT INTO VULNERABILITY (description, "
249                         + "v2Severity, v2ExploitabilityScore, "
250                         + "v2ImpactScore, v2AcInsufInfo, v2ObtainAllPrivilege, "
251                         + "v2ObtainUserPrivilege, v2ObtainOtherPrivilege, v2UserInteractionRequired, "
252                         + "v2Score, v2AccessVector, v2AccessComplexity, "
253                         + "v2Authentication, v2ConfidentialityImpact, v2IntegrityImpact, "
254                         + "v2AvailabilityImpact, v2Version, v3ExploitabilityScore, "
255                         + "v3ImpactScore, v3AttackVector, v3AttackComplexity, "
256                         + "v3PrivilegesRequired, v3UserInteraction, v3Scope, "
257                         + "v3ConfidentialityImpact, v3IntegrityImpact, v3AvailabilityImpact, "
258                         + "v3BaseScore, v3BaseSeverity, v3Version, cve) VALUES (?, ?, ?, ?, ?, ?, "
259                         + "?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
260                         returnedColumns);
261             }
262 
263             merge.setString(1, description);
264 
265             setStringOrNull(merge, 2, v2Severity);
266             setFloatOrNull(merge, 3, v2ExploitabilityScore);
267             setFloatOrNull(merge, 4, v2ImpactScore);
268             setBooleanOrNull(merge, 5, v2AcInsufInfo);
269             setBooleanOrNull(merge, 6, v2ObtainAllPrivilege);
270             setBooleanOrNull(merge, 7, v2ObtainUserPrivilege);
271             setBooleanOrNull(merge, 8, v2ObtainOtherPrivilege);
272             setBooleanOrNull(merge, 9, v2UserInteractionRequired);
273             setFloatOrNull(merge, 10, v2Score);
274             setStringOrNull(merge, 11, v2AccessVector);
275             setStringOrNull(merge, 12, v2AccessComplexity);
276             setStringOrNull(merge, 13, v2Authentication);
277             setStringOrNull(merge, 14, v2ConfidentialityImpact);
278             setStringOrNull(merge, 15, v2IntegrityImpact);
279             setStringOrNull(merge, 16, v2AvailabilityImpact);
280             setStringOrNull(merge, 17, v2Version);
281             setFloatOrNull(merge, 18, v3ExploitabilityScore);
282             setFloatOrNull(merge, 19, v3ImpactScore);
283             setStringOrNull(merge, 20, v3AttackVector);
284             setStringOrNull(merge, 21, v3AttackComplexity);
285             setStringOrNull(merge, 22, v3PrivilegesRequired);
286             setStringOrNull(merge, 23, v3UserInteraction);
287             setStringOrNull(merge, 24, v3Scope);
288             setStringOrNull(merge, 25, v3ConfidentialityImpact);
289             setStringOrNull(merge, 26, v3IntegrityImpact);
290             setStringOrNull(merge, 27, v3AvailabilityImpact);
291             setFloatOrNull(merge, 28, v3BaseScore);
292             setStringOrNull(merge, 29, v3BaseSeverity);
293             setStringOrNull(merge, 30, v3Version);
294 
295             if (vulnerabilityId == 0) {
296                 merge.setString(31, cve);
297             } else {
298                 merge.setInt(31, vulnerabilityId);
299             }
300             final int count = merge.executeUpdate();
301             if (vulnerabilityId == 0) {
302                 try (ResultSet rs = merge.getGeneratedKeys()) {
303                     if (rs.next()) {
304                         vulnerabilityId = rs.getInt(1);
305                     }
306                 }
307             }
308         } finally {
309             if (merge != null) {
310                 merge.close();
311             }
312         }
313         ret.addRow(vulnerabilityId);
314         return ret;
315     }

319     /**
320      * Update or insert a known exploited vulnerability.
321      *
322      * @param conn the connection
323      * @param cveId the id
324      * @param vendorProject the vendor/project
325      * @param product the product
326      * @param vulnerabilityName the vulnerability name
327      * @param dateAdded the date added
328      * @param shortDescription the short description
329      * @param requiredAction the action required
330      * @param dueDate the due date
331      * @param notes notes
332      * @throws SQLException
333      */
334     public static void mergeKnownExploited(final Connection conn, String cveId,
335             String vendorProject, String product, String vulnerabilityName,
336             String dateAdded, String shortDescription, String requiredAction,
337             String dueDate, String notes) throws SQLException {
338 
339         String id = "";
340         try (PreparedStatement selectVulnerabilityId = conn.prepareStatement("SELECT cveID FROM knownExploited cveID WHERE cveID=?")) {
341             selectVulnerabilityId.setString(1, cveId);
342             try (ResultSet rs = selectVulnerabilityId.executeQuery()) {
343                 if (rs.next()) {
344                     id = rs.getString(1);
345                 }
346             }
347         }
348         PreparedStatement merge = null;
349         try {
350             if (Strings.isNullOrEmpty(id)) {
351                 merge = conn.prepareStatement("INSERT INTO knownExploited ("
352                         + "vendorProject, product, vulnerabilityName, "
353                         + "dateAdded, shortDescription, requiredAction, "
354                         + "dueDate, notes, cveID) "
355                         + "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)");
356             } else {
357                 merge = conn.prepareStatement("UPDATE knownExploited SET "
358                         + "vendorProject=?, product=?, vulnerabilityName=?, "
359                         + "dateAdded=?, shortDescription=?, requiredAction=?, "
360                         + "dueDate=?, notes=? WHERE cveID=?");
361             }
362 
363             setStringOrNull(merge, 1, vendorProject);
364             setStringOrNull(merge, 2, product);
365             setStringOrNull(merge, 3, vulnerabilityName);
366             setStringOrNull(merge, 4, dateAdded);
367             setStringOrNull(merge, 5, shortDescription);
368             setStringOrNull(merge, 6, requiredAction);
369             setStringOrNull(merge, 7, dueDate);
370             setStringOrNull(merge, 8, notes);
371             setStringOrNull(merge, 9, cveId);
372             merge.execute();
373         } finally {
374             if (merge != null) {
375                 merge.close();
376             }
377         }
378     }

381     /**
382      * Sets a parameter value on a prepared statement with null checks.
383      *
384      * @param ps the prepared statement
385      * @param i the parameter index
386      * @param value the value
387      * @throws SQLException thrown if there is an error setting the parameter
388      */
389     private static void setStringOrNull(PreparedStatement ps, int i, String value) throws SQLException {
390         if (value == null) {
391             ps.setNull(i, java.sql.Types.NULL);
392         } else {
393             ps.setString(i, value);
394         }
395     }
