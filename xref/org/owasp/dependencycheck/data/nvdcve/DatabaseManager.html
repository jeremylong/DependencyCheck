128     /**
129      * Initializes the connection factory. Ensuring that the appropriate drivers
130      * are loaded and that a connection can be made successfully.
131      *
132      * @throws DatabaseException thrown if we are unable to connect to the
133      * database
134      */
135     private void initialize() throws DatabaseException {
136         final boolean autoUpdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE, true);
137         Connection conn = null;
138         try {
139             //load the driver if necessary
140             final String driverName = settings.getString(Settings.KEYS.DB_DRIVER_NAME, "");
141             if (!driverName.isEmpty()) {
142                 final String driverPath = settings.getString(Settings.KEYS.DB_DRIVER_PATH, "");
143                 LOGGER.debug("Loading driver '{}'", driverName);
144                 try {
145                     if (!driverPath.isEmpty()) {
146                         LOGGER.debug("Loading driver from: {}", driverPath);
147                         driver = DriverLoader.load(driverName, driverPath);
148                     } else {
149                         driver = DriverLoader.load(driverName);
150                     }
151                 } catch (DriverLoadException ex) {
152                     LOGGER.debug("Unable to load database driver", ex);
153                     throw new DatabaseException("Unable to load database driver", ex);
154                 }
155             }
156             userName = settings.getString(Settings.KEYS.DB_USER, "dcuser");
157             //yes, yes - hard-coded password - only if there isn't one in the properties file.
158             password = settings.getString(Settings.KEYS.DB_PASSWORD, "DC-Pass1337!");
159             try {
160                 connectionString = settings.getConnectionString(
161                         Settings.KEYS.DB_CONNECTION_STRING,
162                         Settings.KEYS.DB_FILE_NAME);
163             } catch (IOException ex) {
164                 LOGGER.debug("Unable to retrieve the database connection string", ex);
165                 throw new DatabaseException("Unable to retrieve the database connection string", ex);
166             }
167             isH2 = isH2Connection(connectionString);
168             boolean shouldCreateSchema = false;
169             try {
170                 if (autoUpdate && isH2) {
171                     shouldCreateSchema = !h2DataFileExists();
172                     LOGGER.debug("Need to create DB Structure: {}", shouldCreateSchema);
173                 }
174             } catch (IOException ioex) {
175                 LOGGER.debug("Unable to verify database exists", ioex);
176                 throw new DatabaseException("Unable to verify database exists", ioex);
177             }
178             LOGGER.debug("Loading database connection");
179             LOGGER.debug("Connection String: {}", connectionString);
180             LOGGER.debug("Database User: {}", userName);
181 
182             try {
183                 if (connectionString.toLowerCase().contains("integrated security=true")
184                         || connectionString.toLowerCase().contains("trusted_connection=true")) {
185                     conn = DriverManager.getConnection(connectionString);
186                 } else {
187                     conn = DriverManager.getConnection(connectionString, userName, password);
188                 }
189             } catch (SQLException ex) {
190                 if (ex.getMessage().contains("java.net.UnknownHostException") && connectionString.contains("AUTO_SERVER=TRUE;")) {
191                     connectionString = connectionString.replace("AUTO_SERVER=TRUE;", "");
192                     try {
193                         conn = DriverManager.getConnection(connectionString, userName, password);
194                         settings.setString(Settings.KEYS.DB_CONNECTION_STRING, connectionString);
195                         LOGGER.debug("Unable to start the database in server mode; reverting to single user mode");
196                     } catch (SQLException sqlex) {
197                         LOGGER.debug("Unable to connect to the database", ex);
198                         throw new DatabaseException("Unable to connect to the database", ex);
199                     }
200                 } else {
201                     LOGGER.debug("Unable to connect to the database", ex);
202                     throw new DatabaseException("Unable to connect to the database", ex);
203                 }
204             }
205             databaseProductName = determineDatabaseProductName(conn);
206             isOracle = "oracle".equals(databaseProductName);
207             if (shouldCreateSchema) {
208                 try {
209                     createTables(conn);
210                 } catch (DatabaseException dex) {
211                     LOGGER.debug("", dex);
212                     throw new DatabaseException("Unable to create the database structure", dex);
213                 }
214             }
215             try {
216                 ensureSchemaVersion(conn);
217             } catch (DatabaseException dex) {
218                 LOGGER.debug("", dex);
219                 throw new DatabaseException("Database schema does not match this version of dependency-check", dex);
220             }
221         } finally {
222             if (conn != null) {
223                 try {
224                     conn.close();
225                 } catch (SQLException ex) {
226                     LOGGER.debug("An error occurred closing the connection", ex);
227                 }
228             }
229         }
230     }

232     /**
233      * Tries to determine the product name of the database.
234      *
235      * @param conn the database connection
236      * @return the product name of the database if successful, {@code null} else
237      */
238     private String determineDatabaseProductName(Connection conn) {
239         try {
240             final String databaseProductName = conn.getMetaData().getDatabaseProductName().toLowerCase();
241             LOGGER.debug("Database product: {}", databaseProductName);
242             return databaseProductName;
243         } catch (SQLException se) {
244             LOGGER.warn("Problem determining database product!", se);
245             return null;
246         }
247     }

250      * Cleans up resources and unloads any registered database drivers. This
251      * needs to be called to ensure the driver is unregistered prior to the
252      * finalize method being called as during shutdown the class loader used to
253      * load the driver may be unloaded prior to the driver being de-registered.

257             DriverLoader.cleanup(driver);
258             driver = null;

260         connectionString = null;
261         userName = null;
262         password = null;

360     /**
361      * Creates the database structure (tables and indexes) to store the CVE
362      * data.
363      *
364      * @param conn the database connection
365      * @throws DatabaseException thrown if there is a Database Exception
366      */
367     private void createTables(Connection conn) throws DatabaseException {
368         LOGGER.debug("Creating database structure");
369         final String dbStructure;
370         try {
371             dbStructure = getResource(DB_STRUCTURE_RESOURCE);
372 
373             Statement statement = null;
374             try {
375                 statement = conn.createStatement();
376                 statement.execute(dbStructure);
377             } catch (SQLException ex) {
378                 LOGGER.debug("", ex);
379                 throw new DatabaseException("Unable to create database statement", ex);
380             } finally {
381                 DBUtils.closeStatement(statement);
382             }
383         } catch (IOException ex) {
384             throw new DatabaseException("Unable to create database schema", ex);
385         } catch (LinkageError ex) {
386             LOGGER.debug(new DefaultQuery(ex).call().toString());
387         }
388     }
389 
390     private String getResource(String resource) throws IOException {
391         String dbStructure;
392         try {
393             final URL url = Resources.getResource(resource);
394             dbStructure = Resources.toString(url, StandardCharsets.UTF_8);
395         } catch (IllegalArgumentException ex) {
396             LOGGER.debug("Resources.getResource(String) failed to find the DB Structure Resource", ex);
397             try (InputStream is = FileUtils.getResourceAsStream(resource)) {
398                 dbStructure = IOUtils.toString(is, StandardCharsets.UTF_8);
399             }
400         }
401         return dbStructure;
402     }

404     /**
405      * Updates the database schema by loading the upgrade script for the version
406      * specified. The intended use is that if the current schema version is 2.9
407      * then we would call updateSchema(conn, "2.9"). This would load the
408      * upgrade_2.9.sql file and execute it against the database. The upgrade
409      * script must update the 'version' in the properties table.
410      *
411      * @param conn the database connection object
412      * @param appExpectedVersion the schema version that the application expects
413      * @param currentDbVersion the current schema version of the database
414      * @throws DatabaseException thrown if there is an exception upgrading the
415      * database schema
416      */
417     private void updateSchema(Connection conn, DependencyVersion appExpectedVersion, DependencyVersion currentDbVersion)
418             throws DatabaseException {
419 
420         if (connectionString.startsWith("jdbc:h2:file:")) {
421             LOGGER.debug("Updating database structure");
422             final String updateFile = String.format(DB_STRUCTURE_UPDATE_RESOURCE, currentDbVersion.toString());
423             if ("data/upgrade_4.2.sql".equals(updateFile) && !FileUtils.getResourceAsFile(updateFile).exists()) {
424                 throw new DatabaseException("unable to upgrade the database schema - please run the dependency-check "
425                         + "purge command to remove the existing database");
426             }
427             try {
428                 final String dbStructureUpdate = getResource(updateFile);
429                 Statement statement = null;
430                 try {
431                     statement = conn.createStatement();
432                     statement.execute(dbStructureUpdate);
433                 } catch (SQLException ex) {
434                     throw new DatabaseException(String.format("Unable to upgrade the database schema from %s to %s",
435                             currentDbVersion, appExpectedVersion.toString()), ex);
436                 } finally {
437                     DBUtils.closeStatement(statement);
438                 }
439             } catch (IllegalArgumentException | IOException ex) {
440                 final String msg = String.format("Upgrade SQL file does not exist: %s", updateFile);
441                 throw new DatabaseException(msg, ex);
442             }
443         } else {
444             final int e0 = Integer.parseInt(appExpectedVersion.getVersionParts().get(0));
445             final int c0 = Integer.parseInt(currentDbVersion.getVersionParts().get(0));
446             final int e1 = Integer.parseInt(appExpectedVersion.getVersionParts().get(1));
447             final int c1 = Integer.parseInt(currentDbVersion.getVersionParts().get(1));
448             //CSOFF: EmptyBlock
449             if (e0 == c0 && e1 < c1) {
450                 LOGGER.warn("A new version of dependency-check is available; consider upgrading");
451                 settings.setBoolean(Settings.KEYS.AUTO_UPDATE, false);
452             } else if (e0 == c0 && e1 == c1) {
453                 //do nothing - not sure how we got here, but just in case...
454             } else {
455                 LOGGER.error("The database schema must be upgraded to use this version of dependency-check. Please see {} for more information.",
456                         UPGRADE_HELP_URL);
457                 throw new DatabaseException("Database schema is out of date");
458             }
459             //CSON: EmptyBlock
460         }
461     }

476     /**
477      * Uses the provided connection to check the specified schema version within
478      * the database.
479      *
480      * @param conn the database connection object
481      * @throws DatabaseException thrown if the schema version is not compatible
482      * with this version of dependency-check
483      */
484     private void ensureSchemaVersion(Connection conn) throws DatabaseException {
485         ResultSet rs = null;
486         PreparedStatement ps = null;
487         final ResourceBundle statementBundle = getSqlStatements();
488         final String sql = statementBundle.getString("SELECT_SCHEMA_VERSION");
489         try {
490             ps = conn.prepareStatement(sql);
491             rs = ps.executeQuery();
492             if (rs.next()) {
493                 final String dbSchemaVersion = settings.getString(Settings.KEYS.DB_VERSION);
494                 final DependencyVersion appDbVersion = DependencyVersionUtil.parseVersion(dbSchemaVersion);
495                 if (appDbVersion == null) {
496                     throw new DatabaseException("Invalid application database schema");
497                 }
498                 final DependencyVersion db = DependencyVersionUtil.parseVersion(rs.getString(1));
499                 if (db == null) {
500                     throw new DatabaseException("Invalid database schema");
501                 }
502                 LOGGER.debug("DC Schema: {}", appDbVersion);
503                 LOGGER.debug("DB Schema: {}", db);
504                 if (appDbVersion.compareTo(db) > 0) {
505                     final boolean autoUpdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE, true);
506                     if (autoUpdate) {
507                         updateSchema(conn, appDbVersion, db);
508                         if (++callDepth < 10) {
509                             ensureSchemaVersion(conn);
510                         }
511                     } else {
512                         throw new DatabaseException("Old database schema identified - please execute "
513                                 + "dependency-check without the no-update configuration to continue");
514                     }
515                 }
516             } else {
517                 throw new DatabaseException("Database schema is missing");
518             }
519         } catch (SQLException ex) {
520             LOGGER.debug("", ex);
521             throw new DatabaseException("Unable to check the database schema version", ex);
522         } finally {
523             DBUtils.closeResultSet(rs);
524             DBUtils.closeStatement(ps);
525         }
526     }

528     /**
529      * Opens the database connection pool.
530      */
531     public void open() {
532         connectionPool = new BasicDataSource();
533         if (driver != null) {
534             connectionPool.setDriver(driver);
535         }
536         connectionPool.setUrl(connectionString);
537         connectionPool.setUsername(userName);
538         connectionPool.setPassword(password);
539     }
540 
541     /**
542      * Closes the database connection pool.
543      */
544     public void close() {
545         try {
546             connectionPool.close();
547         } catch (SQLException ex) {
548             LOGGER.debug("Error closing the connection pool", ex);
549         }
550         connectionPool = null;
551     }

554      * Returns if the connection pool is open.
