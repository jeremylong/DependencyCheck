262     /**
263      * Opens the database connection. If the database does not exist, it will
264      * create a new one.
265      *
266      * @throws DatabaseException thrown if there is an error opening the
267      * database connection
268      */
269     private synchronized void open() throws DatabaseException {
270         try {
271             if (!isOpen()) {
272                 connection = connectionFactory.getConnection();
273                 final String databaseProductName = determineDatabaseProductName(this.connection);
274                 isOracle = "oracle".equals(databaseProductName);
275                 statementBundle = databaseProductName != null
276                         ? ResourceBundle.getBundle("data/dbStatements", new Locale(databaseProductName))
277                         : ResourceBundle.getBundle("data/dbStatements");
278                 prepareStatements();
279                 databaseProperties = new DatabaseProperties(this);
280             }
281         } catch (DatabaseException e) {
282             releaseResources();
283             throw e;
284         }
285     }

287     /**
288      * Closes the database connection. Close should be called on this object
289      * when it is done being used.
290      */
291     @Override
292     public synchronized void close() {
293         if (isOpen()) {
294             LOGGER.debug("Closing database");
295             clearCache();
296             LOGGER.debug("Cache cleared");
297             closeStatements();
298             LOGGER.debug("Statments closed");
299             try {
300                 connection.close();
301                 LOGGER.debug("Connection closed");
302             } catch (SQLException ex) {
303                 LOGGER.error("There was an error attempting to close the CveDB, see the log for more details.");
304                 LOGGER.debug("", ex);
305             } catch (Throwable ex) {
306                 LOGGER.error("There was an exception attempting to close the CveDB, see the log for more details.");
307                 LOGGER.debug("", ex);
308             }
309             releaseResources();
310             LOGGER.debug("Resources released");
311             connectionFactory.cleanup();
312         }
313     }

349     /**
350      * Creates a prepared statement from the given key. The SQL is stored in a
351      * properties file and the key is used to lookup the specific query.
352      *
353      * @param key the key to select the prepared statement from the properties
354      * file
355      * @return the prepared statement
356      * @throws DatabaseException throw if there is an error generating the
357      * prepared statement
358      */
359     private PreparedStatement prepareStatement(PreparedStatementCveDb key) throws DatabaseException {
360         PreparedStatement preparedStatement = null;
361         try {
362             final String statementString = statementBundle.getString(key.name());
363 //            if (key == INSERT_CPE) {
364 //                final String[] returnedColumns = {"id"};
365 //                preparedStatement = connection.prepareStatement(statementString, returnedColumns);
366 //            } else {
367             if (isOracle && key == UPDATE_VULNERABILITY) {
368                 preparedStatement = connection.prepareCall(statementString);
369             } else {
370                 preparedStatement = connection.prepareStatement(statementString);
371             }
372             if (isOracle) {
373                 // Oracle has a default fetch-size of 10; MariaDB, MySQL, SQLServer and PostgreSQL by default cache the full
374                 // resultset at the client https://venkatsadasivam.com/2009/02/01/jdbc-performance-tuning-with-optimal-fetch-size/
375                 preparedStatement.setFetchSize(10_000);
376             }
377 //            }
378         } catch (SQLException ex) {
379             throw new DatabaseException(ex);
380         } catch (MissingResourceException ex) {
381             if (!ex.getMessage().contains("key MERGE_PROPERTY")) {
382                 throw new DatabaseException(ex);
383             }
384         }
385         return preparedStatement;
386     }

457     /**
458      * Searches the CPE entries in the database and retrieves all entries for a
459      * given vendor and product combination. The returned list will include all
460      * versions of the product that are registered in the NVD CVE data.
461      *
462      * @param vendor the identified vendor name of the dependency being analyzed
463      * @param product the identified name of the product of the dependency being
464      * analyzed
465      * @return a set of vulnerable software
466      */
467     public synchronized Set<CpePlus> getCPEs(String vendor, String product) {
468         final Set<CpePlus> cpe = new HashSet<>();
469         ResultSet rs = null;
470         try {
471             final PreparedStatement ps = getPreparedStatement(SELECT_CPE_ENTRIES);
472             if (ps == null) {
473                 throw new SQLException("Database query does not exist in the resource bundle: " + SELECT_CPE_ENTRIES);
474             }
475             //part, vendor, product, version, update_version, edition,
476             //lang, sw_edition, target_sw, target_hw, other, ecosystem
477             ps.setString(1, vendor);
478             ps.setString(2, product);
479             rs = ps.executeQuery();
480             final CpeBuilder builder = new CpeBuilder();
481             while (rs.next()) {
482                 final Cpe entry = builder
483                         .part(rs.getString(1))
484                         .vendor(rs.getString(2))
485                         .product(rs.getString(3))
486                         .version(rs.getString(4))
487                         .update(rs.getString(5))
488                         .edition(rs.getString(6))
489                         .language(rs.getString(7))
490                         .swEdition(rs.getString(8))
491                         .targetSw(rs.getString(9))
492                         .targetHw(rs.getString(10))
493                         .other(rs.getString(11)).build();
494                 final CpePlus plus = new CpePlus(entry, rs.getString(12));
495                 cpe.add(plus);
496             }
497         } catch (SQLException | CpeParsingException | CpeValidationException ex) {
498             LOGGER.error("An unexpected SQL Exception occurred; please see the verbose log for more details.");
499             LOGGER.debug("", ex);
500         } finally {
501             DBUtils.closeResultSet(rs);
502         }
503         return cpe;
504     }

595      */
596     public synchronized void saveProperty(String key, String value) {
597         clearCache();
598         try {
599             final PreparedStatement mergeProperty = getPreparedStatement(MERGE_PROPERTY);
600             if (mergeProperty != null) {
601                 mergeProperty.setString(1, key);
602                 mergeProperty.setString(2, value);
603                 mergeProperty.execute();
604             } else {
605                 // No Merge statement, so doing an Update/Insert...
606                 final PreparedStatement updateProperty = getPreparedStatement(UPDATE_PROPERTY);
607                 if (updateProperty == null) {
608                     throw new SQLException("Database query does not exist in the resource bundle: " + UPDATE_PROPERTY);
609                 }
610                 updateProperty.setString(1, value);
611                 updateProperty.setString(2, key);
612                 if (updateProperty.executeUpdate() == 0) {
613                     final PreparedStatement insertProperty = getPreparedStatement(INSERT_PROPERTY);
614                     if (insertProperty == null) {
615                         throw new SQLException("Database query does not exist in the resource bundle: " + INSERT_PROPERTY);
616                     }
617                     insertProperty.setString(1, key);
618                     insertProperty.setString(2, value);
619                     insertProperty.executeUpdate();
620                 }
621             }
622         } catch (SQLException ex) {
623             LOGGER.warn("Unable to save property '{}' with a value of '{}' to the database", key, value);
624             LOGGER.debug("", ex);
625         }
626     }

641     /**
642      * Retrieves the vulnerabilities associated with the specified CPE.
643      *
644      * @param cpe the CPE to retrieve vulnerabilities for
645      * @return a list of Vulnerabilities
646      * @throws DatabaseException thrown if there is an exception retrieving data
647      */
648     public synchronized List<Vulnerability> getVulnerabilities(Cpe cpe) throws DatabaseException {
649         final List<Vulnerability> cachedVulnerabilities = vulnerabilitiesForCpeCache.get(cpe.toCpe23FS());
650         if (cachedVulnerabilities != null) {
651             LOGGER.debug("Cache hit for {}", cpe.toCpe23FS());
652             return cachedVulnerabilities;
653         } else {
654             LOGGER.debug("Cache miss for {}", cpe.toCpe23FS());
655         }
656 
657         final List<Vulnerability> vulnerabilities = new ArrayList<>();
658         ResultSet rs = null;
659         try {
660             final PreparedStatement ps = getPreparedStatement(SELECT_CVE_FROM_SOFTWARE);
661             ps.setString(1, cpe.getVendor());
662             ps.setString(2, cpe.getProduct());
663             rs = ps.executeQuery();
664             String currentCVE = "";
665 
666             final Set<VulnerableSoftware> vulnSoftware = new HashSet<>();
667             while (rs.next()) {
668                 final String cveId = rs.getString(1);
669                 if (currentCVE.isEmpty()) {
670                     //first loop we don't have the cveId
671                     currentCVE = cveId;
672                 }
673                 if (!vulnSoftware.isEmpty() && !currentCVE.equals(cveId)) { //check for match and add
674                     final VulnerableSoftware matchedCPE = getMatchingSoftware(cpe, vulnSoftware);
675                     if (matchedCPE != null) {
676                         final Vulnerability v = getVulnerability(currentCVE);
677                         if (v != null) {
678                             v.setMatchedVulnerableSoftware(matchedCPE);
679                             v.setSource(Vulnerability.Source.NVD);
680                             vulnerabilities.add(v);
681                         }
682                     }
683                     vulnSoftware.clear();
684                     currentCVE = cveId;
685                 }
686                 // 1 cve, 2 part, 3 vendor, 4 product, 5 version, 6 update_version, 7 edition,
687                 // 8 lang, 9 sw_edition, 10 target_sw, 11 target_hw, 12 other, 13 versionEndExcluding,
688                 //14 versionEndIncluding, 15 versionStartExcluding, 16 versionStartIncluding, 17 vulnerable
689                 final VulnerableSoftware vs;
690                 try {
691                     vs = vulnerableSoftwareBuilder.part(rs.getString(2)).vendor(rs.getString(3))
692                             .product(rs.getString(4)).version(rs.getString(5)).update(rs.getString(6))
693                             .edition(rs.getString(7)).language(rs.getString(8)).swEdition(rs.getString(9))
694                             .targetSw(rs.getString(10)).targetHw(rs.getString(11)).other(rs.getString(12))
695                             .versionEndExcluding(rs.getString(13)).versionEndIncluding(rs.getString(14))
696                             .versionStartExcluding(rs.getString(15)).versionStartIncluding(rs.getString(16))
697                             .vulnerable(rs.getBoolean(17)).build();
698                 } catch (CpeParsingException | CpeValidationException ex) {
699                     throw new DatabaseException("Database contains an invalid Vulnerable Software Entry", ex);
700                 }
