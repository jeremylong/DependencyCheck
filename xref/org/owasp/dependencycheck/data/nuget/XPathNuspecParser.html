18  package org.owasp.dependencycheck.data.nuget;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import javax.annotation.concurrent.ThreadSafe;
23  import javax.xml.parsers.DocumentBuilder;
24  import javax.xml.parsers.ParserConfigurationException;
25  import javax.xml.xpath.XPath;
26  import javax.xml.xpath.XPathConstants;
27  import javax.xml.xpath.XPathExpressionException;
28  import javax.xml.xpath.XPathFactory;
29  import org.owasp.dependencycheck.utils.XmlUtils;
30  import org.w3c.dom.Document;
31  import org.w3c.dom.Node;
32  import org.xml.sax.SAXException;
33  
34  /**
35   * Parse a Nuspec file using XPath.
36   *
37   * @author colezlaw
38   */
39  @ThreadSafe
40  public class XPathNuspecParser {
41  
42      /**
43       * Gets the string value of a node or null if it's not present
44       *
45       * @param n the node to test
46       * @return the string content of the node, or null if the node itself is
47       * null
48       */
49      private String getOrNull(Node n) {
50          if (n != null) {
51              return n.getTextContent();
52          } else {
53              return null;
54          }
55      }
56  
57      /**
58       * Parse an input stream and return the resulting {@link NugetPackage}.
59       *
60       * @param stream the input stream to parse
61       * @return the populated bean
62       * @throws NuspecParseException when an exception occurs
63       */
64      public NugetPackage parse(InputStream stream) throws NuspecParseException {
65          try {
66              final DocumentBuilder db = XmlUtils.buildSecureDocumentBuilder();
67              final Document d = db.parse(stream);
68  
69              final XPath xpath = XPathFactory.newInstance().newXPath();
70              final NugetPackage nuspec = new NugetPackage();
71  
72              if (xpath.evaluate("/package/metadata/id", d, XPathConstants.NODE) == null
73                      || xpath.evaluate("/package/metadata/version", d, XPathConstants.NODE) == null
74                      || xpath.evaluate("/package/metadata/authors", d, XPathConstants.NODE) == null
75                      || xpath.evaluate("/package/metadata/description", d, XPathConstants.NODE) == null) {
76                  throw new NuspecParseException("Invalid Nuspec format");
77              }
78  
79              nuspec.setId(xpath.evaluate("/package/metadata/id", d));
80              nuspec.setVersion(xpath.evaluate("/package/metadata/version", d));
81              nuspec.setAuthors(xpath.evaluate("/package/metadata/authors", d));
82              nuspec.setOwners(getOrNull((Node) xpath.evaluate("/package/metadata/owners", d, XPathConstants.NODE)));
83              nuspec.setLicenseUrl(getOrNull((Node) xpath.evaluate("/package/metadata/licenseUrl", d, XPathConstants.NODE)));
84              nuspec.setTitle(getOrNull((Node) xpath.evaluate("/package/metadata/title", d, XPathConstants.NODE)));
85              nuspec.setDescription(xpath.evaluate("/package/metadata/description", d));
86              return nuspec;
87          } catch (ParserConfigurationException | SAXException | IOException | XPathExpressionException | NuspecParseException e) {
88              throw new NuspecParseException("Unable to parse nuspec", e);
89          }
90      }
91  }
