18  package org.owasp.dependencycheck.data.lucene;
19  
20  import java.io.IOException;
21  import javax.annotation.concurrent.NotThreadSafe;
22  import org.apache.commons.lang3.builder.EqualsBuilder;
23  import org.apache.commons.lang3.builder.HashCodeBuilder;
24  import org.apache.lucene.analysis.TokenFilter;
25  import org.apache.lucene.analysis.TokenStream;
26  import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
27  
28  /**
29   * <p>
30   * Takes a TokenStream and adds additional tokens by concatenating pairs of
31   * words.</p>
32   * <p>
33   * <b>Example:</b> "Spring Framework Core" -&gt; "Spring SpringFramework
34   * Framework FrameworkCore Core".</p>
35   *
36   * @author Jeremy Long
37   */
38  @NotThreadSafe
39  public final class TokenPairConcatenatingFilter extends TokenFilter {
40  
41      /**
42       * The char term attribute.
43       */
44      private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
45      /**
46       * The previous word parsed.
47       */
48      private String previousWord;
49      /**
50       * Keeps track if we are adding a single term or concatenating with the
51       * previous.
52       */
53      private boolean addSingleTerm;
54  
55      /**
56       * Constructs a new TokenPairConcatenatingFilter.
57       *
58       * @param stream the TokenStream that this filter will process
59       */
60      public TokenPairConcatenatingFilter(TokenStream stream) {
61          super(stream);
62          addSingleTerm = true;
63          previousWord = null;
64      }
65  
66      /**
67       * Increments the underlying TokenStream and sets CharTermAttributes to
68       * construct an expanded set of tokens by concatenating tokens with the
69       * previous token.
70       *
71       * @return whether or not we have hit the end of the TokenStream
72       * @throws IOException is thrown when an IOException occurs
73       */
74      @Override
75      public boolean incrementToken() throws IOException {
76          if (addSingleTerm && previousWord != null) {
77              addSingleTerm = false;
78              clearAttributes();
79              termAtt.append(previousWord);
80              return true;
81  
82          } else if (input.incrementToken()) {
83              final String word = new String(termAtt.buffer(), 0, termAtt.length());
84              if (word.isEmpty()) {
85                  return true;
86              }
87              if (addSingleTerm) {
88                  clearAttributes();
89                  termAtt.append(word);
90                  previousWord = word;
91                  addSingleTerm = false;
92              } else {
93                  clearAttributes();
94                  termAtt.append(previousWord).append(word);
95                  previousWord = word;
96                  addSingleTerm = true;
97              }
98              return true;
99          }
100         return false;
101     }
102 
103     /**
104      * Resets the filter. This must be manually called between searching and
105      * indexing. Unable to rely on `reset` as it appears to be called between
106      * terms.
107      *
108      * @throws IOException thrown if there is an error reseting the tokenizer
109      */
110     public void clear() throws IOException {
111         previousWord = null;
112         addSingleTerm = true;
113     }
114 
115     /**
116      * {@inheritDoc}
117      */
118     @Override
119     public int hashCode() {
120         return new HashCodeBuilder(13, 27)
121                 .appendSuper(super.hashCode())
122                 .append(addSingleTerm)
123                 .append(previousWord)
124                 .append(termAtt)
125                 .build();
126     }
127 
128     /**
129      * {@inheritDoc}
130      */
131     @Override
132     public boolean equals(Object obj) {
133         if (obj == null || !(obj instanceof TokenPairConcatenatingFilter)) {
134             return false;
135         }
136         if (this == obj) {
137             return true;
138         }
139         final TokenPairConcatenatingFilter rhs = (TokenPairConcatenatingFilter) obj;
140         return new EqualsBuilder()
141                 .appendSuper(super.equals(obj))
142                 .append(addSingleTerm, rhs.addSingleTerm)
143                 .append(previousWord, rhs.previousWord)
144                 .append(termAtt, rhs.termAtt)
145                 .isEquals();
146     }
147 }
