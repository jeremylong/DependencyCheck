18  package org.owasp.dependencycheck.data.nvd.ecosystem;
19  
20  import com.hankcs.algorithm.AhoCorasickDoubleArrayTrie;
21  
22  /**
23   * Add method for String and IHitFull.
24   *
25   * @param <V> the object type
26   * @author skjolber
27   */
28  public class StringAhoCorasickDoubleArrayTrie<V> extends AhoCorasickDoubleArrayTrie<V> {
29  
30      private static final long serialVersionUID = -5923428681217396309L;
31  
32      /**
33       * Parse text.
34       *
35       * @param text The text
36       * @param processor A processor which handles the output
37       */
38      public void parseText(String text, IHitFull<V> processor) {
39          int position = 1;
40          int currentState = 0;
41          for (int i = 0; i < text.length(); i++) {
42              currentState = getState(currentState, text.charAt(i));
43              final int[] hitArray = output[currentState];
44              if (hitArray != null) {
45                  for (int hit : hitArray) {
46                      processor.hit(position - l[hit], position, v[hit], hit);
47                  }
48              }
49              ++position;
50          }
51      }
52  
53      /**
54       * transmit state, supports failure function
55       *
56       * @param currentState the current state
57       * @param character the current character
58       * @return the new state
59       */
60      private int getState(int currentState, char character) {
61          int state = currentState;
62          int newCurrentState = transitionWithRoot(state, character);  //success
63          while (newCurrentState == -1) {
64              //failure
65              state = fail[state];
66              newCurrentState = transitionWithRoot(state, character);
67          }
68          return newCurrentState;
69      }
70  
71      public V[] getValues() {
72          return v;
73      }
74  }
