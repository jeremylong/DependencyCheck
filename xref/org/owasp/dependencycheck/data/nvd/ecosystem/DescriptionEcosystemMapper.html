18  package org.owasp.dependencycheck.data.nvd.ecosystem;
19  
20  import org.apache.commons.lang3.StringUtils;
21  import org.owasp.dependencycheck.data.nvd.json.DefCveItem;
22  
23  import java.util.HashMap;
24  import java.util.Map;
25  import java.util.Map.Entry;
26  import java.util.TreeMap;
27  
28  /**
29   * Helper utility for mapping CVEs to their ecosystems based on the description.
30   *
31   * @author skjolber
32   */
33  public class DescriptionEcosystemMapper {
34  
35      // static fields for thread-safe + hardcoded functionality
36      /**
37       * The array of ecosystems.
38       */
39      private static final String[] ECOSYSTEMS;
40      /**
41       * A helper map to retrieve the index of an ecosystem.
42       */
43      private static final int[] HINT_TO_ECOSYSTEM_LOOKUP;
44      /**
45       * Map of strings to ecosystems.
46       */
47      private static final TreeMap<String, EcosystemHint> ECOSYSTEM_MAP; // thread safe for reading
48  
49      static {
50          ECOSYSTEM_MAP = new TreeMap<>();
51  
52          for (FileExtensionHint fileExtensionHint : FileExtensionHint.values()) {
53              ECOSYSTEM_MAP.put(fileExtensionHint.getValue(), fileExtensionHint);
54          }
55          for (DescriptionKeywordHint descriptionKeywordHint : DescriptionKeywordHint.values()) {
56              ECOSYSTEM_MAP.put(descriptionKeywordHint.getValue(), descriptionKeywordHint);
57          }
58  
59          final Map<String, Integer> ecosystemIndexes = new HashMap<>();
60  
61          HINT_TO_ECOSYSTEM_LOOKUP = new int[ECOSYSTEM_MAP.size()];
62  
63          int index = 0;
64          for (Entry<String, EcosystemHint> entry : ECOSYSTEM_MAP.entrySet()) {
65              final EcosystemHint ecosystemHint = entry.getValue();
66  
67              Integer ecosystemIndex = ecosystemIndexes.get(ecosystemHint.getEcosystem());
68              if (ecosystemIndex == null) {
69                  ecosystemIndex = ecosystemIndexes.size();
70  
71                  ecosystemIndexes.put(ecosystemHint.getEcosystem(), ecosystemIndex);
72              }
73  
74              HINT_TO_ECOSYSTEM_LOOKUP[index] = ecosystemIndex;
75  
76              index++;
77          }
78  
79          ECOSYSTEMS = new String[ecosystemIndexes.size()];
80          ecosystemIndexes.entrySet().forEach((e) -> {
81              ECOSYSTEMS[e.getValue()] = e.getKey();
82          });
83      }
84  
85      // take advantage of chars also being numbers
86      /**
87       * Prefix prefix for matching ecosystems.
88       */
89      private final boolean[] keywordPrefixes = getPrefixesFor(" -(\"'");
90      /**
91       * Postfix prefix for matching ecosystems.
92       */
93      private final boolean[] keywordPostfixes = getPrefixesFor(" -)\"',.:;");
94      /**
95       * Aho Corasick double array trie used for parsing and matching ecosystems.
96       */
97      private final StringAhoCorasickDoubleArrayTrie<EcosystemHint> ahoCorasickDoubleArrayTrie;
98  
99      /**
100      * Constructs a new description ecosystem mapper.
101      */
102     public DescriptionEcosystemMapper() {
103         ahoCorasickDoubleArrayTrie = toAhoCorasickDoubleArrayTrie();
104     }
105 
106     protected static boolean[] getPrefixesFor(String str) {
107         int max = -1;
108         for (int i = 0; i < str.length(); i++) {
109             if (max < str.charAt(i)) {
110                 max = str.charAt(i);
111             }
112         }
113 
114         final boolean[] delimiters = new boolean[max + 1];
115         for (int i = 0; i < str.length(); i++) {
116             delimiters[str.charAt(i)] = true;
117         }
118         return delimiters;
119     }
120 
121     protected static StringAhoCorasickDoubleArrayTrie<EcosystemHint> toAhoCorasickDoubleArrayTrie() {
122         final StringAhoCorasickDoubleArrayTrie<EcosystemHint> exact = new StringAhoCorasickDoubleArrayTrie<>();
123         exact.build(ECOSYSTEM_MAP);
124         return exact;
125     }
126 
127     protected static boolean isExtension(String str, int begin, int end) {
128         if (str.length() != end && Character.isLetterOrDigit(str.charAt(end))) {
129             return false;
130         }
131 
132         return isLowercaseAscii(str, begin + 1, end);
133     }
134 
135     protected static boolean isLowercaseAscii(String multicase, int start, int end) {
136         for (int i = start; i < end; i++) {
137             final char c = multicase.charAt(i);
138 
139             if (c < 'a' || c > 'z') {
140                 return false;
141             }
142         }
143         return true;
144     }
145 
146     /**
147      * Tests if the string is a URL by looking for '://'.
148      *
149      * @param c the text to test.
150      * @param begin the position in the string to begin searching; note the
151      * search is decreasing to 0
152      * @return <code>true</code> if `://` is found; otherwise <code>false</code>
153      */
154     public static boolean isURL(String c, int begin) {
155         int pos = begin - 2;
156 
157         while (pos > 2) {
158             pos--;
159 
160             if (c.charAt(pos) == ' ') {
161                 return false;
162             }
163             if (c.charAt(pos) == ':') {
164                 return c.charAt(pos + 1) == '/' && c.charAt(pos + 2) == '/';
165             }
166         }
167 
168         return false;
169     }
170 
171     protected void increment(int i, int[] ecosystemMap) {
172         ecosystemMap[HINT_TO_ECOSYSTEM_LOOKUP[i]]++;
173     }
174 
175     /**
176      * Returns the ecosystem if identified by English description from the CVE
177      * data.
178      *
179      * @param cve the CVE data
180      * @return the ecosystem if identified
181      */
182     public String getEcosystem(DefCveItem cve) {
183         int[] ecosystemMap = new int[ECOSYSTEMS.length];
184         cve.getCve().getDescription().getDescriptionData().stream()
185                 .filter((langString) -> (langString.getLang().equals("en")))
186                 .forEachOrdered((langString) -> {
187                     search(langString.getValue(), ecosystemMap);
188                 });
189         return getResult(ecosystemMap);
190     }
191 
192     /**
193      * Determines the ecosystem for the given string.
194      *
195      * @param multicase the string to test
196      * @return the ecosystem
197      */
198     public String getEcosystem(String multicase) {
199         int[] ecosystemMap = new int[ECOSYSTEMS.length];
200         search(multicase, ecosystemMap);
201         return getResult(ecosystemMap);
202     }
203 
204     private void search(String multicase, int[] ecosystemMap) {
205         final String c = multicase.toLowerCase();
206         ahoCorasickDoubleArrayTrie.parseText(c, (begin, end, value, index) -> {
207             if (value.getNature() == EcosystemHintNature.FILE_EXTENSION) {
208                 if (!isExtension(multicase, begin, end)) {
209                     return;
210                 }
211 
212                 final String ecosystem = value.getEcosystem();
213                 // real extension, if not part of url
214                 if (Ecosystem.PHP.equals(ecosystem) && c.regionMatches(begin, ".php", 0, 4)) {
215                     if (isURL(c, begin)) {
216                         return;
217                     }
218                 } else if (Ecosystem.JAVA.equals(ecosystem) && c.regionMatches(begin, ".jsp", 0, 4)) {
219                     if (isURL(c, begin)) {
220                         return;
221                     }
222                 }
223             } else { // keyword
224 
225                 // check if full word, i.e. typically space first and then space or dot after
226                 if (begin != 0) {
227                     final char startChar = c.charAt(begin - 1);
228                     if (startChar >= keywordPrefixes.length || !keywordPrefixes[startChar]) {
229                         return;
230                     }
231                 }
232                 if (end != c.length()) {
233                     final char endChar = c.charAt(end);
234                     if (endChar >= keywordPostfixes.length || !keywordPostfixes[endChar]) {
235                         return;
236                     }
237                 }
238 
239                 final String ecosystem = value.getEcosystem();
240                 if (Ecosystem.NATIVE.equals(ecosystem)) { // TODO could be checked afterwards
241                     if (StringUtils.contains(c, "android")) {
242                         return;
243                     }
244                 }
245             }
246             increment(index, ecosystemMap);
247         });
248     }
249 
250     private String getResult(int[] values) {
251         final int best = getBestScore(values);
252         if (best != -1) {
253             return ECOSYSTEMS[best];
254         }
255         return null;
256     }
257 
258     private int getBestScore(int[] values) {
259         int bestIndex = -1;
260         int bestScore = -1;
261         for (int i = 0; i < values.length; i++) {
262             if (values[i] > 0) {
263                 if (values[i] > bestScore) {
264                     bestIndex = i;
265                     bestScore = values[i];
266                 }
267                 values[i] = 0;
268             }
269         }
270         return bestIndex;
271     }
