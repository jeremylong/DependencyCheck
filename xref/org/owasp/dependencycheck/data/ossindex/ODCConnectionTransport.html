18  package org.owasp.dependencycheck.data.ossindex;
19  
20  import java.io.IOException;
21  import java.net.HttpURLConnection;
22  import java.net.URL;
23  import org.owasp.dependencycheck.utils.Settings;
24  import org.owasp.dependencycheck.utils.URLConnectionFactory;
25  import org.sonatype.ossindex.service.client.OssindexClientConfiguration;
26  import org.sonatype.ossindex.service.client.transport.BasicAuthHelper;
27  import org.sonatype.ossindex.service.client.transport.HttpUrlConnectionTransport;
28  import org.sonatype.ossindex.service.client.transport.UserAgentSupplier;
29  
30  /**
31   * ODC connection transport is used instead of HttpUrlConnectionTransport
32   * because the proxy information is already configured.
33   *
34   * @author Jeremy Long
35   */
36  public class ODCConnectionTransport extends HttpUrlConnectionTransport {
37  
38      /**
39       * The authorization header.
40       */
41      private static final String AUTHORIZATION = "Authorization";
42      /**
43       * The OSS Index client configuration.
44       */
45      private final OssindexClientConfiguration configuration;
46      /**
47       * The URL Connection factory.
48       */
49      private final URLConnectionFactory connectionFactory;
50      /**
51       * The user agent to send in the HTTP connection.
52       */
53      private final UserAgentSupplier userAgent;
54  
55      /**
56       * Constructs a new transport object to connect to the OSS Index.
57       *
58       * @param settings the ODC settings
59       * @param config the OSS client configuration
60       * @param userAgent the user agent to send to OSS Index
61       */
62      public ODCConnectionTransport(Settings settings, OssindexClientConfiguration config, UserAgentSupplier userAgent) {
63          super(userAgent);
64          this.userAgent = userAgent;
65          this.configuration = config;
66          connectionFactory = new URLConnectionFactory(settings);
67      }
68  
69      @Override
70      protected HttpURLConnection connect(final URL url) throws IOException {
71          final HttpURLConnection connection = connectionFactory.createHttpURLConnection(url);
72          connection.setRequestProperty("User-Agent", userAgent.get());
73  
74          final String authorization = BasicAuthHelper.authorizationHeader(configuration.getAuthConfiguration());
75          if (authorization != null) {
76              connection.setRequestProperty(AUTHORIZATION, authorization);
77          }
78          return connection;
79      }
80  
81  }
