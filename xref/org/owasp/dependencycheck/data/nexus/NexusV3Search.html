44  /**
45   * Class of methods to search Nexus v3 repositories.
46   *
47   * @author Hans Aikema
48   */
49  @ThreadSafe
50  public class NexusV3Search implements NexusSearch {
51  
52      /**
53       * By default, NexusV3Search accepts only classifier-less artifacts.
54       * <p>
55       * This prevents, among others, sha1-collisions for empty jars on empty javadoc/sources jars.
56       * See e.g. issues #5559 and #5118
57       */
58      private final Set<String> acceptedClassifiers = new HashSet<>();
59  
60      /**
61       * The root URL for the Nexus repository service.
62       */
63      private final URL rootURL;
64  
65      /**
66       * Whether to use the Proxy when making requests.
67       */
68      private final boolean useProxy;
69      /**
70       * The configured settings.
71       */
72      private final Settings settings;
73      /**
74       * Used for logging.
75       */
76      private static final Logger LOGGER = LoggerFactory.getLogger(NexusV3Search.class);
77  
78      /**
79       * Creates a NexusV3Search for the given repository URL.
80       *
81       * @param settings the configured settings
82       * @param useProxy flag indicating if the proxy settings should be used
83       * @throws MalformedURLException thrown if the configured URL is
84       *                               invalid
85       */
86      public NexusV3Search(Settings settings, boolean useProxy) throws MalformedURLException {
87          this.settings = settings;
88          this.useProxy = useProxy;
89          this.acceptedClassifiers.add(null);
90          final String searchUrl = settings.getString(Settings.KEYS.ANALYZER_NEXUS_URL);
91          LOGGER.debug("Nexus Search URL: {}", searchUrl);
92          this.rootURL = new URL(searchUrl);
93  
94      }
95  
96      @Override
97      public MavenArtifact searchSha1(String sha1) throws IOException {
98          if (null == sha1 || !sha1.matches("^[0-9A-Fa-f]{40}$")) {
99              throw new IllegalArgumentException("Invalid SHA1 format");
100         }
101 
102         final List<MavenArtifact> collectedMatchingArtifacts = new ArrayList<>(1);
103 
104         String continuationToken = retrievePageAndAddMatchingArtifact(collectedMatchingArtifacts, sha1, null);
105         while (continuationToken != null && collectedMatchingArtifacts.isEmpty()) {
106             continuationToken = retrievePageAndAddMatchingArtifact(collectedMatchingArtifacts, sha1, continuationToken);
107         }
108         if (collectedMatchingArtifacts.isEmpty()) {
109             throw new FileNotFoundException("Artifact not found in Nexus");
110         } else {
111             return collectedMatchingArtifacts.get(0);
112         }
113     }
114 
115     private String retrievePageAndAddMatchingArtifact(List<MavenArtifact> collectedMatchingArtifacts, String sha1, String continuationToken)
116             throws IOException {
117         final URL url;
118         LOGGER.debug("Search with continuation token {}", continuationToken);
119         if (continuationToken == null) {
120             url = new URL(rootURL, String.format("v1/search/?sha1=%s",
121                     sha1.toLowerCase()));
122         } else {
123             url = new URL(rootURL, String.format("v1/search/?sha1=%s&continuationToken=%s",
124                     sha1.toLowerCase(), continuationToken));
125         }
126 
127         LOGGER.debug("Searching Nexus url {}", url);
128         // Determine if we need to use a proxy. The rules:
129         // 1) If the proxy is set, AND the setting is set to true, use the proxy
130         // 2) Otherwise, don't use the proxy (either the proxy isn't configured,
131         // or proxy is specifically set to false
132         final HttpURLConnection conn;
133         final URLConnectionFactory factory = new URLConnectionFactory(settings);
134         conn = factory.createHttpURLConnection(url, useProxy);
135         conn.setDoOutput(true);
136         final String authHeader = buildHttpAuthHeaderValue();
137         if (!authHeader.isEmpty()) {
138             conn.addRequestProperty("Authorization", authHeader);
139         }
140 
141         conn.addRequestProperty("Accept", "application/json");
142         conn.connect();
143         final String nextContinuationToken;
144         if (conn.getResponseCode() == 200) {
145             nextContinuationToken = parseResponse(conn, sha1, collectedMatchingArtifacts);
146         } else {
147             LOGGER.debug("Could not connect to Nexus received response code: {} {}",
148                     conn.getResponseCode(), conn.getResponseMessage());
149             throw new IOException(String.format("Could not connect to Nexus, HTTP response code %d", conn.getResponseCode()));
150         }
151         return nextContinuationToken;
152     }
153 
154     private String parseResponse(HttpURLConnection conn, String sha1, List<MavenArtifact> matchingArtifacts) throws IOException {
155         try (InputStream in = new BufferedInputStream(conn.getInputStream());
156              JsonReader jsonReader = Json.createReader(in)) {
157             final JsonObject jsonResponse = jsonReader.readObject();
158             final String continuationToken = jsonResponse.getString("continuationToken", null);
159             final JsonArray components = jsonResponse.getJsonArray("items");
160             boolean found = false;
161             for (int i = 0; i < components.size() && !found; i++) {
162                 boolean jarFound = false;
163                 boolean pomFound = false;
164                 String downloadUrl = null;
165                 String groupId = null;
166                 String artifactId = null;
167                 String version = null;
168                 String pomUrl = null;
169 
170                 final JsonObject component = components.getJsonObject(i);
171 
172                 final String format = components.getJsonObject(0).getString("format", "unknown");
173                 if ("maven2".equals(format)) {
174                     final JsonArray assets = component.getJsonArray("assets");
175                     for (int j = 0; !found && j < assets.size(); j++) {
176                         final JsonObject asset = assets.getJsonObject(j);
177                         final JsonObject checksums = asset.getJsonObject("checksum");
178                         final JsonObject maven2 = asset.getJsonObject("maven2");
179                         if (maven2 != null
180                                 && "jar".equals(maven2.getString("extension", null))
181                                 && acceptedClassifiers.contains(maven2.getString("classifier", null))
182                                 && checksums != null && sha1.equals(checksums.getString("sha1", null))
183                         ) {
184                             downloadUrl = asset.getString("downloadUrl");
185                             groupId = maven2.getString("groupId");
186                             artifactId = maven2.getString("artifactId");
187                             version = maven2.getString("version");
188 
189                             jarFound = true;
190                         } else if (maven2 != null && "pom".equals(maven2.getString("extension"))) {
191                             pomFound = true;
192                             pomUrl = asset.getString("downloadUrl");
193                         }
194                         if (pomFound && jarFound) {
195                             found = true;
196                         }
197                     }
198                     if (found) {
199                         matchingArtifacts.add(new MavenArtifact(groupId, artifactId, version, downloadUrl, pomUrl));
200                     } else if (jarFound) {
201                         final MavenArtifact ma = new MavenArtifact(groupId, artifactId, version, downloadUrl);
202                         ma.setPomUrl(MavenArtifact.derivePomUrl(artifactId, version, downloadUrl));
203                         matchingArtifacts.add(ma);
204                         found = true;
205                     }
206                 }
207             }
208             return continuationToken;
209         }
210     }

212     @Override
213     public boolean preflightRequest() {
214         final HttpURLConnection conn;
215         try {
216             final URL url = new URL(rootURL, "v1/status");
217             final URLConnectionFactory factory = new URLConnectionFactory(settings);
218             conn = factory.createHttpURLConnection(url, useProxy);
219             conn.addRequestProperty("Accept", "application/json");
220             final String authHeader = buildHttpAuthHeaderValue();
221             if (!authHeader.isEmpty()) {
222                 conn.addRequestProperty("Authorization", authHeader);
223             }
224             conn.connect();
225             if (conn.getResponseCode() != 200) {
226                 LOGGER.warn("Expected 200 result from Nexus, got {}", conn.getResponseCode());
227                 return false;
228             }
229             if (conn.getContentLength() != 0) {
230                 LOGGER.warn("Expected empty OK response (content-length 0), got content-length {}", conn.getContentLength());
231                 return false;
232             }
233         } catch (IOException e) {
234             LOGGER.warn("Pre-flight request to Nexus failed: ", e);
235             return false;
236         }
237         return true;
238     }
239 
240     /**
241      * Constructs the base64 encoded basic authentication header value.
242      *
243      * @return the base64 encoded basic authentication header value
244      */
245     private String buildHttpAuthHeaderValue() {
246         final String user = settings.getString(Settings.KEYS.ANALYZER_NEXUS_USER, "");
247         final String pass = settings.getString(Settings.KEYS.ANALYZER_NEXUS_PASSWORD, "");
248         String result = "";
249         if (user.isEmpty() || pass.isEmpty()) {
250             LOGGER.debug("Skip authentication as user and/or password for nexus is empty");
251         } else {
252             final String auth = user + ':' + pass;
253             final String base64Auth = Base64.getEncoder().encodeToString(auth.getBytes(StandardCharsets.UTF_8));
254             result = "Basic " + base64Auth;
255         }
256         return result;
257     }
