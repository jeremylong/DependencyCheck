46  /**
47   * Class of methods to search Nexus repositories.
48   *
49   * @author colezlaw
50   */
51  @ThreadSafe
52  public class NexusV2Search implements NexusSearch {
53  
54      /**
55       * The root URL for the Nexus repository service.
56       */
57      private final URL rootURL;
58  
59      /**
60       * Whether to use the Proxy when making requests.
61       */
62      private final boolean useProxy;
63      /**
64       * The configured settings.
65       */
66      private final Settings settings;
67      /**
68       * Used for logging.
69       */
70      private static final Logger LOGGER = LoggerFactory.getLogger(NexusV2Search.class);
71  
72      /**
73       * Creates a NexusSearch for the given repository URL.
74       *
75       * @param settings the configured settings
76       * @param useProxy flag indicating if the proxy settings should be used
77       * @throws java.net.MalformedURLException thrown if the configured URL is
78       * invalid
79       */
80      public NexusV2Search(Settings settings, boolean useProxy) throws MalformedURLException {
81          this.settings = settings;
82          this.useProxy = useProxy;
83  
84          final String searchUrl = settings.getString(Settings.KEYS.ANALYZER_NEXUS_URL);
85          LOGGER.debug("Nexus Search URL: {}", searchUrl);
86          this.rootURL = new URL(searchUrl);
87  
88      }
89  
90      @Override
91      public MavenArtifact searchSha1(String sha1) throws IOException {
92          if (null == sha1 || !sha1.matches("^[0-9A-Fa-f]{40}$")) {
93              throw new IllegalArgumentException("Invalid SHA1 format");
94          }
95  
96          final URL url = new URL(rootURL, String.format("identify/sha1/%s",
97                  sha1.toLowerCase()));
98  
99          LOGGER.debug("Searching Nexus url {}", url);
100 
101         try {
102             // JSON would be more elegant, but there's not currently a dependency
103             // on JSON, so don't want to add one just for this
104             final ToXMLDocumentResponseHandler handler = new ToXMLDocumentResponseHandler();
105             final Document doc = Downloader.getInstance().fetchAndHandle(url, handler, List.of(new BasicHeader(HttpHeaders.ACCEPT,
106                     ContentType.APPLICATION_XML)));
107             final XPath xpath = XPathFactory.newInstance().newXPath();
108             final String groupId = xpath
109                     .evaluate(
110                             "/org.sonatype.nexus.rest.model.NexusArtifact/groupId",
111                             doc);
112             final String artifactId = xpath.evaluate(
113                     "/org.sonatype.nexus.rest.model.NexusArtifact/artifactId",
114                     doc);
115             final String version = xpath
116                     .evaluate(
117                             "/org.sonatype.nexus.rest.model.NexusArtifact/version",
118                             doc);
119             final String link = xpath
120                     .evaluate(
121                             "/org.sonatype.nexus.rest.model.NexusArtifact/artifactLink",
122                             doc);
123             final String pomLink = xpath
124                     .evaluate(
125                             "/org.sonatype.nexus.rest.model.NexusArtifact/pomLink",
126                             doc);
127             final MavenArtifact ma = new MavenArtifact(groupId, artifactId, version);
128             if (link != null && !link.isEmpty()) {
129                 ma.setArtifactUrl(link);
130             }
131             if (pomLink != null && !pomLink.isEmpty()) {
132                 ma.setPomUrl(pomLink);
133             }
134             return ma;
135         } catch (DownloadFailedException | TooManyRequestsException e) {
136                 if (LOGGER.isDebugEnabled()) {
137                     int responseCode = -1;
138                     String responseMessage = "";
139                     if (e.getCause() instanceof HttpResponseException) {
140                         final HttpResponseException cause = (HttpResponseException) e.getCause();
141                         responseCode = cause.getStatusCode();
142                         responseMessage = cause.getReasonPhrase();
143                     }
144                     LOGGER.debug("Could not connect to Nexus received response code: {} {}",
145                             responseCode, responseMessage);
146                 }
147                 throw new IOException("Could not connect to Nexus");
148         } catch (ResourceNotFoundException e) {
149             throw new FileNotFoundException("Artifact not found in Nexus");
150         } catch (XPathExpressionException | URISyntaxException e) {
151             throw new IOException(e.getMessage(), e);
152         }
153     }
154 
155     @Override
156     public boolean preflightRequest() {
157         try {
158             final URL url = new URL(rootURL, "status");
159             final ToXMLDocumentResponseHandler handler = new ToXMLDocumentResponseHandler();
160             final Document doc = Downloader.getInstance().fetchAndHandle(url, handler, List.of(new BasicHeader(HttpHeaders.ACCEPT,
161                     ContentType.APPLICATION_XML)));
162             if (!"status".equals(doc.getDocumentElement().getNodeName())) {
163                 LOGGER.warn("Pre-flight request to Nexus failed; expected root node name of status, got {}", doc.getDocumentElement().getNodeName());
164                 return false;
165             }
166         } catch (IOException | TooManyRequestsException | ResourceNotFoundException | URISyntaxException e) {
167             LOGGER.warn("Pre-flight request to Nexus failed: ", e);
168             return false;
169         }
170         return true;
171     }
