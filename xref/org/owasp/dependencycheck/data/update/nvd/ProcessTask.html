18  package org.owasp.dependencycheck.data.update.nvd;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.sql.SQLException;
23  import java.util.concurrent.Callable;
24  import javax.annotation.concurrent.ThreadSafe;
25  import javax.xml.parsers.ParserConfigurationException;
26  import org.owasp.dependencycheck.data.nvdcve.CveDB;
27  import org.owasp.dependencycheck.data.nvdcve.DatabaseException;
28  import org.owasp.dependencycheck.data.nvdcve.DatabaseProperties;
29  import org.owasp.dependencycheck.data.update.exception.UpdateException;
30  import org.owasp.dependencycheck.utils.Settings;
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  /**
35   * A callable task that will process a given set of NVD CVE xml files and update
36   * the Cve Database accordingly.
37   *
38   * @author Jeremy Long
39   */
40  @ThreadSafe
41  public class ProcessTask implements Callable<ProcessTask> {
42  
43      /**
44       * The logger.
45       */
46      private static final Logger LOGGER = LoggerFactory.getLogger(ProcessTask.class);
47      /**
48       * A field to store any update exceptions that occur during the "call".
49       */
50      private UpdateException exception = null;
51      /**
52       * A reference to the CveDB.
53       */
54      private final CveDB cveDB;
55      /**
56       * A reference to the callable download task.
57       */
58      private final DownloadTask downloadTask;
59      /**
60       * A reference to the properties.
61       */
62      private final DatabaseProperties properties;
63      /**
64       * A reference to the global settings object.
65       */
66      private final Settings settings;
67  
68      /**
69       * Get the value of exception.
70       *
71       * @return the value of exception
72       */
73      public UpdateException getException() {
74          return exception;
75      }
76  
77      /**
78       * Set the value of exception.
79       *
80       * @param exception new value of exception
81       */
82      public void setException(UpdateException exception) {
83          this.exception = exception;
84      }
85  
86      /**
87       * Constructs a new ProcessTask used to process an NVD CVE update.
88       *
89       * @param cveDB the data store object
90       * @param downloadTask the download task that contains the URL references to
91       * download
92       * @param settings a reference to the global settings object; this is
93       * necessary so that when the thread is started the dependencies have a
94       * correct reference to the global settings.
95       */
96      public ProcessTask(final CveDB cveDB, final DownloadTask downloadTask, Settings settings) {
97          this.cveDB = cveDB;
98          this.downloadTask = downloadTask;
99          this.properties = cveDB.getDatabaseProperties();
100         this.settings = settings;
101     }
102 
103     /**
104      * Implements the callable interface.
105      *
106      * @return this object
107      * @throws Exception thrown if there is an exception; note that any
108      * UpdateExceptions are simply added to the tasks exception collection
109      */
110     @Override
111     public ProcessTask call() throws Exception {
112         try {
113             processFiles();
114         } catch (UpdateException ex) {
115             this.exception = ex;
116         } finally {
117             settings.cleanup(false);
118         }
119         return this;
120     }
121 
122     /**
123      * Imports the NVD CVE JSON File into the database.
124      *
125      * @param file the file containing the NVD CVE JSON
126      * @throws ParserConfigurationException is thrown if there is a parser
127      * configuration exception
128      * @throws IOException is thrown if there is a IO Exception
129      * @throws SQLException is thrown if there is a SQL exception
130      * @throws DatabaseException is thrown if there is a database exception
131      * @throws ClassNotFoundException thrown if the h2 database driver cannot be
132      * loaded
133      * @throws UpdateException thrown if the file could not be found
134      */
135     protected void importJSON(File file) throws ParserConfigurationException,
136             IOException, SQLException, DatabaseException, ClassNotFoundException, UpdateException {
137 
138         final NvdCveParser parser = new NvdCveParser(settings, cveDB);
139         parser.parse(file);
140     }
141 
142     /**
143      * Processes the NVD CVE XML file and imports the data into the DB.
144      *
145      * @throws UpdateException thrown if there is an error loading the data into
146      * the database
147      */
148     private void processFiles() throws UpdateException {
149         LOGGER.info("Processing Started for NVD CVE - {}", downloadTask.getNvdCveInfo().getId());
150         final long startProcessing = System.currentTimeMillis();
151         try {
152             importJSON(downloadTask.getFile());
153             cveDB.commit();
154             properties.save(downloadTask.getNvdCveInfo());
155         } catch (ParserConfigurationException | SQLException | DatabaseException | ClassNotFoundException | IOException ex) {
156             throw new UpdateException(ex);
157         } finally {
158             downloadTask.cleanup();
159         }
160         LOGGER.info("Processing Complete for NVD CVE - {}  ({} ms)", downloadTask.getNvdCveInfo().getId(),
161                 System.currentTimeMillis() - startProcessing);
162     }
163 }
