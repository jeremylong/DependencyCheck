40  /**
41   * Checks the gh-pages dependency-check site to determine the current released
42   * version number. If the released version number is greater than the running
43   * version number a warning is printed recommending that an upgrade be
44   * performed.
45   *
46   * @author Jeremy Long
47   */
48  @ThreadSafe
49  public class EngineVersionCheck implements CachedWebDataSource {
50  
51      /**
52       * Static logger.
53       */
54      private static final Logger LOGGER = LoggerFactory.getLogger(EngineVersionCheck.class);
55      /**
56       * The property key indicating when the last version check occurred.
57       */
58      public static final String ENGINE_VERSION_CHECKED_ON = "VersionCheckOn";
59      /**
60       * The property key indicating when the last version check occurred.
61       */
62      public static final String CURRENT_ENGINE_RELEASE = "CurrentEngineRelease";
63      /**
64       * The version retrieved from the database properties or web to check
65       * against.
66       */
67      private String updateToVersion;
68      /**
69       * The configured settings.
70       */
71      private Settings settings;
72  
73      /**
74       * Constructs a new engine version check utility for testing.
75       *
76       * @param settings the configured settings
77       */
78      protected EngineVersionCheck(Settings settings) {
79          this.settings = settings;
80      }
81  
82      /**
83       * Constructs a new engine version check utility.
84       */
85      public EngineVersionCheck() {
86      }
87  
88      /**
89       * Getter for updateToVersion - only used for testing. Represents the
90       * version retrieved from the database.
91       *
92       * @return the version to test
93       */
94      protected String getUpdateToVersion() {
95          return updateToVersion;
96      }
97  
98      /**
99       * Setter for updateToVersion - only used for testing. Represents the
100      * version retrieved from the database.
101      *
102      * @param version the version to test
103      */
104     protected void setUpdateToVersion(String version) {
105         updateToVersion = version;
106     }

108     /**
109      * Downloads the current released version number and compares it to the
110      * running engine's version number. If the released version number is newer
111      * a warning is printed recommending an upgrade.
112      *
113      * @return returns false as no updates are made to the database that would
114      * require compaction
115      * @throws UpdateException thrown if the local database properties could not
116      * be updated
117      */
118     @Override
119     public boolean update(Engine engine) throws UpdateException {
120         this.settings = engine.getSettings();
121         try {
122             final CveDB db = engine.getDatabase();
123             final boolean autoupdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE, true);
124             final boolean enabled = settings.getBoolean(Settings.KEYS.UPDATE_VERSION_CHECK_ENABLED, true);
125             final String original = settings.getString(Settings.KEYS.CVE_ORIGINAL_JSON);
126             final String current = settings.getString(Settings.KEYS.CVE_MODIFIED_JSON);
127             /*
128              * Only update if auto-update is enabled, the engine check is
129              * enabled, and the NVD CVE URLs have not been modified (i.e. the
130              * user has not configured them to point to an internal source).
131              */
132             if (enabled && autoupdate && original != null && original.equals(current)) {
133                 LOGGER.debug("Begin Engine Version Check");
134 
135                 final DatabaseProperties properties = db.getDatabaseProperties();
136 
137                 final long lastChecked = DateUtil.getEpochValueInSeconds(properties.getProperty(ENGINE_VERSION_CHECKED_ON, "0"));
138                 final long now = System.currentTimeMillis() / 1000;
139                 updateToVersion = properties.getProperty(CURRENT_ENGINE_RELEASE, "");
140                 final String currentVersion = settings.getString(Settings.KEYS.APPLICATION_VERSION, "0.0.0");
141                 LOGGER.debug("Last checked: {}", lastChecked);
142                 LOGGER.debug("Now: {}", now);
143                 LOGGER.debug("Current version: {}", currentVersion);
144                 final boolean updateNeeded = shouldUpdate(lastChecked, now, properties, currentVersion);
145                 if (updateNeeded) {
146                     LOGGER.warn("A new version of dependency-check is available. Consider updating to version {}.",
147                             updateToVersion);
148                 }
149             }
150         } catch (DatabaseException ex) {
151             LOGGER.debug("Database Exception opening databases to retrieve properties", ex);
152             throw new UpdateException("Error occurred updating database properties.");
153         }
154         return false;
155     }
156 
157     /**
158      * Determines if a new version of the dependency-check engine has been
159      * released.
160      *
161      * @param lastChecked the epoch time of the last version check
162      * @param now the current epoch time
163      * @param properties the database properties object
164      * @param currentVersion the current version of dependency-check
165      * @return <code>true</code> if a newer version of the database has been
166      * released; otherwise <code>false</code>
167      * @throws UpdateException thrown if there is an error connecting to the
168      * github documentation site or accessing the local database.
169      */
170     protected boolean shouldUpdate(final long lastChecked, final long now, final DatabaseProperties properties,
171             String currentVersion) throws UpdateException {
172         //check every 30 days if we know there is an update, otherwise check every 7 days
173         final int checkRange = 30;
174         if (!DateUtil.withinDateRange(lastChecked, now, checkRange)) {
175             LOGGER.debug("Checking web for new version.");
176             final String currentRelease = getCurrentReleaseVersion();
177             if (currentRelease != null) {
178                 final DependencyVersion v = new DependencyVersion(currentRelease);
179                 if (v.getVersionParts() != null && v.getVersionParts().size() >= 3) {
180                     updateToVersion = v.toString();
181                     if (!currentRelease.equals(updateToVersion)) {
182                         properties.save(CURRENT_ENGINE_RELEASE, updateToVersion);
183                     }
184                     properties.save(ENGINE_VERSION_CHECKED_ON, Long.toString(now));
185                 }
186             }
187             LOGGER.debug("Current Release: {}", updateToVersion);
188         }
189         if (updateToVersion == null) {
190             LOGGER.debug("Unable to obtain current release");
191             return false;
192         }
193         final DependencyVersion running = new DependencyVersion(currentVersion);
194         final DependencyVersion released = new DependencyVersion(updateToVersion);
195         if (running.compareTo(released) < 0) {
196             LOGGER.debug("Upgrade recommended");
197             return true;
198         }
199         LOGGER.debug("Upgrade not needed");
200         return false;
201     }
202 
203     /**
204      * Retrieves the current released version number from the github
205      * documentation site.
206      *
207      * @return the current released version number
208      */
209     protected String getCurrentReleaseVersion() {
210         HttpURLConnection conn = null;
211         try {
212             final String str = settings.getString(Settings.KEYS.ENGINE_VERSION_CHECK_URL, "http://jeremylong.github.io/DependencyCheck/current.txt");
213             final URL url = new URL(str);
214             final URLConnectionFactory factory = new URLConnectionFactory(settings);
215             conn = factory.createHttpURLConnection(url);
216             conn.connect();
217             if (conn.getResponseCode() != 200) {
218                 return null;
219             }
220             final String releaseVersion = new String(ByteStreams.toByteArray(conn.getInputStream()), StandardCharsets.UTF_8);
221             if (releaseVersion != null) {
222                 return releaseVersion.trim();
223             }
224         } catch (MalformedURLException ex) {
225             LOGGER.debug("Unable to retrieve current release version of dependency-check - malformed url?");
226         } catch (URLConnectionFailureException ex) {
227             LOGGER.debug("Unable to retrieve current release version of dependency-check - connection failed");
228         } catch (IOException ex) {
229             LOGGER.debug("Unable to retrieve current release version of dependency-check - i/o exception");
230         } finally {
231             if (conn != null) {
232                 conn.disconnect();
233             }
234         }
235         return null;
236     }

239     public boolean purge(Engine engine) {
240         return true;
241     }
