46  /**
47   * Used to analyze dependencies defined in Pipfile. This analyzer works in
48   * tandem with the `PipfilelockAnalyzer` - and both analyzers use the same key
49   * to enable/disable the analyzers. The PipfilelockAnalyzer will be used over
50   * the Pipfile if the lock file exists.
51   *
52   *
53   * @author fcano
54   */
55  @Experimental
56  @ThreadSafe
57  public class PipfileAnalyzer extends AbstractFileTypeAnalyzer {
58  
59      /**
60       * The logger.
61       */
62      private static final Logger LOGGER = LoggerFactory.getLogger(PipfileAnalyzer.class);
63      /**
64       * "Pipfile" file.
65       */
66      private static final String PIPFILE = "Pipfile";
67      /**
68       * "Pipfile.lock" file.
69       */
70      private static final String LOCKFILE = "Pipfile.lock";
71      /**
72       * The name of the analyzer.
73       */
74      private static final String ANALYZER_NAME = "Pipfile Analyzer";
75  
76      /**
77       * The phase that this analyzer is intended to run in.
78       */
79      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
80  
81      /**
82       * o * Matches AC_INIT variables in the output configure script.
83       */
84      private static final Pattern PACKAGE_VERSION = Pattern.compile("^([^#].*?) = \"(?:[=~>]=([\\.\\*0-9]+?))?\"$", Pattern.MULTILINE);
85  
86      /**
87       * The file filter used to determine which files this analyzer supports.
88       */
89      private static final FileFilter FILTER = FileFilterBuilder.newInstance().addFilenames(PIPFILE).build();
90  
91      /**
92       * Returns the FileFilter
93       *
94       * @return the FileFilter
95       */
96      @Override
97      protected FileFilter getFileFilter() {
98          return FILTER;
99      }
100 
101     /**
102      * Returns the name of the analyzer.
103      *
104      * @return the name of the analyzer.
105      */
106     @Override
107     public String getName() {
108         return ANALYZER_NAME;
109     }
110 
111     /**
112      * Returns the phase that the analyzer is intended to run in.
113      *
114      * @return the phase that the analyzer is intended to run in.
115      */
116     @Override
117     public AnalysisPhase getAnalysisPhase() {
118         return ANALYSIS_PHASE;
119     }
120 
121     /**
122      * Returns the key used in the properties file to reference the analyzer's
123      * enabled property.
124      *
125      * @return the analyzer's enabled property setting key
126      */
127     @Override
128     protected String getAnalyzerEnabledSettingKey() {
129         return Settings.KEYS.ANALYZER_PIPFILE_ENABLED;
130     }
131 
132     @Override
133     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
134         engine.removeDependency(dependency);
135         final File lock = new File(dependency.getActualFile().getParentFile(), LOCKFILE);
136         if (lock.isFile()) {
137             LOGGER.debug("Skipping {} because a lock file was identified", dependency.getActualFilePath());
138             return;
139         } else {
140             LOGGER.debug("Checking file {}", dependency.getActualFilePath());
141         }
142 
143         final File dependencyFile = dependency.getActualFile();
144         if (!dependencyFile.isFile() || dependencyFile.length() == 0) {
145             return;
146         }
147 
148         final File actualFile = dependency.getActualFile();
149 
150         final String contents = getFileContents(actualFile);
151         if (!contents.isEmpty()) {
152             final Matcher matcher = PACKAGE_VERSION.matcher(contents);
153             while (matcher.find()) {
154                 final String identifiedPackage = matcher.group(1);
155                 final String identifiedVersion = matcher.group(2);
156                 LOGGER.debug(String.format("package, version: %s %s", identifiedPackage, identifiedVersion));
157                 final Dependency d = new Dependency(dependency.getActualFile(), true);
158                 d.setName(identifiedPackage);
159                 d.setVersion(identifiedVersion);
160                 try {
161                     final PackageURL purl = PackageURLBuilder.aPackageURL()
162                             .withType("pypi")
163                             .withName(identifiedPackage)
164                             .withVersion(identifiedVersion)
165                             .build();
166                     d.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
167                 } catch (MalformedPackageURLException ex) {
168                     LOGGER.debug("Unable to build package url for pypi", ex);
169                     d.addSoftwareIdentifier(new GenericIdentifier("pypi:" + identifiedPackage + "@" + identifiedVersion, Confidence.HIGH));
170                 }
171                 d.setPackagePath(String.format("%s:%s", identifiedPackage, identifiedVersion));
172                 d.setEcosystem(PythonDistributionAnalyzer.DEPENDENCY_ECOSYSTEM);
173                 final String filePath = String.format("%s:%s/%s", dependency.getFilePath(), identifiedPackage, identifiedVersion);
174                 d.setFilePath(filePath);
175                 d.setSha1sum(Checksum.getSHA1Checksum(filePath));
176                 d.setSha256sum(Checksum.getSHA256Checksum(filePath));
177                 d.setMd5sum(Checksum.getMD5Checksum(filePath));
178                 d.addEvidence(EvidenceType.VENDOR, PIPFILE, "vendor", identifiedPackage, Confidence.HIGHEST);
179                 d.addEvidence(EvidenceType.PRODUCT, PIPFILE, "product", identifiedPackage, Confidence.HIGHEST);
180                 d.addEvidence(EvidenceType.VERSION, PIPFILE, "version", identifiedVersion, Confidence.HIGHEST);
181                 engine.addDependency(d);
182             }
183         }
184     }
185 
186     /**
187      * Retrieves the contents of a given file.
188      *
189      * @param actualFile the file to read
190      * @return the contents of the file
191      * @throws AnalysisException thrown if there is an IO Exception
192      */
193     private String getFileContents(final File actualFile) throws AnalysisException {
194         try {
195             return FileUtils.readFileToString(actualFile, Charset.defaultCharset()).trim();
196         } catch (IOException e) {
197             throw new AnalysisException("Problem occurred while reading dependency file.", e);
198         }
199     }

208     @Override
209     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
210         // No initialization needed.
211     }
