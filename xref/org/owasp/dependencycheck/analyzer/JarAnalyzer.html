288     /**
289      * Loads a specified JAR file and collects information from the manifest and
290      * checksums to identify the correct CPE information.
291      *
292      * @param dependency the dependency to analyze.
293      * @param engine the engine that is scanning the dependencies
294      * @throws AnalysisException is thrown if there is an error reading the JAR
295      * file.
296      */
297     @Override
298     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
299         final List<ClassNameInformation> classNames = collectClassNames(dependency);
300         final String fileName = dependency.getFileName().toLowerCase();
301         if ((classNames.isEmpty()
302                 && (fileName.endsWith("-sources.jar")
303                 || fileName.endsWith("-javadoc.jar")
304                 || fileName.endsWith("-src.jar")
305                 || fileName.endsWith("-doc.jar")
306                 || isMacOSMetaDataFile(dependency, engine)))
307                 || !isZipFile(dependency)) {
308             engine.removeDependency(dependency);
309             return;
310         }
311         Exception exception = null;
312         boolean hasManifest = false;
313         try {
314             hasManifest = parseManifest(dependency, classNames);
315         } catch (IOException ex) {
316             LOGGER.debug("Invalid Manifest", ex);
317             exception = ex;
318         }
319         boolean hasPOM = false;
320         try {
321             hasPOM = analyzePOM(dependency, classNames, engine);
322         } catch (AnalysisException ex) {
323             LOGGER.debug("Error parsing pom.xml", ex);
324             exception = ex;
325         }
326         final boolean addPackagesAsEvidence = !(hasManifest && hasPOM);
327         analyzePackageNames(classNames, dependency, addPackagesAsEvidence);
328         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
329 
330         if (exception != null) {
331             throw new AnalysisException(String.format("An error occurred extracting evidence from "
332                     + "%s, analysis may be incomplete; please see the log for more details.",
333                     dependency.getDisplayFileName()), exception);
334         }
335     }

337     /**
338      * Checks if the given dependency appears to be a macOS meta-data file,
339      * returning true if its filename starts with a ._ prefix and if there is
340      * another dependency with the same filename minus the ._ prefix, otherwise
341      * it returns false.
342      *
343      * @param dependency the dependency to check if it's a macOS meta-data file
344      * @param engine the engine that is scanning the dependencies
345      * @return whether or not the given dependency appears to be a macOS
346      * meta-data file
347      */
348     @SuppressFBWarnings(justification = "If actual file path is not null the path will have elements and getFileName will not be called on a null",
349             value = {"NP_NULL_ON_SOME_PATH_FROM_RETURN_VALUE"})
350     private boolean isMacOSMetaDataFile(final Dependency dependency, final Engine engine) {
351         if (dependency.getActualFilePath() != null) {
352             final String fileName = Paths.get(dependency.getActualFilePath()).getFileName().toString();
353             return fileName.startsWith("._") && hasDependencyWithFilename(engine.getDependencies(), fileName.substring(2));
354         }
355         return false;
356     }

358     /**
359      * Iterates through the given list of dependencies and returns true when it
360      * finds a dependency with a filename matching the given filename, otherwise
361      * returns false.
362      *
363      * @param dependencies the dependencies to search within
364      * @param fileName the filename to search for
365      * @return whether or not the given dependencies contain a dependency with
366      * the given filename
367      */
368     @SuppressFBWarnings(justification = "If actual file path is not null the path will have elements and getFileName will not be called on a null",
369             value = {"NP_NULL_ON_SOME_PATH_FROM_RETURN_VALUE"})
370     private boolean hasDependencyWithFilename(final Dependency[] dependencies, final String fileName) {
371         for (final Dependency dependency : dependencies) {
372             if (dependency.getActualFilePath() != null
373                     && Paths.get(dependency.getActualFilePath()).getFileName().toString().equalsIgnoreCase(fileName)) {
374                 return true;
375             }
376         }
377         return false;
378     }

380     /**
381      * Attempts to read the first bytes of the given dependency (using its
382      * actual file path) and returns true if they match the expected first bytes
383      * of a zip file, which may be empty or spanned. If they don't match, or if
384      * the file could not be read, then it returns false.
385      *
386      * @param dependency the dependency to check if it's a zip file
387      * @return whether or not the given dependency appears to be a zip file from
388      * its first bytes
389      */
390     @SuppressFBWarnings(justification = "try with resources will clean up the output stream", value = {"OBL_UNSATISFIED_OBLIGATION"})
391     private boolean isZipFile(final Dependency dependency) {
392         final byte[] buffer = new byte[4];
393         try (FileInputStream fileInputStream = new FileInputStream(dependency.getActualFilePath())) {
394             if (fileInputStream.read(buffer) > 0
395                     && (Arrays.equals(buffer, ZIP_FIRST_BYTES)
396                     || Arrays.equals(buffer, ZIP_EMPTY_FIRST_BYTES)
397                     || Arrays.equals(buffer, ZIP_SPANNED_FIRST_BYTES))) {
398                 return true;
399             }
400         } catch (Exception e) {
401             LOGGER.warn("Unable to check if '{}' is a zip file", dependency.getActualFilePath());
402             LOGGER.trace("", e);
403         }
404         return false;
405     }

407     /**
408      * Attempts to find a pom.xml within the JAR file. If found it extracts
409      * information and adds it to the evidence. This will attempt to interpolate
410      * the strings contained within the pom.properties if one exists.
411      *
412      * @param dependency the dependency being analyzed
413      * @param classes a collection of class name information
414      * @param engine the analysis engine, used to add additional dependencies
415      * @throws AnalysisException is thrown if there is an exception parsing the
416      * pom
417      * @return whether or not evidence was added to the dependency
418      */
419     protected boolean analyzePOM(Dependency dependency, List<ClassNameInformation> classes, Engine engine) throws AnalysisException {
420 
421         //TODO add breakpoint on groov-all to find out why commons-cli is not added as a new dependency?
422         boolean evidenceAdded = false;
423         try (JarFile jar = new JarFile(dependency.getActualFilePath())) {
424             //check if we are scanning in a repo directory - so the pom is adjacent to the jar
425             final String repoPomName = FilenameUtils.removeExtension(dependency.getActualFilePath()) + ".pom";
426             final File repoPom = new File(repoPomName);
427             if (repoPom.isFile()) {
428                 final Model pom = PomUtils.readPom(repoPom);
429                 evidenceAdded |= setPomEvidence(dependency, pom, classes, true);
430             }
431 
432             final List<String> pomEntries = retrievePomListing(jar);
433 
434             for (String path : pomEntries) {
435                 LOGGER.debug("Reading pom entry: {}", path);
436                 try {
437                     //extract POM to its own directory and add it as its own dependency
438                     final Properties pomProperties = retrievePomProperties(path, jar);
439                     final File pomFile = extractPom(path, jar);
440                     final Model pom = PomUtils.readPom(pomFile);
441                     pom.processProperties(pomProperties);
442 
443                     final String artifactId = new File(path).getParentFile().getName();
444                     if (dependency.getActualFile().getName().startsWith(artifactId)) {
445                         evidenceAdded |= setPomEvidence(dependency, pom, classes, true);
446                     } else {
447                         final String displayPath = String.format("%s%s%s",
448                                 dependency.getFilePath(),
449                                 File.separator,
450                                 path);
451                         final String displayName = String.format("%s%s%s",
452                                 dependency.getFileName(),
453                                 File.separator,
454                                 path);
455                         final Dependency newDependency = new Dependency();
456                         newDependency.setActualFilePath(pomFile.getAbsolutePath());
457                         newDependency.setFileName(displayName);
458                         newDependency.setFilePath(displayPath);
459                         newDependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
460                         String groupId = pom.getGroupId();
461                         String version = pom.getVersion();
462                         if (groupId == null) {
463                             groupId = pom.getParentGroupId();
464                         }
465                         if (version == null) {
466                             version = pom.getParentVersion();
467                         }
468                         if (groupId == null) {
469                             newDependency.setName(pom.getArtifactId());
470                             newDependency.setPackagePath(String.format("%s:%s", pom.getArtifactId(), version));
471                         } else {
472                             newDependency.setName(String.format("%s:%s", groupId, pom.getArtifactId()));
473                             newDependency.setPackagePath(String.format("%s:%s:%s", groupId, pom.getArtifactId(), version));
474                         }
475                         newDependency.setDisplayFileName(String.format("%s (shaded: %s)",
476                                 dependency.getDisplayFileName(), newDependency.getPackagePath()));
477                         newDependency.setVersion(version);
478                         setPomEvidence(newDependency, pom, null, true);
479                         if (dependency.getProjectReferences().size() > 0) {
480                             newDependency.addAllProjectReferences(dependency.getProjectReferences());
481                         }
482                         engine.addDependency(newDependency);
483                     }
484                 } catch (AnalysisException ex) {
485                     LOGGER.warn("An error occurred while analyzing '{}'.", dependency.getActualFilePath());
486                     LOGGER.trace("", ex);
487                 }
488             }
489         } catch (IOException ex) {
490             LOGGER.warn("Unable to read JarFile '{}'.", dependency.getActualFilePath());
491             LOGGER.trace("", ex);
492         }
493         return evidenceAdded;
494     }

496     /**
497      * Given a path to a pom.xml within a JarFile, this method attempts to load
498      * a sibling pom.properties if one exists.
499      *
500      * @param path the path to the pom.xml within the JarFile
501      * @param jar the JarFile to load the pom.properties from
502      * @return a Properties object or null if no pom.properties was found
503      */
504     private Properties retrievePomProperties(String path, final JarFile jar) {
505         Properties pomProperties = null;
506         final String propPath = path.substring(0, path.length() - 7) + "pom.properties";
507         final ZipEntry propEntry = jar.getEntry(propPath);
508         if (propEntry != null) {
509             try (Reader reader = new InputStreamReader(jar.getInputStream(propEntry), StandardCharsets.UTF_8)) {
510                 pomProperties = new Properties();
511                 pomProperties.load(reader);
512                 LOGGER.debug("Read pom.properties: {}", propPath);
513             } catch (UnsupportedEncodingException ex) {
514                 LOGGER.trace("UTF-8 is not supported", ex);
515             } catch (IOException ex) {
516                 LOGGER.trace("Unable to read the POM properties", ex);
517             }
518         }
519         return pomProperties;
520     }

522     /**
523      * Searches a JarFile for pom.xml entries and returns a listing of these
524      * entries.
525      *
526      * @param jar the JarFile to search
527      * @return a list of pom.xml entries
528      * @throws IOException thrown if there is an exception reading a JarEntry
529      */
530     private List<String> retrievePomListing(final JarFile jar) throws IOException {
531         final List<String> pomEntries = new ArrayList<>();
532         final Enumeration<JarEntry> entries = jar.entries();
533         while (entries.hasMoreElements()) {
534             final JarEntry entry = entries.nextElement();
535             final String entryName = new File(entry.getName()).getName().toLowerCase();
536             if (!entry.isDirectory() && "pom.xml".equals(entryName)
537                     && entry.getName().toUpperCase().startsWith("META-INF")) {
538                 pomEntries.add(entry.getName());
539             }
540         }
541         return pomEntries;
542     }

544     /**
545      * Retrieves the specified POM from a jar.
546      *
547      * @param path the path to the pom.xml file within the jar file
548      * @param jar the jar file to extract the pom from
549      * @return returns the POM file
550      * @throws AnalysisException is thrown if there is an exception extracting
551      * the file
552      */
553     private File extractPom(String path, JarFile jar) throws AnalysisException {
554         final File tmpDir = getNextTempDirectory();
555         final File file = new File(tmpDir, "pom.xml");
556         final ZipEntry entry = jar.getEntry(path);
557         if (entry == null) {
558             throw new AnalysisException(String.format("Pom (%s) does not exist in %s", path, jar.getName()));
559         }
560         try (InputStream input = jar.getInputStream(entry);
561                 FileOutputStream fos = new FileOutputStream(file)) {
562             ByteStreams.copy(input, fos);
563         } catch (IOException ex) {
564             LOGGER.warn("An error occurred reading '{}' from '{}'.", path, jar.getName());
565             LOGGER.error("", ex);
566         }
567         return file;
568     }

570     /**
571      * Sets evidence from the pom on the supplied dependency.
572      *
573      * @param dependency the dependency to set data on
574      * @param pom the information from the pom
575      * @param classes a collection of ClassNameInformation - containing data
576      * about the fully qualified class names within the JAR file being analyzed
577      * @param isMainPom a flag indicating if this is the primary pom.
578      * @return true if there was evidence within the pom that we could use;
579      * otherwise false
580      */
581     public static boolean setPomEvidence(Dependency dependency, Model pom, List<ClassNameInformation> classes, boolean isMainPom) {
582         if (pom == null) {
583             return false;
584         }
585         boolean foundSomething = false;
586         boolean addAsIdentifier = true;
587         String groupid = pom.getGroupId();
588         String parentGroupId = pom.getParentGroupId();
589         String artifactid = pom.getArtifactId();
590         String parentArtifactId = pom.getParentArtifactId();
591         String version = pom.getVersion();
592         String parentVersion = pom.getParentVersion();
593 
594         if (("org.sonatype.oss".equals(parentGroupId) && "oss-parent".equals(parentArtifactId))
595                 || ("org.springframework.boot".equals(parentGroupId) && "spring-boot-starter-parent".equals(parentArtifactId))) {
596             parentGroupId = null;
597             parentArtifactId = null;
598             parentVersion = null;
599         }
600 
601         if ((groupid == null || groupid.isEmpty()) && parentGroupId != null && !parentGroupId.isEmpty()) {
602             groupid = parentGroupId;
603         }
604 
605         final String originalGroupID = groupid;
606         if (groupid != null && (groupid.startsWith("org.") || groupid.startsWith("com."))) {
607             groupid = groupid.substring(4);
608         }
609 
610         if ((artifactid == null || artifactid.isEmpty()) && parentArtifactId != null && !parentArtifactId.isEmpty()) {
611             artifactid = parentArtifactId;
612         }
613 
614         final String originalArtifactID = artifactid;
615         if (artifactid != null && (artifactid.startsWith("org.") || artifactid.startsWith("com."))) {
616             artifactid = artifactid.substring(4);
617         }
618 
619         if ((version == null || version.isEmpty()) && parentVersion != null && !parentVersion.isEmpty()) {
620             version = parentVersion;
621         }
622 
623         if (isMainPom && dependency.getName() == null && originalArtifactID != null && !originalArtifactID.isEmpty()) {
624             if (originalGroupID != null && !originalGroupID.isEmpty()) {
625                 dependency.setName(String.format("%s:%s", originalGroupID, originalArtifactID));
626             } else {
627                 dependency.setName(originalArtifactID);
628             }
629         }
630         if (isMainPom && dependency.getVersion() == null && version != null && !version.isEmpty()) {
631             dependency.setVersion(version);
632         }
633 
634         if (groupid != null && !groupid.isEmpty()) {
635             foundSomething = true;
636             dependency.addEvidence(EvidenceType.VENDOR, "pom", "groupid", groupid, Confidence.HIGHEST);
637             //In several cases we are seeing the product name at the end of the group identifier.
638             // This may cause several FP on products that have a collection of dependencies (e.g. jetty).
639             //dependency.addEvidence(EvidenceType.PRODUCT, "pom", "groupid", groupid, Confidence.LOW);
640             dependency.addEvidence(EvidenceType.PRODUCT, "pom", "groupid", groupid, Confidence.HIGHEST);
641             addMatchingValues(classes, groupid, dependency, EvidenceType.VENDOR);
642             addMatchingValues(classes, groupid, dependency, EvidenceType.PRODUCT);
643             if (parentGroupId != null && !parentGroupId.isEmpty() && !parentGroupId.equals(groupid)) {
644                 dependency.addEvidence(EvidenceType.VENDOR, "pom", "parent-groupid", parentGroupId, Confidence.MEDIUM);
645                 //see note above for groupid
646                 //dependency.addEvidence(EvidenceType.PRODUCT, "pom", "parent-groupid", parentGroupId, Confidence.LOW);
647                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "parent-groupid", parentGroupId, Confidence.MEDIUM);
648                 addMatchingValues(classes, parentGroupId, dependency, EvidenceType.VENDOR);
649                 addMatchingValues(classes, parentGroupId, dependency, EvidenceType.PRODUCT);
650             }
651         } else {
652             addAsIdentifier = false;
653         }
654 
655         if (artifactid != null && !artifactid.isEmpty()) {
656             foundSomething = true;
657             dependency.addEvidence(EvidenceType.PRODUCT, "pom", "artifactid", artifactid, Confidence.HIGHEST);
658             dependency.addEvidence(EvidenceType.VENDOR, "pom", "artifactid", artifactid, Confidence.LOW);
659             addMatchingValues(classes, artifactid, dependency, EvidenceType.VENDOR);
660             addMatchingValues(classes, artifactid, dependency, EvidenceType.PRODUCT);
661             if (parentArtifactId != null && !parentArtifactId.isEmpty() && !parentArtifactId.equals(artifactid)) {
662                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "parent-artifactid", parentArtifactId, Confidence.MEDIUM);
663                 dependency.addEvidence(EvidenceType.VENDOR, "pom", "parent-artifactid", parentArtifactId, Confidence.LOW);
664                 addMatchingValues(classes, parentArtifactId, dependency, EvidenceType.VENDOR);
665                 addMatchingValues(classes, parentArtifactId, dependency, EvidenceType.PRODUCT);
666             }
667         } else {
668             addAsIdentifier = false;
669         }
670 
671         if (version != null && !version.isEmpty()) {
672             foundSomething = true;
673             dependency.addEvidence(EvidenceType.VERSION, "pom", "version", version, Confidence.HIGHEST);
674             if (parentVersion != null && !parentVersion.isEmpty() && !parentVersion.equals(version)) {
675                 dependency.addEvidence(EvidenceType.VERSION, "pom", "parent-version", version, Confidence.LOW);
676             }
677         } else {
678             addAsIdentifier = false;
679         }
680 
681         if (addAsIdentifier && isMainPom) {
682             Identifier id = null;
683             try {
684                 if (originalArtifactID != null && originalArtifactID.matches(VALID_NAME)
685                         && originalGroupID != null && originalGroupID.matches(VALID_NAME)) {
686                     final PackageURL purl = PackageURLBuilder.aPackageURL().withType("maven").withNamespace(originalGroupID)
687                             .withName(originalArtifactID).withVersion(version).build();
688                     id = new PurlIdentifier(purl, Confidence.HIGH);
689                 } else {
690                     LOGGER.debug("Invalid maven identifier identified: " + originalGroupID + ":" + originalArtifactID);
691 //                    final String gav = String.format("%s:%s:%s", originalGroupID, originalArtifactID, version);
692 //                    id = new GenericIdentifier("generic:" + gav, Confidence.LOW);
693                 }
694             } catch (MalformedPackageURLException ex) {
695                 final String gav = String.format("%s:%s:%s", originalGroupID, originalArtifactID, version);
696                 LOGGER.debug("Error building package url for " + gav + "; using generic identifier instead.", ex);
697                 id = new GenericIdentifier("maven:" + gav, Confidence.HIGH);
698             }
699             if (id != null) {
700                 dependency.addSoftwareIdentifier(id);
701             }
702         }
703 
704         // org name
705         final String org = pom.getOrganization();
706         if (org != null && !org.isEmpty()) {
707             dependency.addEvidence(EvidenceType.VENDOR, "pom", "organization name", org, Confidence.HIGH);
708             dependency.addEvidence(EvidenceType.PRODUCT, "pom", "organization name", org, Confidence.LOW);
709             addMatchingValues(classes, org, dependency, EvidenceType.VENDOR);
710             addMatchingValues(classes, org, dependency, EvidenceType.PRODUCT);
711         }
712         // org name
713         String orgUrl = pom.getOrganizationUrl();
714         if (orgUrl != null && !orgUrl.isEmpty()) {
715             if (orgUrl.startsWith("https://github.com/") || orgUrl.startsWith("https://gitlab.com/")) {
716                 orgUrl = orgUrl.substring(19);
717                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "url", orgUrl, Confidence.HIGH);
718             } else {
719                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "organization url", orgUrl, Confidence.LOW);
720             }
721             dependency.addEvidence(EvidenceType.VENDOR, "pom", "organization url", orgUrl, Confidence.MEDIUM);
722         }
723         //pom name
724         final String pomName = pom.getName();
725         if (pomName != null && !pomName.isEmpty() && !"${project.groupId}:${project.artifactId}".equals(pomName)) {
726             foundSomething = true;
727             dependency.addEvidence(EvidenceType.PRODUCT, "pom", "name", pomName, Confidence.HIGH);
728             dependency.addEvidence(EvidenceType.VENDOR, "pom", "name", pomName, Confidence.HIGH);
729             addMatchingValues(classes, pomName, dependency, EvidenceType.VENDOR);
730             addMatchingValues(classes, pomName, dependency, EvidenceType.PRODUCT);
731         }
732 
733         //Description
734         final String description = pom.getDescription();
735         if (description != null && !description.isEmpty()
736                 && !description.startsWith("POM was created by")
737                 && !description.startsWith("Sonatype helps open source projects")
738                 && !description.endsWith("project for Spring Boot")) {
739             foundSomething = true;
740             final String trimmedDescription = addDescription(dependency, description, "pom", "description");
741             addMatchingValues(classes, trimmedDescription, dependency, EvidenceType.VENDOR);
742             addMatchingValues(classes, trimmedDescription, dependency, EvidenceType.PRODUCT);
743         }
744 
745         String projectURL = pom.getProjectURL();
746         if (projectURL != null && !projectURL.trim().isEmpty()) {
747             if (projectURL.startsWith("https://github.com/") || projectURL.startsWith("https://gitlab.com/")) {
748                 projectURL = projectURL.substring(19);
749                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "url", projectURL, Confidence.HIGH);
750             } else {
751                 dependency.addEvidence(EvidenceType.PRODUCT, "pom", "url", projectURL, Confidence.MEDIUM);
752             }
753             dependency.addEvidence(EvidenceType.VENDOR, "pom", "url", projectURL, Confidence.HIGHEST);
754 
755         }
756 
757         extractLicense(pom, dependency);
758         return foundSomething;
759     }

761     /**
762      * Analyzes the path information of the classes contained within the
763      * JarAnalyzer to try and determine possible vendor or product names. If any
764      * are found they are stored in the packageVendor and packageProduct
765      * hashSets.
766      *
767      * @param classNames a list of class names
768      * @param dependency a dependency to analyze
769      * @param addPackagesAsEvidence a flag indicating whether or not package
770      * names should be added as evidence.
771      */
772     protected void analyzePackageNames(List<ClassNameInformation> classNames,
773             Dependency dependency, boolean addPackagesAsEvidence) {
774         final Map<String, Integer> vendorIdentifiers = new HashMap<>();
775         final Map<String, Integer> productIdentifiers = new HashMap<>();
776         analyzeFullyQualifiedClassNames(classNames, vendorIdentifiers, productIdentifiers);
777 
778         final int classCount = classNames.size();
779 
780         vendorIdentifiers.forEach((key, value) -> {
781             final float ratio = value / (float) classCount;
782             if (ratio > 0.5) {
783                 //TODO remove weighting?
784                 dependency.addVendorWeighting(key);
785                 if (addPackagesAsEvidence && key.length() > 1) {
786                     dependency.addEvidence(EvidenceType.VENDOR, "jar", "package name", key, Confidence.LOW);
787                 }
788             }
789         });
790         productIdentifiers.forEach((key, value) -> {
791             final float ratio = value / (float) classCount;
792             if (ratio > 0.5) {
793                 //todo remove weighting
794                 dependency.addProductWeighting(key);
795                 if (addPackagesAsEvidence && key.length() > 1) {
796                     dependency.addEvidence(EvidenceType.PRODUCT, "jar", "package name", key, Confidence.LOW);
797                 }
798             }
799         });
800     }

802     /**
803      * <p>
804      * Reads the manifest from the JAR file and collects the entries. Some
805      * vendorKey entries are:</p>
806      * <ul><li>Implementation Title</li>
807      * <li>Implementation Version</li> <li>Implementation Vendor</li>
808      * <li>Implementation VendorId</li> <li>Bundle Name</li> <li>Bundle
809      * Version</li> <li>Bundle Vendor</li> <li>Bundle Description</li> <li>Main
810      * Class</li> </ul>
811      * However, all but a handful of specific entries are read in.
812      *
813      * @param dependency A reference to the dependency
814      * @param classInformation a collection of class information
815      * @return whether evidence was identified parsing the manifest
816      * @throws IOException if there is an issue reading the JAR file
817      */
818     //CSOFF: MethodLength
819     protected boolean parseManifest(Dependency dependency, List<ClassNameInformation> classInformation)
820             throws IOException {
821         boolean foundSomething = false;
822         try (JarFile jar = new JarFile(dependency.getActualFilePath())) {
823             final Manifest manifest = jar.getManifest();
824             if (manifest == null) {
825                 if (!dependency.getFileName().toLowerCase().endsWith("-sources.jar")
826                         && !dependency.getFileName().toLowerCase().endsWith("-javadoc.jar")
827                         && !dependency.getFileName().toLowerCase().endsWith("-src.jar")
828                         && !dependency.getFileName().toLowerCase().endsWith("-doc.jar")) {
829                     LOGGER.debug("Jar file '{}' does not contain a manifest.", dependency.getFileName());
830                 }
831                 return false;
832             }
833             String source = "Manifest";
834             String specificationVersion = null;
835             boolean hasImplementationVersion = false;
836             Attributes atts = manifest.getMainAttributes();
837             for (Entry<Object, Object> entry : atts.entrySet()) {
838                 String key = entry.getKey().toString();
839                 String value = atts.getValue(key);
840                 if (HTML_DETECTION_PATTERN.matcher(value).find()) {
841                     value = Jsoup.parse(value).text();
842                 }
843                 if (value.startsWith("git@github.com:") || value.startsWith("git@gitlab.com:")) {
844                     value = value.substring(15);
845                 }
846                 if (IGNORE_VALUES.contains(value)) {
847                     continue;
848                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_TITLE.toString())) {
849                     foundSomething = true;
850                     dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.HIGH);
851                     addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
852                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VERSION.toString())) {
853                     hasImplementationVersion = true;
854                     foundSomething = true;
855                     dependency.addEvidence(EvidenceType.VERSION, source, key, value, Confidence.HIGH);
856                 } else if ("specification-version".equalsIgnoreCase(key)) {
857                     specificationVersion = value;
858                 } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VENDOR.toString())) {
859                     foundSomething = true;
860                     dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.HIGH);
861                     addMatchingValues(classInformation, value, dependency, EvidenceType.VENDOR);
862                 } else if (key.equalsIgnoreCase(IMPLEMENTATION_VENDOR_ID)) {
863                     foundSomething = true;
864                     dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.MEDIUM);
865                     addMatchingValues(classInformation, value, dependency, EvidenceType.VENDOR);
866                 } else if (key.equalsIgnoreCase(BUNDLE_DESCRIPTION)) {
867                     if (!value.startsWith("Sonatype helps open source projects")) {
868                         foundSomething = true;
869                         addDescription(dependency, value, "manifest", key);
870                         addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
871                     }
872                 } else if (key.equalsIgnoreCase(BUNDLE_NAME)) {
873                     foundSomething = true;
874                     dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.MEDIUM);
875                     addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
876 //                //the following caused false positives.
877 //                } else if (key.equalsIgnoreCase(BUNDLE_VENDOR)) {
878                 } else if (key.equalsIgnoreCase(BUNDLE_VERSION)) {
879                     foundSomething = true;
880                     dependency.addEvidence(EvidenceType.VERSION, source, key, value, Confidence.HIGH);
881                 } else if (key.equalsIgnoreCase(Attributes.Name.MAIN_CLASS.toString())) {
882                     //noinspection UnnecessaryContinue
883                     continue;
884                     //skipping main class as if this has important information to add it will be added during class name analysis...
885                 } else if ("implementation-url".equalsIgnoreCase(key)
886                         && value != null
887                         && value.startsWith("https://projects.spring.io/spring-boot/#/spring-boot-starter-parent/parent/")) {
888                     continue;
889                 } else {
890                     key = key.toLowerCase();
891                     if (!IGNORE_KEYS.contains(key)
892                             && !key.endsWith("jdk")
893                             && !key.contains("lastmodified")
894                             && !key.endsWith("package")
895                             && !key.endsWith("classpath")
896                             && !key.endsWith("class-path")
897                             && !key.endsWith("-scm") //todo change this to a regex?
898                             && !key.startsWith("scm-")
899                             && !value.trim().startsWith("scm:")
900                             && !isImportPackage(key, value)
901                             && !isPackage(key, value)) {
902                         foundSomething = true;
903                         if (key.contains("version")) {
904                             if (!key.contains("specification")) {
905                                 dependency.addEvidence(EvidenceType.VERSION, source, key, value, Confidence.MEDIUM);
906                             }
907                         } else if ("build-id".equals(key)) {
908                             int pos = value.indexOf('(');
909                             if (pos > 0) {
910                                 value = value.substring(0, pos - 1);
911                             }
912                             pos = value.indexOf('[');
913                             if (pos > 0) {
914                                 value = value.substring(0, pos - 1);
915                             }
916                             dependency.addEvidence(EvidenceType.VERSION, source, key, value, Confidence.MEDIUM);
917                         } else if (key.contains("title")) {
918                             dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.MEDIUM);
919                             addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
920                         } else if (key.contains("vendor")) {
921                             if (key.contains("specification")) {
922                                 dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.LOW);
923                             } else {
924                                 dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.MEDIUM);
925                                 addMatchingValues(classInformation, value, dependency, EvidenceType.VENDOR);
926                             }
927                         } else if (key.contains("name")) {
928                             dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.MEDIUM);
929                             dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.MEDIUM);
930                             addMatchingValues(classInformation, value, dependency, EvidenceType.VENDOR);
931                             addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
932                         } else if (key.contains("license")) {
933                             addLicense(dependency, value);
934                         } else if (key.contains("description")) {
935                             if (!value.startsWith("Sonatype helps open source projects")) {
936                                 final String trimmedDescription = addDescription(dependency, value, "manifest", key);
937                                 addMatchingValues(classInformation, trimmedDescription, dependency, EvidenceType.VENDOR);
938                                 addMatchingValues(classInformation, trimmedDescription, dependency, EvidenceType.PRODUCT);
939                             }
940                         } else {
941                             dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.LOW);
942                             dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.LOW);
943                             addMatchingValues(classInformation, value, dependency, EvidenceType.VERSION);
944                             addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
945                             if (value.matches(".*\\d.*")) {
946                                 final StringTokenizer tokenizer = new StringTokenizer(value, " ");
947                                 while (tokenizer.hasMoreElements()) {
948                                     final String s = tokenizer.nextToken();
949                                     if (s.matches("^[0-9.]+$")) {
950                                         dependency.addEvidence(EvidenceType.VERSION, source, key, s, Confidence.LOW);
951                                     }
952                                 }
953                             }
954                         }
955                     }
956                 }
957             }
958             for (Map.Entry<String, Attributes> item : manifest.getEntries().entrySet()) {
959                 final String name = item.getKey();
960                 source = "manifest: " + name;
961                 atts = item.getValue();
962                 for (Entry<Object, Object> entry : atts.entrySet()) {
963                     final String key = entry.getKey().toString();
964                     final String value = atts.getValue(key);
965                     if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_TITLE.toString())) {
966                         foundSomething = true;
967                         dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.MEDIUM);
968                         addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
969                     } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VERSION.toString())) {
970                         foundSomething = true;
971                         dependency.addEvidence(EvidenceType.VERSION, source, key, value, Confidence.MEDIUM);
972                     } else if (key.equalsIgnoreCase(Attributes.Name.IMPLEMENTATION_VENDOR.toString())) {
973                         foundSomething = true;
974                         dependency.addEvidence(EvidenceType.VENDOR, source, key, value, Confidence.MEDIUM);
975                         addMatchingValues(classInformation, value, dependency, EvidenceType.VENDOR);
976                     } else if (key.equalsIgnoreCase(Attributes.Name.SPECIFICATION_TITLE.toString())) {
977                         foundSomething = true;
978                         dependency.addEvidence(EvidenceType.PRODUCT, source, key, value, Confidence.MEDIUM);
979                         addMatchingValues(classInformation, value, dependency, EvidenceType.PRODUCT);
980                     }
981                 }
982             }
983             if (specificationVersion != null && !hasImplementationVersion) {
984                 foundSomething = true;
985                 dependency.addEvidence(EvidenceType.VERSION, source, "specification-version", specificationVersion, Confidence.HIGH);
986             }
987         }
988         return foundSomething;
989     }

992     /**
993      * Adds a description to the given dependency. If the description contains
994      * one of the following strings beyond 100 characters, then the description
995      * used will be trimmed to that position:
996      * <ul><li>"such as"</li><li>"like "</li><li>"will use "</li><li>"* uses
997      * "</li></ul>
998      *
999      * @param dependency a dependency
1000      * @param description the description
1001      * @param source the source of the evidence
1002      * @param key the "name" of the evidence
1003      * @return if the description is trimmed, the trimmed version is returned;
1004      * otherwise the original description is returned
1005      */
1006     public static String addDescription(Dependency dependency, String description, String source, String key) {
1007         if (dependency.getDescription() == null) {
1008             dependency.setDescription(description);
1009         }
1010         String desc;
1011         if (HTML_DETECTION_PATTERN.matcher(description).find()) {
1012             desc = Jsoup.parse(description).text();
1013         } else {
1014             desc = description;
1015         }
1016         dependency.setDescription(desc);
1017         if (desc.length() > 100) {
1018             desc = desc.replaceAll("\\s\\s+", " ");
1019             final int posSuchAs = desc.toLowerCase().indexOf("such as ", 100);
1020             final int posLike = desc.toLowerCase().indexOf("like ", 100);
1021             final int posWillUse = desc.toLowerCase().indexOf("will use ", 100);
1022             final int posUses = desc.toLowerCase().indexOf(" uses ", 100);
1023 
1024             int pos = -1;
1025             pos = Math.max(pos, posSuchAs);
1026             if (pos >= 0 && posLike >= 0) {
1027                 pos = Math.min(pos, posLike);
1028             } else {
1029                 pos = Math.max(pos, posLike);
1030             }
1031             if (pos >= 0 && posWillUse >= 0) {
1032                 pos = Math.min(pos, posWillUse);
1033             } else {
1034                 pos = Math.max(pos, posWillUse);
1035             }
1036             if (pos >= 0 && posUses >= 0) {
1037                 pos = Math.min(pos, posUses);
1038             } else {
1039                 pos = Math.max(pos, posUses);
1040             }
1041             if (pos > 0) {
1042                 desc = desc.substring(0, pos) + "...";
1043             }
1044 //            //no longer add description directly. Use matching terms in other parts of the evidence collection
1045 //            //but description adds too many FP
1046 //            dependency.addEvidence(EvidenceType.PRODUCT, source, key, desc, Confidence.LOW);
1047 //            dependency.addEvidence(EvidenceType.VENDOR, source, key, desc, Confidence.LOW);
1048 //        } else {
1049 //            dependency.addEvidence(EvidenceType.PRODUCT, source, key, desc, Confidence.MEDIUM);
1050 //            dependency.addEvidence(EvidenceType.VENDOR, source, key, desc, Confidence.MEDIUM);
1051         }
1052         return desc;
1053     }

1130     /**
1131      * Cycles through an enumeration of JarEntries, contained within the
1132      * dependency, and returns a list of the class names. This does not include
1133      * core Java package names (i.e. java.* or javax.*).
1134      *
1135      * @param dependency the dependency being analyzed
1136      * @return an list of fully qualified class names
1137      */
1138     protected List<ClassNameInformation> collectClassNames(Dependency dependency) {
1139         final List<ClassNameInformation> classNames = new ArrayList<>();
1140         try (JarFile jar = new JarFile(dependency.getActualFilePath())) {
1141             final Enumeration<JarEntry> entries = jar.entries();
1142             while (entries.hasMoreElements()) {
1143                 final JarEntry entry = entries.nextElement();
1144                 final String name = entry.getName().toLowerCase();
1145                 //no longer stripping "|com\\.sun" - there are some com.sun jar files with CVEs.
1146                 if (name.endsWith(".class") && !name.matches("^javax?\\..*$")) {
1147                     final ClassNameInformation className = new ClassNameInformation(name.substring(0, name.length() - 6));
1148                     classNames.add(className);
1149                 }
1150             }
1151         } catch (IOException ex) {
1152             LOGGER.warn("Unable to open jar file '{}'.", dependency.getFileName());
1153             LOGGER.debug("", ex);
1154         }
1155         return classNames;
1156     }

1159      * Cycles through the list of class names and places the package levels 0-3
1160      * into the provided maps for vendor and product. This is helpful when
1161      * analyzing vendor/product as many times this is included in the package
1162      * name.

1164      * @param classNames a list of class names

1170     private void analyzeFullyQualifiedClassNames(List<ClassNameInformation> classNames,
1171             Map<String, Integer> vendor, Map<String, Integer> product) {
1172         for (ClassNameInformation entry : classNames) {
1173             final List<String> list = entry.getPackageStructure();
1174             addEntry(vendor, list.get(0));

1178             } else if (list.size() == 3) {

1183                 addEntry(vendor, list.get(1));
1184                 addEntry(vendor, list.get(2));
1185                 addEntry(product, list.get(1));
1186                 addEntry(product, list.get(2));
1187                 addEntry(product, list.get(3));

1209      * Cycles through the collection of class name information to see if parts
1210      * of the package names are contained in the provided value. If found, it
1211      * will be added as the HIGHEST confidence evidence because we have more
1212      * then one source corroborating the value.
1213      *
1214      * @param classes a collection of class name information
1215      * @param value the value to check to see if it contains a package name
1216      * @param dep the dependency to add new entries too
1217      * @param type the type of evidence (vendor, product, or version)
1218      */
1219     protected static void addMatchingValues(List<ClassNameInformation> classes, String value, Dependency dep, EvidenceType type) {
1220         if (value == null || value.isEmpty() || classes == null || classes.isEmpty()) {
1221             return;
1222         }
1223         final HashSet<String> tested = new HashSet<>();
1224         //TODO add a hashSet and only analyze any given key once.
1225         for (ClassNameInformation cni : classes) {
1226             //classes.forEach((cni) -> {
1227             for (String key : cni.getPackageStructure()) {
1228                 //cni.getPackageStructure().forEach((key) -> {
1229                 if (!tested.contains(key)) {
1230                     tested.add(key);
1231                     final int pos = StringUtils.indexOfIgnoreCase(value, key);
1232                     if ((pos == 0 && (key.length() == value.length() || (key.length() < value.length()
1233                             && !Character.isLetterOrDigit(value.charAt(key.length())))))
1234                             || (pos > 0 && !Character.isLetterOrDigit(value.charAt(pos - 1))
1235                             && (pos + key.length() == value.length() || (key.length() < value.length()
1236                             && !Character.isLetterOrDigit(value.charAt(pos + key.length())))))) {
1237                         dep.addEvidence(type, "jar", "package name", key, Confidence.HIGHEST);

1301     /**
1302      * Stores information about a class name.
1303      */
1304     protected static class ClassNameInformation {
1305 
1306         /**
1307          * The fully qualified class name.
1308          */
1309         private String name;
1310         /**
1311          * Up to the first four levels of the package structure, excluding a
1312          * leading "org" or "com".
1313          */
1314         private final ArrayList<String> packageStructure = new ArrayList<>();
1315 
1316         /**
1317          * <p>
1318          * Stores information about a given class name. This class will keep the
1319          * fully qualified class name and a list of the important parts of the
1320          * package structure. Up to the first four levels of the package
1321          * structure are stored, excluding a leading "org" or "com".
1322          * Example:</p>
1323          * <code>ClassNameInformation obj = new ClassNameInformation("org/owasp/dependencycheck/analyzer/JarAnalyzer");
1324          * System.out.println(obj.getName());
1325          * for (String p : obj.getPackageStructure())
1326          *     System.out.println(p);
1327          * </code>
1328          * <p>
1329          * Would result in:</p>
1330          * <code>org.owasp.dependencycheck.analyzer.JarAnalyzer
1331          * owasp
1332          * dependencycheck
1333          * analyzer
1334          * jaranalyzer</code>
1335          *
1336          * @param className a fully qualified class name
1337          */
1338         ClassNameInformation(String className) {
1339             name = className;
1340             if (name.contains("/")) {
1341                 final String[] tmp = StringUtils.split(className.toLowerCase(), '/');
1342                 int start = 0;
1343                 int end = 3;
1344                 if ("com".equals(tmp[0]) || "org".equals(tmp[0])) {
1345                     start = 1;
1346                     end = 4;
1347                 }
1348                 if (tmp.length <= end) {
1349                     end = tmp.length - 1;
1350                 }
1351                 packageStructure.addAll(Arrays.asList(tmp).subList(start, end + 1));
1352             } else {
1353                 packageStructure.add(name);
1354             }
1355         }
1356 
1357         /**
1358          * Get the value of name
1359          *
1360          * @return the value of name
1361          */
1362         public String getName() {
1363             return name;
1364         }
1365 
1366         /**
1367          * Set the value of name
1368          *
1369          * @param name new value of name
1370          */
1371         public void setName(String name) {
1372             this.name = name;
1373         }
1374 
1375         /**
1376          * Get the value of packageStructure
1377          *
1378          * @return the value of packageStructure
1379          */
1380         public ArrayList<String> getPackageStructure() {
1381             return packageStructure;
1382         }
1383     }
