194      * @throws InitializationException is thrown if there is an exception
195      * deleting or creating temporary files

197     @Override
198     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
199         try {
200             final File baseDir = getSettings().getTempDirectory();
201             tempFileLocation = File.createTempFile("check", "tmp", baseDir);
202             if (!tempFileLocation.delete()) {
203                 setEnabled(false);
204                 final String msg = String.format("Unable to delete temporary file '%s'.", tempFileLocation.getAbsolutePath());
205                 throw new InitializationException(msg);
206             }
207             if (!tempFileLocation.mkdirs()) {
208                 setEnabled(false);
209                 final String msg = String.format("Unable to create directory '%s'.", tempFileLocation.getAbsolutePath());
210                 throw new InitializationException(msg);
211             }
212         } catch (IOException ex) {
213             setEnabled(false);
214             throw new InitializationException("Unable to create a temporary file", ex);
215         }
216     }

235                 }

240     /**
241      * Determines if the file can be analyzed by the analyzer. If the npm
242      * analyzer are enabled the archive analyzer will skip the node_modules and
243      * bower_modules directories.
244      *
245      * @param pathname the path to the file
246      * @return true if the file can be analyzed by the given analyzer; otherwise
247      * false
248      */
249     @Override
250     public boolean accept(File pathname) {
251         boolean accept = super.accept(pathname);
252         final boolean npmEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED, false);
253         final boolean yarnEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_YARN_AUDIT_ENABLED, false);
254         final boolean pnpmEnabled = getSettings().getBoolean(Settings.KEYS.ANALYZER_PNPM_AUDIT_ENABLED, false);
255         if (accept && (npmEnabled || yarnEnabled || pnpmEnabled)) {
256             try {
257                 accept = shouldProcess(pathname);
258             } catch (AnalysisException ex) {
259                 throw new UnexpectedAnalysisException(ex.getMessage(), ex.getCause());
260             }
261         }
262         return accept;
263     }

280     /**
281      * Extracts the contents of the archive dependency and scans for additional
282      * dependencies.
283      *
284      * @param dependency the dependency being analyzed
285      * @param engine the engine doing the analysis
286      * @param scanDepth the current scan depth; extracctAndAnalyze is recursive
287      * and will, be default, only go 3 levels deep
288      * @throws AnalysisException thrown if there is a problem analyzing the
289      * dependencies
290      */
291     private void extractAndAnalyze(Dependency dependency, Engine engine, int scanDepth) throws AnalysisException {
292         final File f = new File(dependency.getActualFilePath());
293         final File tmpDir = getNextTempDirectory();
294         extractFiles(f, tmpDir, engine);
295 
296         //make a copy
297         final List<Dependency> dependencySet = findMoreDependencies(engine, tmpDir);
298 
299         if (dependencySet != null && !dependencySet.isEmpty()) {
300             for (Dependency d : dependencySet) {
301                 if (d.getFilePath().startsWith(tmpDir.getAbsolutePath())) {
302                     //fix the dependency's display name and path
303                     final String displayPath = String.format("%s%s",
304                             dependency.getFilePath(),
305                             d.getActualFilePath().substring(tmpDir.getAbsolutePath().length()));
306                     final String displayName = String.format("%s: %s",
307                             dependency.getFileName(),
308                             d.getFileName());
309                     d.setFilePath(displayPath);
310                     d.setFileName(displayName);
311                     d.addAllProjectReferences(dependency.getProjectReferences());
312 
313                     //TODO - can we get more evidence from the parent? EAR contains module name, etc.
314                     //analyze the dependency (i.e. extract files) if it is a supported type.
315                     if (this.accept(d.getActualFile()) && scanDepth < maxScanDepth) {
316                         extractAndAnalyze(d, engine, scanDepth + 1);
317                     }
318                 } else {
319                     dependencySet.stream().filter((sub) -> sub.getFilePath().startsWith(tmpDir.getAbsolutePath())).forEach((sub) -> {
320                         final String displayPath = String.format("%s%s",
321                                 dependency.getFilePath(),
322                                 sub.getActualFilePath().substring(tmpDir.getAbsolutePath().length()));
323                         final String displayName = String.format("%s: %s",
324                                 dependency.getFileName(),
325                                 sub.getFileName());
326                         sub.setFilePath(displayPath);
327                         sub.setFileName(displayName);
328                     });
329                 }
330             }
331         }
332         if (REMOVE_FROM_ANALYSIS.accept(dependency.getActualFile())) {
333             addDisguisedJarsToDependencies(dependency, engine);
334             engine.removeDependency(dependency);
335         }
336     }

338     /**
339      * If a zip file was identified as a possible JAR, this method will add the
340      * zip to the list of dependencies.
341      *
342      * @param dependency the zip file
343      * @param engine the engine
344      * @throws AnalysisException thrown if there is an issue
345      */
346     private void addDisguisedJarsToDependencies(Dependency dependency, Engine engine) throws AnalysisException {
347         if (ZIP_FILTER.accept(dependency.getActualFile()) && isZipFileActuallyJarFile(dependency)) {
348             final File tempDir = getNextTempDirectory();
349             final String fileName = dependency.getFileName();
350 
351             LOGGER.info("The zip file '{}' appears to be a JAR file, making a copy and analyzing it as a JAR.", fileName);
352             final File tmpLoc = new File(tempDir, fileName.substring(0, fileName.length() - 3) + "jar");
353             //store the archives sha1 and change it so that the engine doesn't think the zip and jar file are the same
354             // and add it is a related dependency.
355             final String archiveMd5 = dependency.getMd5sum();
356             final String archiveSha1 = dependency.getSha1sum();
357             final String archiveSha256 = dependency.getSha256sum();
358             try {
359                 dependency.setMd5sum("");
360                 dependency.setSha1sum("");
361                 dependency.setSha256sum("");
362                 org.apache.commons.io.FileUtils.copyFile(dependency.getActualFile(), tmpLoc);
363                 final List<Dependency> dependencySet = findMoreDependencies(engine, tmpLoc);
364                 if (dependencySet != null && !dependencySet.isEmpty()) {
365                     dependencySet.forEach((d) -> {
366                         //fix the dependency's display name and path
367                         if (d.getActualFile().equals(tmpLoc)) {
368                             d.setFilePath(dependency.getFilePath());
369                             d.setDisplayFileName(dependency.getFileName());
370                         } else {
371                             d.getRelatedDependencies().stream().filter((rel) -> rel.getActualFile().equals(tmpLoc)).forEach((rel) -> {
372                                 rel.setFilePath(dependency.getFilePath());
373                                 rel.setDisplayFileName(dependency.getFileName());
374                             });
375                         }
376                     });
377                 }
378             } catch (IOException ex) {
379                 LOGGER.debug("Unable to perform deep copy on '{}'", dependency.getActualFile().getPath(), ex);
380             } finally {
381                 dependency.setMd5sum(archiveMd5);
382                 dependency.setSha1sum(archiveSha1);
383                 dependency.setSha256sum(archiveSha256);
384             }
385         }
386     }

400      * Retrieves the next temporary directory to extract an archive too.
401      *
402      * @return a directory
403      * @throws AnalysisException thrown if unable to create temporary directory
404      */
405     private File getNextTempDirectory() throws AnalysisException {
406         final File directory = new File(tempFileLocation, String.valueOf(DIRECTORY_COUNT.incrementAndGet()));
407         //getting an exception for some directories not being able to be created; might be because the directory already exists?
408         if (directory.exists()) {
409             return getNextTempDirectory();
410         }
411         if (!directory.mkdirs()) {
412             final String msg = String.format("Unable to create temp directory '%s'.", directory.getAbsolutePath());
413             throw new AnalysisException(msg);
414         }
415         return directory;
416     }

418     /**
419      * Extracts the contents of an archive into the specified directory.
420      *
421      * @param archive an archive file such as a WAR or EAR
422      * @param destination a directory to extract the contents to
423      * @param engine the scanning engine
424      * @throws AnalysisException thrown if the archive is not found
425      */
426     private void extractFiles(File archive, File destination, Engine engine) throws AnalysisException {
427         if (archive != null && destination != null) {
428             String archiveExt = FileUtils.getFileExtension(archive.getName());
429             if (archiveExt == null) {
430                 return;
431             }
432             archiveExt = archiveExt.toLowerCase();
433 
434             final FileInputStream fis;
435             try {
436                 fis = new FileInputStream(archive);
437             } catch (FileNotFoundException ex) {
438                 final String msg = String.format("Error extracting file `%s`: %s", archive.getAbsolutePath(), ex.getMessage());
439                 LOGGER.debug(msg, ex);
440                 throw new AnalysisException(msg);
441             }
442             BufferedInputStream in = null;
443             //ZipArchiveInputStream zin = null;
444             ZipInputStream zin = null;
445             TarArchiveInputStream tin = null;
446             GzipCompressorInputStream gin = null;
447             BZip2CompressorInputStream bzin = null;
448             RpmInputStream rin = null;
449             CpioArchiveInputStream cain = null;
450             try {
451                 if (KNOWN_ZIP_EXT.contains(archiveExt) || ADDITIONAL_ZIP_EXT.contains(archiveExt)) {
452                     in = new BufferedInputStream(fis);
453                     ensureReadableJar(archiveExt, in);
454                     //zin = new ZipArchiveInputStream(in);
455                     zin = new ZipInputStream(in);
456                     extractArchive(zin, destination, engine);
457                 } else if ("tar".equals(archiveExt)) {
458                     in = new BufferedInputStream(fis);
459                     tin = new TarArchiveInputStream(in);
460                     extractArchive(tin, destination, engine);
461                 } else if ("gz".equals(archiveExt) || "tgz".equals(archiveExt)) {
462                     final String uncompressedName = GzipUtils.getUncompressedFilename(archive.getName());
463                     final File f = new File(destination, uncompressedName);
464                     if (engine.accept(f)) {
465                         final String destPath = destination.getCanonicalPath();
466                         if (!f.getCanonicalPath().startsWith(destPath)) {
467                             final String msg = String.format(
468                                     "Archive (%s) contains a file that would be written outside of the destination directory",
469                                     archive.getPath());
470                             throw new AnalysisException(msg);
471                         }
472                         in = new BufferedInputStream(fis);
473                         gin = new GzipCompressorInputStream(in);
474                         decompressFile(gin, f);
475                     }
476                 } else if ("bz2".equals(archiveExt) || "tbz2".equals(archiveExt)) {
477                     final String uncompressedName = BZip2Utils.getUncompressedFilename(archive.getName());
478                     final File f = new File(destination, uncompressedName);
479                     if (engine.accept(f)) {
480                         final String destPath = destination.getCanonicalPath();
481                         if (!f.getCanonicalPath().startsWith(destPath)) {
482                             final String msg = String.format(
483                                     "Archive (%s) contains a file that would be written outside of the destination directory",
484                                     archive.getPath());
485                             throw new AnalysisException(msg);
486                         }
487                         in = new BufferedInputStream(fis);
488                         bzin = new BZip2CompressorInputStream(in);
489                         decompressFile(bzin, f);
490                     }
491                 } else if ("rpm".equals(archiveExt)) {
492                     rin = new RpmInputStream(fis);
493                     //return of getTag is not used - but the call is a
494                     //necassary step in reading from the stream
495                     rin.getPayloadHeader().getTag(RpmTag.NAME);
496                     cain = new CpioArchiveInputStream(rin);
497                     extractArchive(cain, destination, engine);
498                 }
499             } catch (ArchiveExtractionException ex) {
500                 LOGGER.error("Exception extracting archive '{}'.", archive.getName());
501                 LOGGER.debug("", ex);
502                 throw new AnalysisException(ex.getMessage(), ex);
503             } catch (IOException ex) {
504                 LOGGER.error("Exception reading archive '{}'.", archive.getName());
505                 LOGGER.debug("", ex);
506                 throw new AnalysisException(ex.getMessage(), ex);
507             } finally {
508                 //overly verbose and not needed... but keeping it anyway due to
509                 //having issue with file handles being left open
510                 FileUtils.close(fis);
511                 FileUtils.close(in);
512                 FileUtils.close(zin);
513                 FileUtils.close(tin);
514                 FileUtils.close(gin);
515                 FileUtils.close(bzin);
516             }
517         }
518     }

520     /**
521      * Checks if the file being scanned is a JAR or WAR that begins with
522      * '#!/bin' which indicates it is a fully executable jar. If a fully
523      * executable JAR is identified the input stream will be advanced to the
524      * start of the actual JAR file ( skipping the script).
525      *
526      * @see
527      * <a href="http://docs.spring.io/spring-boot/docs/1.3.0.BUILD-SNAPSHOT/reference/htmlsingle/#deployment-install">Installing
528      * Spring Boot Applications</a>
529      * @param archiveExt the file extension
530      * @param in the input stream
531      * @throws IOException thrown if there is an error reading the stream
532      */
533     private void ensureReadableJar(final String archiveExt, BufferedInputStream in) throws IOException {
534         if (("war".equals(archiveExt) || "jar".equals(archiveExt)) && in.markSupported()) {
535             in.mark(7);
536             final byte[] b = new byte[7];
537             final int read = in.read(b);
538             if (read == 7
539                     && b[0] == '#'
540                     && b[1] == '!'
541                     && b[2] == '/'
542                     && b[3] == 'b'
543                     && b[4] == 'i'
544                     && b[5] == 'n'
545                     && b[6] == '/') {
546                 boolean stillLooking = true;
547                 int chr;
548                 int nxtChr;
549                 //CSOFF: InnerAssignment
550                 //CSOFF: NestedIfDepth
551                 while (stillLooking && (chr = in.read()) != -1) {
552                     if (chr == '\n' || chr == '\r') {
553                         in.mark(4);
554                         if ((chr = in.read()) != -1) {
555                             if (chr == 'P' && (chr = in.read()) != -1) {
556                                 if (chr == 'K' && (chr = in.read()) != -1) {
557                                     if ((chr == 3 || chr == 5 || chr == 7) && (nxtChr = in.read()) != -1) {
558                                         if (nxtChr == chr + 1) {
559                                             stillLooking = false;
560                                             in.reset();
561                                         }
562                                     }
563                                 }
564                             }
565                         }
566                     }
567                 }
568                 //CSON: InnerAssignment
569                 //CSON: NestedIfDepth
570             } else {
571                 in.reset();
572             }
573         }
574     }

576     private void extractArchive(ZipInputStream input, File destination, Engine engine) throws ArchiveExtractionException {
577         ZipEntry entry;
578         try {
579             //final String destPath = destination.getCanonicalPath();
580             final Path d = destination.toPath();
581             while ((entry = input.getNextEntry()) != null) {
582                 //final File file = new File(destination, entry.getName());
583                 final Path f = d.resolve(entry.getName()).normalize();
584                 if (!f.startsWith(d)) {
585                     LOGGER.debug("ZipSlip detected\n-Destination: " + d + "\n-Path: " + f);
586                     final String msg = String.format(
587                             "Archive contains a file (%s) that would be extracted outside of the target directory.",
588                             entry.getName());
589                     throw new ArchiveExtractionException(msg);
590                 }
591                 final File file = f.toFile();
592                 if (entry.isDirectory()) {
593                     if (!file.exists() && !file.mkdirs()) {
594                         final String msg = String.format("Unable to create directory '%s'.", file.getAbsolutePath());
595                         throw new AnalysisException(msg);
596                     }
597                 } else if (engine.accept(file)) {
598                     extractAcceptedFile(input, file);
599                 }
600             }
601         } catch (IOException | AnalysisException ex) {
602             throw new ArchiveExtractionException(ex);
603         } finally {
604             FileUtils.close(input);
605         }
606     }

608     /**
609      * Extracts files from an archive.
610      *
611      * @param input the archive to extract files from
612      * @param destination the location to write the files too
613      * @param engine the dependency-check engine
614      * @throws ArchiveExtractionException thrown if there is an exception
615      * extracting files from the archive
616      */
617     private void extractArchive(ArchiveInputStream input, File destination, Engine engine) throws ArchiveExtractionException {
618         ArchiveEntry entry;
619         try {
620             //final String destPath = destination.getCanonicalPath();
621             final Path d = destination.toPath();
622             while ((entry = input.getNextEntry()) != null) {
623                 //final File file = new File(destination, entry.getName());
624                 final Path f = d.resolve(entry.getName()).normalize();
625                 if (!f.startsWith(d)) {
626                     LOGGER.debug("ZipSlip detected\n-Destination: " + d + "\n-Path: " + f);
627                     final String msg = String.format(
628                             "Archive contains a file (%s) that would be extracted outside of the target directory.",
629                             entry.getName());
630                     throw new ArchiveExtractionException(msg);
631                 }
632                 final File file = f.toFile();
633                 if (entry.isDirectory()) {
634                     if (!file.exists() && !file.mkdirs()) {
635                         final String msg = String.format("Unable to create directory '%s'.", file.getAbsolutePath());
636                         throw new AnalysisException(msg);
637                     }
638                 } else if (engine.accept(file)) {
639                     extractAcceptedFile(input, file);
640                 }
641             }
642         } catch (IOException | AnalysisException ex) {
643             throw new ArchiveExtractionException(ex);
644         } finally {
645             FileUtils.close(input);
646         }
647     }

650      * Extracts a file from an archive.
651      *
652      * @param input the archives input stream
653      * @param file the file to extract
654      * @throws AnalysisException thrown if there is an error
655      */
656     private static void extractAcceptedFile(ZipInputStream input, File file) throws AnalysisException {
657         LOGGER.debug("Extracting '{}'", file.getPath());
658         final File parent = file.getParentFile();
659         if (!parent.isDirectory() && !parent.mkdirs()) {
660             final String msg = String.format("Unable to build directory '%s'.", parent.getAbsolutePath());
661             throw new AnalysisException(msg);
662         }
663         try (FileOutputStream fos = new FileOutputStream(file)) {
664             IOUtils.copy(input, fos);
665         } catch (FileNotFoundException ex) {
666             LOGGER.debug("", ex);
667             final String msg = String.format("Unable to find file '%s'.", file.getName());
668             throw new AnalysisException(msg, ex);
669         } catch (IOException ex) {
670             LOGGER.debug("", ex);
671             final String msg = String.format("IO Exception while parsing file '%s'.", file.getName());
672             throw new AnalysisException(msg, ex);
673         }
674     }

721     /**
722      * Attempts to determine if a zip file is actually a JAR file.
723      *
724      * @param dependency the dependency to check
725      * @return true if the dependency appears to be a JAR file; otherwise false
726      */
727     private boolean isZipFileActuallyJarFile(Dependency dependency) {
728         boolean isJar = false;
729         ZipFile zip = null;
730         try {
731             zip = new ZipFile(dependency.getActualFilePath());
732             if (zip.getEntry("META-INF/MANIFEST.MF") != null
733                     || zip.getEntry("META-INF/maven") != null) {
734                 final Enumeration<ZipArchiveEntry> entries = zip.getEntries();
735                 while (entries.hasMoreElements()) {
736                     final ZipArchiveEntry entry = entries.nextElement();
737                     if (!entry.isDirectory()) {
738                         final String name = entry.getName().toLowerCase();
739                         if (name.endsWith(".class")) {
740                             isJar = true;
741                             break;
742                         }
743                     }
744                 }
745             }
746         } catch (IOException ex) {
747             LOGGER.debug("Unable to unzip zip file '{}'", dependency.getFilePath(), ex);
748         } finally {
749             ZipFile.closeQuietly(zip);
750         }
751         return isJar;
752     }

754     /**
755      * Initializes settings used by the scanning functions of the archive
756      * analyzer.
757      */
758     private void initializeSettings() {
759         maxScanDepth = getSettings().getInt("archive.scan.depth", 3);
760         final Set<String> extensions = new HashSet<>(EXTENSIONS);
761         extensions.addAll(KNOWN_ZIP_EXT);
762         final String additionalZipExt = getSettings().getString(Settings.KEYS.ADDITIONAL_ZIP_EXTENSIONS);
763         if (additionalZipExt != null) {
764             final String[] ext = additionalZipExt.split("\\s*,\\s*");
765             Collections.addAll(extensions, ext);
766             Collections.addAll(ADDITIONAL_ZIP_EXT, ext);
767         }
768         fileFilter = FileFilterBuilder.newInstance().addExtensions(extensions).build();
769     }
