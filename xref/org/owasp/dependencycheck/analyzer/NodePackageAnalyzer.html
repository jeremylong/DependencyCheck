110     /**
111      * Performs validation on the configuration to ensure that the correct
112      * analyzers are in place.
113      *
114      * @param engine the dependency-check engine
115      * @throws InitializationException thrown if there is a configuration error
116      */
117     @Override
118     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
119         if (engine.getMode() != Mode.EVIDENCE_COLLECTION) {
120             try {
121                 final Settings settings = engine.getSettings();
122                 final String[] tmp = settings.getArray(Settings.KEYS.ECOSYSTEM_SKIP_CPEANALYZER);
123                 if (tmp != null) {
124                     final List<String> skipEcosystems = Arrays.asList(tmp);
125                     if (skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)
126                             && !settings.getBoolean(Settings.KEYS.ANALYZER_OSSINDEX_ENABLED)) {
127                         if (!settings.getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED)) {
128                             final String msg = "Invalid Configuration: enabling the Node Package Analyzer without "
129                                     + "using the Node Audit Analyzer or OSS Index Analyzer is not supported.";
130                             throw new InitializationException(msg);
131                         } else if (!isNodeAuditEnabled(engine)) {
132                             final String msg = "Missing package.lock or npm-shrinkwrap.lock file: Unable to scan a node "
133                                     + "project without a package-lock.json or npm-shrinkwrap.json.";
134                             throw new InitializationException(msg);
135                         }
136                     } else if (skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)
137                             && !settings.getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED)) {
138                         LOGGER.warn("Using only the OSS Index Analyzer with Node.js can result in many false positives "
139                                 + "- please enable the Node Audit Analyzer.");
140                     }
141                 }
142             } catch (InvalidSettingException ex) {
143                 throw new InitializationException("Unable to read configuration settings", ex);
144             }
145         }
146     }

179     /**
180      * Determines if the Node Audit analyzer is enabled.
181      *
182      * @param engine a reference to the dependency-check engine
183      * @return <code>true</code> if the Node Audit Analyzer is enabled;
184      * otherwise <code>false</code>
185      */
186     private boolean isNodeAuditEnabled(Engine engine) {
187         for (Analyzer a : engine.getAnalyzers()) {
188             if (a instanceof NodeAuditAnalyzer) {
189                 if (a.isEnabled()) {
190                     try {
191                         ((NodeAuditAnalyzer) a).prepareFileTypeAnalyzer(engine);
192                     } catch (InitializationException ex) {
193                         LOGGER.debug("Error initializing the NodeAuditAnalyzer");
194                     }
195                 }
196                 return a.isEnabled();
197             }
198         }
199         return false;
200     }

202     /**
203      * Checks if a package lock file or equivalent exists for the NPM project.
204      *
205      * @param dependencyFile a reference to the `package.json` file
206      * @return <code>true</code> if no lock file is found; otherwise
207      * <code>true</code>
208      */
209     private boolean noLockFileExists(File dependencyFile) {
210         //TODO if we support yarn we need to add the check for the yarn.yml file
211         final File lock = new File(dependencyFile.getParentFile(), "package-lock.json");
212         final File shrinkwrap = new File(dependencyFile.getParentFile(), "npm-shrinkwrap.json");
213         return !(lock.isFile() || shrinkwrap.isFile());
214     }

216     @Override
217     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
218         if (isNodeAuditEnabled(engine)
219                 && !(PACKAGE_LOCK_JSON.equals(dependency.getFileName()) || SHRINKWRAP_JSON.equals(dependency.getFileName()))) {
220             engine.removeDependency(dependency);
221         }
222         final File dependencyFile = dependency.getActualFile();
223         if (!dependencyFile.isFile() || dependencyFile.length() == 0 || !shouldProcess(dependencyFile)) {
224             return;
225         }
226         if (noLockFileExists(dependency.getActualFile())) {
227             LOGGER.warn("No lock file exists - this will result in false negatives; please run `npm install --package-lock`");
228         }
229         final File baseDir = dependencyFile.getParentFile();
230         if (PACKAGE_JSON.equals(dependency.getFileName())) {
231             final File lockfile = new File(baseDir, PACKAGE_LOCK_JSON);
232             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
233             if (shrinkwrap.exists() || lockfile.exists()) {
234                 return;
235             }
236         } else if (PACKAGE_LOCK_JSON.equals(dependency.getFileName())) {
237             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
238             if (shrinkwrap.exists()) {
239                 return;
240             }
241         }
242         final File nodeModules = new File(baseDir, "node_modules");
243         if (!nodeModules.isDirectory()) {
244             LOGGER.warn("Analyzing `{}` - however, the node_modules directory does not exist. "
245                     + "Please run `npm install` prior to running dependency-check", dependencyFile.toString());
246             return;
247         }
248 
249         try (JsonReader jsonReader = Json.createReader(FileUtils.openInputStream(dependencyFile))) {
250             final JsonObject json = jsonReader.readObject();
251             final String parentName = json.getString("name", "");
252             final String parentVersion = json.getString("version", "");
253             if (parentName.isEmpty()) {
254                 return;
255             }
256             dependency.setName(parentName);
257             final String parentPackage;
258             if (!parentVersion.isEmpty()) {
259                 dependency.setVersion(parentVersion);
260                 parentPackage = String.format("%s:%s", parentName, parentVersion);
261             } else {
262                 parentPackage = parentName;
263             }
264             processDependencies(json, baseDir, dependencyFile, parentPackage, engine);
265         } catch (JsonException e) {
266             LOGGER.warn("Failed to parse package.json file.", e);
267         } catch (IOException e) {
268             throw new AnalysisException("Problem occurred while reading dependency file.", e);
269         }
270     }

272     /**
273      * should process the dependency ? Will return true if you need to skip it .
274      * (e.g. dependency can't be read, or if npm audit doesn't handle it)
275      *
276      * @param name the name of the dependency
277      * @param version the version of the dependency
278      * @param optional is the dependency optional ?
279      * @param fileExist is the package.json available for this file ?
280      * @return should you skip this dependency ?
281      */
282     public static boolean shouldSkipDependency(String name, String version, boolean optional, boolean fileExist) {
283         // some package manager can handle alias, yarn for example, but npm doesn't support it
284         if (version.startsWith("npm:")) {
285             //TODO make this an error that gets logged
286             LOGGER.warn("dependency skipped: package.json contain an alias for {} => {} npm audit doesn't "
287                     + "support aliases", name, version.replace("npm:", ""));
288             return true;
289         }
290 
291         if (optional && !fileExist) {
292             LOGGER.warn("dependency skipped: node module {} seems optional and not installed", name);
293             return true;
294         }
295 
296         // this seems to produce crash sometimes, I need to tests
297         // using a local node_module is not supported by npm audit, it crash
298         if (version.startsWith("file:")) {
299             LOGGER.warn("dependency skipped: package.json contain an local node_module for {} seems to be "
300                     + "located {} npm audit doesn't support locally referenced modules",
301                     name, version.replace("file:", ""));
302             return true;
303         }
304         return false;
305     }

307     /**
308      * Checks if the given dependency should be skipped.
309      *  @see NodePackageAnalyzer#shouldSkipDependency(java.lang.String,
310      * java.lang.String, boolean, boolean)
311      * @param name the name of the dependency to test
312      * @param version the version of the dependency to test
313      * @return <code>true</code> if the dependency should be skipped; otherwise
314      * <code>false</code>
315      */
316     public static boolean shouldSkipDependency(String name, String version) {
317         return shouldSkipDependency(name, version, false, true);
318     }

320     /**
321      * Process the dependencies in the lock file by first parsing its
322      * dependencies and then finding the package.json for the module and adding
323      * it as a dependency.
324      *
325      * @param json the data to process
326      * @param baseDir the base directory being scanned
327      * @param rootFile the root package-lock/npm-shrinkwrap being analyzed
328      * @param parentPackage the parent package name of the current node
329      * @param engine a reference to the dependency-check engine
330      * @throws AnalysisException thrown if there is an exception
331      */
332     private void processDependencies(JsonObject json, File baseDir, File rootFile,
333             String parentPackage, Engine engine) throws AnalysisException {
334         if (json.containsKey("dependencies")) {
335             final JsonObject deps = json.getJsonObject("dependencies");
336             for (Map.Entry<String, JsonValue> entry : deps.entrySet()) {
337                 final String name = entry.getKey();
338                 final String version;
339                 boolean optional = false;
340 
341                 final File base = Paths.get(baseDir.getPath(), "node_modules", name).toFile();
342                 final File f = new File(base, PACKAGE_JSON);
343                 JsonObject jo = null;
344 
345                 if (entry.getValue() instanceof JsonObject) {
346                     jo = (JsonObject) entry.getValue();
347                     version = jo.getString("version");
348                     optional = jo.getBoolean("optional", false);
349                 } else {
350                     version = ((JsonString) entry.getValue()).getString();
351                 }
352 
353                 if (shouldSkipDependency(name, version, optional, f.exists())) {
354                     continue;
355                 }
356 
357                 if (null != jo && jo.containsKey("dependencies")) {
358                     final String subPackageName = String.format("%s/%s:%s", parentPackage, name, version);
359                     processDependencies(jo, base, rootFile, subPackageName, engine);
360                 }
361 
362                 final Dependency child;
363                 if (f.exists()) {
364                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
365                     child = new Dependency(f);
366                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
367 
368                     try (JsonReader jr = Json.createReader(FileUtils.openInputStream(f))) {
369                         final JsonObject childJson = jr.readObject();
370                         gatherEvidence(childJson, child);
371 
372                     } catch (JsonException e) {
373                         LOGGER.warn("Failed to parse package.json file from dependency.", e);
374                     } catch (IOException e) {
375                         throw new AnalysisException("Problem occurred while reading dependency file.", e);
376                     }
377                 } else {
378                     LOGGER.warn("Unable to find node module: {}", f.toString());
379                     child = new Dependency(rootFile, true);
380                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
381                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
382                     child.setSha1sum(Checksum.getSHA1Checksum(String.format("%s:%s", name, version)));
383                     child.setSha256sum(Checksum.getSHA256Checksum(String.format("%s:%s", name, version)));
384                     child.setMd5sum(Checksum.getMD5Checksum(String.format("%s:%s", name, version)));
385                     child.addEvidence(EvidenceType.VENDOR, rootFile.getName(), "name", name, Confidence.HIGHEST);
386                     child.addEvidence(EvidenceType.PRODUCT, rootFile.getName(), "name", name, Confidence.HIGHEST);
387                     child.addEvidence(EvidenceType.VERSION, rootFile.getName(), "version", version, Confidence.HIGHEST);
388                     child.setName(name);
389                     child.setVersion(version);
390                     final String packagePath = String.format("%s:%s", name, version);
391                     child.setDisplayFileName(packagePath);
392                     child.setPackagePath(packagePath);
393                     try {
394                         final PackageURL purl = PackageURLBuilder.aPackageURL().withType("npm").withName(name).withVersion(version).build();
395                         final PurlIdentifier id = new PurlIdentifier(purl, Confidence.HIGHEST);
396                         child.addSoftwareIdentifier(id);
397                     } catch (MalformedPackageURLException ex) {
398                         LOGGER.debug("Unable to build package url for `" + packagePath + "`", ex);
399                     }
400                 }
401 
402                 child.addProjectReference(parentPackage);
403                 child.setEcosystem(DEPENDENCY_ECOSYSTEM);
404 
405                 final Dependency existing = findDependency(engine, name, version);
406                 if (existing != null) {
407                     if (existing.isVirtual()) {
408                         DependencyMergingAnalyzer.mergeDependencies(child, existing, null);
409                         engine.removeDependency(existing);
410                         engine.addDependency(child);
411                     } else {
412                         DependencyBundlingAnalyzer.mergeDependencies(existing, child, null);
413                     }
414                 } else {
415                     engine.addDependency(child);
416                 }
417             }
418         }
