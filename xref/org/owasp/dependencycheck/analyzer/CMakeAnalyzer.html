50  /**
51   * <p>
52   * Used to analyze CMake build files, and collect information that can be used
53   * to determine the associated CPE.</p>
54   * <p>
55   * Note: This analyzer catches straightforward invocations of the project
56   * command, plus some other observed patterns of version inclusion in real CMake
57   * projects. Many projects make use of older versions of CMake and/or use custom
58   * "homebrew" ways to insert version information. Hopefully as the newer CMake
59   * call pattern grows in usage, this analyzer allow more CPEs to be
60   * identified.</p>
61   *
62   * @author Dale Visser
63   */
64  @Experimental
65  public class CMakeAnalyzer extends AbstractFileTypeAnalyzer {

78      /**
79       * Used when compiling file scanning regex patterns.
80       */
81      private static final int REGEX_OPTIONS = Pattern.DOTALL | Pattern.CASE_INSENSITIVE | Pattern.MULTILINE;
82      /**
83       * Regex to obtain the project version.
84       */
85      private static final Pattern PROJECT_VERSION = Pattern.compile("^\\s*set\\s*\\(\\s*VERSION\\s*\"([^\"]*)\"\\)",
86              REGEX_OPTIONS);
87      /**
88       * Regex to obtain variables.
89       */
90      private static final Pattern SET_VAR_REGEX = Pattern.compile(
91              "^\\s*set\\s*\\(\\s*([a-zA-Z\\d_\\-]*)\\s+\"?([a-zA-Z\\d_\\-.${}]*)\"?\\s*\\)", REGEX_OPTIONS);
92      /**
93       * Regex to find inlined variables to replace them.
94       */
95      private static final Pattern INL_VAR_REGEX = Pattern.compile("(\\$\\s*\\{([^}]*)\\s*})", REGEX_OPTIONS);
96      /**
97       * Regex to extract the product information.
98       */
99      private static final Pattern PROJECT = Pattern.compile("^ *project *\\([ \\n]*(\\w+)[ \\n]*.*?\\)", REGEX_OPTIONS);
100 
101     /**
102      * Regex to extract product and version information.
103      *
104      * <p>Group 1: Product</p>
105      * <p>Group 2: Version</p>
106      */
107     private static final Pattern SET_VERSION = Pattern
108             .compile("^\\s*set\\s*\\(\\s*(\\w+)_version\\s+\"?([^\")]*)\\s*\"?\\)", REGEX_OPTIONS);

158     /**
159      * Analyzes python packages and adds evidence to the dependency.
160      *
161      * @param dependency the dependency being analyzed
162      * @param engine the engine being used to perform the scan
163      * @throws AnalysisException thrown if there is an unrecoverable error
164      * analyzing the dependency
165      */
166     @Override
167     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
168         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
169         final File file = dependency.getActualFile();
170         final String name = file.getName();
171         final String contents;
172         try {
173             contents = FileUtils.readFileToString(file, Charset.defaultCharset()).trim();
174         } catch (IOException e) {
175             throw new AnalysisException(
176                     "Problem occurred while reading dependency file.", e);
177         }
178         if (StringUtils.isNotBlank(contents)) {
179             final Map<String, String> vars = collectDefinedVariables(contents);
180 
181             String contentsReplacer = contents;
182             Matcher r = INL_VAR_REGEX.matcher(contents);
183             while (r.find()) {
184                 boolean leastOne = false;
185                 if (vars.containsKey(r.group(2))) {
186                     if (!vars.get(r.group(2)).contains(r.group(2))) {
187                         contentsReplacer = contentsReplacer.replace(r.group(1), vars.get(r.group(2)));
188                         r = INL_VAR_REGEX.matcher(contentsReplacer);
189                         leastOne = true;
190                     }
191                 }
192                 while (r.find()) {
193                     if (vars.containsKey(r.group(2))) {
194                         if (!vars.get(r.group(2)).contains(r.group(2))) {
195                             contentsReplacer = contentsReplacer.replace(r.group(1), vars.get(r.group(2)));
196                             r = INL_VAR_REGEX.matcher(contentsReplacer);
197                             leastOne = true;
198                         }
199                     }
200                 }
201                 if (!leastOne) {
202                     break;
203                 }
204                 r = INL_VAR_REGEX.matcher(contentsReplacer);
205             }
206             final String contentsReplaced = contentsReplacer;
207             final Matcher m = PROJECT.matcher(contentsReplaced);
208             int count = 0;
209             while (m.find()) {
210                 count++;
211                 LOGGER.debug(String.format(
212                         "Found project command match with %d groups: %s",
213                         m.groupCount(), m.group(0)));
214                 final String group = m.group(1);
215                 LOGGER.debug("Group 1: {}", group);
216                 dependency.addEvidence(EvidenceType.PRODUCT, name, "Project", group, Confidence.HIGH);
217                 dependency.addEvidence(EvidenceType.VENDOR, name, "Project", group, Confidence.HIGH);
218                 dependency.setName(group);
219                 dependency.setDisplayFileName(group);
220             }
221             if (count > 0) {
222                 dependency.addEvidence(EvidenceType.VENDOR, "CmakeAnalyzer", "hint", "gnu", Confidence.MEDIUM);
223             }
224             LOGGER.debug("Found {} matches.", count);
225             final Matcher mVersion = PROJECT_VERSION.matcher(contentsReplaced);
226             while (mVersion.find()) {
227                 LOGGER.debug(String.format(
228                         "Found set version command match with %d groups: %s",
229                         mVersion.groupCount(), mVersion.group(0)));
230                 final String group = mVersion.group(1);
231                 LOGGER.debug("Group 1: {}", group);
232                 dependency.addEvidence(EvidenceType.VERSION, name, "VERSION", group, Confidence.HIGH);
233                 final DependencyVersion vers = DependencyVersionUtil.parseVersion(group, true);
234                 if (vers != null) {
235                     dependency.setVersion(vers.toString());
236                 }
237             }
238 
239             analyzeSetVersionCommand(dependency, engine, contentsReplaced);
240         }
241     }

243     /**
244      * Collect defined CMake variables
245      *
246      * @param contents the version information
247      *
248      * @return a map referencing identified variables
249      */
250     private Map<String, String> collectDefinedVariables(String contents) {
251         final Map<String, String> vars = new HashMap<>();
252         final Matcher m = SET_VAR_REGEX.matcher(contents);
253         int count = 0;
254         while (m.find()) {
255             count++;
256             LOGGER.debug("Found set variable command match with {} groups: {}",
257                     m.groupCount(), m.group(0));
258             final String name = m.group(1);
259             final String value = m.group(2);
260             LOGGER.debug("Group 1: {}", name);
261             LOGGER.debug("Group 2: {}", value);
262             vars.put(name, value);
263         }
264         LOGGER.debug("Found {} matches.", count);
265         return removeSelfReferences(vars);
266     }

268     /**
269      * Extracts the version information from the contents. If more then one
270      * version is found additional dependencies are added to the dependency
271      * list.
272      *
273      * @param dependency the dependency being analyzed
274      * @param engine the dependency-check engine
275      * @param contents the version information
276      */
277     private void analyzeSetVersionCommand(Dependency dependency, Engine engine, String contents) {
278         Dependency currentDep = dependency;
279 
280         final Matcher m = SET_VERSION.matcher(contents);
281         int count = 0;
282         while (m.find()) {
283             count++;
284             LOGGER.debug("Found project command match with {} groups: {}",
285                     m.groupCount(), m.group(0));
286             String product = m.group(1);
287             final String version = m.group(2);
288             LOGGER.debug("Group 1: {}", product);
289             LOGGER.debug("Group 2: {}", version);
290             final String aliasPrefix = "ALIASOF_";
291             if (product.startsWith(aliasPrefix)) {
292                 product = product.replaceFirst(aliasPrefix, "");
293             }
294             if (product.startsWith("_")) {
295                 product = product.substring(1);
296             }
297             if (count > 1) {
298                 currentDep = new Dependency(dependency.getActualFile(), true);
299                 currentDep.setEcosystem(DEPENDENCY_ECOSYSTEM);
300                 final String filePath = String.format("%s:%s", dependency.getFilePath(), product);
301                 currentDep.setFilePath(filePath);
302 
303                 currentDep.setSha1sum(Checksum.getSHA1Checksum(filePath));
304                 currentDep.setSha256sum(Checksum.getSHA256Checksum(filePath));
305                 currentDep.setMd5sum(Checksum.getMD5Checksum(filePath));
306                 engine.addDependency(currentDep);
307             }
308             final String source = currentDep.getFileName();
309             currentDep.addEvidence(EvidenceType.PRODUCT, source, "Product", product, Confidence.MEDIUM);
310             currentDep.addEvidence(EvidenceType.VENDOR, source, "Vendor", product, Confidence.MEDIUM);
311             currentDep.addEvidence(EvidenceType.VERSION, source, "Version", version, Confidence.MEDIUM);
312             if (product.toLowerCase().endsWith("lib")) {
313                 currentDep = new Dependency(dependency.getActualFile(), true);
314                 currentDep.setEcosystem(DEPENDENCY_ECOSYSTEM);
315                 final String filePath = String.format("%s:%s", dependency.getFilePath(), product);
316                 currentDep.setFilePath(filePath);
317 
318                 currentDep.setSha1sum(Checksum.getSHA1Checksum(filePath));
319                 currentDep.setSha256sum(Checksum.getSHA256Checksum(filePath));
320                 currentDep.setMd5sum(Checksum.getMD5Checksum(filePath));
321                 engine.addDependency(currentDep);
322                 product = "lib" + product.toLowerCase().substring(0, product.length() - 3);
323                 currentDep.addEvidence(EvidenceType.PRODUCT, source, "Product", product, Confidence.MEDIUM);
324                 currentDep.addEvidence(EvidenceType.VENDOR, source, "Vendor", product, Confidence.MEDIUM);
325                 currentDep.addEvidence(EvidenceType.VERSION, source, "Version", version, Confidence.MEDIUM);
326             }
327             if (StringUtils.isBlank(currentDep.getName())) {
328                 currentDep.setName(product);
329                 currentDep.setDisplayFileName(product);
330             }
331             if (StringUtils.isBlank(currentDep.getVersion())) {
332                 final DependencyVersion vers = DependencyVersionUtil.parseVersion(version, true);
333                 if (vers != null) {
334                     currentDep.setVersion(vers.toString());
335                 }
336             }
337         }
338         LOGGER.debug("Found {} matches.", count);
339     }

346     /**
347      * This method prevents to generate an infinite loop when variables are
348      * initialized by other variables and end up forming an unresolvable
349      * chain.
350      *
351      * <p>This method takes the resolved variables map as an input and will return
352      * a new map, without the keys generating an infinite resolution chain.</p>
353      *
354      * @param vars variables initialization detected in the CMake build file
355      *
356      * @return a new map without infinite chain variables
357      */
358     Map<String, String> removeSelfReferences(final Map<String, String> vars) {
359         final Map<String, String> resolvedVars = new HashMap<>();
360 
361         vars.forEach((key, value) -> {
362             if (!isVariableSelfReferencing(vars, key)) {
363                 resolvedVars.put(key, value);
364             }
365         });
366 
367         return resolvedVars;
368     }
369 
370     private boolean isVariableSelfReferencing(Map<String, String> vars, String key) {
371         final List<String> resolutionChain = new ArrayList<>();
372         resolutionChain.add(key);
373 
374         String nextKey = resolutionChain.get(0);
375         do {
376             final Matcher matcher = INL_VAR_REGEX.matcher(vars.get(nextKey));
377             if (!matcher.find()) {
378                 break;
379             }
380             nextKey = matcher.group(2);
381             if (Objects.nonNull(nextKey) && resolutionChain.contains(nextKey)) {
382                 return true;
383             }
384             resolutionChain.add(nextKey);
385         } while (Objects.nonNull(nextKey) && vars.containsKey(nextKey) && !key.equals(nextKey));
386 
387         return resolutionChain.size() != 1 && key.equals(nextKey);
388     }
