95      @Override
96      protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
97          // Here we check if mix_audit actually runs from this location. We do this by running the
98          // `mix_audit --version` command and seeing whether or not it succeeds (if it returns with an exit value of 0)
99          final Process process;
100         try {
101             final List<String> mixAuditArgs = Collections.singletonList("--version");
102             process = launchMixAudit(getSettings().getTempDirectory(), mixAuditArgs);
103         } catch (AnalysisException ae) {
104             setEnabled(false);
105             final String msg = String.format("Exception from mix_audit process: %s. Disabling %s", ae.getCause(), ANALYZER_NAME);
106             throw new InitializationException(msg, ae);
107         } catch (IOException ex) {
108             setEnabled(false);
109             throw new InitializationException("Unable to create temporary file, the Mix Audit Analyzer will be disabled", ex);
110         }
111 
112         final int exitValue;
113         final String mixAuditVersionDetails;
114         try (ProcessReader processReader = new ProcessReader(process)) {
115             processReader.readAll();
116             exitValue = process.exitValue();
117 
118             if (exitValue != 0) {
119                 if (StringUtils.isBlank(processReader.getError())) {
120                     LOGGER.warn("Unexpected exit value from mix_audit process and error stream unexpectedly not ready to capture error details. "
121                             + "Disabling {}. Exit value was: {}", ANALYZER_NAME, exitValue);
122                     setEnabled(false);
123                     throw new InitializationException("mix_audit error stream unexpectedly not ready.");
124                 } else {
125                     setEnabled(false);
126                     LOGGER.warn("Unexpected exit value from mix_audit process. Disabling {}. Exit value was: {}. "
127                             + "error stream output from mix_audit process was: {}", ANALYZER_NAME, exitValue, processReader.getError());
128                     throw new InitializationException("Unexpected exit value from bundle-audit process.");
129                 }
130             } else {
131                 if (StringUtils.isBlank(processReader.getOutput())) {
132                     LOGGER.warn("mix_audit input stream unexpectedly not ready to capture version details. Disabling {}", ANALYZER_NAME);
133                     setEnabled(false);
134                     throw new InitializationException("mix_audit input stream unexpectedly not ready to capture version details.");
135                 } else {
136                     mixAuditVersionDetails = processReader.getOutput();
137                 }
138             }
139         } catch (InterruptedException ex) {
140             setEnabled(false);
141             final String msg = String.format("mix_audit process was interrupted. Disabling %s", ANALYZER_NAME);
142             Thread.currentThread().interrupt();
143             throw new InitializationException(msg);
144         } catch (IOException ex) {
145             setEnabled(false);
146             final String msg = String.format("IOException '%s' during mix_audit process was interrupted. Disabling %s",
147                     ex.getMessage(), ANALYZER_NAME);
148             throw new InitializationException(msg);
149         }
150 
151         if (isEnabled()) {
152             LOGGER.debug("{} is enabled and is using mix_audit with version: {}.", ANALYZER_NAME, mixAuditVersionDetails);
153         }
154     }

187     /**
188      * Launch mix audit.
189      *
190      * @param folder directory that contains the mix.lock file
191      * @param mixAuditArgs the arguments to pass to mix audit
192      * @return a handle to the process
193      * @throws AnalysisException thrown when there is an issue launching mix
194      * audit
195      */
196     private Process launchMixAudit(File folder, List<String> mixAuditArgs) throws AnalysisException {
197         if (!folder.isDirectory()) {
198             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
199         }
200 
201         final List<String> args = new ArrayList<>();
202         final String mixAuditPath = getSettings().getString(Settings.KEYS.ANALYZER_MIX_AUDIT_PATH);
203         File mixAudit = null;
204 
205         if (mixAuditPath != null) {
206             mixAudit = new File(mixAuditPath);
207             if (!mixAudit.isFile()) {
208                 LOGGER.warn("Supplied `mixAudit` path is incorrect: {}", mixAuditPath);
209                 mixAudit = null;
210             }
211         } else {
212             final Path homePath = Paths.get(System.getProperty("user.home"));
213             final Path escriptPath = Paths.get(homePath.toString(), ".mix", "escripts", "mix_audit");
214             mixAudit = escriptPath.toFile();
215         }
216 
217         args.add(mixAudit != null ? mixAudit.getAbsolutePath() : "mix_audit");
218         args.addAll(mixAuditArgs);
219         final ProcessBuilder builder = new ProcessBuilder(args);
220 
221         builder.directory(folder);
222         try {
223             LOGGER.info("Launching: {} from {}", args, folder);
224             return builder.start();
225         } catch (IOException ioe) {
226             throw new AnalysisException("mix_audit initialization failure; this error can be ignored if you are not analyzing Elixir. "
227                     + "Otherwise ensure that mix_audit is installed and the path to mix_audit is correctly specified", ioe);
228         }
229     }

231     /**
232      * Determines if the analyzer can analyze the given file type.
233      *
234      * @param dependency the dependency to determine if it can analyze
235      * @param engine the dependency-check engine
236      * @throws AnalysisException thrown if there is an analysis exception.
237      */
238     @Override
239     protected void analyzeDependency(Dependency dependency, Engine engine)
240             throws AnalysisException {
241         final File parentFile = dependency.getActualFile().getParentFile();
242         final List<String> mixAuditArgs = Arrays.asList("--format", "json");
243 
244         final Process process = launchMixAudit(parentFile, mixAuditArgs);
245         final int exitValue;
246         try (MixAuditProcessor processor = new MixAuditProcessor(dependency, engine);
247                 ProcessReader processReader = new ProcessReader(process, processor)) {
248             processReader.readAll();
249             exitValue = process.exitValue();
250             if (exitValue < 0 || exitValue > 1) {
251                 final String msg = String.format("Unexpected exit code from mix_audit process; exit code: %s", exitValue);
252                 throw new AnalysisException(msg);
253             }
254         } catch (InterruptedException ie) {
255             Thread.currentThread().interrupt();
256             throw new AnalysisException("mix_audit process interrupted", ie);
257         } catch (IOException | CpeValidationException ioe) {
258             LOGGER.warn("mix_audit failure", ioe);
259             throw new AnalysisException("mix_audit failure", ioe);
260         }
261     }
