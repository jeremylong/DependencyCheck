51  /**

116     /**
117      * The prepare method does nothing for this Analyzer.
118      *
119      * @param engine a reference the dependency-check engine
120      * @throws InitializationException thrown if there is an exception
121      */
122     @Override
123     public void prepareAnalyzer(Engine engine) throws InitializationException {
124         try {
125             loadHintRules();
126         } catch (HintParseException ex) {
127             LOGGER.debug("Unable to parse hint file", ex);
128             throw new InitializationException("Unable to parse the hint file", ex);
129         }
130     }

132     /**
133      * The HintAnalyzer uses knowledge about a dependency to add additional
134      * information to help in identification of identifiers or vulnerabilities.
135      *
136      * @param dependency The dependency being analyzed
137      * @param engine The scanning engine
138      * @throws AnalysisException is thrown if there is an exception analyzing
139      * the dependency.
140      */
141     @Override
142     @SuppressWarnings("StringSplitter")
143     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
144         for (HintRule hint : hints) {
145             boolean matchFound = false;
146             for (EvidenceMatcher given : hint.getGivenVendor()) {
147                 if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.VENDOR), given)) {
148                     matchFound = true;
149                     break;
150                 }
151             }
152             if (!matchFound) {
153                 for (EvidenceMatcher given : hint.getGivenProduct()) {
154                     if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.PRODUCT), given)) {
155                         matchFound = true;
156                         break;
157                     }
158                 }
159             }
160             if (!matchFound) {
161                 for (EvidenceMatcher given : hint.getGivenVersion()) {
162                     if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.VERSION), given)) {
163                         matchFound = true;
164                         break;
165                     }
166                 }
167             }
168             if (!matchFound) {
169                 for (PropertyType pt : hint.getFileNames()) {
170                     if (pt.matches(dependency.getFileName())) {
171                         matchFound = true;
172                         break;
173                     }
174                 }
175             }
176             if (matchFound) {
177                 hint.getAddVendor().forEach((e) -> {
178                     dependency.addEvidence(EvidenceType.VENDOR, e);
179                     for (String weighting : e.getValue().split(" ")) {
180                         dependency.addVendorWeighting(weighting);
181                     }
182                 });
183                 hint.getAddProduct().forEach((e) -> {
184                     dependency.addEvidence(EvidenceType.PRODUCT, e);
185                     for (String weighting : e.getValue().split(" ")) {
186                         dependency.addProductWeighting(weighting);
187                     }
188                 });
189                 hint.getAddVersion().forEach((e) -> dependency.addEvidence(EvidenceType.VERSION, e));
190 
191                 hint.getRemoveVendor().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.VENDOR, e));
192                 hint.getRemoveProduct().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.PRODUCT, e));
193                 hint.getRemoveVersion().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.VERSION, e));
194             }
195         }
196 
197         for (Evidence e : dependency.getEvidence(EvidenceType.VENDOR)) {
198             for (VendorDuplicatingHintRule dhr : vendorHints) {
199                 if (dhr.getValue().equalsIgnoreCase(e.getValue())) {
200                     dependency.addEvidence(EvidenceType.VENDOR, new Evidence(e.getSource() + " (hint)",
201                             e.getName(), dhr.getDuplicate(), e.getConfidence(), true));
202                 }
203             }
204         }
205     }
206 
207     /**
208      * Determine if there is matching evidence.
209      *
210      * @param evidences the evidence to test
211      * @param criterion the criteria for a match
212      * @return true if the evidence matches, otherwise false
213      */
214     private boolean hasMatchingEvidence(Set<Evidence> evidences, EvidenceMatcher criterion) {
215         for (Evidence evidence : evidences) {
216             if (criterion.matches(evidence)) {
217                 return true;
218             }
219         }
220         return false;
221     }
222 
223     /**
224      * Removes any matching evidence from the dependency.
225      *
226      * @param dependency the dependency to update
227      * @param type the type of evidence to inspect and possibly remove
228      * @param e the evidence matcher
229      */
230     private void removeMatchingEvidences(Dependency dependency, EvidenceType type, EvidenceMatcher e) {
231         for (Evidence evidence : dependency.getEvidence(type)) {
232             if (e.matches(evidence)) {
233                 dependency.removeEvidence(type, evidence);
234             }
235         }
236     }

238     /**
239      * Loads the hint rules file.
240      *
241      * @throws HintParseException thrown if the XML cannot be parsed.
242      */
243     private void loadHintRules() throws HintParseException {
244         final List<HintRule> localHints;
245         final List<VendorDuplicatingHintRule> localVendorHints;
246         final HintParser parser = new HintParser();
247         File file = null;
248         try (InputStream in = FileUtils.getResourceAsStream(HINT_RULE_FILE_NAME)) {
249             if (in == null) {
250                 throw new HintParseException("Hint rules `" + HINT_RULE_FILE_NAME + "` could not be found");
251             }
252             parser.parseHints(in);
253         } catch (SAXException | IOException ex) {
254             throw new HintParseException("Error parsing hints: " + ex.getMessage(), ex);
255         }
256         localHints = parser.getHintRules();
257         localVendorHints = parser.getVendorDuplicatingHintRules();
258 
259         final String filePath = getSettings().getString(Settings.KEYS.HINTS_FILE);
260         if (filePath != null) {
261             boolean deleteTempFile = false;
262             try {
263                 final Pattern uriRx = Pattern.compile("^(https?|file):.*", Pattern.CASE_INSENSITIVE);
264                 if (uriRx.matcher(filePath).matches()) {
265                     deleteTempFile = true;
266                     file = getSettings().getTempFile("hint", "xml");
267                     final URL url = new URL(filePath);
268                     final Downloader downloader = new Downloader(getSettings());
269                     try {
270                         downloader.fetchFile(url, file, false);
271                     } catch (DownloadFailedException ex) {
272                         try {
273                             Thread.sleep(500);
274                             downloader.fetchFile(url, file, true);
275                         } catch (TooManyRequestsException ex1) {
276                             throw new HintParseException("Unable to download hint file `" + file + "`; received 429 - too many requests", ex1);
277                         } catch (ResourceNotFoundException ex1) {
278                             throw new HintParseException("Unable to download hint file `" + file + "`; received 404 - resource not found", ex1);
279                         } catch (InterruptedException ex1) {
280                             Thread.currentThread().interrupt();
281                             throw new HintParseException("Unable to download hint file `" + file + "`", ex1);
282                         }
283                     } catch (TooManyRequestsException ex) {
284                         throw new HintParseException("Unable to download hint file `" + file + "`; received 429 - too many requests", ex);
285                     } catch (ResourceNotFoundException ex) {
286                         throw new HintParseException("Unable to download hint file `" + file + "`; received 404 - resource not found", ex);
287                     }
288                 } else {
289                     file = new File(filePath);
290                     if (!file.exists()) {
291                         try (InputStream fromClasspath = FileUtils.getResourceAsStream(filePath)) {
292                             deleteTempFile = true;
293                             file = getSettings().getTempFile("hint", "xml");
294                             org.apache.commons.io.FileUtils.copyInputStreamToFile(fromClasspath, file);
295                         } catch (IOException ex) {
296                             throw new HintParseException("Unable to locate hints file in classpath", ex);
297                         }
298                     }
299                 }
300 
301                 if (file == null) {
302                     throw new HintParseException("Unable to locate hints file:" + filePath);
303                 } else {
304                     try {
305                         parser.parseHints(file);
306                         if (parser.getHintRules() != null && !parser.getHintRules().isEmpty()) {
307                             localHints.addAll(parser.getHintRules());
308                         }
309                         if (parser.getVendorDuplicatingHintRules() != null && !parser.getVendorDuplicatingHintRules().isEmpty()) {
310                             localVendorHints.addAll(parser.getVendorDuplicatingHintRules());
311                         }
312                     } catch (HintParseException ex) {
313                         LOGGER.warn("Unable to parse hint rule xml file '{}'", file.getPath());
314                         LOGGER.warn(ex.getMessage());
315                         LOGGER.debug("", ex);
316                         throw ex;
317                     }
318                 }
319             } catch (DownloadFailedException ex) {
320                 throw new HintParseException("Unable to fetch the configured hint file", ex);
321             } catch (MalformedURLException ex) {
322                 throw new HintParseException("Configured hint file has an invalid URL", ex);
323             } catch (IOException ex) {
324                 throw new HintParseException("Unable to create temp file for hints", ex);
325             } finally {
326                 if (deleteTempFile && file != null) {
327                     FileUtils.delete(file);
328                 }
329             }
330         }
331         hints = localHints.toArray(new HintRule[0]);
332         vendorHints = localVendorHints.toArray(new VendorDuplicatingHintRule[0]);
333         LOGGER.debug("{} hint rules were loaded.", hints.length);
334         LOGGER.debug("{} duplicating hint rules were loaded.", vendorHints.length);
335     }
