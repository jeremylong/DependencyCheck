117     /**
118      * The prepare method does nothing for this Analyzer.
119      *
120      * @param engine a reference the dependency-check engine
121      * @throws InitializationException thrown if there is an exception
122      */
123     @Override
124     public void prepareAnalyzer(Engine engine) throws InitializationException {
125         try {
126             loadHintRules();
127         } catch (HintParseException ex) {
128             LOGGER.debug("Unable to parse hint file", ex);
129             throw new InitializationException("Unable to parse the hint file", ex);
130         }
131     }

133     /**
134      * The HintAnalyzer uses knowledge about a dependency to add additional
135      * information to help in identification of identifiers or vulnerabilities.
136      *
137      * @param dependency The dependency being analyzed
138      * @param engine The scanning engine
139      * @throws AnalysisException is thrown if there is an exception analyzing
140      * the dependency.
141      */
142     @Override
143     @SuppressWarnings("StringSplitter")
144     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
145         for (HintRule hint : hints) {
146             boolean matchFound = false;
147             for (EvidenceMatcher given : hint.getGivenVendor()) {
148                 if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.VENDOR), given)) {
149                     matchFound = true;
150                     break;
151                 }
152             }
153             if (!matchFound) {
154                 for (EvidenceMatcher given : hint.getGivenProduct()) {
155                     if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.PRODUCT), given)) {
156                         matchFound = true;
157                         break;
158                     }
159                 }
160             }
161             if (!matchFound) {
162                 for (EvidenceMatcher given : hint.getGivenVersion()) {
163                     if (hasMatchingEvidence(dependency.getEvidence(EvidenceType.VERSION), given)) {
164                         matchFound = true;
165                         break;
166                     }
167                 }
168             }
169             if (!matchFound) {
170                 for (PropertyType pt : hint.getFileNames()) {
171                     if (pt.matches(dependency.getFileName())) {
172                         matchFound = true;
173                         break;
174                     }
175                 }
176             }
177             if (matchFound) {
178                 hint.getAddVendor().forEach((e) -> {
179                     dependency.addEvidence(EvidenceType.VENDOR, e);
180                     for (String weighting : e.getValue().split(" ")) {
181                         dependency.addVendorWeighting(weighting);
182                     }
183                 });
184                 hint.getAddProduct().forEach((e) -> {
185                     dependency.addEvidence(EvidenceType.PRODUCT, e);
186                     for (String weighting : e.getValue().split(" ")) {
187                         dependency.addProductWeighting(weighting);
188                     }
189                 });
190                 hint.getAddVersion().forEach((e) -> dependency.addEvidence(EvidenceType.VERSION, e));
191 
192                 hint.getRemoveVendor().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.VENDOR, e));
193                 hint.getRemoveProduct().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.PRODUCT, e));
194                 hint.getRemoveVersion().forEach((e) -> removeMatchingEvidences(dependency, EvidenceType.VERSION, e));
195             }
196         }
197 
198         final Iterator<Evidence> itr = dependency.getEvidence(EvidenceType.VENDOR).iterator();
199         while (itr.hasNext()) {
200             final Evidence e = itr.next();
201             for (VendorDuplicatingHintRule dhr : vendorHints) {
202                 if (dhr.getValue().equalsIgnoreCase(e.getValue())) {
203                     dependency.addEvidence(EvidenceType.VENDOR, new Evidence(e.getSource() + " (hint)",
204                             e.getName(), dhr.getDuplicate(), e.getConfidence()));
205                 }
206             }
207         }
208     }

210     /**
211      * Determine if there is matching evidence.
212      *
213      * @param evidences the evidence to test
214      * @param criterion the criteria for a match
215      * @return true if the evidence matches, otherwise false
216      */
217     private boolean hasMatchingEvidence(Set<Evidence> evidences, EvidenceMatcher criterion) {
218         for (Evidence evidence : evidences) {
219             if (criterion.matches(evidence)) {
220                 return true;
221             }
222         }
223         return false;
224     }
225 
226     /**
227      * Removes any matching evidence from the dependency.
228      *
229      * @param dependency the dependency to update
230      * @param type the type of evidence to inspect and possibly remove
231      * @param e the evidence matcher
232      */
233     private void removeMatchingEvidences(Dependency dependency, EvidenceType type, EvidenceMatcher e) {
234         for (Evidence evidence : dependency.getEvidence(type)) {
235             if (e.matches(evidence)) {
236                 dependency.removeEvidence(type, evidence);
237             }
238         }
239     }

241     /**
242      * Loads the hint rules file.
243      *
244      * @throws HintParseException thrown if the XML cannot be parsed.
245      */
246     private void loadHintRules() throws HintParseException {
247         final List<HintRule> localHints;
248         final List<VendorDuplicatingHintRule> localVendorHints;
249         final HintParser parser = new HintParser();
250         File file = null;
251         try (InputStream in = FileUtils.getResourceAsStream(HINT_RULE_FILE_NAME)) {
252             if (in == null) {
253                 throw new HintParseException("Hint rules `" + HINT_RULE_FILE_NAME + "` could not be found");
254             }
255             parser.parseHints(in);
256         } catch (SAXException | IOException ex) {
257             throw new HintParseException("Error parsing hints: " + ex.getMessage(), ex);
258         }
259         localHints = parser.getHintRules();
260         localVendorHints = parser.getVendorDuplicatingHintRules();
261 
262         final String filePath = getSettings().getString(Settings.KEYS.HINTS_FILE);
263         if (filePath != null) {
264             boolean deleteTempFile = false;
265             try {
266                 final Pattern uriRx = Pattern.compile("^(https?|file):.*", Pattern.CASE_INSENSITIVE);
267                 if (uriRx.matcher(filePath).matches()) {
268                     deleteTempFile = true;
269                     file = getSettings().getTempFile("hint", "xml");
270                     final URL url = new URL(filePath);
271                     final Downloader downloader = new Downloader(getSettings());
272                     try {
273                         downloader.fetchFile(url, file, false);
274                     } catch (DownloadFailedException ex) {
275                         try {
276                             Thread.sleep(500);
277                             downloader.fetchFile(url, file, true);
278                         } catch (TooManyRequestsException ex1) {
279                             throw new HintParseException("Unable to download hint file `" + file + "`; received 429 - too many requests", ex1);
280                         } catch (ResourceNotFoundException ex1) {
281                             throw new HintParseException("Unable to download hint file `" + file + "`; received 404 - resource not found", ex1);
282                         } catch (InterruptedException ex1) {
283                             Thread.currentThread().interrupt();
284                             throw new HintParseException("Unable to download hint file `" + file + "`", ex1);
285                         }
286                     } catch (TooManyRequestsException ex) {
287                         throw new HintParseException("Unable to download hint file `" + file + "`; received 429 - too many requests", ex);
288                     } catch (ResourceNotFoundException ex) {
289                         throw new HintParseException("Unable to download hint file `" + file + "`; received 404 - resource not found", ex);
290                     }
291                 } else {
292                     file = new File(filePath);
293                     if (!file.exists()) {
294                         try (InputStream fromClasspath = FileUtils.getResourceAsStream(filePath)) {
295                             if (fromClasspath != null) {
296                                 deleteTempFile = true;
297                                 file = getSettings().getTempFile("hint", "xml");
298                                 try {
299                                     org.apache.commons.io.FileUtils.copyInputStreamToFile(fromClasspath, file);
300                                 } catch (IOException ex) {
301                                     throw new HintParseException("Unable to locate hints file in classpath", ex);
302                                 }
303                             }
304                         }
305                     }
306                 }
307 
308                 if (file == null) {
309                     throw new HintParseException("Unable to locate hints file:" + filePath);
310                 } else {
311                     try {
312                         parser.parseHints(file);
313                         if (parser.getHintRules() != null && !parser.getHintRules().isEmpty()) {
314                             localHints.addAll(parser.getHintRules());
315                         }
316                         if (parser.getVendorDuplicatingHintRules() != null && !parser.getVendorDuplicatingHintRules().isEmpty()) {
317                             localVendorHints.addAll(parser.getVendorDuplicatingHintRules());
318                         }
319                     } catch (HintParseException ex) {
320                         LOGGER.warn("Unable to parse hint rule xml file '{}'", file.getPath());
321                         LOGGER.warn(ex.getMessage());
322                         LOGGER.debug("", ex);
323                         throw ex;
324                     }
325                 }
326             } catch (DownloadFailedException ex) {
327                 throw new HintParseException("Unable to fetch the configured hint file", ex);
328             } catch (MalformedURLException ex) {
329                 throw new HintParseException("Configured hint file has an invalid URL", ex);
330             } catch (IOException ex) {
331                 throw new HintParseException("Unable to create temp file for hints", ex);
332             } finally {
333                 if (deleteTempFile && file != null) {
334                     FileUtils.delete(file);
335                 }
336             }
337         }
338         hints = localHints.toArray(new HintRule[0]);
339         vendorHints = localVendorHints.toArray(new VendorDuplicatingHintRule[0]);
340         LOGGER.debug("{} hint rules were loaded.", hints.length);
341         LOGGER.debug("{} duplicating hint rules were loaded.", vendorHints.length);
342     }
