127     /**
128      * Analyzes the dependencies and removes bad/incorrect CPE associations
129      * based on various heuristics.
130      *
131      * @param dependency the dependency to analyze.
132      * @param engine the engine that is scanning the dependencies
133      * @throws AnalysisException is thrown if there is an error reading the JAR
134      * file.
135      */
136     @Override
137     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
138         removeJreEntries(dependency);
139         removeBadMatches(dependency);
140         removeBadSpringMatches(dependency);
141         removeWrongVersionMatches(dependency);
142         removeSpuriousCPE(dependency);
143         removeDuplicativeEntriesFromJar(dependency, engine);
144         addFalseNegativeCPEs(dependency);
145     }

147     /**
148      * Removes inaccurate matches on springframework CPEs.
149      *
150      * @param dependency the dependency to test for and remove known inaccurate
151      * CPE matches
152      */
153     private void removeBadSpringMatches(Dependency dependency) {
154         String mustContain = null;
155         for (Identifier i : dependency.getSoftwareIdentifiers()) {
156             if (i.getValue() != null && i.getValue().startsWith("org.springframework.")) {
157                 final int endPoint = i.getValue().indexOf(':', 19);
158                 if (endPoint >= 0) {
159                     mustContain = i.getValue().substring(19, endPoint).toLowerCase();
160                     break;
161                 }
162             }
163         }
164         if (mustContain != null) {
165             final Set<Identifier> removalSet = new HashSet<>();
166             for (Identifier i : dependency.getVulnerableSoftwareIdentifiers()) {
167                 if (i.getValue() != null
168                         && i.getValue().startsWith("cpe:/a:springsource:")
169                         && !i.getValue().toLowerCase().contains(mustContain)) {
170                     removalSet.add(i);
171                 }
172             }
173             removalSet.forEach(dependency::removeVulnerableSoftwareIdentifier);
174         }
175     }

177     /**
178      * <p>
179      * Intended to remove spurious CPE entries. By spurious we mean duplicate,
180      * less specific CPE entries.</p>
181      * <p>
182      * Example:</p>
183      * <code>
184      * cpe:/a:some-vendor:some-product
185      * cpe:/a:some-vendor:some-product:1.5
186      * cpe:/a:some-vendor:some-product:1.5.2
187      * </code>
188      * <p>
189      * Should be trimmed to:</p>
190      * <code>
191      * cpe:/a:some-vendor:some-product:1.5.2
192      * </code>
193      *
194      * @param dependency the dependency being analyzed
195      */
196     //CSOFF: NestedIfDepth
197     @SuppressWarnings("null")
198     @SuppressFBWarnings(justification = "null checks are working correctly to prevent NPE", value = {"NP_NULL_ON_SOME_PATH_MIGHT_BE_INFEASIBLE"})
199     private void removeSpuriousCPE(Dependency dependency) {
200         final List<Identifier> ids = new ArrayList<>(dependency.getVulnerableSoftwareIdentifiers());
201         Collections.sort(ids);
202         final ListIterator<Identifier> mainItr = ids.listIterator();
203         while (mainItr.hasNext()) {
204             final Identifier temp = mainItr.next();
205             if (temp instanceof CpeIdentifier) {
206                 final CpeIdentifier currentId = (CpeIdentifier) temp;
207                 final Cpe currentCpe = currentId.getCpe();
208                 final ListIterator<Identifier> subItr = ids.listIterator(mainItr.nextIndex());
209                 while (subItr.hasNext()) {
210                     final Identifier nextId = subItr.next();
211                     if (nextId instanceof CpeIdentifier) {
212                         final CpeIdentifier nextCpeId = (CpeIdentifier) nextId;
213                         final Cpe nextCpe = nextCpeId.getCpe();
214                         //TODO fix the version problem below
215                         if (currentCpe.getVendor().equals(nextCpe.getVendor())) {
216                             if (currentCpe.getProduct().equals(nextCpe.getProduct())) {
217                                 // see if one is contained in the other.. remove the contained one from dependency.getIdentifier
218                                 final String currentVersion = currentCpe.getVersion();
219                                 final String nextVersion = nextCpe.getVersion();
220                                 if (currentVersion == null && nextVersion == null) {
221                                     //how did we get here?
222                                     LOGGER.debug("currentVersion and nextVersion are both null?");
223                                 } else if (currentVersion == null && nextVersion != null) {
224                                     dependency.removeVulnerableSoftwareIdentifier(currentId);
225                                 } else if (nextVersion == null && currentVersion != null) {
226                                     dependency.removeVulnerableSoftwareIdentifier(nextId);
227                                 } else if (currentVersion.length() < nextVersion.length()) {
228                                     if (nextVersion.startsWith(currentVersion) || "-".equals(currentVersion)) {
229                                         dependency.removeVulnerableSoftwareIdentifier(currentId);
230                                     }
231                                 } else if (currentVersion.startsWith(nextVersion) || "-".equals(nextVersion)) {
232                                     dependency.removeVulnerableSoftwareIdentifier(nextId);
233                                 }
234                             }
235                         }
236                     }
237                 }
238             }
239         }
240     }

243     /**
244      * Removes any CPE entries for the JDK/JRE unless the filename ends with
245      * rt.jar
246      *
247      * @param dependency the dependency to remove JRE CPEs from
248      */
249     private void removeJreEntries(Dependency dependency) {
250         final Set<Identifier> removalSet = new HashSet<>();
251         dependency.getVulnerableSoftwareIdentifiers().forEach(i -> {
252             final Matcher coreCPE = CORE_JAVA.matcher(i.getValue());
253             final Matcher coreFiles = CORE_FILES.matcher(dependency.getFileName());
254             final Matcher coreJsfCPE = CORE_JAVA_JSF.matcher(i.getValue());
255             final Matcher coreJsfFiles = CORE_JSF_FILES.matcher(dependency.getFileName());
256             if ((coreCPE.matches() && !coreFiles.matches())
257                     || (coreJsfCPE.matches() && !coreJsfFiles.matches())) {
258                 removalSet.add(i);
259             }
260 
261         });
262         removalSet.forEach((i) -> dependency.removeVulnerableSoftwareIdentifier(i));
263     }

372     /**
373      * Removes CPE matches for the wrong version of a dependency. Currently,
374      * this only covers Axis 1 & 2.
375      *
376      * @param dependency the dependency to analyze
377      */
378     private void removeWrongVersionMatches(Dependency dependency) {
379         final Set<Identifier> identifiersToRemove = new HashSet<>();
380         final String fileName = dependency.getFileName();
381         if (fileName != null && fileName.contains("axis2")) {
382             dependency.getVulnerableSoftwareIdentifiers().stream()
383                     .filter((i) -> (i instanceof CpeIdentifier))
384                     .map(i -> (CpeIdentifier) i)
385                     .forEach((i) -> {
386                         final Cpe cpe = i.getCpe();
387                         if ("apache".equals(cpe.getVendor()) && "axis".equals(cpe.getProduct())) {
388                             identifiersToRemove.add(i);
389                         }
390                     });
391         } else if (fileName != null && fileName.contains("axis")) {
392             dependency.getVulnerableSoftwareIdentifiers().stream()
393                     .filter((i) -> (i instanceof CpeIdentifier))
394                     .map(i -> (CpeIdentifier) i)
395                     .forEach((i) -> {
396                         final Cpe cpe = i.getCpe();
397                         if ("apache".equals(cpe.getVendor()) && "axis2".equals(cpe.getProduct())) {
398                             identifiersToRemove.add(i);
399                         }
400                     });
401         }
402         identifiersToRemove.forEach(dependency::removeVulnerableSoftwareIdentifier);
403     }

405     /**
406      * There are some known CPE entries, specifically regarding sun and oracle
407      * products due to the acquisition and changes in product names, that based
408      * on given evidence we can add the related CPE entries to ensure a complete
409      * list of CVE entries.
410      *
411      * @param dependency the dependency being analyzed
412      */
413     @SuppressWarnings("UnnecessaryParentheses")
414     private void addFalseNegativeCPEs(Dependency dependency) {
415         final CpeBuilder builder = new CpeBuilder();
416         //TODO move this to the hint analyzer
417         dependency.getVulnerableSoftwareIdentifiers().stream()
418                 .filter((i) -> (i instanceof CpeIdentifier))
419                 .map(i -> (CpeIdentifier) i)
420                 .forEach((i) -> {
421                     final Cpe cpe = i.getCpe();
422                     if ((("oracle".equals(cpe.getVendor())
423                             && ("opensso".equals(cpe.getProduct()) || "opensso_enterprise".equals(cpe.getProduct()))))
424                             || ("sun".equals(cpe.getVendor())
425                             && ("opensso".equals(cpe.getProduct()) || "opensso_enterprise".equals(cpe.getProduct())))) {
426 
427                         try {
428                             final Cpe newCpe1 = builder.part(Part.APPLICATION).vendor("sun")
429                                     .product("opensso_enterprise").version(cpe.getVersion()).build();
430                             final Cpe newCpe2 = builder.part(Part.APPLICATION).vendor("oracle")
431                                     .product("opensso_enterprise").version(cpe.getVersion()).build();
432                             final Cpe newCpe3 = builder.part(Part.APPLICATION).vendor("sun")
433                                     .product("opensso").version(cpe.getVersion()).build();
434                             final Cpe newCpe4 = builder.part(Part.APPLICATION).vendor("oracle")
435                                     .product("opensso").version(cpe.getVersion()).build();
436                             final CpeIdentifier newCpeId1 = new CpeIdentifier(newCpe1, i.getConfidence());
437                             final CpeIdentifier newCpeId2 = new CpeIdentifier(newCpe2, i.getConfidence());
438                             final CpeIdentifier newCpeId3 = new CpeIdentifier(newCpe3, i.getConfidence());
439                             final CpeIdentifier newCpeId4 = new CpeIdentifier(newCpe4, i.getConfidence());
440                             dependency.addVulnerableSoftwareIdentifier(newCpeId1);
441                             dependency.addVulnerableSoftwareIdentifier(newCpeId2);
442                             dependency.addVulnerableSoftwareIdentifier(newCpeId3);
443                             dependency.addVulnerableSoftwareIdentifier(newCpeId4);
444 
445                         } catch (CpeValidationException ex) {
446                             LOGGER.warn("Unable to add oracle and sun CPEs", ex);
447                         }
448                     }
449                     if ("apache".equals(cpe.getVendor()) && "santuario_xml_security_for_java".equals(cpe.getProduct())) {
450                         try {
451                             final Cpe newCpe1 = builder.part(Part.APPLICATION).vendor("apache")
452                                     .product("xml_security_for_java").version(cpe.getVersion()).build();
453                             final CpeIdentifier newCpeId1 = new CpeIdentifier(newCpe1, i.getConfidence());
454                             dependency.addVulnerableSoftwareIdentifier(newCpeId1);
455                         } catch (CpeValidationException ex) {
456                             LOGGER.warn("Unable to add apache xml_security_for_java CPE", ex);
457                         }
458                     }
459                 });
460     }

499     /**
500      * Retrieves a given dependency, based on a given path, from a list of
501      * dependencies.
502      *
503      * @param dependencyPath the path of the dependency to return
504      * @param dependencies the array of dependencies to search
505      * @return the dependency object for the given path, otherwise null
506      */
507     private Dependency findDependency(String dependencyPath, Dependency[] dependencies) {
508         for (Dependency d : dependencies) {
509             if (d.getFilePath().equalsIgnoreCase(dependencyPath)) {
510                 return d;
511             }
512         }
513         return null;
514     }
