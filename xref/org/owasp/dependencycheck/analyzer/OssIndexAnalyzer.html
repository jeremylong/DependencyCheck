61  /**
62   * Enrich dependency information from Sonatype OSS index.
63   *
64   * @author Jason Dillon
65   * @since 5.0.0
66   */
67  public class OssIndexAnalyzer extends AbstractAnalyzer {
68  
69      /**
70       * A reference to the logger.
71       */
72      private static final Logger LOG = LoggerFactory.getLogger(OssIndexAnalyzer.class);
73  
74      /**
75       * A pattern to match CVE identifiers.
76       */
77      private static final Pattern CVE_PATTERN = Pattern.compile("\\bCVE-\\d{4}-\\d{4,10}\\b");
78  
79      /**
80       * The reference type.
81       */
82      public static final String REFERENCE_TYPE = "OSSINDEX";
83  
84      /**
85       * Fetched reports.
86       */
87      private static Map<PackageUrl, ComponentReport> reports;
88  
89      /**
90       * Flag to indicate if fetching reports failed.
91       */
92      private static boolean failed = false;
93  
94      /**
95       * Lock to protect fetching state.
96       */
97      private static final Object FETCH_MUTIX = new Object();
98  
99      @Override
100     public String getName() {
101         return "Sonatype OSS Index Analyzer";
102     }
103 
104     @Override
105     public AnalysisPhase getAnalysisPhase() {
106         return AnalysisPhase.FINDING_ANALYSIS_PHASE2;
107     }

109     @Override
110     protected String getAnalyzerEnabledSettingKey() {
111         return Settings.KEYS.ANALYZER_OSSINDEX_ENABLED;
112     }

114     /**
115      * Run without parallel support.
116      *
117      * @return false
118      */
119     @Override
120     public boolean supportsParallelProcessing() {
121         return true;
122     }

124     @Override
125     protected void closeAnalyzer() throws Exception {
126         synchronized (FETCH_MUTIX) {
127             reports = null;
128             failed = false;
129         }
130     }
131 
132     @Override
133     protected void analyzeDependency(final Dependency dependency, final Engine engine) throws AnalysisException {
134         // batch request component-reports for all dependencies
135         synchronized (FETCH_MUTIX) {
136             if (!failed && reports == null) {
137                 try {
138                     reports = requestReports(engine.getDependencies());
139                 } catch (TransportException ex) {
140                     failed = true;
141                     if (ex.getMessage() != null && ex.getMessage().endsWith("401")) {
142                         throw new AnalysisException("Invalid credentails provided for OSS Index", ex);
143                     }
144                     LOG.debug("Error requesting component reports", ex);
145                     throw new AnalysisException("Failed to request component-reports", ex);
146                 } catch (Exception e) {
147                     LOG.debug("Error requesting component reports", e);
148                     failed = true;
149                     throw new AnalysisException("Failed to request component-reports", e);
150                 }
151             }
152         }
153 
154         // skip enrichment if we failed to fetch reports
155         if (!failed) {
156             enrich(dependency);
157         }
158     }
159 
160     /**
161      * Helper to complain if unable to parse Package-URL.
162      *
163      * @param value the url to parse
164      * @return the package url
165      */
166     @Nullable
167     private PackageUrl parsePackageUrl(final String value) {
168         try {
169             return PackageUrl.parse(value);
170         } catch (InvalidException e) {
171             LOG.debug("Invalid Package-URL: {}", value, e);
172             return null;
173         }
174     }
175 
176     /**
177      * Batch request component-reports for all dependencies.
178      *
179      * @param dependencies the collection of dependencies
180      * @return the map of dependency to OSS Index's component-report
181      * @throws Exception thrown if there is an exception requesting the report
182      */
183     private Map<PackageUrl, ComponentReport> requestReports(final Dependency[] dependencies) throws Exception {
184         LOG.debug("Requesting component-reports for {} dependencies", dependencies.length);
185         // create requests for each dependency which has a PURL identifier
186         final List<PackageUrl> packages = new ArrayList<>();
187         Arrays.stream(dependencies).forEach(dependency -> {
188             dependency.getSoftwareIdentifiers().stream()
189                     .filter(id -> id instanceof PurlIdentifier)
190                     .map(id -> parsePackageUrl(id.getValue()))
191                     .filter(id -> id != null && StringUtils.isNotBlank(id.getVersion()))
192                     .forEach(id -> packages.add(id));
193         });
194         // only attempt if we have been able to collect some packages
195         if (!packages.isEmpty()) {
196             try (OssindexClient client = OssindexClientFactory.create(getSettings())) {
197                 LOG.debug("OSS Index Analyzer submitting: " + packages.toString());
198                 return client.requestComponentReports(packages);
199             }
200         }
201         LOG.warn("Unable to determine Package-URL identifiers for {} dependencies", dependencies.length);
202         return Collections.emptyMap();
203     }

205     /**
206      * Attempt to enrich given dependency with vulnerability details from OSS
207      * Index component-report.
208      *
209      * @param dependency the dependency to enrich
210      */
211     private void enrich(final Dependency dependency) {
212         LOG.debug("Enrich dependency: {}", dependency);
213 
214         for (Identifier id : dependency.getSoftwareIdentifiers()) {
215             if (id instanceof PurlIdentifier) {
216                 LOG.debug("  Package: {} -> {}", id, id.getConfidence());
217 
218                 final PackageUrl purl = parsePackageUrl(id.getValue());
219                 if (purl != null && StringUtils.isNotBlank(purl.getVersion())) {
220                     try {
221                         final ComponentReport report = reports.get(purl);
222                         if (report == null) {
223                             LOG.debug("Missing component-report for: " + purl);
224                             continue;
225                         }
226 
227                         // expose the URL to the package details for report generation
228                         id.setUrl(report.getReference().toString());
229 
230                         report.getVulnerabilities().stream()
231                                 .map((vuln) -> transform(report, vuln))
232                                 .forEachOrdered((v) -> {
233                                     final Vulnerability existing = dependency.getVulnerabilities().stream()
234                                             .filter(e -> e.getName().equals(v.getName())).findFirst()
235                                             .orElse(null);
236                                     if (existing != null) {
237                                         //TODO - can we enhance anything other than the references?
238                                         existing.getReferences().addAll(v.getReferences());
239                                     } else {
240                                         dependency.addVulnerability(v);
241                                     }
242                                 });
243                     } catch (Exception e) {
244                         LOG.warn("Failed to fetch component-report for: {}", purl, e);
245                     }
246                 }
247             }
248         }
249     }

251     /**
252      * Transform OSS Index component-report to ODC vulnerability.
253      *
254      * @param report the component report
255      * @param source the vulnerability from the report to transform
256      * @return the transformed vulnerability
257      */
258     private Vulnerability transform(final ComponentReport report, final ComponentReportVulnerability source) {
259         final Vulnerability result = new Vulnerability();
260         result.setSource(Vulnerability.Source.OSSINDEX);
261 
262         if (source.getCve() != null) {
263             result.setName(source.getCve());
264         } else {
265             String cve = null;
266             if (source.getTitle() != null) {
267                 final Matcher matcher = CVE_PATTERN.matcher(source.getTitle());
268                 if (matcher.find()) {
269                     cve = matcher.group();
270                 } else {
271                     cve = source.getTitle();
272                 }
273             }
274             if (cve == null && source.getReference() != null) {
275                 final Matcher matcher = CVE_PATTERN.matcher(source.getReference().toString());
276                 if (matcher.find()) {
277                     cve = matcher.group();
278                 }
279             }
280             result.setName(cve != null ? cve : source.getId());
281         }
282         result.setDescription(source.getDescription());
283         result.addCwe(source.getCwe());
284 
285         final float cvssScore = source.getCvssScore() != null ? source.getCvssScore() : -1;
286 
287         if (source.getCvssVector() != null) {
288             // convert cvss details
289             final CvssVector cvssVector = CvssVectorFactory.create(source.getCvssVector());
290 
291             final Map<String, String> metrics = cvssVector.getMetrics();
292             if (cvssVector instanceof Cvss2Vector) {
293                 result.setCvssV2(new CvssV2(
294                         cvssScore,
295                         metrics.get(Cvss2Vector.ACCESS_VECTOR),
296                         metrics.get(Cvss2Vector.ACCESS_COMPLEXITY),
297                         metrics.get(Cvss2Vector.AUTHENTICATION),
298                         metrics.get(Cvss2Vector.CONFIDENTIALITY_IMPACT),
299                         metrics.get(Cvss2Vector.INTEGRITY_IMPACT),
300                         metrics.get(Cvss2Vector.AVAILABILITY_IMPACT),
301                         Cvss2Severity.of(cvssScore).name()
302                 ));
303             } else if (cvssVector instanceof Cvss3Vector) {
304                 result.setCvssV3(new CvssV3(
305                         metrics.get(Cvss3Vector.ATTACK_VECTOR),
306                         metrics.get(Cvss3Vector.ATTACK_COMPLEXITY),
307                         metrics.get(Cvss3Vector.PRIVILEGES_REQUIRED),
308                         metrics.get(Cvss3Vector.USER_INTERACTION),
309                         metrics.get(Cvss3Vector.SCOPE),
310                         metrics.get(Cvss3Vector.CONFIDENTIALITY_IMPACT),
311                         metrics.get(Cvss3Vector.INTEGRITY_IMPACT),
312                         metrics.get(Cvss3Vector.AVAILABILITY_IMPACT),
313                         cvssScore,
314                         Cvss3Severity.of(cvssScore).name()
315                 ));
316             } else {
317                 LOG.warn("Unsupported CVSS vector: {}", cvssVector);
318                 result.setUnscoredSeverity(Float.toString(cvssScore));
319             }
320         } else {
321             LOG.debug("OSS has no vector for {}", result.getName());
322             result.setUnscoredSeverity(Float.toString(cvssScore));
323         }
324         // generate a reference to the vulnerability details on OSS Index
325         result.addReference(REFERENCE_TYPE, source.getTitle(), source.getReference().toString());
326 
327         // attach vulnerable software details as best we can
328         final PackageUrl purl = report.getCoordinates();
329         try {
330             final VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder()
331                     .part(Part.APPLICATION)
332                     .vendor(purl.getNamespaceAsString())
333                     .product(purl.getName())
334                     .version(purl.getVersion());
335 
336             // TODO: consider if we want/need to extract version-ranges to apply to vulnerable-software?
337             final VulnerableSoftware software = builder.build();
338             result.addVulnerableSoftware(software);
339             result.setMatchedVulnerableSoftware(software);
340         } catch (CpeValidationException e) {
341             LOG.warn("Unable to construct vulnerable-software for: {}", purl, e);
342         }
343 
344         return result;
345     }
