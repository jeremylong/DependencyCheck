149     /**
150      * Launch bundle-audit.
151      *
152      * @param folder directory that contains bundle audit
153      * @param bundleAuditArgs the arguments to pass to bundle audit
154      * @return a handle to the process
155      * @throws AnalysisException thrown when there is an issue launching bundle
156      * audit
157      */
158     private Process launchBundleAudit(File folder, List<String> bundleAuditArgs) throws AnalysisException {
159         if (!folder.isDirectory()) {
160             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
161         }
162         final List<String> args = new ArrayList<>();
163         final String bundleAuditPath = getSettings().getString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_PATH);
164         File bundleAudit = null;
165         if (bundleAuditPath != null) {
166             bundleAudit = new File(bundleAuditPath);
167             if (!bundleAudit.isFile()) {
168                 LOGGER.warn("Supplied `bundleAudit` path is incorrect: {}", bundleAuditPath);
169                 bundleAudit = null;
170             }
171         }
172         args.add(bundleAudit != null ? bundleAudit.getAbsolutePath() : "bundle-audit");
173         args.addAll(bundleAuditArgs);
174         final ProcessBuilder builder = new ProcessBuilder(args);
175 
176         final String bundleAuditWorkingDirectoryPath = getSettings().getString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_WORKING_DIRECTORY);
177         File bundleAuditWorkingDirectory = null;
178         if (bundleAuditWorkingDirectoryPath != null) {
179             bundleAuditWorkingDirectory = new File(bundleAuditWorkingDirectoryPath);
180             if (!bundleAuditWorkingDirectory.isDirectory()) {
181                 LOGGER.warn("Supplied `bundleAuditWorkingDirectory` path is incorrect: {}",
182                         bundleAuditWorkingDirectoryPath);
183                 bundleAuditWorkingDirectory = null;
184             }
185         }
186         final File launchBundleAuditFromDirectory = bundleAuditWorkingDirectory != null ? bundleAuditWorkingDirectory : folder;
187         builder.directory(launchBundleAuditFromDirectory);
188         try {
189             LOGGER.info("Launching: {} from {}", args, launchBundleAuditFromDirectory);
190             return builder.start();
191         } catch (IOException ioe) {
192             throw new AnalysisException("bundle-audit initialization failure; this error "
193                     + "can be ignored if you are not analyzing Ruby. Otherwise ensure that "
194                     + "bundle-audit is installed and the path to bundle audit is correctly "
195                     + "specified", ioe);
196         }
197     }

199     /**
200      * Initialize the analyzer.
201      *
202      * @param engine a reference to the dependency-checkException engine
203      * @throws InitializationException if anything goes wrong
204      */
205     @Override
206     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
207         if (engine != null) {
208             this.cvedb = engine.getDatabase();
209         }
210         String bundleAuditVersionDetails = null;
211         try {
212             final List<String> bundleAuditArgs = Arrays.asList("version");
213             final Process process = launchBundleAudit(getSettings().getTempDirectory(), bundleAuditArgs);
214             try (ProcessReader processReader = new ProcessReader(process)) {
215                 processReader.readAll();
216                 final String error = processReader.getError();
217                 if (error != null) {
218                     LOGGER.warn("Warnings from bundle-audit {}", error);
219                 }
220                 bundleAuditVersionDetails = processReader.getOutput();
221                 final int exitValue = process.exitValue();
222                 if (exitValue != 0) {
223                     setEnabled(false);
224                     final String msg = String.format("bundle-audit execution failed - "
225                             + "exit code: %d; error: %s ", exitValue, error);
226                     throw new InitializationException(msg);
227                 }
228             }
229         } catch (AnalysisException ae) {
230             setEnabled(false);
231             final String msg = String.format("Exception from bundle-audit process: %s. "
232                     + "Disabling %s", ae.getCause(), ANALYZER_NAME);
233             throw new InitializationException(msg, ae);
234         } catch (UnsupportedEncodingException ex) {
235             setEnabled(false);
236             throw new InitializationException("Unexpected bundle-audit encoding when "
237                     + "reading input stream.", ex);
238         } catch (IOException ex) {
239             setEnabled(false);
240             throw new InitializationException("Unable to read bundle-audit output.", ex);
241         } catch (InterruptedException ex) {
242             setEnabled(false);
243             final String msg = String.format("Bundle-audit process was interrupted. "
244                     + "Disabling %s", ANALYZER_NAME);
245             Thread.currentThread().interrupt();
246             throw new InitializationException(msg);
247         }
248         LOGGER.info("{} is enabled and is using bundle-audit with version details: {}. "
249                 + "Note: It is necessary to manually run \"bundle-audit update\" "
250                 + "occasionally to keep its database up to date.", ANALYZER_NAME,
251                 bundleAuditVersionDetails);
252     }

254     /**
255      * Determines if the analyzer can analyze the given file type.
256      *
257      * @param dependency the dependency to determine if it can analyze
258      * @param engine the dependency-checkException engine
259      * @throws AnalysisException thrown if there is an analysis exception.
260      */
261     @Override
262     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
263         if (needToDisableGemspecAnalyzer) {
264             boolean failed = true;
265             final String className = RubyGemspecAnalyzer.class.getName();
266             for (FileTypeAnalyzer analyzer : engine.getFileTypeAnalyzers()) {
267                 if (analyzer instanceof RubyBundlerAnalyzer) {
268                     ((RubyBundlerAnalyzer) analyzer).setEnabled(false);
269                     LOGGER.info("Disabled {} to avoid noisy duplicate results.",
270                             RubyBundlerAnalyzer.class.getName());
271                 } else if (analyzer instanceof RubyGemspecAnalyzer) {
272                     ((RubyGemspecAnalyzer) analyzer).setEnabled(false);
273                     LOGGER.info("Disabled {} to avoid noisy duplicate results.", className);
274                     failed = false;
275                 }
276             }
277             needToDisableGemspecAnalyzer = false;
278         }
279         final File parentFile = dependency.getActualFile().getParentFile();
280         final List<String> bundleAuditArgs = Arrays.asList("check", "--verbose");
281 
282         final Process process = launchBundleAudit(parentFile, bundleAuditArgs);
283         try (BundlerAuditProcessor processor = new BundlerAuditProcessor(dependency, engine);
284                 ProcessReader processReader = new ProcessReader(process, processor)) {
285 
286             processReader.readAll();
287             final String error = processReader.getError();
288             if (StringUtils.isNoneBlank(error)) {
289                 LOGGER.warn("Warnings from bundle-audit {}", error);
290             }
291             final int exitValue = process.exitValue();
292             if (exitValue < 0 || exitValue > 1) {
293                 final String msg = String.format("Unexpected exit code from bundle-audit "
294                         + "process; exit code: %s", exitValue);
295                 throw new AnalysisException(msg);
296             }
297         } catch (InterruptedException ie) {
298             Thread.currentThread().interrupt();
299             throw new AnalysisException("bundle-audit process interrupted", ie);
300         } catch (IOException | CpeValidationException ioe) {
301             LOGGER.warn("bundle-audit failure", ioe);
302             throw new AnalysisException("bunder-audit error: " + ioe.getMessage(), ioe);
303         }
304     }
