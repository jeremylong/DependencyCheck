132     /**
133      * Launch bundle-audit.
134      *
135      * @param folder directory that contains bundle audit
136      * @param bundleAuditArgs the arguments to pass to bundle audit
137      * @return a handle to the process
138      * @throws AnalysisException thrown when there is an issue launching bundle
139      * audit
140      */
141     private Process launchBundleAudit(File folder, List<String> bundleAuditArgs) throws AnalysisException {
142         if (!folder.isDirectory()) {
143             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
144         }
145         final List<String> args = new ArrayList<>();
146         final String bundleAuditPath = getSettings().getString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_PATH);
147         File bundleAudit = null;
148         if (bundleAuditPath != null) {
149             bundleAudit = new File(bundleAuditPath);
150             if (!bundleAudit.isFile()) {
151                 LOGGER.warn("Supplied `bundleAudit` path is incorrect: {}", bundleAuditPath);
152                 bundleAudit = null;
153             }
154         }
155         args.add(bundleAudit != null ? bundleAudit.getAbsolutePath() : "bundle-audit");
156         args.addAll(bundleAuditArgs);
157         final ProcessBuilder builder = new ProcessBuilder(args);
158 
159         final String bundleAuditWorkingDirectoryPath = getSettings().getString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_WORKING_DIRECTORY);
160         File bundleAuditWorkingDirectory = null;
161         if (bundleAuditWorkingDirectoryPath != null) {
162             bundleAuditWorkingDirectory = new File(bundleAuditWorkingDirectoryPath);
163             if (!bundleAuditWorkingDirectory.isDirectory()) {
164                 LOGGER.warn("Supplied `bundleAuditWorkingDirectory` path is incorrect: {}", bundleAuditWorkingDirectoryPath);
165                 bundleAuditWorkingDirectory = null;
166             }
167         }
168         final File launchBundleAuditFromDirectory = bundleAuditWorkingDirectory != null ? bundleAuditWorkingDirectory : folder;
169         builder.directory(launchBundleAuditFromDirectory);
170         try {
171             LOGGER.info("Launching: {} from {}", args, launchBundleAuditFromDirectory);
172             return builder.start();
173         } catch (IOException ioe) {
174             throw new AnalysisException("bundle-audit initialization failure; this error can be ignored if you are not analyzing Ruby. "
175                     + "Otherwise ensure that bundle-audit is installed and the path to bundle audit is correctly specified", ioe);
176         }
177     }

179     /**
180      * Initialize the analyzer.
181      *
182      * @param engine a reference to the dependency-check engine
183      * @throws InitializationException if anything goes wrong
184      */
185     @Override
186     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
187         if (engine != null) {
188             this.cvedb = engine.getDatabase();
189         }
190 
191         // Here we check if bundle-audit actually runs from this location. We do this by running the
192         // `bundle-audit version` command and seeing whether or not it succeeds (if it returns with an exit value of 0)
193         final Process process;
194         try {
195             final List<String> bundleAuditArgs = ImmutableList.of("version");
196             process = launchBundleAudit(getSettings().getTempDirectory(), bundleAuditArgs);
197         } catch (AnalysisException ae) {
198             setEnabled(false);
199             final String msg = String.format("Exception from bundle-audit process: %s. Disabling %s", ae.getCause(), ANALYZER_NAME);
200             throw new InitializationException(msg, ae);
201         } catch (IOException ex) {
202             setEnabled(false);
203             throw new InitializationException("Unable to create temporary file, the Ruby Bundle Audit Analyzer will be disabled", ex);
204         }
205 
206         final int exitValue;
207         try {
208             exitValue = process.waitFor();
209         } catch (InterruptedException ex) {
210             setEnabled(false);
211             final String msg = String.format("Bundle-audit process was interrupted. Disabling %s", ANALYZER_NAME);
212             Thread.currentThread().interrupt();
213             throw new InitializationException(msg);
214         }
215 
216         final String bundleAuditVersionDetails;
217         if (exitValue != 0) {
218             try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
219                 if (!reader.ready()) {
220                     LOGGER.warn("Unexpected exit value from bundle-audit process and error stream unexpectedly not ready to capture error details. "
221                             + "Disabling {}. Exit value was: {}", ANALYZER_NAME, exitValue);
222                     setEnabled(false);
223                     throw new InitializationException("Bundle-audit error stream unexpectedly not ready.");
224                 } else {
225                     final String line = reader.readLine();
226                     setEnabled(false);
227                     LOGGER.warn("Unexpected exit value from bundle-audit process. Disabling {}. Exit value was: {}. "
228                             + "error stream output from bundle-audit process was: {}", ANALYZER_NAME, exitValue, line);
229                     throw new InitializationException("Unexpected exit value from bundle-audit process.");
230                 }
231             } catch (UnsupportedEncodingException ex) {
232                 setEnabled(false);
233                 throw new InitializationException("Unexpected bundle-audit encoding when reading error stream.", ex);
234             } catch (IOException ex) {
235                 setEnabled(false);
236                 throw new InitializationException("Unable to read bundle-audit output from error stream.", ex);
237             }
238         } else {
239             try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
240                 if (!reader.ready()) {
241                     LOGGER.warn("Bundle-audit input stream unexpectedly not ready to capture version details. Disabling {}", ANALYZER_NAME);
242                     setEnabled(false);
243                     throw new InitializationException("Bundle-audit input stream unexpectedly not ready to capture version details.");
244                 } else {
245                     bundleAuditVersionDetails = reader.readLine();
246                 }
247             } catch (UnsupportedEncodingException ex) {
248                 setEnabled(false);
249                 throw new InitializationException("Unexpected bundle-audit encoding when reading input stream.", ex);
250             } catch (IOException ex) {
251                 setEnabled(false);
252                 throw new InitializationException("Unable to read bundle-audit output from input stream.", ex);
253             }
254         }
255 
256         if (isEnabled()) {
257             LOGGER.info("{} is enabled and is using bundle-audit with version details: {}. Note: It is necessary to manually run "
258                     + "\"bundle-audit update\" occasionally to keep its database up to date.", ANALYZER_NAME, bundleAuditVersionDetails);
259         }
260     }

293     /**
294      * Determines if the analyzer can analyze the given file type.
295      *
296      * @param dependency the dependency to determine if it can analyze
297      * @param engine the dependency-check engine
298      * @throws AnalysisException thrown if there is an analysis exception.
299      */
300     @Override
301     protected void analyzeDependency(Dependency dependency, Engine engine)
302             throws AnalysisException {
303         if (needToDisableGemspecAnalyzer) {
304             boolean failed = true;
305             final String className = RubyGemspecAnalyzer.class.getName();
306             for (FileTypeAnalyzer analyzer : engine.getFileTypeAnalyzers()) {
307                 if (analyzer instanceof RubyBundlerAnalyzer) {
308                     ((RubyBundlerAnalyzer) analyzer).setEnabled(false);
309                     LOGGER.info("Disabled {} to avoid noisy duplicate results.", RubyBundlerAnalyzer.class.getName());
310                 } else if (analyzer instanceof RubyGemspecAnalyzer) {
311                     ((RubyGemspecAnalyzer) analyzer).setEnabled(false);
312                     LOGGER.info("Disabled {} to avoid noisy duplicate results.", className);
313                     failed = false;
314                 }
315             }
316             if (failed) {
317                 LOGGER.warn("Did not find {}.", className);
318             }
319             needToDisableGemspecAnalyzer = false;
320         }
321         final File parentFile = dependency.getActualFile().getParentFile();
322         final List<String> bundleAuditArgs = ImmutableList.of("check", "--verbose");
323 
324         final Process process = launchBundleAudit(parentFile, bundleAuditArgs);
325         final int exitValue;
326         try {
327             exitValue = process.waitFor();
328         } catch (InterruptedException ie) {
329             Thread.currentThread().interrupt();
330             throw new AnalysisException("bundle-audit process interrupted", ie);
331         }
332         if (exitValue < 0 || exitValue > 1) {
333             final String msg = String.format("Unexpected exit code from bundle-audit process; exit code: %s", exitValue);
334             throw new AnalysisException(msg);
335         }
336         try {
337             try (BufferedReader errReader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
338                 while (errReader.ready()) {
339                     final String error = errReader.readLine();
340                     LOGGER.warn(error);
341                 }
342             }
343             try (BufferedReader rdr = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
344                 processBundlerAuditOutput(dependency, engine, rdr);
345             }
346         } catch (IOException | CpeValidationException ioe) {
347             LOGGER.warn("bundle-audit failure", ioe);
348         }
349     }

351     /**
352      * Processes the bundler audit output.
353      *
354      * @param original the dependency
355      * @param engine the dependency-check engine
356      * @param rdr the reader of the report
357      * @throws IOException thrown if the report cannot be read
358      * @throws CpeValidationException if there is an error building the
359      * CPE/VulnerableSoftware object
360      */
361     private void processBundlerAuditOutput(Dependency original, Engine engine, BufferedReader rdr) throws IOException, CpeValidationException {
362         final String parentName = original.getActualFile().getParentFile().getName();
363         final String fileName = original.getFileName();
364         final String filePath = original.getFilePath();
365         Dependency dependency = null;
366         Vulnerability vulnerability = null;
367         String gem = null;
368         final Map<String, Dependency> map = new HashMap<>();
369         boolean appendToDescription = false;
370         while (rdr.ready()) {
371             final String nextLine = rdr.readLine();
372             if (null == nextLine) {
373                 break;
374             } else if (nextLine.startsWith(NAME)) {
375                 appendToDescription = false;
376                 gem = nextLine.substring(NAME.length());
377                 if (!map.containsKey(gem)) {
378                     map.put(gem, createDependencyForGem(engine, parentName, fileName, filePath, gem));
379                 }
380                 dependency = map.get(gem);
381                 LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
382             } else if (nextLine.startsWith(VERSION)) {
383                 vulnerability = createVulnerability(parentName, dependency, gem, nextLine);
384             } else if (nextLine.startsWith(ADVISORY)) {
385                 setVulnerabilityName(parentName, dependency, vulnerability, nextLine);
386             } else if (nextLine.startsWith(CRITICALITY)) {
387                 addCriticalityToVulnerability(parentName, vulnerability, nextLine);
388             } else if (nextLine.startsWith("URL: ")) {
389                 addReferenceToVulnerability(parentName, vulnerability, nextLine);
390             } else if (nextLine.startsWith("Description:")) {
391                 appendToDescription = true;
392                 if (null != vulnerability) {
393                     vulnerability.setDescription("*** Vulnerability obtained from bundle-audit verbose report. "
394                             + "Title link may not work. CPE below is guessed. CVSS score is estimated (-1.0 "
395                             + " indicates unknown). See link below for full details. *** ");
396                 }
397             } else if (appendToDescription && null != vulnerability) {
398                 vulnerability.setDescription(vulnerability.getDescription() + nextLine + "\n");
399             }
400         }
401     }

403     /**
404      * Sets the vulnerability name.
405      *
406      * @param parentName the parent name
407      * @param dependency the dependency
408      * @param vulnerability the vulnerability
409      * @param nextLine the line to parse
410      */
411     private void setVulnerabilityName(String parentName, Dependency dependency, Vulnerability vulnerability, String nextLine) {
412         final String advisory = nextLine.substring(ADVISORY.length());
413         if (null != vulnerability) {
414             vulnerability.setName(advisory);
415         }
416         if (null != dependency) {
417             dependency.addVulnerability(vulnerability);
418         }
419         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
420     }

422     /**
423      * Adds a reference to the vulnerability.
424      *
425      * @param parentName the parent name
426      * @param vulnerability the vulnerability
427      * @param nextLine the line to parse
428      */
429     private void addReferenceToVulnerability(String parentName, Vulnerability vulnerability, String nextLine) {
430         final String url = nextLine.substring("URL: ".length());
431         if (null != vulnerability) {
432             final Reference ref = new Reference();
433             ref.setName(vulnerability.getName());
434             ref.setSource("bundle-audit");
435             ref.setUrl(url);
436             vulnerability.getReferences().add(ref);
437         }
438         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
439     }

441     /**
442      * Adds the criticality to the vulnerability
443      *
444      * @param parentName the parent name
445      * @param vulnerability the vulnerability
446      * @param nextLine the line to parse
447      */
448     private void addCriticalityToVulnerability(String parentName, Vulnerability vulnerability, String nextLine) {
449         if (null != vulnerability) {
450             final String criticality = nextLine.substring(CRITICALITY.length()).trim();
451             float score = -1.0f;
452             Vulnerability v = null;
453             if (cvedb != null) {
454                 try {
455                     v = cvedb.getVulnerability(vulnerability.getName());
456                 } catch (DatabaseException ex) {
457                     LOGGER.debug("Unable to look up vulnerability {}", vulnerability.getName());
458                 }
459             }
460             if (v != null && (v.getCvssV2() != null || v.getCvssV3() != null)) {
461                 if (v.getCvssV2() != null) {
462                     vulnerability.setCvssV2(v.getCvssV2());
463                 }
464                 if (v.getCvssV3() != null) {
465                     vulnerability.setCvssV3(v.getCvssV3());
466                 }
467             } else {
468                 if ("High".equalsIgnoreCase(criticality)) {
469                     score = 8.5f;
470                 } else if ("Medium".equalsIgnoreCase(criticality)) {
471                     score = 5.5f;
472                 } else if ("Low".equalsIgnoreCase(criticality)) {
473                     score = 2.0f;
474                 }
475                 vulnerability.setCvssV2(new CvssV2(score, "-", "-", "-", "-", "-", "-", criticality));
476             }
477         }
478         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
479     }

481     /**
482      * Creates a vulnerability.
483      *
484      * @param parentName the parent name
485      * @param dependency the dependency
486      * @param gem the gem name
487      * @param nextLine the line to parse
488      * @return the vulnerability
489      * @throws CpeValidationException thrown if there is an error building the
490      * CPE vulnerability object
491      */
492     private Vulnerability createVulnerability(String parentName, Dependency dependency, String gem, String nextLine) throws CpeValidationException {
493         Vulnerability vulnerability = null;
494         if (null != dependency) {
495             final String version = nextLine.substring(VERSION.length());
496             dependency.addEvidence(EvidenceType.VERSION,
497                     "bundler-audit",
498                     "Version",
499                     version,
500                     Confidence.HIGHEST);
501             dependency.setVersion(version);
502             dependency.setName(gem);
503             try {
504                 final PackageURL purl = PackageURLBuilder.aPackageURL().withType("gem").withName(dependency.getName())
505                         .withVersion(dependency.getVersion()).build();
506                 dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
507             } catch (MalformedPackageURLException ex) {
508                 LOGGER.debug("Unable to build package url for python", ex);
509                 final GenericIdentifier id = new GenericIdentifier("gem:" + dependency.getName() + "@" + dependency.getVersion(),
510                         Confidence.HIGHEST);
511                 dependency.addSoftwareIdentifier(id);
512             }
513 
514             vulnerability = new Vulnerability(); // don't add to dependency until we have name set later
515             vulnerability.setSource(Vulnerability.Source.BUNDLEAUDIT);
516             final VulnerableSoftwareBuilder builder = new VulnerableSoftwareBuilder();
517             final VulnerableSoftware vs = builder.part(Part.APPLICATION)
518                     .vendor(gem)
519                     .product(String.format("%s_project", gem))
520                     .version(version).build();
521             vulnerability.addVulnerableSoftware(vs);
522             vulnerability.setMatchedVulnerableSoftware(vs);
523             vulnerability.setCvssV2(new CvssV2(-1, "-", "-", "-", "-", "-", "-", "unknown"));
524         }
525         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
526         return vulnerability;
527     }

529     /**
530      * Creates the dependency based off of the gem.
531      *
532      * @param engine the engine used for scanning
533      * @param parentName the gem parent
534      * @param fileName the file name
535      * @param filePath the file path
536      * @param gem the gem name
537      * @return the dependency to add
538      * @throws IOException thrown if a temporary gem file could not be written
539      */
540     private Dependency createDependencyForGem(Engine engine, String parentName, String fileName, String filePath, String gem) throws IOException {
541         final File gemFile;
542         try {
543             gemFile = File.createTempFile(gem, "_Gemfile.lock", getSettings().getTempDirectory());
544         } catch (IOException ioe) {
545             throw new IOException("Unable to create temporary gem file");
546         }
547         final String displayFileName = String.format("%s%c%s:%s", parentName, File.separatorChar, fileName, gem);
548 
549         FileUtils.write(gemFile, displayFileName, Charset.defaultCharset()); // unique contents to avoid dependency bundling
550         final Dependency dependency = new Dependency(gemFile);
551         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
552         dependency.addEvidence(EvidenceType.PRODUCT, "bundler-audit", "Name", gem, Confidence.HIGHEST);
553         //TODO add package URL - note, this may require parsing the gemfile.lock and getting the version for each entry
554         dependency.setDisplayFileName(displayFileName);
555         dependency.setFileName(fileName);
556         dependency.setFilePath(filePath);
557         engine.addDependency(dependency);
558         return dependency;
559     }
