35  /**
36   * Takes a dependency and analyzes the filename and determines the hashes.
37   *
38   * @author Jeremy Long
39   */
40  @ThreadSafe
41  public class FileNameAnalyzer extends AbstractAnalyzer {
42  
43      /**
44       * Python init files
45       */
46      //CSOFF: WhitespaceAfter
47      private static final NameFileFilter IGNORED_FILES = new NameFileFilter(new String[]{
48          "__init__.py",
49          "__init__.pyc",
50          "__init__.pyo",
51          "composer.lock",
52          "configure.in",
53          "configure.ac",
54          "Gemfile.lock",
55          "METADATA",
56          "PKG-INFO",
57          "package.json",
58          "packages.config",
59          "Package.swift",
60          "classes.jar",
61          "build.gradle",
62          "CMakeLists.txt"}, IOCase.INSENSITIVE);
63      //CSON: WhitespaceAfter
64  
65      //<editor-fold defaultstate="collapsed" desc="All standard implementation details of Analyzer">
66      /**
67       * The name of the analyzer.
68       */
69      private static final String ANALYZER_NAME = "File Name Analyzer";
70      /**
71       * The phase that this analyzer is intended to run in.
72       */
73      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
74  
75      /**
76       * Returns the name of the analyzer.
77       *
78       * @return the name of the analyzer.
79       */
80      @Override
81      public String getName() {
82          return ANALYZER_NAME;
83      }
84  
85      /**
86       * Returns the phase that the analyzer is intended to run in.
87       *
88       * @return the phase that the analyzer is intended to run in.
89       */
90      @Override
91      public AnalysisPhase getAnalysisPhase() {
92          return ANALYSIS_PHASE;
93      }
94  
95      /**
96       * <p>
97       * Returns the setting key to determine if the analyzer is enabled.</p>
98       *
99       * @return the key for the analyzer's enabled property
100      */
101     @Override
102     protected String getAnalyzerEnabledSettingKey() {
103         return Settings.KEYS.ANALYZER_FILE_NAME_ENABLED;
104     }
105     //</editor-fold>
106 
107     /**
108      * Collects information about the file name.
109      *
110      * @param dependency the dependency to analyze.
111      * @param engine the engine that is scanning the dependencies
112      * @throws AnalysisException is thrown if there is an error reading the JAR
113      * file.
114      */
115     @Override
116     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
117         //strip any path information that may get added by ArchiveAnalyzer, etc.
118         final File f = dependency.getActualFile();
119         final String fileName = FilenameUtils.removeExtension(f.getName());
120         final String ext = FilenameUtils.getExtension(f.getName());
121         if (!IGNORED_FILES.accept(f) && !"js".equals(ext)) {
122             //add version evidence
123             final DependencyVersion version = DependencyVersionUtil.parseVersion(fileName);
124             final String packageName = DependencyVersionUtil.parsePreVersion(fileName);
125 
126             if (version != null) {
127                 // If the version number is just a number like 2 or 23, reduce the confidence
128                 // a shade. This should hopefully correct for cases like log4j.jar or
129                 // struts2-core.jar
130                 if (version.getVersionParts() == null || version.getVersionParts().size() < 2) {
131                     dependency.addEvidence(EvidenceType.VERSION, "file", "version", version.toString(), Confidence.MEDIUM);
132                 } else {
133                     dependency.addEvidence(EvidenceType.VERSION, "file", "version", version.toString(), Confidence.HIGH);
134                 }
135                 dependency.addEvidence(EvidenceType.VERSION, "file", "name", packageName, Confidence.MEDIUM);
136             }
137 
138             dependency.addEvidence(EvidenceType.PRODUCT, "file", "name", packageName, Confidence.HIGH);
139             dependency.addEvidence(EvidenceType.VENDOR, "file", "name", packageName, Confidence.HIGH);
140         }
141     }
