179     @Override
180     protected void analyzeDependency(Dependency dependency, Engine engine)
181             throws AnalysisException {
182 
183         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
184         final File actualFile = dependency.getActualFile();
185         if (WHL_FILTER.accept(actualFile)) {
186             collectMetadataFromArchiveFormat(dependency, DIST_INFO_FILTER,
187                     METADATA_FILTER);
188         } else if (EGG_OR_ZIP.accept(actualFile)) {
189             collectMetadataFromArchiveFormat(dependency, EGG_INFO_FILTER,
190                     PKG_INFO_FILTER);
191         } else {
192             final String name = actualFile.getName();
193             final boolean metadata = METADATA.equals(name);
194             if (metadata || PKG_INFO.equals(name)) {
195                 final File parent = actualFile.getParentFile();
196                 final String parentName = parent.getName();
197                 if (parent.isDirectory()
198                         && ((metadata && parentName.endsWith(".dist-info"))
199                         || parentName.endsWith(".egg-info") || "EGG-INFO"
200                         .equals(parentName))) {
201                     collectWheelMetadata(dependency, actualFile);
202                 }
203             }
204         }
205     }
206 
207     /**
208      * Collects the meta data from an archive.
209      *
210      * @param dependency the archive being scanned
211      * @param folderFilter the filter to apply to the folder
212      * @param metadataFilter the filter to apply to the meta data
213      * @throws AnalysisException thrown when there is a problem analyzing the
214      * dependency
215      */
216     private void collectMetadataFromArchiveFormat(Dependency dependency,
217             FilenameFilter folderFilter, FilenameFilter metadataFilter)
218             throws AnalysisException {
219         final File temp = getNextTempDirectory();
220         LOGGER.debug("{} exists? {}", temp, temp.exists());
221         try {
222             ExtractionUtil.extractFilesUsingFilter(
223                     new File(dependency.getActualFilePath()), temp,
224                     metadataFilter);
225         } catch (ExtractionException ex) {
226             throw new AnalysisException(ex);
227         }
228 
229         File matchingFile = getMatchingFile(temp, folderFilter);
230         if (matchingFile != null) {
231             matchingFile = getMatchingFile(matchingFile, metadataFilter);
232             if (matchingFile != null) {
233                 collectWheelMetadata(dependency, matchingFile);
234             }
235         }
236     }

238     /**
239      * Makes sure a usable temporary directory is available.
240      *
241      * @param engine a reference to the dependency-check engine
242      * @throws InitializationException an AnalyzeException is thrown when the
243      * temp directory cannot be created
244      */
245     @Override
246     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
247         try {
248             final File baseDir = getSettings().getTempDirectory();
249             tempFileLocation = File.createTempFile("check", "tmp", baseDir);
250             if (!tempFileLocation.delete()) {
251                 setEnabled(false);
252                 final String msg = String.format(
253                         "Unable to delete temporary file '%s'.",
254                         tempFileLocation.getAbsolutePath());
255                 throw new InitializationException(msg);
256             }
257             if (!tempFileLocation.mkdirs()) {
258                 setEnabled(false);
259                 final String msg = String.format(
260                         "Unable to create directory '%s'.",
261                         tempFileLocation.getAbsolutePath());
262                 throw new InitializationException(msg);
263             }
264         } catch (IOException ex) {
265             setEnabled(false);
266             throw new InitializationException("Unable to create a temporary file", ex);
267         }
268     }

270     /**
271      * Deletes any files extracted from the Wheel during analysis.
272      */
273     @Override
274     public void closeAnalyzer() {
275         if (tempFileLocation != null && tempFileLocation.exists()) {
276             LOGGER.debug("Attempting to delete temporary files");
277             final boolean success = FileUtils.delete(tempFileLocation);
278             if (!success && tempFileLocation.exists()) {
279                 final String[] l = tempFileLocation.list();
280                 if (l != null && l.length > 0) {
281                     LOGGER.warn("Failed to delete some temporary files, see the log for more details");
282                 }
283             }
284         }
285     }

287     /**
288      * Gathers evidence from the METADATA file.
289      *
290      * @param dependency the dependency being analyzed
291      * @param file a reference to the manifest/properties file
292      */
293     private static void collectWheelMetadata(Dependency dependency, File file) {
294         final InternetHeaders headers = getManifestProperties(file);
295         final String version = addPropertyToEvidence(dependency, EvidenceType.VERSION, Confidence.HIGHEST, headers, "Version");
296         final String name = addPropertyToEvidence(dependency, EvidenceType.VENDOR, Confidence.HIGHEST, headers, "Name");
297         addPropertyToEvidence(dependency, EvidenceType.PRODUCT, Confidence.HIGHEST, headers, "Name");
298 
299         final String packagePath = String.format("%s:%s", name, version);
300         dependency.setName(name);
301         dependency.setVersion(version);
302         dependency.setPackagePath(packagePath);
303         dependency.setDisplayFileName(packagePath);
304         final String url = headers.getHeader("Home-page", null);
305         if (StringUtils.isNotBlank(url)) {
306             if (UrlStringUtils.isUrl(url)) {
307                 dependency.addEvidence(EvidenceType.VENDOR, METADATA, "vendor", url, Confidence.MEDIUM);
308             }
309         }
310         addPropertyToEvidence(dependency, EvidenceType.VENDOR, Confidence.LOW, headers, "Author");
311         final String summary = headers.getHeader("Summary", null);
312         if (StringUtils.isNotBlank(summary)) {
313             JarAnalyzer.addDescription(dependency, summary, METADATA, "summary");
314         }
315 
316         try {
317             final PackageURL purl = PackageURLBuilder.aPackageURL().withType("pypi")
318                     .withName(name).withVersion(version).build();
319             dependency.addSoftwareIdentifier(new PurlIdentifier(purl, Confidence.HIGHEST));
320         } catch (MalformedPackageURLException ex) {
321             LOGGER.debug("Unable to build package url for python", ex);
322             final GenericIdentifier id = new GenericIdentifier("generic:" + name + "@" + version, Confidence.HIGHEST);
323             dependency.addSoftwareIdentifier(id);
324         }
325     }

327     /**
328      * Adds a value to the evidence collection.
329      *
330      * @param dependency the dependency being analyzed
331      * @param type the type of evidence to add
332      * @param confidence the confidence in the evidence being added
333      * @param headers the properties collection
334      * @param property the property name
335      * @return returns the value of the property if found; otherwise
336      * <code>null</code>
337      */
338     private static String addPropertyToEvidence(Dependency dependency, EvidenceType type, Confidence confidence,
339             InternetHeaders headers, String property) {
340         final String value = headers.getHeader(property, null);
341         LOGGER.debug("Property: {}, Value: {}", property, value);
342         if (StringUtils.isNotBlank(value)) {
343             dependency.addEvidence(type, METADATA, property, value, confidence);
344         }
345         return value;
346     }

348     /**
349      * Returns a list of files that match the given filter, this does not
350      * recursively scan the directory.
351      *
352      * @param folder the folder to filter
353      * @param filter the filter to apply to the files in the directory
354      * @return the list of Files in the directory that match the provided filter
355      */
356     private static File getMatchingFile(File folder, FilenameFilter filter) {
357         File result = null;
358         final File[] matches = folder.listFiles(filter);
359         if (null != matches && 1 == matches.length) {
360             result = matches[0];
361         }
362         return result;
363     }

365     /**
366      * Reads the manifest entries from the provided file.
367      *
368      * @param manifest the manifest
369      * @return the manifest entries
370      */
371     private static InternetHeaders getManifestProperties(File manifest) {
372         final InternetHeaders result = new InternetHeaders();
373         if (null == manifest) {
374             LOGGER.debug("Manifest file not found.");
375         } else {
376             try (InputStream in = new BufferedInputStream(new FileInputStream(manifest))) {
377                 result.load(in);
378             } catch (MessagingException | FileNotFoundException e) {
379                 LOGGER.warn(e.getMessage(), e);
380             } catch (IOException ex) {
381                 LOGGER.warn(ex.getMessage(), ex);
382             }
383         }
384         return result;
385     }

387     /**
388      * Retrieves the next temporary destination directory for extracting an
389      * archive.
390      *
391      * @return a directory
392      * @throws AnalysisException thrown if unable to create temporary directory
393      */
394     private File getNextTempDirectory() throws AnalysisException {
395         File directory;
396 
397         // getting an exception for some directories not being able to be
398         // created; might be because the directory already exists?
399         do {
400             final int dirCount = DIR_COUNT.incrementAndGet();
401             directory = new File(tempFileLocation, String.valueOf(dirCount));
402         } while (directory.exists());
403         if (!directory.mkdirs()) {
404             throw new AnalysisException(String.format(
405                     "Unable to create temp directory '%s'.",
406                     directory.getAbsolutePath()));
407         }
408         return directory;
409     }
