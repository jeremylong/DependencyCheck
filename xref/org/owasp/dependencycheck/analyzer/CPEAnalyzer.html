177     /**
178      * Creates the CPE Lucene Index.
179      *
180      * @param engine a reference to the dependency-check engine
181      * @throws InitializationException is thrown if there is an issue opening
182      * the index.
183      */
184     @Override
185     public void prepareAnalyzer(Engine engine) throws InitializationException {
186         super.prepareAnalyzer(engine);
187         try {
188             this.open(engine.getDatabase());
189         } catch (IOException ex) {
190             LOGGER.debug("Exception initializing the Lucene Index", ex);
191             throw new InitializationException("An exception occurred initializing the Lucene Index", ex);
192         } catch (DatabaseException ex) {
193             LOGGER.debug("Exception accessing the database", ex);
194             throw new InitializationException("An exception occurred accessing the database", ex);
195         }
196         final String[] tmp = engine.getSettings().getArray(Settings.KEYS.ECOSYSTEM_SKIP_CPEANALYZER);
197         if (tmp == null) {
198             skipEcosystems = new ArrayList<>();
199         } else {
200             LOGGER.debug("Skipping CPE Analysis for {}", StringUtils.join(tmp, ","));
201             skipEcosystems = Arrays.asList(tmp);
202         }
203         ecosystemTools = new Ecosystem(engine.getSettings());
204         suppression = new CpeSuppressionAnalyzer();
205         suppression.initialize(engine.getSettings());
206         suppression.prepareAnalyzer(engine);
207     }

243     /**
244      * Searches the data store of CPE entries, trying to identify the CPE for
245      * the given dependency based on the evidence contained within. The
246      * dependency passed in is updated with any identified CPE values.
247      *
248      * @param dependency the dependency to search for CPE entries on
249      * @throws CorruptIndexException is thrown when the Lucene index is corrupt
250      * @throws IOException is thrown when an IOException occurs
251      * @throws ParseException is thrown when the Lucene query cannot be parsed
252      * @throws AnalysisException thrown if the suppression rules failed
253      */
254     protected void determineCPE(Dependency dependency) throws CorruptIndexException, IOException, ParseException, AnalysisException {
255         boolean identifierAdded;
256 
257         final Set<String> majorVersions = dependency.getSoftwareIdentifiers()
258                 .stream()
259                 .filter(i -> i instanceof PurlIdentifier)
260                 .map(i -> {
261                     final PurlIdentifier p = (PurlIdentifier) i;
262                     final DependencyVersion depVersion = DependencyVersionUtil.parseVersion(p.getVersion(), false);
263                     if (depVersion != null) {
264                         return depVersion.getVersionParts().get(0);
265                     }
266                     return null;
267                 }).collect(Collectors.toSet());
268 
269         final Map<String, MutableInt> vendors = new HashMap<>();
270         final Map<String, MutableInt> products = new HashMap<>();
271         final Set<Integer> previouslyFound = new HashSet<>();
272 
273         for (Confidence confidence : Confidence.values()) {
274             collectTerms(vendors, dependency.getIterator(EvidenceType.VENDOR, confidence));
275             LOGGER.debug("vendor search: {}", vendors);
276             collectTerms(products, dependency.getIterator(EvidenceType.PRODUCT, confidence));
277             addMajorVersionToTerms(majorVersions, products);
278             LOGGER.debug("product search: {}", products);
279             if (!vendors.isEmpty() && !products.isEmpty()) {
280                 final List<IndexEntry> entries = searchCPE(vendors, products,
281                         dependency.getVendorWeightings(), dependency.getProductWeightings(),
282                         dependency.getEcosystem());
283                 if (entries == null) {
284                     continue;
285                 }
286 
287                 identifierAdded = false;
288                 for (IndexEntry e : entries) {
289                     if (previouslyFound.contains(e.getDocumentId()) /*|| (filter > 0 && e.getSearchScore() < filter)*/) {
290                         continue;
291                     }
292                     previouslyFound.add(e.getDocumentId());
293                     if (verifyEntry(e, dependency, majorVersions)) {
294                         final String vendor = e.getVendor();
295                         final String product = e.getProduct();
296                         LOGGER.debug("identified vendor/product: {}/{}", vendor, product);
297                         identifierAdded |= determineIdentifiers(dependency, vendor, product, confidence);
298                     }
299                 }
300                 if (identifierAdded) {
301                     break;
302                 }
303             }
304         }
305     }

307     /**
308      * <p>
309      * Returns the text created by concatenating the text and the values from
310      * the EvidenceCollection (filtered for a specific confidence). This
311      * attempts to prevent duplicate terms from being added.</p>
312      * <p>
313      * Note, if the evidence is longer then 1000 characters it will be
314      * truncated.</p>
315      *
316      * @param terms the collection of terms
317      * @param evidence an iterable set of evidence to concatenate
318      */
319     @SuppressWarnings("null")
320 
321     protected void collectTerms(Map<String, MutableInt> terms, Iterable<Evidence> evidence) {
322         for (Evidence e : evidence) {
323             String value = cleanseText(e.getValue());
324             if (StringUtils.isBlank(value)) {
325                 continue;
326             }
327             if (value.length() > 1000) {
328                 boolean trimmed = false;
329                 int pos = value.lastIndexOf(" ", 1000);
330                 if (pos > 0) {
331                     value = value.substring(0, pos);
332                     trimmed = true;
333                 } else {
334                     pos = value.lastIndexOf(".", 1000);
335                 }
336                 if (!trimmed) {
337                     if (pos > 0) {
338                         value = value.substring(0, pos);
339                         trimmed = true;
340                     } else {
341                         pos = value.lastIndexOf("-", 1000);
342                     }
343                 }
344                 if (!trimmed) {
345                     if (pos > 0) {
346                         value = value.substring(0, pos);
347                         trimmed = true;
348                     } else {
349                         pos = value.lastIndexOf("_", 1000);
350                     }
351                 }
352                 if (!trimmed) {
353                     if (pos > 0) {
354                         value = value.substring(0, pos);
355                         trimmed = true;
356                     } else {
357                         pos = value.lastIndexOf("/", 1000);
358                     }
359                 }
360                 if (!trimmed && pos > 0) {
361                     value = value.substring(0, pos);
362                     trimmed = true;
363                 }
364                 if (!trimmed) {
365                     value = value.substring(0, 1000);
366                 }
367             }
368             addTerm(terms, value);
369         }
370     }

372     private void addMajorVersionToTerms(Set<String> majorVersions, Map<String, MutableInt> products) {
373         final Map<String, MutableInt> temp = new HashMap<>();
374         products.entrySet().stream()
375                 .filter(term -> term.getKey() != null)
376                 .forEach(term -> {
377                     majorVersions.stream()
378                             .filter(version -> version != null
379                             && (!term.getKey().endsWith(version)
380                             && !Character.isDigit(term.getKey().charAt(term.getKey().length() - 1))
381                             && !products.containsKey(term.getKey() + version)))
382                             .forEach(version -> {
383                                 addTerm(temp, term.getKey() + version);
384                             });
385                 });
386         products.entrySet().stream()
387                 .filter(term -> term.getKey() != null)
388                 .forEach(term -> {
389                     majorVersions.stream()
390                             .filter(version -> version != null)
391                             .map(version -> "v" + version)
392                             .filter(version -> (!term.getKey().endsWith(version)
393                             && !Character.isDigit(term.getKey().charAt(term.getKey().length() - 1))
394                             && !products.containsKey(term.getKey() + version)))
395                             .forEach(version -> {
396                                 addTerm(temp, term.getKey() + version);
397                             });
398                 });
399         products.putAll(temp);
400     }

403      * Adds a term to the map of terms.
404      *
405      * @param terms the map of terms
406      * @param value the value of the term to add
407      */
408     private void addTerm(Map<String, MutableInt> terms, String value) {
409         final MutableInt count = terms.get(value);
410         if (count == null) {
411             terms.put(value, new MutableInt(1));
412         } else {
413             count.add(1);
414         }

417     /**
418      * <p>
419      * Searches the Lucene CPE index to identify possible CPE entries associated
420      * with the supplied vendor, product, and version.</p>
421      *
422      * <p>
423      * If either the vendorWeightings or productWeightings lists have been
424      * populated this data is used to add weighting factors to the search.</p>
425      *
426      * @param vendor the text used to search the vendor field
427      * @param product the text used to search the product field
428      * @param vendorWeightings a list of strings to use to add weighting factors
429      * to the vendor field
430      * @param productWeightings Adds a list of strings that will be used to add
431      * weighting factors to the product search
432      * @param ecosystem the dependency's ecosystem
433      * @return a list of possible CPE values
434      */
435     protected List<IndexEntry> searchCPE(Map<String, MutableInt> vendor, Map<String, MutableInt> product,
436             Set<String> vendorWeightings, Set<String> productWeightings, String ecosystem) {
437 
438         final int maxQueryResults = ecosystemTools.getLuceneMaxQueryLimitFor(ecosystem);
439         final List<IndexEntry> ret = new ArrayList<>(maxQueryResults);
440 
441         final String searchString = buildSearch(vendor, product, vendorWeightings, productWeightings);
442         if (searchString == null) {
443             return ret;
444         }
445         try {
446             final Query query = cpe.parseQuery(searchString);
447             final TopDocs docs = cpe.search(query, maxQueryResults);
448 
449             for (ScoreDoc d : docs.scoreDocs) {
450                 //if (d.score >= minLuceneScore) {
451                 final Document doc = cpe.getDocument(d.doc);
452                 final IndexEntry entry = new IndexEntry();
453                 entry.setDocumentId(d.doc);
454                 entry.setVendor(doc.get(Fields.VENDOR));
455                 entry.setProduct(doc.get(Fields.PRODUCT));
456                 entry.setSearchScore(d.score);
457 
458 //                LOGGER.error("Explanation: ---------------------");
459 //                LOGGER.error("Explanation: " + entry.getVendor() + " " + entry.getProduct() + " " + entry.getSearchScore());
460 //                LOGGER.error("Explanation: " + searchString);
461 //                LOGGER.error("Explanation: " + cpe.explain(query, d.doc));
462                 if (!ret.contains(entry)) {
463                     ret.add(entry);
464                 }
465                 //}
466             }
467             return ret;
468         } catch (ParseException ex) {
469             LOGGER.warn("An error occurred querying the CPE data. See the log for more details.");
470             LOGGER.info("Unable to parse: {}", searchString, ex);
471         } catch (IndexException ex) {
472             LOGGER.warn("An error occurred resetting the CPE index searcher. See the log for more details.");
473             LOGGER.info("Unable to reset the search analyzer", ex);
474         } catch (IOException ex) {
475             LOGGER.warn("An error occurred reading CPE data. See the log for more details.");
476             LOGGER.info("IO Error with search string: {}", searchString, ex);
477         }
478         return null;
479     }

481     /**
482      * <p>
483      * Builds a Lucene search string by properly escaping data and constructing
484      * a valid search query.</p>
485      *
486      * <p>
487      * If either the possibleVendor or possibleProducts lists have been
488      * populated this data is used to add weighting factors to the search string
489      * generated.</p>
490      *
491      * @param vendor text to search the vendor field
492      * @param product text to search the product field
493      * @param vendorWeighting a list of strings to apply to the vendor to boost
494      * the terms weight
495      * @param productWeightings a list of strings to apply to the product to
496      * boost the terms weight
497      * @return the Lucene query
498      */
499     protected String buildSearch(Map<String, MutableInt> vendor, Map<String, MutableInt> product,
500             Set<String> vendorWeighting, Set<String> productWeightings) {
501 
502         final StringBuilder sb = new StringBuilder();
503 
504         if (!appendWeightedSearch(sb, Fields.PRODUCT, product, productWeightings)) {
505             return null;
506         }
507         sb.append(" AND ");
508         if (!appendWeightedSearch(sb, Fields.VENDOR, vendor, vendorWeighting)) {
509             return null;
510         }
511         return sb.toString();
512     }

514     /**
515      * This method constructs a Lucene query for a given field. The searchText
516      * is split into separate words and if the word is within the list of
517      * weighted words then an additional weighting is applied to the term as it
518      * is appended into the query.
519      *
520      * @param sb a StringBuilder that the query text will be appended to.
521      * @param field the field within the Lucene index that the query is
522      * searching.
523      * @param terms text used to construct the query.
524      * @param weightedText a list of terms that will be considered higher
525      * importance when searching.
526      * @return if the append was successful.
527      */
528     @SuppressWarnings("StringSplitter")
529     private boolean appendWeightedSearch(StringBuilder sb, String field, Map<String, MutableInt> terms, Set<String> weightedText) {
530         if (terms.isEmpty()) {
531             return false;
532         }
533         sb.append(field).append(":(");
534         boolean addSpace = false;
535         boolean addedTerm = false;
536 
537         for (Map.Entry<String, MutableInt> entry : terms.entrySet()) {
538             final StringBuilder boostedTerms = new StringBuilder();
539             final int weighting = entry.getValue().intValue();
540             final String[] text = entry.getKey().split(" ");
541             for (String word : text) {
542                 if (word.isEmpty()) {
543                     continue;
544                 }
545                 if (addSpace) {
546                     sb.append(" ");
547                 } else {
548                     addSpace = true;
549                 }
550                 addedTerm = true;
551                 if (LuceneUtils.isKeyword(word)) {
552                     sb.append("\"");
553                     LuceneUtils.appendEscapedLuceneQuery(sb, word);
554                     sb.append("\"");
555                 } else {
556                     LuceneUtils.appendEscapedLuceneQuery(sb, word);
557                 }
558                 final String boostTerm = findBoostTerm(word, weightedText);
559 
560                 //The weighting is on a full phrase rather then at a term level for vendor or products
561                 //TODO - should the weighting be at a "word" level as opposed to phrase level? Or combined word and phrase?
562                 //remember the reason we are counting the frequency of "phrases" as opposed to terms is that
563                 //we need to keep the correct sequence of terms from the evidence so the term concatenating analyzer
564                 //works correctly and will causes searches to take spring framework and produce: spring springframework framework
565                 if (boostTerm != null) {
566                     sb.append("^").append(weighting + WEIGHTING_BOOST);
567                     if (!boostTerm.equals(word)) {
568                         boostedTerms.append(" ");
569                         LuceneUtils.appendEscapedLuceneQuery(boostedTerms, boostTerm);
570                         boostedTerms.append("^").append(weighting + WEIGHTING_BOOST);
571                     }
572                 } else if (weighting > 1) {
573                     sb.append("^").append(weighting);
574                 }
575             }
576             if (boostedTerms.length() > 0) {
577                 sb.append(boostedTerms);
578             }
579         }
580         sb.append(")");
581         return addedTerm;
582     }

595     /**
596      * Searches the collection of boost terms for the given term. The elements
597      * are case insensitive matched using only the alpha-numeric contents of the
598      * terms; all other characters are removed.
599      *
600      * @param term the term to search for
601      * @param boost the collection of boost terms
602      * @return the value identified
603      */
604     private String findBoostTerm(String term, Set<String> boost) {
605         for (String entry : boost) {
606             if (equalsIgnoreCaseAndNonAlpha(term, entry)) {
607                 return entry;
608             }
609         }
610         return null;
611     }

613     /**
614      * Compares two strings after lower casing them and removing the non-alpha
615      * characters.
616      *
617      * @param l string one to compare.
618      * @param r string two to compare.
619      * @return whether or not the two strings are similar.
620      */
621     private boolean equalsIgnoreCaseAndNonAlpha(String l, String r) {
622         if (l == null || r == null) {
623             return false;
624         }
625 
626         final String left = l.replaceAll(CLEANSE_NONALPHA_RX, "");
627         final String right = r.replaceAll(CLEANSE_NONALPHA_RX, "");
628         return left.equalsIgnoreCase(right);
629     }

637      * @param dependency the dependency that the CPE entries could be for
638      * @param majorVersions the major versions detected for the dependency
639      * @return whether or not the entry is valid.
640      */
641     private boolean verifyEntry(final IndexEntry entry, final Dependency dependency,
642             final Set<String> majorVersions) {
643         boolean isValid = false;
644         //TODO - does this nullify some of the fuzzy matching that happens in the lucene search?
645         // for instance CPE some-component and in the evidence we have SomeComponent.
646 
647         //TODO - should this have a package manager only flag instead of just looking for NPM
648         if (Ecosystem.NODEJS.equals(dependency.getEcosystem())) {
649             for (Identifier i : dependency.getSoftwareIdentifiers()) {
650                 if (i instanceof PurlIdentifier) {
651                     final PurlIdentifier p = (PurlIdentifier) i;
652                     if (cleanPackageName(p.getName()).equals(cleanPackageName(entry.getProduct()))) {
653                         isValid = true;
654                     }
655                 }
656             }
657         } else if (collectionContainsString(dependency.getEvidence(EvidenceType.VENDOR), entry.getVendor())) {
658             if (collectionContainsString(dependency.getEvidence(EvidenceType.PRODUCT), entry.getProduct())) {
659                 isValid = true;
660             } else {
661                 isValid = majorVersions.stream().filter(version
662                         -> version != null && entry.getProduct().endsWith("v" + version) && entry.getProduct().length() > version.length() + 1)
663                         .anyMatch(version
664                                 -> collectionContainsString(dependency.getEvidence(EvidenceType.PRODUCT),
665                                 entry.getProduct().substring(0, entry.getProduct().length() - version.length() - 1))
666                         );
667                 isValid |= majorVersions.stream().filter(version
668                         -> version != null && entry.getProduct().endsWith(version) && entry.getProduct().length() > version.length())
669                         .anyMatch(version
670                                 -> collectionContainsString(dependency.getEvidence(EvidenceType.PRODUCT),
671                                 entry.getProduct().substring(0, entry.getProduct().length() - version.length()))
672                         );
673             }
674         }
675         return isValid;
676     }

691     /**
692      * Used to determine if the EvidenceCollection contains a specific string.
693      *
694      * @param evidence an of evidence object to check
695      * @param text the text to search for
696      * @return whether or not the EvidenceCollection contains the string
697      */
698     @SuppressWarnings("StringSplitter")
699     private boolean collectionContainsString(Set<Evidence> evidence, String text) {
700         //TODO - likely need to change the split... not sure if this will work for CPE with special chars
701         if (text == null) {
702             return false;
703         }
704         // Check if we have an exact match
705         final String textLC = text.toLowerCase();
706         for (Evidence e : evidence) {
707             if (e.getValue().toLowerCase().equals(textLC)) {
708                 return true;
709             }
710         }
711 
712         final String[] words = text.split("[\\s_-]+");
713         final List<String> list = new ArrayList<>();
714         String tempWord = null;
715         final CharArraySet stopWords = SearchFieldAnalyzer.getStopWords();
716         for (String word : words) {
717             /*
718              single letter words should be concatenated with the next word.
719              so { "m", "core", "sample" } -> { "mcore", "sample" }
720              */
721             if (tempWord != null) {
722                 list.add(tempWord + word);
723                 tempWord = null;
724             } else if (word.length() <= 2) {
725                 tempWord = word;
726             } else {
727                 if (stopWords.contains(word)) {
728                     continue;
729                 }
730                 list.add(word);
731             }
732         }
733         if (tempWord != null) {
734             if (!list.isEmpty()) {
735                 final String tmp = list.get(list.size() - 1) + tempWord;
736                 list.add(tmp);
737             } else {
738                 list.add(tempWord);
739             }
740         }
741         if (list.isEmpty()) {
742             return false;
743         }
744         boolean isValid = true;
745 
746         // Prepare the evidence values, e.g. remove the characters we used for splitting
747         final List<String> evidenceValues = new ArrayList<>(evidence.size());
748         evidence.forEach((e) -> {
749             evidenceValues.add(e.getValue().toLowerCase().replaceAll("[\\s_-]+", ""));
750         });
751 
752         for (String word : list) {
753             word = word.toLowerCase();
754             boolean found = false;
755             for (String e : evidenceValues) {
756                 if (e.contains(word)) {
757                     if ("http".equals(word) && e.contains("http:")) {
758                         continue;
759                     }
760                     found = true;
761                     break;
762                 }
763             }
764             isValid &= found;
765 //            if (!isValid) {
766 //                break;
767 //            }
768         }
769         return isValid;
770     }

797     /**
798      * Retrieves a list of CPE values from the CveDB based on the vendor and
799      * product passed in. The list is then validated to find only CPEs that are
800      * valid for the given dependency. It is possible that the CPE identified is
801      * a best effort "guess" based on the vendor, product, and version
802      * information.
803      *
804      * @param dependency the Dependency being analyzed
805      * @param vendor the vendor for the CPE being analyzed
806      * @param product the product for the CPE being analyzed
807      * @param currentConfidence the current confidence being used during
808      * analysis
809      * @return <code>true</code> if an identifier was added to the dependency;
810      * otherwise <code>false</code>
811      * @throws UnsupportedEncodingException is thrown if UTF-8 is not supported
812      * @throws AnalysisException thrown if the suppression rules failed
813      */
814     @SuppressWarnings("StringSplitter")
815     protected boolean determineIdentifiers(Dependency dependency, String vendor, String product,
816             Confidence currentConfidence) throws UnsupportedEncodingException, AnalysisException {
817 
818         final CpeBuilder cpeBuilder = new CpeBuilder();
819 
820         final Set<CpePlus> cpePlusEntries = cve.getCPEs(vendor, product);
821         final Set<Cpe> cpes = filterEcosystem(dependency.getEcosystem(), cpePlusEntries);
822         if (cpes == null || cpes.isEmpty()) {
823             return false;
824         }
825 
826         DependencyVersion bestGuess;
827         if ("Golang".equals(dependency.getEcosystem()) && dependency.getVersion() == null) {
828             bestGuess = new DependencyVersion("*");
829         } else {
830             bestGuess = new DependencyVersion("-");
831         }
832         String bestGuessUpdate = null;
833         Confidence bestGuessConf = null;
834         String bestGuessURL = null;
835         final Set<IdentifierMatch> collected = new HashSet<>();
836 
837         considerDependencyVersion(dependency, vendor, product, currentConfidence, collected, bestGuess);
838 
839         //TODO the following algorithm incorrectly identifies things as a lower version
840         // if there lower confidence evidence when the current (highest) version number
841         // is newer then anything in the NVD.
842         for (Confidence conf : Confidence.values()) {
843             for (Evidence evidence : dependency.getIterator(EvidenceType.VERSION, conf)) {
844                 final DependencyVersion evVer = DependencyVersionUtil.parseVersion(evidence.getValue(), true);
845                 if (evVer == null) {
846                     continue;
847                 }
848                 DependencyVersion evBaseVer = null;
849                 String evBaseVerUpdate = null;
850                 final int idx = evVer.getVersionParts().size() - 1;
851                 if (evVer.getVersionParts().get(idx)
852                         .matches("^(v|release|final|snapshot|beta|alpha|u|rc|m|20\\d\\d).*$")) {
853                     //store the update version
854                     final String checkUpdate = evVer.getVersionParts().get(idx);
855                     if (checkUpdate.matches("^(v|release|final|snapshot|beta|alpha|u|rc|m|20\\d\\d).*$")) {
856                         evBaseVerUpdate = checkUpdate;
857                         evBaseVer = new DependencyVersion();
858                         evBaseVer.setVersionParts(evVer.getVersionParts().subList(0, idx));
859                     }
860                 }
861                 //TODO - review and update for new JSON data
862                 for (Cpe vs : cpes) {
863                     final DependencyVersion dbVer = DependencyVersionUtil.parseVersion(vs.getVersion());
864                     DependencyVersion dbVerUpdate = dbVer;
865                     if (vs.getUpdate() != null && !vs.getUpdate().isEmpty() && !vs.getUpdate().startsWith("*") && !vs.getUpdate().startsWith("-")) {
866                         dbVerUpdate = DependencyVersionUtil.parseVersion(vs.getVersion() + '.' + vs.getUpdate(), true);
867                     }
868                     if (dbVer == null) { //special case, no version specified - everything is vulnerable
869                         final String url = String.format(NVD_SEARCH_BROAD_URL, URLEncoder.encode(vs.getVendor(), UTF8),
870                                 URLEncoder.encode(vs.getProduct(), UTF8));
871                         final IdentifierMatch match = new IdentifierMatch(vs, url, IdentifierConfidence.BROAD_MATCH, conf);
872                         collected.add(match);
873                     } else if (evVer.equals(dbVer)) {
874                         addExactMatch(vs, evBaseVerUpdate, conf, collected);
875                     } else if (evBaseVer != null && evBaseVer.equals(dbVer)
876                             && (bestGuessConf == null || bestGuessConf.compareTo(conf) > 0)) {
877                         bestGuessConf = conf;
878                         bestGuess = dbVer;
879                         bestGuessUpdate = evBaseVerUpdate;
880                         bestGuessURL = String.format(NVD_SEARCH_URL, URLEncoder.encode(vs.getVendor(), UTF8),
881                                 URLEncoder.encode(vs.getProduct(), UTF8), URLEncoder.encode(vs.getVersion(), UTF8));
882                     } else if (dbVerUpdate != null && evVer.getVersionParts().size() <= dbVerUpdate.getVersionParts().size()
883                             && evVer.matchesAtLeastThreeLevels(dbVerUpdate)) {
884                         if (bestGuessConf == null || bestGuessConf.compareTo(conf) > 0) {
885                             if (bestGuess.getVersionParts().size() < dbVer.getVersionParts().size()) {
886                                 bestGuess = dbVer;
887                                 bestGuessUpdate = evBaseVerUpdate;
888                                 bestGuessConf = conf;
889                             }
890                         }
891                     }
892                 }
893                 if ((bestGuessConf == null || bestGuessConf.compareTo(conf) > 0)
894                         && bestGuess.getVersionParts().size() < evVer.getVersionParts().size()) {
895                     bestGuess = evVer;
896                     bestGuessUpdate = evBaseVerUpdate;
897                     bestGuessConf = conf;
898                 }
899             }
900         }
901 
902         cpeBuilder.part(Part.APPLICATION).vendor(vendor).product(product);
903         final int idx = bestGuess.getVersionParts().size() - 1;
904         if (bestGuess.getVersionParts().get(idx)
905                 .matches("^(v|release|final|snapshot|beta|alpha|u|rc|m|20\\d\\d).*$")) {
906             cpeBuilder.version(StringUtils.join(bestGuess.getVersionParts().subList(0, idx), "."));
907             //when written - no update versions in the NVD start with v### - they all strip the v off
908             if (bestGuess.getVersionParts().get(idx).matches("^v\\d.*$")) {
909                 cpeBuilder.update(bestGuess.getVersionParts().get(idx).substring(1));
910             } else {
911                 cpeBuilder.update(bestGuess.getVersionParts().get(idx));
912             }
913         } else {
914             cpeBuilder.version(bestGuess.toString());
915             if (bestGuessUpdate != null) {
916                 cpeBuilder.update(bestGuessUpdate);
917             }
918         }
919         final Cpe guessCpe;
920 
921         try {
922             guessCpe = cpeBuilder.build();
923         } catch (CpeValidationException ex) {
924             throw new AnalysisException(String.format("Unable to create a CPE for %s:%s:%s", vendor, product, bestGuess.toString()));
925         }
926         if (!"-".equals(guessCpe.getVersion())) {
927             String url = null;
928             if (bestGuessURL != null) {
929                 url = bestGuessURL;
930             }
931             if (bestGuessConf == null) {
932                 bestGuessConf = Confidence.LOW;
933             }
934             final IdentifierMatch match = new IdentifierMatch(guessCpe, url, IdentifierConfidence.BEST_GUESS, bestGuessConf);
935 
936             collected.add(match);
937         }
938         boolean identifierAdded = false;
939         if (!collected.isEmpty()) {
940             final List<IdentifierMatch> items = new ArrayList<>(collected);
941 
942             Collections.sort(items);
943             final IdentifierConfidence bestIdentifierQuality = items.get(0).getIdentifierConfidence();
944             final Confidence bestEvidenceQuality = items.get(0).getEvidenceConfidence();
945             boolean addedNonGuess = false;
946             final Confidence prevAddedConfidence = dependency.getVulnerableSoftwareIdentifiers().stream().map(id -> id.getConfidence())
947                     .min(Comparator.comparing(Confidence::ordinal))
948                     .orElse(Confidence.LOW);
949 
950             for (IdentifierMatch m : items) {
951                 if (bestIdentifierQuality.equals(m.getIdentifierConfidence())
952                         && bestEvidenceQuality.equals(m.getEvidenceConfidence())) {
953                     final CpeIdentifier i = m.getIdentifier();
954                     if (bestIdentifierQuality == IdentifierConfidence.BEST_GUESS) {
955                         if (addedNonGuess) {
956                             continue;
957                         }
958                         i.setConfidence(Confidence.LOW);
959                     } else {
960                         i.setConfidence(bestEvidenceQuality);
961                     }
962                     if (prevAddedConfidence.compareTo(i.getConfidence()) < 0) {
963                         continue;
964                     }
965 
966                     //TODO - while this gets the job down it is slow; consider refactoring
967                     dependency.addVulnerableSoftwareIdentifier(i);
968                     suppression.analyze(dependency, null);
969                     if (dependency.getVulnerableSoftwareIdentifiers().contains(i)) {
970                         identifierAdded = true;
971                         if (!addedNonGuess && bestIdentifierQuality != IdentifierConfidence.BEST_GUESS) {
972                             addedNonGuess = true;
973                         }
974                     }
975                 }
976             }
977         }
978         return identifierAdded;
979     }

984      * @param vs a reference to the vulnerable software
985      * @param updateVersion the update version
986      * @param conf the current confidence
987      * @param collected a reference to the collected identifiers

990     private void addExactMatch(Cpe vs, String updateVersion, Confidence conf,
991             final Set<IdentifierMatch> collected) throws UnsupportedEncodingException {
992

994         final String url = String.format(NVD_SEARCH_URL, URLEncoder.encode(vs.getVendor(), UTF8),
995                 URLEncoder.encode(vs.getProduct(), UTF8), URLEncoder.encode(vs.getVersion(), UTF8));

997         if (updateVersion != null && "*".equals(vs.getUpdate())) {

999                 useCpe = cpeBuilder.part(vs.getPart()).wfVendor(vs.getWellFormedVendor())
1000                         .wfProduct(vs.getWellFormedProduct()).wfVersion(vs.getWellFormedVersion())
1001                         .wfEdition(vs.getWellFormedEdition()).wfLanguage(vs.getWellFormedLanguage())

1003                         .update(updateVersion).build();

1006                 useCpe = vs;
1007             }
1008         } else {
1009             useCpe = vs;
1010         }
1011         final IdentifierMatch match = new IdentifierMatch(useCpe, url, IdentifierConfidence.EXACT_MATCH, conf);
1012         collected.add(match);

1015     /**
1016      * Evaluates whether or not to use the `version` of the dependency instead
1017      * of the version evidence. The dependency should not always be used as it
1018      * can cause FP.
1019      *
1020      * @param dependency the dependency being analyzed
1021      * @param product the product name
1022      * @param vendor the vendor name
1023      * @param confidence the current confidence level
1024      * @param collected a reference to the identifiers matched
1025      * @param bestGuess the current best guess as to the dependency version
1026      * @throws AnalysisException thrown if aliens attacked and valid input could
1027      * not be used to construct a CPE
1028      * @throws UnsupportedEncodingException thrown if run on a system that
1029      * doesn't support UTF-8
1030      */
1031     private void considerDependencyVersion(Dependency dependency,
1032             String vendor, String product, Confidence confidence,
1033             final Set<IdentifierMatch> collected, DependencyVersion bestGuess)
1034             throws AnalysisException, UnsupportedEncodingException {
1035 
1036         if (dependency.getVersion() != null && !dependency.getVersion().isEmpty()) {
1037             final CpeBuilder cpeBuilder = new CpeBuilder();
1038             boolean useDependencyVersion = true;
1039             final CharArraySet stopWords = SearchFieldAnalyzer.getStopWords();
1040             if (dependency.getName() != null && !dependency.getName().isEmpty()) {
1041                 final String name = dependency.getName();
1042                 for (String word : product.split("[^a-zA-Z0-9]")) {
1043                     useDependencyVersion &= name.contains(word) || stopWords.contains(word);
1044                 }
1045             }
1046 
1047             if (useDependencyVersion) {
1048                 //TODO - we need to filter this so that we only use this if something in the
1049                 //dependency.getName() matches the vendor/product in some way
1050                 final DependencyVersion depVersion = new DependencyVersion(dependency.getVersion());
1051                 if (depVersion.getVersionParts().size() > 0) {
1052                     cpeBuilder.part(Part.APPLICATION).vendor(vendor).product(product);
1053                     addVersionAndUpdate(depVersion, cpeBuilder);
1054                     try {
1055                         final Cpe depCpe = cpeBuilder.build();
1056                         final String url = String.format(NVD_SEARCH_URL, URLEncoder.encode(vendor, UTF8),
1057                                 URLEncoder.encode(product, UTF8), URLEncoder.encode(depCpe.getVersion(), UTF8));
1058                         final IdentifierMatch match = new IdentifierMatch(depCpe, url, IdentifierConfidence.EXACT_MATCH, confidence);
1059                         collected.add(match);
1060                     } catch (CpeValidationException ex) {
1061                         throw new AnalysisException(String.format("Unable to create a CPE for %s:%s:%s", vendor, product, bestGuess.toString()));
1062                     }
1063                 }
1064             }
1065         }
1066     }

1306             final CPEAnalyzer analyzer = new CPEAnalyzer();
1307             analyzer.initialize(props);
1308             analyzer.prepareAnalyzer(en);
1309             LOGGER.error("test");
1310             System.out.println("Memory index query for ODC");
1311             try (BufferedReader br = new BufferedReader(new InputStreamReader(System.in, StandardCharsets.UTF_8))) {
1312                 while (true) {
1313 
1314                     final Map<String, MutableInt> vendor = new HashMap<>();
1315                     final Map<String, MutableInt> product = new HashMap<>();
1316                     System.out.print("Vendor: ");
1317                     String[] parts = br.readLine().split(" ");
1318                     for (String term : parts) {
1319                         final MutableInt count = vendor.get(term);
1320                         if (count == null) {
1321                             vendor.put(term, new MutableInt(0));
1322                         } else {
1323                             count.add(1);
1324                         }
1325                     }
1326                     System.out.print("Product: ");
1327                     parts = br.readLine().split(" ");
1328                     for (String term : parts) {
1329                         final MutableInt count = product.get(term);
1330                         if (count == null) {
1331                             product.put(term, new MutableInt(0));
1332                         } else {
1333                             count.add(1);
1334                         }
1335                     }
1336                     final List<IndexEntry> list = analyzer.searchCPE(vendor, product, new HashSet<>(), new HashSet<>(), "default");
1337                     if (list == null || list.isEmpty()) {
1338                         System.out.println("No results found");
1339                     } else {
1340                         list.forEach((e) -> System.out.println(String.format("%s:%s (%f)", e.getVendor(), e.getProduct(),

1344                     System.out.println();
1345                 }
1346             }
