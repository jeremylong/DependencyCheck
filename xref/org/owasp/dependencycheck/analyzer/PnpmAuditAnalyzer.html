74      /**
75       * Analyzes the pnpm lock file to determine vulnerable dependencies. Uses
76       * pnpm audit --json to vulnerabilities report from NPM API.
77       *
78       * @param dependency the pnpm lock file
79       * @param engine the analysis engine
80       * @throws AnalysisException thrown if there is an error analyzing the file
81       */
82      @Override
83      protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
84          if (dependency.getDisplayFileName().equals(dependency.getFileName())) {
85              engine.removeDependency(dependency);
86          }
87          final File packageLock = dependency.getActualFile();
88          if (!packageLock.isFile() || packageLock.length() == 0 || !shouldProcess(packageLock)) {
89              return;
90          }
91          final List<Advisory> advisories;
92          final MultiValuedMap<String, String> dependencyMap = new HashSetValuedHashMap<>();
93          advisories = analyzePackage(packageLock, dependency);
94          try {
95              processResults(advisories, engine, dependency, dependencyMap);
96          } catch (CpeValidationException ex) {
97              throw new UnexpectedAnalysisException(ex);
98          }
99      }

121     /**
122      * Initializes the analyzer once before any analysis is performed.
123      *
124      * @param engine a reference to the dependency-check engine
125      * @throws InitializationException if there's an error during initialization
126      */
127     @Override
128     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
129         super.prepareFileTypeAnalyzer(engine);
130         if (!isEnabled()) {
131             LOGGER.debug("{} Analyzer is disabled skipping pnpm executable check", getName());
132             return;
133         }
134         final List<String> args = new ArrayList<>();
135         args.add(getPnpm());
136         args.add("--help");
137         final ProcessBuilder builder = new ProcessBuilder(args);
138         LOGGER.debug("Launching: {}", args);
139         try {
140             final Process process = builder.start();
141             try (ProcessReader processReader = new ProcessReader(process)) {
142                 processReader.readAll();
143                 final int exitValue = process.waitFor();
144                 final int expectedExitValue = 0;
145                 final int executableNotFoundExitValue = 127;
146                 switch (exitValue) {
147                     case expectedExitValue:
148                         LOGGER.debug("{} is enabled.", getName());
149                         break;
150                     case executableNotFoundExitValue:
151                     default:
152                         this.setEnabled(false);
153                         LOGGER.warn("The {} has been disabled. Pnpm executable was not found.", getName());
154                 }
155             }
156         } catch (Exception ex) {
157             this.setEnabled(false);
158             LOGGER.warn("The {} has been disabled. Pnpm executable was not found.", getName());
159             throw new InitializationException("Unable to read pnpm audit output.", ex);
160         }
161     }

163     /**
164      * Attempts to determine the path to `pnpm`.
165      *
166      * @return the path to `pnpm`
167      */
168     private String getPnpm() {
169         final String value;
170         synchronized (this) {
171             if (pnpmPath == null) {
172                 final String path = getSettings().getString(Settings.KEYS.ANALYZER_PNPM_PATH);
173                 if (path == null) {
174                     pnpmPath = "pnpm";
175                 } else {
176                     final File pnpmFile = new File(path);
177                     if (pnpmFile.isFile()) {
178                         pnpmPath = pnpmFile.getAbsolutePath();
179                     } else {
180                         LOGGER.warn("Provided path to `pnpm` executable is invalid.");
181                         pnpmPath = "pnpm";
182                     }
183                 }
184             }
185             value = pnpmPath;
186         }
187         return value;
188     }

190     private JSONObject fetchPnpmAuditJson(Dependency dependency, boolean skipDevDependencies) throws AnalysisException {
191         final File folder = dependency.getActualFile().getParentFile();
192         if (!folder.isDirectory()) {
193             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
194         }
195         try {
196             final List<String> args = new ArrayList<>();
197 
198             args.add(getPnpm());
199             args.add("audit");
200             if (skipDevDependencies) {
201                 args.add("--prod");
202             }
203             // pnpm audit returns a json compliant with NpmAuditParser
204             args.add("--json");
205             // ensure we are using the right registry despite .npmrc
206             args.add("--registry");
207             args.add("https://registry.npmjs.org/");
208             final ProcessBuilder builder = new ProcessBuilder(args);
209             builder.directory(folder);
210             // Workaround 64k limitation of InputStream, redirect stdout to a file that we will read later
211             // instead of reading directly stdout from Process's InputStream which is topped at 64k
212             final File tmpFile = File.createTempFile("pnpm_audit", null);
213             builder.redirectOutput(tmpFile);
214             LOGGER.debug("Launching: {}", args);
215             final Process process = builder.start();
216             try (ProcessReader processReader = new ProcessReader(process)) {
217                 processReader.readAll();
218                 final String errOutput = processReader.getError();
219                 if (!StringUtils.isBlank(errOutput)) {
220                     LOGGER.error("Process error output: {}", errOutput);
221                 }
222                 String verboseJson = FileUtils.readFileToString(tmpFile, StandardCharsets.UTF_8);
223                 // Workaround implicit creation of .pnpm-debug.log, see https://github.com/pnpm/pnpm/issues/3832
224                 // affects usage of docker container to analyze mounted directories without privileges
225                 if (verboseJson.contains("EACCES: permission denied, open 'node_modules/.pnpm-debug.log'")) {
226                     verboseJson = verboseJson.substring(0, verboseJson.indexOf("EACCES: permission denied, open 'node_modules/.pnpm-debug.log'"));
227                 }
228                 LOGGER.debug("Audit report: {}", verboseJson);
229                 return new JSONObject(verboseJson);
230             } catch (InterruptedException ex) {
231                 Thread.currentThread().interrupt();
232                 throw new AnalysisException("Pnpm audit process was interrupted.", ex);
233             } catch (JSONException e) {
234                 Thread.currentThread().interrupt();
235                 throw new AnalysisException("Pnpm audit returned an invalid response.", e);
236             } finally {
237                 if (!tmpFile.delete()) {
238                     LOGGER.debug("Unable to delete temp file: {}", tmpFile);
239                 }
240             }
241         } catch (IOException ioe) {
242             throw new AnalysisException("pnpm audit failure; this error can be ignored if you are not analyzing projects with a pnpm lockfile.", ioe);
243         }
244     }

246     /**
247      * Analyzes the package and pnpm lock files by extracting dependency
248      * information, creating a payload to submit to the npm audit API,
249      * submitting the payload, and returning the identified advisories.
250      *
251      * @param lockFile a reference to the pnpm-lock.yaml
252      * @param dependency a reference to the dependency-object for the
253      * pnpm-lock.yaml
254      * @return a list of advisories
255      * @throws AnalysisException thrown when there is an error creating or
256      * submitting the npm audit API payload
257      */
258     private List<Advisory> analyzePackage(final File lockFile,
259             Dependency dependency)
260             throws AnalysisException {
261         try {
262             final boolean skipDevDependencies = getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_SKIPDEV, false);
263 
264             // Use pnpm directly to fetch audit.json
265             // Retrieves the contents of package-lock.json from the Dependency
266             final JSONObject auditJson = fetchPnpmAuditJson(dependency, skipDevDependencies);
267             // Submits the package payload to the nsp check service
268             return getAuditParser().parse(auditJson);
269 
270         } catch (JSONException e) {
271             throw new AnalysisException(String.format("Failed to parse %s file from the NPM Audit API "
272                     + "(PnpmAuditAnalyzer).", lockFile.getPath()), e);
273         } catch (SearchException ex) {
274             LOGGER.error("PnpmAuditAnalyzer failed on {}", dependency.getActualFilePath());
275             throw ex;
276         }
277     }
