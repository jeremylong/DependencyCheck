29  /**
30   * <p>
31   * This analyzer will merge dependencies, created from different source, into a
32   * single dependency.</p>
33   *
34   * @author Jeremy Long
35   */
36  public class DependencyMergingAnalyzer extends AbstractDependencyComparingAnalyzer {

82      /**
83       * Evaluates the dependencies
84       *
85       * @param dependency a dependency to compare
86       * @param nextDependency a dependency to compare
87       * @param dependenciesToRemove a set of dependencies that will be removed
88       * @return true if a dependency is removed; otherwise false
89       */
90      @Override
91      @SuppressWarnings("ReferenceEquality")
92      protected boolean evaluateDependencies(final Dependency dependency, final Dependency nextDependency, final Set<Dependency> dependenciesToRemove) {
93          Dependency main;
94          //CSOFF: InnerAssignment
95          if ((main = getMainGemspecDependency(dependency, nextDependency)) != null) {
96              if (main == dependency) {
97                  mergeDependencies(dependency, nextDependency, dependenciesToRemove);
98              } else {
99                  mergeDependencies(nextDependency, dependency, dependenciesToRemove);
100                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
101             }
102         } else if ((main = getMainSwiftDependency(dependency, nextDependency)) != null) {
103             if (main == dependency) {
104                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
105             } else {
106                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
107                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
108             }
109         } else if ((main = getMainAndroidDependency(dependency, nextDependency)) != null) {
110             if (main == dependency) {
111                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
112             } else {
113                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
114                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
115             }
116         } else if ((main = getMainDotnetDependency(dependency, nextDependency)) != null) {
117             if (main == dependency) {
118                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
119             } else {
120                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
121                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
122             }
123         }
124         //CSON: InnerAssignment
125         return false;
126     }

128     /**
129      * Adds the relatedDependency to the dependency's related dependencies.
130      *
131      * @param dependency the main dependency
132      * @param relatedDependency a collection of dependencies to be removed from
133      * the main analysis loop, this is the source of dependencies to remove
134      * @param dependenciesToRemove a collection of dependencies that will be
135      * removed from the main analysis loop, this function adds to this
136      * collection
137      */
138     public static void mergeDependencies(final Dependency dependency, final Dependency relatedDependency,
139             final Set<Dependency> dependenciesToRemove) {
140         LOGGER.debug("Merging '{}' into '{}'", relatedDependency.getFilePath(), dependency.getFilePath());
141         dependency.addRelatedDependency(relatedDependency);
142         relatedDependency.getEvidence(EvidenceType.VENDOR).forEach((e) -> dependency.addEvidence(EvidenceType.VENDOR, e));
143         relatedDependency.getEvidence(EvidenceType.PRODUCT).forEach((e) -> dependency.addEvidence(EvidenceType.PRODUCT, e));
144         relatedDependency.getEvidence(EvidenceType.VERSION).forEach((e) -> dependency.addEvidence(EvidenceType.VERSION, e));
145 
146         relatedDependency.getRelatedDependencies().stream()
147                 .peek(dependency::addRelatedDependency)
148                 .forEach((d) -> relatedDependency.removeRelatedDependencies(d));
149         dependency.addAllProjectReferences(relatedDependency.getProjectReferences());
150         if (dependenciesToRemove != null) {
151             dependenciesToRemove.add(relatedDependency);
152         }
153     }

155     /**
156      * Bundling Ruby gems that are identified from different .gemspec files but
157      * denote the same package path. This happens when Ruby bundler installs an
158      * application's dependencies by running "bundle install".
159      *
160      * @param dependency1 dependency to compare
161      * @param dependency2 dependency to compare
162      * @return true if the the dependencies being analyzed appear to be the
163      * same; otherwise false
164      */
165     protected boolean isSameRubyGem(Dependency dependency1, Dependency dependency2) {
166         if (dependency1 == null || dependency2 == null
167                 || !dependency1.getFileName().endsWith(".gemspec")
168                 || !dependency2.getFileName().endsWith(".gemspec")
169                 || dependency1.getPackagePath() == null
170                 || dependency2.getPackagePath() == null) {
171             return false;
172         }
173         return dependency1.getPackagePath().equalsIgnoreCase(dependency2.getPackagePath());
174     }
175 
176     /**
177      * Ruby gems installed by "bundle install" can have zero or more *.gemspec
178      * files, all of which have the same packagePath and should be grouped. If
179      * one of these gemspec is from &lt;parent&gt;/specifications/*.gemspec,
180      * because it is a stub with fully resolved gem meta-data created by Ruby
181      * bundler, this dependency should be the main one. Otherwise, use
182      * dependency2 as main.
183      *
184      * This method returns null if any dependency is not from *.gemspec, or the
185      * two do not have the same packagePath. In this case, they should not be
186      * grouped.
187      *
188      * @param dependency1 dependency to compare
189      * @param dependency2 dependency to compare
190      * @return the main dependency; or null if a gemspec is not included in the
191      * analysis
192      */
193     protected Dependency getMainGemspecDependency(Dependency dependency1, Dependency dependency2) {
194         if (isSameRubyGem(dependency1, dependency2)) {
195             final File lFile = dependency1.getActualFile();
196             final File left = lFile.getParentFile();
197             if (left != null && left.getName().equalsIgnoreCase("specifications")) {
198                 return dependency1;
199             }
200             return dependency2;
201         }
202         return null;
203     }

205     /**
206      * Bundling same swift dependencies with the same packagePath but identified
207      * by different file type analyzers.
208      *
209      * @param dependency1 dependency to test
210      * @param dependency2 dependency to test
211      * @return <code>true</code> if the dependencies appear to be the same;
212      * otherwise <code>false</code>
213      */
214     protected boolean isSameSwiftPackage(Dependency dependency1, Dependency dependency2) {
215         if (dependency1 == null || dependency2 == null
216                 || (!dependency1.getFileName().endsWith(".podspec")
217                 && !dependency1.getFileName().equals("Package.swift"))
218                 || (!dependency2.getFileName().endsWith(".podspec")
219                 && !dependency2.getFileName().equals("Package.swift"))
220                 || dependency1.getPackagePath() == null
221                 || dependency2.getPackagePath() == null) {
222             return false;
223         }
224         return dependency1.getPackagePath().equalsIgnoreCase(dependency2.getPackagePath());
225     }
226 
227     /**
228      * Determines which of the swift dependencies should be considered the
229      * primary.
230      *
231      * @param dependency1 the first swift dependency to compare
232      * @param dependency2 the second swift dependency to compare
233      * @return the primary swift dependency
234      */
235     protected Dependency getMainSwiftDependency(Dependency dependency1, Dependency dependency2) {
236         if (isSameSwiftPackage(dependency1, dependency2)) {
237             if (dependency1.getFileName().endsWith(".podspec")) {
238                 return dependency1;
239             }
240             return dependency2;
241         }
242         return null;
243     }
244 
245     /**
246      * Determines which of the android dependencies should be considered the
247      * primary.
248      *
249      * @param dependency1 the first android dependency to compare
250      * @param dependency2 the second android dependency to compare
251      * @return the primary swift dependency
252      */
253     protected Dependency getMainAndroidDependency(Dependency dependency1, Dependency dependency2) {
254         if (dependency1.isVirtual() || dependency2.isVirtual()) {
255             return null;
256         }
257         if ("classes.jar".equals(dependency2.getActualFile().getName())
258                 && "aar".equals(FileUtils.getFileExtension(dependency1.getActualFile().getName()))
259                 && dependency2.getFileName().contains(dependency1.getActualFile().getName())) {
260             return dependency1;
261         }
262         if ("classes.jar".equals(dependency1.getActualFile().getName())
263                 && "aar".equals(FileUtils.getFileExtension(dependency2.getActualFile().getName()))
264                 && dependency1.getFileName().contains(dependency2.getActualFile().getName())) {
265             return dependency2;
266         }
267         return null;
268     }
269 
270     /**
271      * Determines which of the dotnet dependencies should be considered the
272      * primary.
273      *
274      * @param dependency1 the first dotnet dependency to compare
275      * @param dependency2 the second dotnet dependency to compare
276      * @return the primary swift dependency
277      */
278     protected Dependency getMainDotnetDependency(Dependency dependency1, Dependency dependency2) {
279         if (dependency1.getName() != null && dependency1.getVersion() != null
280                 && dependency2.getName() != null && dependency2.getVersion() != null
281                 && (AssemblyAnalyzer.DEPENDENCY_ECOSYSTEM.equals(dependency1.getEcosystem())
282                 || NugetconfAnalyzer.DEPENDENCY_ECOSYSTEM.equals(dependency1.getEcosystem()))
283                 && (AssemblyAnalyzer.DEPENDENCY_ECOSYSTEM.equals(dependency2.getEcosystem())
284                 || NugetconfAnalyzer.DEPENDENCY_ECOSYSTEM.equals(dependency2.getEcosystem()))) {
285             if (dependency1.getName().equals(dependency2.getName()) && dependency1.getVersion().equals(dependency2.getVersion())) {
286                 if (dependency2.isVirtual()) {
287                     return dependency2;
288                 }
289                 return dependency1;
290             }
291         }
292         return null;
293     }
