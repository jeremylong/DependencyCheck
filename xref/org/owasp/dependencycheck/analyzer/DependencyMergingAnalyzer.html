32   * This analyzer will merge dependencies, created from different source, into a
33   * single dependency.</p>

83      /**
84       * Evaluates the dependencies
85       *
86       * @param dependency a dependency to compare
87       * @param nextDependency a dependency to compare
88       * @param dependenciesToRemove a set of dependencies that will be removed
89       * @return true if a dependency is removed; otherwise false
90       */
91      @Override
92      @SuppressWarnings("ReferenceEquality")
93      protected boolean evaluateDependencies(final Dependency dependency, final Dependency nextDependency, final Set<Dependency> dependenciesToRemove) {
94          Dependency main;
95          //CSOFF: InnerAssignment
96          if ((main = getMainGemspecDependency(dependency, nextDependency)) != null) {
97              if (main == dependency) {
98                  mergeDependencies(dependency, nextDependency, dependenciesToRemove);
99              } else {
100                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
101                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
102             }
103         } else if ((main = getMainSwiftDependency(dependency, nextDependency)) != null) {
104             if (main == dependency) {
105                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
106             } else {
107                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
108                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
109             }
110         } else if ((main = getMainAndroidDependency(dependency, nextDependency)) != null) {
111             if (main == dependency) {
112                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
113             } else {
114                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
115                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
116             }
117         } else if ((main = getMainDotnetDependency(dependency, nextDependency)) != null) {
118             if (main == dependency) {
119                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
120             } else {
121                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
122                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
123             }
124         } else if ((main = getMainVirtualDependency(dependency, nextDependency)) != null) {
125             if (main == dependency) {
126                 mergeDependencies(dependency, nextDependency, dependenciesToRemove);
127             } else {
128                 mergeDependencies(nextDependency, dependency, dependenciesToRemove);
129                 return true; //since we merged into the next dependency - skip forward to the next in mainIterator
130             }
131         }
132         //CSON: InnerAssignment
133         return false;
134     }

136     /**
137      * Adds the relatedDependency to the dependency's related dependencies.
138      *
139      * @param dependency the main dependency
140      * @param relatedDependency a collection of dependencies to be removed from
141      * the main analysis loop, this is the source of dependencies to remove
142      * @param dependenciesToRemove a collection of dependencies that will be
143      * removed from the main analysis loop, this function adds to this
144      * collection
145      */
146     public static void mergeDependencies(final Dependency dependency, final Dependency relatedDependency,
147             final Set<Dependency> dependenciesToRemove) {
148         synchronized (dependency) {
149             LOGGER.debug("Merging '{}' into '{}'", relatedDependency.getFilePath(), dependency.getFilePath());
150             dependency.addRelatedDependency(relatedDependency);
151             relatedDependency.getEvidence(EvidenceType.VENDOR).forEach((e) -> dependency.addEvidence(EvidenceType.VENDOR, e));
152             relatedDependency.getEvidence(EvidenceType.PRODUCT).forEach((e) -> dependency.addEvidence(EvidenceType.PRODUCT, e));
153             relatedDependency.getEvidence(EvidenceType.VERSION).forEach((e) -> dependency.addEvidence(EvidenceType.VERSION, e));
154 
155             relatedDependency.getRelatedDependencies()
156                     .forEach(dependency::addRelatedDependency);
157             relatedDependency.clearRelatedDependencies();
158             dependency.addAllProjectReferences(relatedDependency.getProjectReferences());
159             if (dependenciesToRemove != null) {
160                 dependenciesToRemove.add(relatedDependency);
161             }
162         }
163     }

165     /**
166      * Bundling Ruby gems that are identified from different .gemspec files but
167      * denote the same package path. This happens when Ruby bundler installs an
168      * application's dependencies by running "bundle install".
169      *
170      * @param dependency1 dependency to compare
171      * @param dependency2 dependency to compare
172      * @return true if the the dependencies being analyzed appear to be the
173      * same; otherwise false
174      */
175     protected boolean isSameRubyGem(Dependency dependency1, Dependency dependency2) {
176         if (dependency1 == null || dependency2 == null
177                 || !dependency1.getFileName().endsWith(".gemspec")
178                 || !dependency2.getFileName().endsWith(".gemspec")
179                 || dependency1.getPackagePath() == null
180                 || dependency2.getPackagePath() == null) {
181             return false;
182         }
183         return dependency1.getPackagePath().equalsIgnoreCase(dependency2.getPackagePath());
184     }
185 
186     /**
187      * Ruby gems installed by "bundle install" can have zero or more *.gemspec
188      * files, all of which have the same packagePath and should be grouped. If
189      * one of these gemspec is from &lt;parent&gt;/specifications/*.gemspec,
190      * because it is a stub with fully resolved gem meta-data created by Ruby
191      * bundler, this dependency should be the main one. Otherwise, use
192      * dependency2 as main.
193      *
194      * This method returns null if any dependency is not from *.gemspec, or the
195      * two do not have the same packagePath. In this case, they should not be
196      * grouped.
197      *
198      * @param dependency1 dependency to compare
199      * @param dependency2 dependency to compare
200      * @return the main dependency; or null if a gemspec is not included in the
201      * analysis
202      */
203     protected Dependency getMainGemspecDependency(Dependency dependency1, Dependency dependency2) {
204         if (dependency1 != null && dependency2 != null
205                 && Ecosystem.RUBY.equals(dependency1.getEcosystem())
206                 && Ecosystem.RUBY.equals(dependency2.getEcosystem())
207                 && isSameRubyGem(dependency1, dependency2)) {
208             final File lFile = dependency1.getActualFile();
209             final File left = lFile.getParentFile();
210             if (left != null && left.getName().equalsIgnoreCase("specifications")) {
211                 return dependency1;
212             }
213             return dependency2;
214         }
215         return null;
216     }

218     /**
219      * Bundling same swift dependencies with the same packagePath but identified
220      * by different file type analyzers.
221      *
222      * @param dependency1 dependency to test
223      * @param dependency2 dependency to test
224      * @return <code>true</code> if the dependencies appear to be the same;
225      * otherwise <code>false</code>
226      */
227     protected boolean isSameSwiftPackage(Dependency dependency1, Dependency dependency2) {
228         if (dependency1 == null || dependency2 == null
229                 || (!dependency1.getFileName().endsWith(".podspec")
230                 && !dependency1.getFileName().equals("Package.swift"))
231                 || (!dependency2.getFileName().endsWith(".podspec")
232                 && !dependency2.getFileName().equals("Package.swift"))
233                 || dependency1.getPackagePath() == null
234                 || dependency2.getPackagePath() == null) {
235             return false;
236         }
237         return dependency1.getPackagePath().equalsIgnoreCase(dependency2.getPackagePath());
238     }

240     /**
241      * Determines which of the swift dependencies should be considered the
242      * primary.
243      *
244      * @param dependency1 the first swift dependency to compare
245      * @param dependency2 the second swift dependency to compare
246      * @return the primary swift dependency
247      */
248     protected Dependency getMainSwiftDependency(Dependency dependency1, Dependency dependency2) {
249         if (dependency1 != null && dependency2 != null
250                 && Ecosystem.IOS.equals(dependency1.getEcosystem())
251                 && Ecosystem.IOS.equals(dependency2.getEcosystem())
252                 && isSameSwiftPackage(dependency1, dependency2)) {
253             if (dependency1.getFileName().endsWith(".podspec")) {
254                 return dependency1;
255             }
256             return dependency2;
257         }
258         return null;
259     }
260 
261     /**
262      * Determines which of the android dependencies should be considered the
263      * primary.
264      *
265      * @param dependency1 the first android dependency to compare
266      * @param dependency2 the second android dependency to compare
267      * @return the primary swift dependency
268      */
269     protected Dependency getMainAndroidDependency(Dependency dependency1, Dependency dependency2) {
270         if (!dependency1.isVirtual()
271                 && !dependency2.isVirtual()
272                 && Ecosystem.JAVA.equals(dependency1.getEcosystem())
273                 && Ecosystem.JAVA.equals(dependency2.getEcosystem())) {
274             final String name1 = dependency1.getActualFile().getName();
275             final String name2 = dependency2.getActualFile().getName();
276             if ("classes.jar".equals(name2)
277                     && "aar".equals(FileUtils.getFileExtension(name1))
278                     && dependency2.getFileName().contains(name1)) {
279                 return dependency1;
280             }
281             if ("classes.jar".equals(name1)
282                     && "aar".equals(FileUtils.getFileExtension(name2))
283                     && dependency1.getFileName().contains(name2)) {
284                 return dependency2;
285             }
286         }
287         return null;
288     }

290     /**
291      * Determines which of the dotnet dependencies should be considered the
292      * primary.
293      *
294      * @param dependency1 the first dotnet dependency to compare
295      * @param dependency2 the second dotnet dependency to compare
296      * @return the primary swift dependency
297      */
298     protected Dependency getMainDotnetDependency(Dependency dependency1, Dependency dependency2) {
299         if (dependency1.getName() != null && dependency1.getVersion() != null
300                 && dependency2.getName() != null && dependency2.getVersion() != null
301                 && Ecosystem.DOTNET.equals(dependency1.getEcosystem())
302                 && Ecosystem.DOTNET.equals(dependency2.getEcosystem())
303                 && dependency1.getName().equals(dependency2.getName())
304                 && dependency1.getVersion().equals(dependency2.getVersion())) {
305             if (dependency1.isVirtual()) {
306                 return dependency2;
307             }
308             return dependency1;
309         }
310         return null;
311     }
312 
313     /**
314      * Determines which of the virtual dependencies should be considered the
315      * primary.
316      *
317      * @param dependency1 the first virtual dependency to compare
318      * @param dependency2 the second virtual dependency to compare
319      *
320      * @return the first virtual dependency (or {code null} if they are not to
321      * be considered mergeable virtual dependencies)
322      */
323     protected Dependency getMainVirtualDependency(Dependency dependency1, Dependency dependency2) {
324         if (dependency1.isVirtual() && dependency2.isVirtual()
325                 && dependency1.getName() != null && dependency2.getName() != null
326                 && dependency1.getVersion() != null && dependency2.getVersion() != null
327                 && dependency1.getActualFilePath() != null && dependency2.getActualFilePath() != null
328                 && dependency1.getName().equals(dependency2.getName())
329                 && dependency1.getVersion().equals(dependency2.getVersion())
330                 && dependency1.getActualFilePath().equals(dependency2.getActualFilePath())) {
331             return dependency1;
332         }
333         return null;
334     }
