55  public class GolangDepAnalyzer extends AbstractFileTypeAnalyzer {

57      /**
58       * The logger.
59       */
60      private static final Logger LOGGER = LoggerFactory.getLogger(GolangDepAnalyzer.class);
61      /**
62       * A descriptor for the type of dependencies processed or added by this
63       * analyzer.
64       */
65      public static final String DEPENDENCY_ECOSYSTEM = Ecosystem.GOLANG;
66  
67      /**
68       * Lock file name.
69       */
70      private static final String GOPKG_LOCK = "Gopkg.lock";
71  
72      /**
73       * The file filter for Gopkg.lock
74       */
75      private static final FileFilter GOPKG_LOCK_FILTER = FileFilterBuilder.newInstance()
76              .addFilenames(GOPKG_LOCK)
77              .build();

131     /**
132      * Analyzes go packages and adds evidence to the dependency.
133      *
134      * @param dependency the dependency being analyzed
135      * @param engine the engine being used to perform the scan
136      *
137      * @throws AnalysisException thrown if there is an unrecoverable error
138      * analyzing the dependency
139      */
140     @Override
141     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
142         //do not report on the build file itself
143         engine.removeDependency(dependency);
144 
145         final Toml result = new Toml().read(dependency.getActualFile());
146         final List<Toml> projectsLocks = result.getTables("projects");
147         if (projectsLocks == null) {
148             return;
149         }
150         projectsLocks.forEach((project) -> {
151             final String name = project.getString("name");
152             final String version = project.getString("version");
153             final String revision = project.getString("revision");
154 
155             Dependency dep = createDependency(dependency, name, version, revision, null);
156             engine.addDependency(dep);
157             final List<String> packages = project.getList("packages");
158             for (String pkg : packages) {
159                 if (StringUtils.isNotBlank(pkg) && !".".equals(pkg)) {
160                     dep = createDependency(dependency, name, version, revision, pkg);
161                     engine.addDependency(dep);
162                 }
163             }
164         });
165     }
166 
167     /**
168      * Builds a dependency object based on the given data.
169      *
170      * @param parentDependency a reference to the parent dependency
171      * @param name the name of the dependency
172      * @param version the version of the dependency
173      * @param revision the revision of the dependency
174      * @param subPath the sub-path of the dependency
175      * @return a new dependency object
176      */
177     private Dependency createDependency(Dependency parentDependency, String name, String version, String revision, String subPath) {
178         final Dependency dep = new Dependency(parentDependency.getActualFile(), true);
179         dep.setEcosystem(DEPENDENCY_ECOSYSTEM);
180         if (StringUtils.isNotBlank(subPath)) {
181             dep.setDisplayFileName(name + "/" + subPath);
182             dep.setName(name + "/" + subPath);
183         } else {
184             dep.setDisplayFileName(name);
185             dep.setName(name);
186         }
187         final PackageURLBuilder packageBuilder = PackageURLBuilder.aPackageURL().withType("golang");
188 
189         String baseNamespace = null;
190         String depNamespace = null;
191         String depName = null;
192         if (StringUtils.isNotBlank(name)) {
193             final int slashPos = name.indexOf("/");
194             if (slashPos > 0) {
195                 baseNamespace = name.substring(0, slashPos);
196                 final int lastSlash = name.lastIndexOf("/");
197                 depName = name.substring(lastSlash + 1);
198                 if (lastSlash != slashPos) {
199                     depNamespace = name.substring(slashPos + 1, lastSlash);
200                     dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "namespace", depNamespace, Confidence.HIGH);
201                     dep.addEvidence(EvidenceType.VENDOR, GOPKG_LOCK, "namespace", depNamespace, Confidence.HIGH);
202                     packageBuilder.withNamespace(baseNamespace + "/" + depNamespace);
203                 } else {
204                     packageBuilder.withNamespace(baseNamespace);
205                 }
206                 packageBuilder.withName(depName);
207                 if (!"golang.org".equals(baseNamespace)) {
208                     dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "namespace", baseNamespace, Confidence.LOW);
209                     dep.addEvidence(EvidenceType.VENDOR, GOPKG_LOCK, "namespace", baseNamespace, Confidence.LOW);
210                 }
211 
212                 dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "name", depName, Confidence.HIGHEST);
213                 dep.addEvidence(EvidenceType.VENDOR, GOPKG_LOCK, "name", depName, Confidence.HIGHEST);
214             } else {
215                 packageBuilder.withName(name);
216                 dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "namespace", name, Confidence.HIGHEST);
217                 dep.addEvidence(EvidenceType.VENDOR, GOPKG_LOCK, "namespace", name, Confidence.HIGHEST);
218             }
219         }
220         if (StringUtils.isNotBlank(version)) {
221             packageBuilder.withVersion(version);
222             dep.setVersion(version);
223             dep.addEvidence(EvidenceType.VERSION, GOPKG_LOCK, "version", version, Confidence.HIGHEST);
224         }
225         if (StringUtils.isNotBlank(revision)) {
226             if (version == null) {
227                 //this is used to help determine the actual version in the NVD - a commit hash doesn't work
228                 // instead we need to make it an asterik for the CPE...
229                 //dep.setVersion(revision);
230                 packageBuilder.withVersion(revision);
231             }
232             //Revision (which appears to be a commit hash) won't be of any value in the analysis.
233             //dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "revision", revision, Confidence.HIGHEST);
234         }
235         if (StringUtils.isNotBlank(subPath)) {
236             packageBuilder.withSubpath(subPath);
237             dep.addEvidence(EvidenceType.PRODUCT, GOPKG_LOCK, "package", subPath, Confidence.HIGH);
238             dep.addEvidence(EvidenceType.VENDOR, GOPKG_LOCK, "package", subPath, Confidence.MEDIUM);
239         }
240 
241         Identifier id;
242         try {
243             final PackageURL purl = packageBuilder.build();
244             id = new PurlIdentifier(purl, Confidence.HIGHEST);
245         } catch (MalformedPackageURLException ex) {
246             LOGGER.warn("Unable to create package-url identifier for `{}` in `{}` - reason: {}",
247                     name, parentDependency.getFilePath(), ex.getMessage());
248             final StringBuilder value = new StringBuilder(name);
249             if (StringUtils.isNotBlank(subPath)) {
250                 value.append("/").append(subPath);
251             }
252             if (StringUtils.isNotBlank(version)) {
253                 value.append("@").append(version);
254             }
255             id = new GenericIdentifier(value.toString(), Confidence.HIGH);
256         }
257         dep.addSoftwareIdentifier(id);
258         dep.setSha1sum(Checksum.getSHA1Checksum(id.toString()));
259         dep.setMd5sum(Checksum.getMD5Checksum(id.toString()));
260         dep.setSha256sum(Checksum.getSHA256Checksum(id.toString()));
261 
262         return dep;
263     }
