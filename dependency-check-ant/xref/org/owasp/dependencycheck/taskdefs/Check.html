506     /**
507      * If this is a reference, this method will add the referenced resource
508      * collection to the collection of paths.
509      *
510      * @throws BuildException if the reference is not to a resource collection
511      */
512     //declaring a throw that extends runtime exception may be a bad practice
513     //but seems to be an ingrained practice within Ant as even the base `Task`
514     //contains an `execute() throws BuildExecption`.
515     @SuppressWarnings("squid:RedundantThrowsDeclarationCheck")
516     private void dealWithReferences() throws BuildException {
517         if (isReference()) {
518             final Object o = refId.getReferencedObject(getProject());
519             if (!(o instanceof ResourceCollection)) {
520                 throw new BuildException("refId '" + refId.getRefId()
521                         + "' does not refer to a resource collection.");
522             }
523             getPath().add((ResourceCollection) o);
524         }
525     }
