64       * Test of mergeProperties method, of class Settings.
65       *
66       * @throws java.io.IOException thrown when the test fails
67       * @throws java.net.URISyntaxException thrown when the test fails
68       */
69      @Test
70      public void testMergeProperties_String() throws IOException, URISyntaxException {
71          String key = Settings.KEYS.PROXY_PORT;
72          String expResult = getSettings().getString(key);
73          File f = new File(this.getClass().getClassLoader().getResource("test.properties").toURI());
74          //InputStream in = this.getClass().getClassLoader().getResourceAsStream("test.properties");
75          getSettings().mergeProperties(f.getAbsolutePath());
76          String result = getSettings().getString(key);
77          Assert.assertTrue("setting didn't change?", (expResult == null && result != null) || !expResult.equals(result));
78      }

236     /**
237      * Assert {@link Settings#getArray(String)} from a delimited string returns
238      * multiple values in an array.
239      */
240     @Test
241     public void testGetArrayFromADelimitedString() {
242         // GIVEN a delimited string
243         final String delimitedString = "value1,value2";
244         getSettings().setString("key", delimitedString);
245 
246         // WHEN getting the array
247         final String[] array = getSettings().getArray("key");
248 
249         // THEN the split array is returned
250         assertThat("Expected the array to be non-null", array, notNullValue());
251         assertThat("Expected the array to have two values", array.length, is(2));
252         assertThat("Expected the first array value to be value1", array[0], is("value1"));
253         assertThat("Expected the second array value to be value2", array[1], is("value2"));
254     }
