18  package org.owasp.dependencycheck.utils;
19  
20  import java.io.Closeable;
21  import java.io.File;
22  import java.io.FileInputStream;
23  import java.io.FileNotFoundException;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.net.URL;
27  import java.util.UUID;
28  
29  import org.jetbrains.annotations.NotNull;
30  import org.jetbrains.annotations.Nullable;
31  
32  import org.apache.commons.io.FilenameUtils;
33  import org.apache.commons.lang3.StringUtils;
34  import org.apache.commons.lang3.SystemUtils;
35  
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  /**
40   * A collection of utilities for processing information about files.
41   *
42   * @author Jeremy Long
43   */
44  public final class FileUtils {
45  
46      /**
47       * The logger.
48       */
49      private static final Logger LOGGER = LoggerFactory.getLogger(FileUtils.class);
50      /**
51       * Bit bucket for non-Windows systems
52       */
53      private static final String BIT_BUCKET_UNIX = "/dev/null";
54  
55      /**
56       * Bit bucket for Windows systems (yes, only one 'L')
57       */
58      private static final String BIT_BUCKET_WIN = "NUL";
59  
60      /**
61       * Private constructor for a utility class.
62       */
63      private FileUtils() {
64      }
65  
66      /**
67       * Returns the (lowercase) file extension for a specified file.
68       *
69       * @param fileName the file name to retrieve the file extension from.
70       * @return the file extension.
71       */
72      @Nullable
73      public static String getFileExtension(@NotNull String fileName) {
74          @Nullable
75          final String fileExt = FilenameUtils.getExtension(fileName);
76          return StringUtils.isNoneEmpty(fileExt) ? StringUtils.lowerCase(fileExt) : null;
77      }
78  
79      /**
80       * Deletes a file. If the File is a directory it will recursively delete the
81       * contents.
82       *
83       * @param file the File to delete
84       * @return true if the file was deleted successfully, otherwise false
85       */
86      public static boolean delete(@Nullable File file) {
87          if (file == null) {
88              LOGGER.warn("cannot delete null File");
89              return false;
90          }
91  
92          try {
93              org.apache.commons.io.FileUtils.forceDelete(file);
94          } catch (IOException ex) {
95              LOGGER.trace(ex.getMessage(), ex);
96              LOGGER.debug("Failed to delete file: {} (error message: {}); attempting to delete on exit.", file.getPath(), ex.getMessage());
97              file.deleteOnExit();
98              return false;
99          }
100 
101         return true;
102     }
103 
104     /**
105      * Creates a unique temporary directory in the given directory.
106      *
107      * @param base the base directory to create a temporary directory within
108      * @return the temporary directory
109      * @throws java.io.IOException thrown when a directory cannot be created
110      * within the base directory
111      */
112     @NotNull
113     public static File createTempDirectory(@Nullable final File base) throws IOException {
114         final File tempDir = new File(base, "dctemp" + UUID.randomUUID());
115         if (tempDir.exists()) {
116             return createTempDirectory(base);
117         }
118         if (!tempDir.mkdirs()) {
119             throw new IOException("Could not create temp directory `" + tempDir.getAbsolutePath() + "`");
120         }
121         LOGGER.debug("Temporary directory is `{}`", tempDir.getAbsolutePath());
122         return tempDir;
123     }
124 
125     /**
126      * Return the bit bucket for the OS. '/dev/null' for Unix and 'NUL' for
127      * Windows
128      *
129      * @return a String containing the bit bucket
130      */
131     @NotNull
132     public static String getBitBucket() {
133         return SystemUtils.IS_OS_WINDOWS ? BIT_BUCKET_WIN : BIT_BUCKET_UNIX;
134     }
135 
136     /**
137      * Close the given {@link java.io.Closeable} instance, ignoring nulls, and
138      * logging any thrown {@link java.io.IOException}.
139      *
140      * @param closeable to be closed
141      */
142     public static void close(@Nullable final Closeable closeable) {
143         if (null != closeable) {
144             try {
145                 closeable.close();
146             } catch (IOException ex) {
147                 LOGGER.trace("", ex);
148             }
149         }
150     }
151 
152     /**
153      * Gets the {@link java.io.InputStream} for this resource.
154      *
155      * @param resource path
156      * @return the input stream for the given resource
157      * @throws FileNotFoundException if the file could not be found
158      */
159     @Nullable
160     public static InputStream getResourceAsStream(@NotNull String resource) throws FileNotFoundException {
161         final ClassLoader classLoader = FileUtils.class.getClassLoader();
162         final InputStream inputStream = classLoader != null
163                 ? classLoader.getResourceAsStream(resource)
164                 : ClassLoader.getSystemResourceAsStream(resource);
165 
166         if (inputStream == null) {
167             return new FileInputStream(resource);
168         }
169         return inputStream;
170     }
171 
172     /**
173      * Returns a File object for the given resource. The resource is attempted
174      * to be loaded from the class loader.
175      *
176      * @param resource path
177      * @return the file reference for the resource
178      */
179     public static File getResourceAsFile(final String resource) {
180         final ClassLoader classLoader = FileUtils.class.getClassLoader();
181         String path = null;
182         if (classLoader != null) {
183             final URL url = classLoader.getResource(resource);
184             if (url != null) {
185                 path = url.getFile();
186             }
187         } else {
188             path = ClassLoader.getSystemResource(resource).getFile();
189         }
190 
191         if (path == null) {
192             return new File(resource);
193         }
194         return new File(path);
195     }
196 }
