136     /**
137      * Initialize the Downloader from the settings.
138      * Extracts the configured proxy- and credential information from the settings and system properties and
139      * caches those for future use by the Downloader.
140      *
141      * @param settings The settings to configure from
142      * @throws InvalidSettingException When improper configurations are found.
143      */
144     public void configure(Settings settings) throws InvalidSettingException {
145         this.settings = settings;
146 
147         final SystemDefaultCredentialsProvider credentialsProvider = new SystemDefaultCredentialsProvider();
148         if (settings.getString(Settings.KEYS.PROXY_SERVER) != null) {
149             // Legacy proxy configuration present
150             // So don't rely on the system properties for proxy; use the legacy settings configuration
151             final String proxyHost = settings.getString(Settings.KEYS.PROXY_SERVER);
152             final int proxyPort = settings.getInt(Settings.KEYS.PROXY_PORT, -1);
153             httpClientBuilder.setProxy(new HttpHost(proxyHost, proxyPort));
154             if (settings.getString(Settings.KEYS.PROXY_USERNAME) != null) {
155                 final String proxyuser = settings.getString(Settings.KEYS.PROXY_USERNAME);
156                 final char[] proxypass = settings.getString(Settings.KEYS.PROXY_PASSWORD).toCharArray();
157                 credentialsProvider.setCredentials(
158                         new AuthScope(null, proxyHost, proxyPort, null, null),
159                         new UsernamePasswordCredentials(proxyuser, proxypass)
160                 );
161             }
162         }
163         tryAddRetireJSCredentials(settings, credentialsProvider);
164         tryAddHostedSuppressionCredentials(settings, credentialsProvider);
165         tryAddKEVCredentials(settings, credentialsProvider);
166         tryAddNexusAnalyzerCredentials(settings, credentialsProvider);
167         tryAddNVDApiDatafeed(settings, credentialsProvider);
168         httpClientBuilder.setDefaultCredentialsProvider(credentialsProvider);
169         httpClientBuilderExplicitNoproxy.setDefaultCredentialsProvider(credentialsProvider);
170     }

222     /**
223      * Add user/password credentials for the host/port of the URL, all configured in the settings, to the credential-store.
224      *
225      * @param settings The settings to retrieve the values from
226      * @param store The credentialStore
227      * @param userKey The key for a configured username credential part
228      * @param passwordKey The key for a configured password credential part
229      * @param urlKey The key for a configured url for which the credentials hold
230      * @param desc A descriptive text for use in error messages for this credential
231      * @throws InvalidSettingException When the password is empty or one of the other keys are not found in the settings.
232      */
233     private void addUserPasswordCreds(Settings settings, CredentialsStore store, String userKey, String urlKey, String passwordKey, String desc)
234             throws InvalidSettingException {
235         final String theUser = settings.getString(userKey);
236         final String theURL = settings.getString(urlKey);
237         final char[] thePass = settings.getString(passwordKey, "").toCharArray();
238         if (theUser == null || theURL == null || thePass.length == 0) {
239             throw new InvalidSettingException(desc + " URL and username are required when setting " + desc + " password");
240         }
241         try {
242             final URL parsedURL = new URL(theURL);
243             addCredentials(store, desc, parsedURL, theUser, thePass);
244         } catch (MalformedURLException e) {
245             throw new InvalidSettingException(desc + " URL must be a valid URL", e);
246         }
247     }
248 
249     private static void addCredentials(CredentialsStore credentialsStore, String messageScope, URL parsedURL, String theUser, char[] thePass)
250             throws InvalidSettingException {
251         final String theProtocol = parsedURL.getProtocol();
252         if ("file".equals(theProtocol)) {
253             LOGGER.warn("Credentials are not supported for file-protocol, double-check your configuration options for {}.", messageScope);
254             return;
255         } else if ("http".equals(theProtocol)) {
256             LOGGER.warn("Insecure configuration: Basic Credentials are configured to be used over a plain http connection for {}. "
257                     + "Consider migrating to https to guard the credentials.", messageScope);
258         } else if (!"https".equals(theProtocol)) {
259             throw new InvalidSettingException("Unsupported protocol in the " + messageScope
260                     + " URL; only file, http and https are supported");
261         }
262         final String theHost = parsedURL.getHost();
263         final int thePort = parsedURL.getPort();
264         final Credentials creds = new UsernamePasswordCredentials(theUser, thePass);
265         final AuthScope scope = new AuthScope(theProtocol, theHost, thePort, null, null);
266         credentialsStore.setCredentials(scope, creds);
267     }

284     /**
285      * Retrieves a file from a given URL and saves it to the outputPath.
286      *
287      * @param url        the URL of the file to download
288      * @param outputPath the path to the save the file to
289      * @param useProxy   whether to use the configured proxy when downloading
290      *                   files
291      * @throws DownloadFailedException       is thrown if there is an error downloading the file
292      * @throws URLConnectionFailureException is thrown when certificate-chain trust errors occur downloading the file
293      * @throws TooManyRequestsException      thrown when a 429 is received
294      * @throws ResourceNotFoundException     thrown when a 404 is received
295      */
296     public void fetchFile(URL url, File outputPath, boolean useProxy) throws DownloadFailedException,
297             TooManyRequestsException, ResourceNotFoundException, URLConnectionFailureException {
298         try {
299             if ("file".equals(url.getProtocol())) {
300                 final Path p = Paths.get(url.toURI());
301                 Files.copy(p, outputPath.toPath(), StandardCopyOption.REPLACE_EXISTING);
302             } else {
303                 final BasicClassicHttpRequest req;
304                 req = new BasicClassicHttpRequest(Method.GET, url.toURI());
305                 try (CloseableHttpClient hc = useProxy ? httpClientBuilder.build() : httpClientBuilderExplicitNoproxy.build()) {
306                     final SaveToFileResponseHandler responseHandler = new SaveToFileResponseHandler(outputPath);
307                     hc.execute(req, responseHandler);
308                 }
309             }
310         } catch (HttpResponseException hre) {
311             wrapAndThrowHttpResponseException(url.toString(), hre);
312         } catch (SSLHandshakeException ex) {
313             if (ex.getMessage().contains("unable to find valid certification path to requested target")) {
314                 final String msg = String.format("Unable to connect to '%s' - the Java trust store does not contain a trusted root for the cert. "
315                         + "Please see https://github.com/jeremylong/InstallCert for one method of updating the trusted certificates.", url);
316                 throw new URLConnectionFailureException(msg, ex);
317             }
318             final String msg = format("Download failed, unable to copy '%s' to '%s'; %s", url, outputPath.getAbsolutePath(), ex.getMessage());
319             throw new DownloadFailedException(msg, ex);
320         } catch (RuntimeException | URISyntaxException | IOException ex) {
321             final String msg = format("Download failed, unable to copy '%s' to '%s'; %s", url, outputPath.getAbsolutePath(), ex.getMessage());
322             throw new DownloadFailedException(msg, ex);
323         }
324     }

326     private static void wrapAndThrowHttpResponseException(String url, HttpResponseException hre)
327             throws ResourceNotFoundException, TooManyRequestsException, DownloadFailedException {
328         final String messageFormat = "%s - Server status: %d - Server reason: %s";
329         switch (hre.getStatusCode()) {
330             case 404:
331                 throw new ResourceNotFoundException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()), hre);
332             case 429:
333                 throw new TooManyRequestsException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()), hre);
334             default:
335                 throw new DownloadFailedException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()), hre);
336         }
337     }

339     /**
340      * Retrieves a file from a given URL using an ad-hoc created CredentialsProvider if needed
341      * and saves it to the outputPath.
342      *
343      * @param url         the URL of the file to download
344      * @param outputPath  the path to the save the file to
345      * @param useProxy    whether to use the configured proxy when downloading
346      *                    files
347      * @param userKey     the settings key for the username to be used
348      * @param passwordKey the settings key for the password to be used
349      * @throws DownloadFailedException       is thrown if there is an error downloading the file
350      * @throws URLConnectionFailureException is thrown when certificate-chain trust errors occur downloading the file
351      * @throws TooManyRequestsException      thrown when a 429 is received
352      * @throws ResourceNotFoundException     thrown when a 404 is received
353      * @implNote This method should only be used in cases where the target host cannot be determined beforehand from settings, so that ad-hoc
354      * Credentials needs to be constructed for the target URL when the user/password keys point to configured credentials. The method delegates to
355      * {@link #fetchFile(URL, File, boolean)} when credentials are not configured for the given keys or the resource points to a file.
356      */
357     public void fetchFile(URL url, File outputPath, boolean useProxy, String userKey, String passwordKey) throws DownloadFailedException,
358             TooManyRequestsException, ResourceNotFoundException, URLConnectionFailureException {
359         if ("file".equals(url.getProtocol())
360                 || userKey == null || settings.getString(userKey) == null
361                 || passwordKey == null || settings.getString(passwordKey) == null
362         ) {
363             // no credentials configured, so use the default fetchFile
364             fetchFile(url, outputPath, useProxy);
365             return;
366         }
367         final String theProtocol = url.getProtocol();
368         if (!("http".equals(theProtocol) || "https".equals(theProtocol))) {
369             throw new DownloadFailedException("Unsupported protocol in the URL; only file, http and https are supported");
370         }
371         try {
372             final HttpClientContext context = HttpClientContext.create();
373             final BasicCredentialsProvider localCredentials = new BasicCredentialsProvider();
374             addCredentials(localCredentials, url.toString(), url, settings.getString(userKey), settings.getString(passwordKey).toCharArray());
375             context.setCredentialsProvider(localCredentials);
376             try (CloseableHttpClient hc = useProxy ? httpClientBuilder.build() : httpClientBuilderExplicitNoproxy.build()) {
377                 final BasicClassicHttpRequest req = new BasicClassicHttpRequest(Method.GET, url.toURI());
378                 final SaveToFileResponseHandler responseHandler = new SaveToFileResponseHandler(outputPath);
379                 hc.execute(req, context, responseHandler);
380             }
381         } catch (HttpResponseException hre) {
382             wrapAndThrowHttpResponseException(url.toString(), hre);
383         } catch (SSLHandshakeException ex) {
384             if (ex.getMessage().contains("unable to find valid certification path to requested target")) {
385                 final String msg = String.format("Unable to connect to '%s' - the Java trust store does not contain a trusted root for the cert. "
386                         + "Please see https://github.com/jeremylong/InstallCert for one method of updating the trusted certificates.", url);
387                 throw new URLConnectionFailureException(msg, ex);
388             }
389             final String msg = format("Download failed, unable to copy '%s' to '%s'; %s", url, outputPath.getAbsolutePath(), ex.getMessage());
390             throw new DownloadFailedException(msg, ex);
391         } catch (RuntimeException | URISyntaxException | IOException ex) {
392             final String msg = format("Download failed, unable to copy '%s' to '%s'; %s", url, outputPath.getAbsolutePath(), ex.getMessage());
393             throw new DownloadFailedException(msg, ex);
394         }
395     }

397     /**
398      * Posts a payload to the URL and returns the response as a string.
399      *
400      * @param url         the URL to POST to
401      * @param payload     the Payload to post
402      * @param payloadType the string describing the payload's mime-type
403      * @param hdr         Additional headers to add to the HTTP request
404      * @return the content of the response
405      * @throws DownloadFailedException       is thrown if there is an error downloading the file
406      * @throws URLConnectionFailureException is thrown when certificate-chain trust errors occur downloading the file
407      * @throws TooManyRequestsException      thrown when a 429 is received
408      * @throws ResourceNotFoundException     thrown when a 404 is received
409      */
410     public String postBasedFetchContent(URI url, String payload, ContentType payloadType, List<Header> hdr)
411             throws DownloadFailedException, TooManyRequestsException, ResourceNotFoundException, URLConnectionFailureException {
412         try {
413             if (url.getScheme() == null || !url.getScheme().toLowerCase(Locale.ROOT).matches("^https?")) {
414                 throw new IllegalArgumentException("Unsupported protocol in the URL; only http and https are supported");
415             } else {
416                 final BasicClassicHttpRequest req;
417                 req = new BasicClassicHttpRequest(Method.POST, url);
418                 req.setEntity(new StringEntity(payload, payloadType));
419                 for (Header h : hdr) {
420                     req.addHeader(h);
421                 }
422                 final String result;
423                 try (CloseableHttpClient hc = httpClientBuilder.build()) {
424                     result = hc.execute(req, new BasicHttpClientResponseHandler());
425                 }
426                 return result;
427             }
428         } catch (HttpResponseException hre) {
429             wrapAndThrowHttpResponseException(url.toString(), hre);
430             throw new InternalError("wrapAndThrowHttpResponseException will always throw an exception but Java compiler fails to spot it");
431         } catch (SSLHandshakeException ex) {
432             if (ex.getMessage().contains("unable to find valid certification path to requested target")) {
433                 final String msg = String.format("Unable to connect to '%s' - the Java trust store does not contain a trusted root for the cert. "
434                         + "Please see https://github.com/jeremylong/InstallCert for one method of updating the trusted certificates.", url);
435                 throw new URLConnectionFailureException(msg, ex);
436             }
437             final String msg = format("Download failed, error downloading '%s'; %s", url, ex.getMessage());
438             throw new DownloadFailedException(msg, ex);
439         } catch (IOException | RuntimeException ex) {
440             final String msg = format("Download failed, error downloading '%s'; %s", url, ex.getMessage());
441             throw new DownloadFailedException(msg, ex);
442         }
443     }

460     /**
461      * Retrieves a file from a given URL and returns the contents.
462      *
463      * @param url      the URL of the file to download
464      * @param useProxy whether to use the configured proxy when downloading
465      *                 files
466      * @param charset  The characterset to use to interpret the binary content of the file
467      * @return the content of the file
468      * @throws DownloadFailedException   is thrown if there is an error
469      *                                   downloading the file
470      * @throws TooManyRequestsException  thrown when a 429 is received
471      * @throws ResourceNotFoundException thrown when a 404 is received
472      */
473     public String fetchContent(URL url, boolean useProxy, Charset charset)
474             throws DownloadFailedException, TooManyRequestsException, ResourceNotFoundException {
475         try {
476             final String result;
477             if ("file".equals(url.getProtocol())) {
478                 final Path p = Paths.get(url.toURI());
479                 result = Files.readString(p, charset);
480             } else {
481                 final BasicClassicHttpRequest req;
482                 req = new BasicClassicHttpRequest(Method.GET, url.toURI());
483                 try (CloseableHttpClient hc = useProxy ? httpClientBuilder.build() : httpClientBuilderExplicitNoproxy.build()) {
484                     final ExplicitEncodingToStringResponseHandler responseHandler = new ExplicitEncodingToStringResponseHandler(charset);
485                     result = hc.execute(req, responseHandler);
486                 }
487             }
488             return result;
489         } catch (HttpResponseException hre) {
490             wrapAndThrowHttpResponseException(url.toString(), hre);
491             throw new InternalError("wrapAndThrowHttpResponseException will always throw an exception but Java compiler fails to spot it");
492         } catch (RuntimeException | URISyntaxException | IOException ex) {
493             final String msg = format("Download failed, error downloading '%s'; %s", url, ex.getMessage());
494             throw new DownloadFailedException(msg, ex);
495         }
496     }

542      * @throws ResourceNotFoundException When HTTP status 404 is encountered
543      */
544     public <T> T fetchAndHandle(@NotNull URL url, @NotNull HttpClientResponseHandler<T> handler, @NotNull List<Header> hdr, boolean useProxy)
545             throws IOException, TooManyRequestsException, ResourceNotFoundException {
546         try {
547             final T data;
548             if ("file".equals(url.getProtocol())) {
549                 final Path p = Paths.get(url.toURI());
550                 try (InputStream is = Files.newInputStream(p)) {
551                     final HttpEntity dummyEntity = new BasicHttpEntity(is, ContentType.APPLICATION_JSON);
552                     final ClassicHttpResponse dummyResponse = new BasicClassicHttpResponse(200);
553                     dummyResponse.setEntity(dummyEntity);
554                     data = handler.handleResponse(dummyResponse);
555                 } catch (HttpException e) {
556                     throw new IllegalStateException("HttpException encountered without HTTP traffic", e);
557                 }
558             } else {
559                 final String theProtocol = url.getProtocol();
560                 if (!("http".equals(theProtocol) || "https".equals(theProtocol))) {
561                     throw new DownloadFailedException("Unsupported protocol in the URL; only file, http and https are supported");
562                 }
563                 try (CloseableHttpClient hc = useProxy ? httpClientBuilder.build() : httpClientBuilderExplicitNoproxy.build()) {
564                     final BasicClassicHttpRequest req = new BasicClassicHttpRequest(Method.GET, url.toURI());
565                     for (Header h : hdr) {
566                         req.addHeader(h);
567                     }
568                     data = hc.execute(req, handler);
569                 }
570             }
571             return data;
572         } catch (HttpResponseException hre) {
573             final String messageFormat = "%s - Server status: %d - Server reason: %s";
574             switch (hre.getStatusCode()) {
575                 case 404:
576                     throw new ResourceNotFoundException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()));
577                 case 429:
578                     throw new TooManyRequestsException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()));
579                 default:
580                     throw new IOException(String.format(messageFormat, url, hre.getStatusCode(), hre.getReasonPhrase()));
581             }
582         } catch (RuntimeException | URISyntaxException ex) {
583             final String msg = format("Download failed, unable to retrieve and parse '%s'; %s", url, ex.getMessage());
584             throw new IOException(msg, ex);
585         }
586     }
