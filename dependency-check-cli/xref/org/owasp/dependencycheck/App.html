107     /**
108      * Main CLI entry-point into the application.
109      *
110      * @param args the command line arguments
111      * @return the exit code to return
112      */
113     public int run(String[] args) {
114         int exitCode = 0;
115         final CliParser cli = new CliParser(settings);
116 
117         try {
118             cli.parse(args);
119         } catch (FileNotFoundException ex) {
120             System.err.println(ex.getMessage());
121             cli.printHelp();
122             return -1;
123         } catch (ParseException ex) {
124             System.err.println(ex.getMessage());
125             cli.printHelp();
126             return -2;
127         }
128         final String verboseLog = cli.getStringArgument(CliParser.ARGUMENT.VERBOSE_LOG);
129         if (verboseLog != null) {
130             prepareLogger(verboseLog);
131         }
132 
133         if (cli.isPurge()) {
134             final String connStr = cli.getStringArgument(CliParser.ARGUMENT.CONNECTION_STRING);
135             if (connStr != null) {
136                 LOGGER.error("Unable to purge the database when using a non-default connection string");
137                 exitCode = -3;
138             } else {
139                 try {
140                     populateSettings(cli);
141                 } catch (InvalidSettingException ex) {
142                     LOGGER.error(ex.getMessage());
143                     LOGGER.debug(ERROR_LOADING_PROPERTIES_FILE, ex);
144                     exitCode = -4;
145                     return exitCode;
146                 }
147                 try (Engine engine = new Engine(Engine.Mode.EVIDENCE_PROCESSING, settings)) {
148                     if (!engine.purge()) {
149                         exitCode = -7;
150                         return exitCode;
151                     }
152                 } finally {
153                     settings.cleanup();
154                 }
155             }
156         } else if (cli.isGetVersion()) {
157             cli.printVersionInfo();
158         } else if (cli.isUpdateOnly()) {
159             try {
160                 populateSettings(cli);
161                 settings.setBoolean(Settings.KEYS.AUTO_UPDATE, true);
162             } catch (InvalidSettingException ex) {
163                 LOGGER.error(ex.getMessage());
164                 LOGGER.debug(ERROR_LOADING_PROPERTIES_FILE, ex);
165                 exitCode = -4;
166                 return exitCode;
167             }
168             try {
169                 runUpdateOnly();
170             } catch (UpdateException ex) {
171                 LOGGER.error(ex.getMessage(), ex);
172                 exitCode = -8;
173             } catch (DatabaseException ex) {
174                 LOGGER.error(ex.getMessage(), ex);
175                 exitCode = -9;
176             } finally {
177                 settings.cleanup();
178             }
179         } else if (cli.isRunScan()) {
180             try {
181                 populateSettings(cli);
182             } catch (InvalidSettingException ex) {
183                 LOGGER.error(ex.getMessage(), ex);
184                 LOGGER.debug(ERROR_LOADING_PROPERTIES_FILE, ex);
185                 exitCode = -4;
186                 return exitCode;
187             }
188             try {
189                 final String[] scanFiles = cli.getScanFiles();
190                 if (scanFiles != null) {
191                     exitCode = runScan(cli.getReportDirectory(), cli.getReportFormat(), cli.getProjectName(), scanFiles,
192                             cli.getExcludeList(), cli.getSymLinkDepth(), cli.getFailOnCVSS());
193                 } else {
194                     LOGGER.error("No scan files configured");
195                 }
196             } catch (DatabaseException ex) {
197                 LOGGER.error(ex.getMessage());
198                 LOGGER.debug("database exception", ex);
199                 exitCode = -11;
200             } catch (ReportException ex) {
201                 LOGGER.error(ex.getMessage());
202                 LOGGER.debug("report exception", ex);
203                 exitCode = -12;
204             } catch (ExceptionCollection ex) {
205                 if (ex.isFatal()) {
206                     exitCode = -13;
207                     LOGGER.error("One or more fatal errors occurred");
208                 } else {
209                     exitCode = -14;
210                 }
211                 for (Throwable e : ex.getExceptions()) {
212                     if (e.getMessage() != null) {
213                         LOGGER.error(e.getMessage());
214                         LOGGER.debug("unexpected error", e);
215                     }
216                 }
217             } finally {
218                 settings.cleanup();
219             }
220         } else {
221             cli.printHelp();
222         }
223         return exitCode;
224     }

226     /**
227      * Scans the specified directories and writes the dependency reports to the
228      * reportDirectory.
229      *
230      * @param reportDirectory the path to the directory where the reports will
231      * be written
232      * @param outputFormats String[] of output formats of the report
233      * @param applicationName the application name for the report
234      * @param files the files/directories to scan
235      * @param excludes the patterns for files/directories to exclude
236      * @param symLinkDepth the depth that symbolic links will be followed
237      * @param cvssFailScore the score to fail on if a vulnerability is found
238      * @return the exit code if there was an error
239      * @throws ReportException thrown when the report cannot be generated
240      * @throws DatabaseException thrown when there is an error connecting to the
241      * database
242      * @throws ExceptionCollection thrown when an exception occurs during
243      * analysis; there may be multiple exceptions contained within the
244      * collection.
245      */
246     private int runScan(String reportDirectory, String[] outputFormats, String applicationName, String[] files,
247             String[] excludes, int symLinkDepth, float cvssFailScore) throws DatabaseException,
248             ExceptionCollection, ReportException {
249         Engine engine = null;
250         try {
251             final List<String> antStylePaths = getPaths(files);
252             final Set<File> paths = scanAntStylePaths(antStylePaths, symLinkDepth, excludes);
253 
254             engine = new Engine(settings);
255             engine.scan(paths);
256 
257             ExceptionCollection exCol = null;
258             try {
259                 engine.analyzeDependencies();
260             } catch (ExceptionCollection ex) {
261                 if (ex.isFatal()) {
262                     throw ex;
263                 }
264                 exCol = ex;
265             }
266 
267             try {
268                 for (String outputFormat : outputFormats) {
269                     engine.writeReports(applicationName, new File(reportDirectory), outputFormat, exCol);
270                 }
271             } catch (ReportException ex) {
272                 if (exCol != null) {
273                     exCol.addException(ex);
274                     throw exCol;
275                 } else {
276                     throw ex;
277                 }
278             }
279             if (exCol != null && !exCol.getExceptions().isEmpty()) {
280                 throw exCol;
281             }
282             return determineReturnCode(engine, cvssFailScore);
283         } finally {
284             if (engine != null) {
285                 engine.close();
286             }
287         }
288     }

290     /**
291      * Determines the return code based on if one of the dependencies scanned
292      * has a vulnerability with a CVSS score above the cvssFailScore.
293      *
294      * @param engine the engine used during analysis
295      * @param cvssFailScore the max allowed CVSS score
296      * @return returns <code>1</code> if a severe enough vulnerability is
297      * identified; otherwise <code>0</code>
298      */
299     private int determineReturnCode(Engine engine, float cvssFailScore) {
300         int retCode = 0;
301         //Set the exit code based on whether we found a high enough vulnerability
302         final StringBuilder ids = new StringBuilder();
303         for (Dependency d : engine.getDependencies()) {
304             boolean addName = true;
305             for (Vulnerability v : d.getVulnerabilities()) {
306                 final float cvssV2 = v.getCvssV2() != null ? v.getCvssV2().getScore() : -1;
307                 final float cvssV3 = v.getCvssV3() != null ? v.getCvssV3().getBaseScore() : -1;
308                 final float unscoredCvss = v.getUnscoredSeverity() != null ? SeverityUtil.estimateCvssV2(v.getUnscoredSeverity()) : -1;
309 
310                 if (cvssV2 >= cvssFailScore
311                         || cvssV3 >= cvssFailScore
312                         || unscoredCvss >= cvssFailScore
313                         //safety net to fail on any if for some reason the above misses on 0
314                         || (cvssFailScore <= 0.0f)) {
315                     float score = 0.0f;
316                     if (cvssV3 >= 0.0f) {
317                         score = cvssV3;
318                     } else if (cvssV2 >= 0.0f) {
319                         score = cvssV2;
320                     } else if (unscoredCvss >= 0.0f) {
321                         score = unscoredCvss;
322                     }
323                     if (addName) {
324                         addName = false;
325                         ids.append(NEW_LINE).append(d.getFileName()).append(": ");
326                         ids.append(v.getName()).append('(').append(score).append(')');
327                     } else {
328                         ids.append(", ").append(v.getName()).append('(').append(score).append(')');
329                     }
330                 }
331             }
332         }
333         if (ids.length() > 0) {
334             LOGGER.error(
335                     String.format("%n%nOne or more dependencies were identified with vulnerabilities that have a CVSS score greater than or "
336                             + "equal to '%.1f': %n%s%n%nSee the dependency-check report for more details.%n%n", cvssFailScore, ids)
337             );
338 
339             retCode = 1;
340         }
341 
342         return retCode;
343     }

345     /**
346      * Scans the give Ant Style paths and collects the actual files.
347      *
348      * @param antStylePaths a list of ant style paths to scan for actual files
349      * @param symLinkDepth the depth to traverse symbolic links
350      * @param excludes an array of ant style excludes
351      * @return returns the set of identified files
352      */
353     private Set<File> scanAntStylePaths(List<String> antStylePaths, int symLinkDepth, String[] excludes) {
354         final Set<File> paths = new TreeSet<>();
355         for (String file : antStylePaths) {
356             LOGGER.debug("Scanning {}", file);
357             final DirectoryScanner scanner = new DirectoryScanner();
358             String include = file.replace('\\', '/');
359             final File baseDir;
360             final int pos = getLastFileSeparator(include);
361             final String tmpBase = include.substring(0, pos);
362             final String tmpInclude = include.substring(pos + 1);
363             if (tmpInclude.indexOf('*') >= 0 || tmpInclude.indexOf('?') >= 0
364                     || new File(include).isFile()) {
365                 baseDir = new File(tmpBase);
366                 include = tmpInclude;
367             } else {
368                 baseDir = new File(tmpBase, tmpInclude);
369                 include = "**/*";
370             }
371             LOGGER.debug("BaseDir: " + baseDir);
372             LOGGER.debug("Include: " + include);
373             scanner.setBasedir(baseDir);
374             final String[] includes = {include};
375             scanner.setIncludes(includes);
376             scanner.setMaxLevelsOfSymlinks(symLinkDepth);
377             if (symLinkDepth <= 0) {
378                 scanner.setFollowSymlinks(false);
379             }
380             if (excludes != null && excludes.length > 0) {
381                 for (String e : excludes) {
382                     LOGGER.debug("Exclude: " + e);
383                 }
384                 scanner.addExcludes(excludes);
385             }
386             scanner.scan();
387             if (scanner.getIncludedFilesCount() > 0) {
388                 for (String s : scanner.getIncludedFiles()) {
389                     final File f = new File(baseDir, s);
390                     LOGGER.debug("Found file {}", f);
391                     paths.add(f);
392                 }
393             }
394         }
395         return paths;
396     }

398     /**
399      * Determines the ant style paths from the given array of files.
400      *
401      * @param files an array of file paths
402      * @return a list containing ant style paths
403      */
404     private List<String> getPaths(String[] files) {
405         final List<String> antStylePaths = new ArrayList<>();
406         for (String file : files) {
407             final String antPath = ensureCanonicalPath(file);
408             antStylePaths.add(antPath);
409         }
410         return antStylePaths;
411     }

438             try {
439                 settings.mergeProperties(propertiesFile);

441                 throw new InvalidSettingException("Unable to find properties file '" + propertiesFile.getPath() + "'", ex);
442             } catch (IOException ex) {
443                 throw new InvalidSettingException("Error reading properties file '" + propertiesFile.getPath() + "'", ex);
444             }
445         }
446         final String dataDirectory = cli.getStringArgument(CliParser.ARGUMENT.DATA_DIRECTORY);
447         if (dataDirectory != null) {
448             settings.setString(Settings.KEYS.DATA_DIRECTORY, dataDirectory);
449         } else if (System.getProperty("basedir") != null) {

658     /**
659      * Creates a file appender and adds it to logback.
660      *
661      * @param verboseLog the path to the verbose log file
662      */
663     private void prepareLogger(String verboseLog) {
664         final LoggerContext context = (LoggerContext) LoggerFactory.getILoggerFactory();
665         final PatternLayoutEncoder encoder = new PatternLayoutEncoder();
666         encoder.setPattern("%d %C:%L%n%-5level - %msg%n");
667         encoder.setContext(context);
668         encoder.start();
669         final FileAppender<ILoggingEvent> fa = new FileAppender<>();
670         fa.setAppend(true);
671         fa.setEncoder(encoder);
672         fa.setContext(context);
673         fa.setFile(verboseLog);
674         final File f = new File(verboseLog);
675         String name = f.getName();
676         final int i = name.lastIndexOf('.');
677         if (i > 1) {
678             name = name.substring(0, i);
679         }
680         fa.setName(name);
681         fa.start();
682         final ch.qos.logback.classic.Logger rootLogger = context.getLogger(ch.qos.logback.classic.Logger.ROOT_LOGGER_NAME);
683         rootLogger.setLevel(Level.DEBUG);
684         final ThresholdFilter filter = new ThresholdFilter();
685         filter.setLevel(LogLevel.INFO.getValue());
686         filter.setContext(context);
687         filter.start();
688         rootLogger.iteratorForAppenders().forEachRemaining(action -> action.addFilter(filter));
689         rootLogger.addAppender(fa);
690     }

692     /**
693      * Takes a path and resolves it to be a canonical &amp; absolute path. The
694      * caveats are that this method will take an Ant style file selector path
695      * (../someDir/**\/*.jar) and convert it to an absolute/canonical path (at
696      * least to the left of the first * or ?).
697      *
698      * @param path the path to canonicalize
699      * @return the canonical path
700      */
701     protected String ensureCanonicalPath(String path) {
702         final String basePath;
703         String wildCards = null;
704         final String file = path.replace('\\', '/');
705         if (file.contains("*") || file.contains("?")) {
706 
707             int pos = getLastFileSeparator(file);
708             if (pos < 0) {
709                 return file;
710             }
711             pos += 1;
712             basePath = file.substring(0, pos);
713             wildCards = file.substring(pos);
714         } else {
715             basePath = file;
716         }
717 
718         File f = new File(basePath);
719         try {
720             f = f.getCanonicalFile();
721             if (wildCards != null) {
722                 f = new File(f, wildCards);
723             }
724         } catch (IOException ex) {
725             LOGGER.warn("Invalid path '{}' was provided.", path);
726             LOGGER.debug("Invalid path provided", ex);
727         }
728         return f.getAbsolutePath().replace('\\', '/');
729     }

731     /**
732      * Returns the position of the last file separator.
733      *
734      * @param file a file path
735      * @return the position of the last file separator
736      */
737     @SuppressWarnings("ManualMinMaxCalculation")
738     private int getLastFileSeparator(String file) {
739         if (file.contains("*") || file.contains("?")) {
740             int p1 = file.indexOf('*');
741             int p2 = file.indexOf('?');
742             p1 = p1 > 0 ? p1 : file.length();
743             p2 = p2 > 0 ? p2 : file.length();
744             int pos = p1 < p2 ? p1 : p2;
745             pos = file.lastIndexOf('/', pos);
746             return pos;
747         } else {
748             return file.lastIndexOf('/');
749         }
750     }
